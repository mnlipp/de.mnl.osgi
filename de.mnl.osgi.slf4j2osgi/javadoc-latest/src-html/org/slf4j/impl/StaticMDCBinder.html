026package org.slf4j.impl;
027
028import org.slf4j.helpers.NOPMDCAdapter;
029import org.slf4j.spi.MDCAdapter;
030
031/**
032 * This implementation is bound to {@link NOPMDCAdapter}.
033 *
034 * @author Ceki G&uuml;lc&uuml;
035 */
036
037public final class StaticMDCBinder {
038
039    /**
040     * The unique instance of this class.
041     */
042    public static final StaticMDCBinder SINGLETON = new StaticMDCBinder();
043
044    private StaticMDCBinder() {
045    }
046
047    /**
048     * Return the singleton of this class.
049     * 
050     * @return the StaticMDCBinder singleton
051     * @since 1.7.14
052     */
053    public static StaticMDCBinder getSingleton() {
054        return SINGLETON;
055    }
056
057    /**
058     * Currently this method always returns an instance of 
059     * {@link StaticMDCBinder}.
060     *
061     * @return the result
062     */
063    public MDCAdapter getMDCA() {
064        return new NOPMDCAdapter();
065    }
066
067    /**
068     * Gets the MDC adapter class as String
069     *
070     * @return the MDC adapter class name
071     */
072    public String getMDCAdapterClassStr() {
073        return NOPMDCAdapter.class.getName();
074    }
075}
