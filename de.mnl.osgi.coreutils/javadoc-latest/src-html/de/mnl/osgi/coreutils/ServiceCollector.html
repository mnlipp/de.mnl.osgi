047/**
048 * Maintains a collection of services matching some criteria.
049 * <P>
050 * An instance can be created with different criteria for matching
051 * services of a given type. While the instance is open, it 
052 * maintains a collection of all matching services provided
053 * by the framework. Changes of the collection are reported using
054 * the registered handlers.
055 * <P>
056 * Because OSGi is a threaded environment, the registered services 
057 * can vary any time. Results from queries on the collection are 
058 * therefore only reliable while synchronizing on the 
059 * {@code ServiceCollector} instance. The synchronization puts 
060 * other threads that attempt to change the collection on hold. 
061 * Note that this implies a certain risk of creating deadlocks.
062 * Also note that this does not prevent modifications by the
063 * holder of the lock.
064 * <P>
065 * Callbacks are always synchronized on this collector, else
066 * the state of the collector during the execution of the
067 * callback might no longer be the state that triggered
068 * the callback.
069 * <P>
070 * Instead of using the services from the framework directly,
071 * they may optionally be adapted to another type by setting a
072 * service getter function. The invocation of this function is 
073 * also synchronized on this collector.
074 *
075 * @param <S> the type of the service
076 * @param <T> the type of service instances returned by queries, 
077 *      usually the same type as the type of the service
078 *      (see {@link #setServiceGetter(BiFunction)})
079 */
080@SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.GodClass",
081    "PMD.TooManyFields", "PMD.DataflowAnomalyAnalysis" })
082public class ServiceCollector<S, T> implements AutoCloseable {

343    private void modified() {
344        synchronized (modificationCount) {
345            modificationCount[0] = modificationCount[0] + 1;
346            cachedService = null;
347            modificationCount.notifyAll();
348        }

364    /**
365     * Starts collecting services. Short for calling {@code open(false)}.
366     * 
367     * @param collectAllServices if <code>true</code>, then this 
368     *     {@code ServiceCollector} will collect all matching services 
369     *     regardless of class loader
370     *     accessibility. If <code>false</code>, then 
371     *     this {@code ServiceCollector} will only collect matching services 
372     *     which are class loader
373     *     accessible to the bundle whose <code>BundleContext</code> is 
374     *     used by this {@code ServiceCollector}.
375     * @throws IllegalStateException If the {@code BundleConetxt}
376     *     with which this {@code ServiceCollector} was created is no 
377     *     longer valid.
378     */
379    @SuppressWarnings({ "PMD.AvoidUncheckedExceptionsInSignatures",
380        "PMD.ConfusingTernary", "PMD.AvoidThrowingRawExceptionTypes" })
381    public void open(boolean collectAllServices) throws IllegalStateException {
382        synchronized (this) {
383            if (isOpen()) {
384                // Already open, don't treat as error.
385                return;
386            }
387            modificationCount[0] = 0;
388            try {
389                registerListener(collectAllServices);
390                if (collectClass != null) {
391                    initialReferences.addAll(getInitialReferences(
392                        collectAllServices, collectClass, null));
393                } else if (collectReference != null) {
394                    if (collectReference.getBundle() != null) {
395                        initialReferences.add(collectReference);
396                    }
397                } else { /* user supplied filter */
398                    initialReferences.addAll(getInitialReferences(
399                        collectAllServices, null, listenerFilter));
400                }
401                processInitial();
402            } catch (InvalidSyntaxException e) {
403                throw new RuntimeException(
404                    "unexpected InvalidSyntaxException: " + e.getMessage(),
405                    e);
406            }
407        }
408    }
409
410    private void registerListener(boolean collectAllServices)
411            throws InvalidSyntaxException {
412        if (collectAllServices) {
413            listener = new AllServiceListener() {
414                @Override
415                public void serviceChanged(ServiceEvent event) {
416                    ServiceCollector.this.serviceChanged(event);
417                }
418            };
419        } else {
420            listener = new ServiceListener() {
421
422                @Override
423                public void serviceChanged(ServiceEvent event) {
424                    ServiceCollector.this.serviceChanged(event);
425                }
426
427            };
428        }
429        context.addServiceListener(listener, listenerFilter);
430    }

432    /**
433     * Returns the list of initial {@code ServiceReference}s that will be
434     * collected by this {@code ServiceCollector}.
435     * 
436     * @param collectAllServices If {@code true}, use
437     *        {@code getAllServiceReferences}.
438     * @param className The class name with which the service was registered, or
439     *        {@code null} for all services.
440     * @param filterString The filter criteria or {@code null} for all services.
441     * @return The list of initial {@code ServiceReference}s.
442     * @throws InvalidSyntaxException If the specified filterString has an
443     *         invalid syntax.
444     */
445    private List<ServiceReference<S>> getInitialReferences(
446            boolean collectAllServices, String className, String filterString)
447            throws InvalidSyntaxException {
448        @SuppressWarnings("unchecked")
449        ServiceReference<S>[] result
450            = (ServiceReference<S>[]) (collectAllServices
451                ? context.getAllServiceReferences(className, filterString)
452                : context.getServiceReferences(className, filterString));
453        if (result == null) {
454            return Collections.emptyList();
455        }
456        return Arrays.asList(result);
457    }

459    private void processInitial() {
460        while (true) {
461            // Process one by one so that we do not keep holding the lock.
462            synchronized (this) {
463                if (!isOpen() || initialReferences.isEmpty()) {
464                    return; /* we are done */
465                }
466                // Get one...
467                Iterator<ServiceReference<S>> iter
468                    = initialReferences.iterator();
469                ServiceReference<S> reference = iter.next();
470                iter.remove();
471                // Process (as if it had been registered).
472                addToCollected(reference);
473            }
474        }
475    }

477    private void serviceChanged(ServiceEvent event) {
478        /*
479         * Check if we had a delayed call (which could happen when we
480         * close).
481         */
482        if (!isOpen()) {
483            return;
484        }
485        @SuppressWarnings("unchecked")
486        final ServiceReference<S> reference
487            = (ServiceReference<S>) event.getServiceReference();
488
489        switch (event.getType()) {
490        case ServiceEvent.REGISTERED:
491            synchronized (this) {
492                initialReferences.remove(reference);
493                addToCollected(reference);
494            }
495            break;
496        case ServiceEvent.MODIFIED:
497            synchronized (this) {
498                T service = collected.get(reference);
499                if (service == null) {
500                    // Probably still in initialReferences, ignore.
501                    return;
502                }
503                modified();
504                if (onModified != null
505                    && reference.equals(collected.firstKey())) {
506                    onModified.accept(reference, service);
507                }
508            }
509            break;
510        case ServiceEvent.MODIFIED_ENDMATCH:
511        case ServiceEvent.UNREGISTERING:
512            synchronized (this) {
513                // May occur while processing the initial set of references.
514                initialReferences.remove(reference);
515                removeFromCollected(reference);
516            }
517            break;
518        default:
519            break;
520        }
521    }

523    /**
524     * Add the given reference to the collected services.
525     * Must be called from synchronized block.
526     * 
527     * @param reference reference to be collected.
528     */
529    private void addToCollected(final ServiceReference<S> reference) {
530        if (!isOpen()) {
531            // We have been closed.
532            return;
533        }
534        if (collected.get(reference) != null) {
535            /* if we are already collecting this reference */
536            return; /* skip this reference */
537        }
538
539        @SuppressWarnings("unchecked")
540        T service = (svcGetter == null)
541            ? (T) context.getService(reference)
542            : svcGetter.apply(context, reference);
543        if (service == null) {
544            // Has either vanished in the meantime (should not happen
545            // when processing a REGISTERED event, but may happen when
546            // processing a reference from initialReferences) or was
547            // filtered by the service getter.
548            return;
549        }
550        boolean wasEmpty = collected.isEmpty();
551        collected.put(reference, service);
552        modified();
553        if (wasEmpty) {
554            Optional.ofNullable(onBound)
555                .ifPresent(cb -> cb.accept(reference, service));
556        }
557        Optional.ofNullable(onAdded)
558            .ifPresent(cb -> cb.accept(reference, service));
559        // If added is first, first has changed
560        if (onModified != null && collected.size() > 1
561            && collected.firstKey().equals(reference)) {
562            onModified.accept(reference, service);
563        }
564    }

566    /**
567     * Removes a service reference from the collection.
568     * Must be called from synchronized block.
569     *
570     * @param reference the reference
571     */
572    @SuppressWarnings("PMD.AvoidLiteralsInIfCondition")
573    private void removeFromCollected(ServiceReference<S> reference) {
574        synchronized (this) {
575            // Only proceed if collected
576            T service = collected.get(reference);
577            if (service == null) {
578                return;
579            }
580            Optional.ofNullable(onRemoving)
581                .ifPresent(cb -> cb.accept(reference, service));
582            if (collected.size() == 1) {
583                Optional.ofNullable(onUnbinding)
584                    .ifPresent(cb -> cb.accept(reference, service));
585            }
586            context.ungetService(reference);
587            boolean firstChanges = reference.equals(collected.firstKey());
588            collected.remove(reference);
589            if (isOpen()) {
590                modified();
591            }
592            // Has first changed?
593            if (onModified != null && firstChanges && !collected.isEmpty()) {
594                onModified.accept(reference, service);
595            }
596        }
597
598    }

600    /**
601     * Stops collecting services.
602     */
603    public void close() {
604        synchronized (this) {
605            context.removeServiceListener(listener);
606            synchronized (modificationCount) {
607                modificationCount[0] = -1;
608                modificationCount.notifyAll();
609            }
610        }
611        while (!collected.isEmpty()) {
612            synchronized (this) {
613                removeFromCollected(collected.lastKey());
614            }
615        }
616        cachedService = null;
617    }

619    /**
620     * Wait for at least one service to be collected by this
621     * {@code ServiceCollector}. This method will also return when this
622     * {@code ServiceCollector} is closed from another thread.
623     * <p>
624     * It is strongly recommended that {@code waitForService} is not used 
625     * during the calling of the {@code BundleActivator} methods.
626     * {@code BundleActivator} methods are expected to complete in a short
627     * period of time.
628     * 
629     * @param timeout The time interval in milliseconds to wait. If zero, the
630     *     method will wait indefinitely.
631     * @return Returns the result of {@link #service()}.
632     * @throws InterruptedException If another thread has interrupted the
633     *     current thread.
634     * @throws IllegalArgumentException If the value of timeout is negative.
635     */
636    public Optional<T> waitForService(long timeout)
637            throws InterruptedException {
638        if (timeout < 0) {
639            throw new IllegalArgumentException("timeout value is negative");
640        }
641
642        T service = service().orElse(null);
643        if (service != null) {
644            return Optional.of(service);
645        }
646
647        @SuppressWarnings("PMD.UselessParentheses")
648        final long endTime
649            = (timeout == 0) ? 0 : (System.currentTimeMillis() + timeout);
650        while (true) {
651            synchronized (modificationCount) {
652                if (modificationCount() < 0) {
653                    return Optional.empty();
654                }
655                modificationCount.wait(timeout);
656            }
657            Optional<T> found = service();
658            if (found.isPresent()) {
659                return found;
660            }
661            if (endTime > 0) { // if we have a timeout
662                timeout = endTime - System.currentTimeMillis();
663                if (timeout <= 0) { // that has expired
664                    return Optional.empty();
665                }
666            }
667        }
668    }

684     * Returns a {@code ServiceReference} for one of the services collected
685     * by this {@code ServiceCollector}.
686     * <p>
687     * If multiple services have been collected, the service with the highest
688     * ranking (as specified in its {@code service.ranking} property) is
689     * returned. If there is a tie in ranking, the service with the lowest
690     * service id (as specified in its {@code service.id} property); that is,
691     * the service that was registered first is returned. This is the same
692     * algorithm used by {@code BundleContext.getServiceReference}.
693     * 
694     * @return an optional {@code ServiceReference}
695     */
696    public Optional<ServiceReference<S>> serviceReference() {
697        try {
698            synchronized (this) {
699                return Optional.of(collected.firstKey());
700            }
701        } catch (NoSuchElementException e) {
702            return Optional.empty();
703        }
704    }

720    /**
721     * Returns a service object for one of the services collected by this
722     * {@code ServiceCollector}. This is effectively the same as 
723     * {@code serviceReference().flatMap(ref -> service(ref))}.
724     * <P>
725     * The result value is cached, so refrain from implementing another
726     * cache in the invoking code.
727     * 
728     * @return an optional service object
729     */
730    public Optional<T> service() {
731        final T cached = cachedService;
732        if (cached != null) {
733            return Optional.of(cached);
734        }
735        synchronized (this) {
736            Iterator<T> iter = collected.values().iterator();
737            if (iter.hasNext()) {
738                cachedService = iter.next();
739                return Optional.of(cachedService);
740            }
741        }
742        return Optional.empty();
743    }

750     * @return a list of service objects which may be empty
751     */
752    public List<T> services() {
753        synchronized (this) {
754            return Collections
755                .unmodifiableList(new ArrayList<>(collected.values()));
756        }
757    }

759    /**
760     * Return the number of services collected by this
761     * {@code ServiceCollector}. This value may not be correct during the
762     * execution of handlers.
763     * 
764     * @return The number of services collected
765     */
766    public int size() {
767        synchronized (this) {
768            return collected.size();
769        }
770    }

772    /**
773     * Returns the modification count for this {@code ServiceCollector}.
774     * 
775     * The modification count is initialized to 0 when this 
776     * {@code ServiceCollector} is opened. Every time a service is added, 
777     * modified or removed from this {@code ServiceCollector}, 
778     * the modification count is incremented.
779     * <p>
780     * The modification count can be used to determine if this
781     * {@code ServiceCollector} has added, modified or removed a service by
782     * comparing a modification count value previously collected with the 
783     * current modification count value. If the value has not changed, 
784     * then no service has been added, modified or removed from this 
785     * {@code ServiceCollector} since the previous modification count 
786     * was collected.
787     * 
788     * @return The modification count for this {@code ServiceCollector} or
789     *      -1 if this {@code ServiceCollector} is not open.
790     */
791    public int modificationCount() {
792        return modificationCount[0];
793    }
794
795    /**
796     * Checks if this {@code ServiceCollector} is open.
797     *
798     * @return true, if is open
799     */
800    public boolean isOpen() {
801        return modificationCount[0] >= 0;
802    }
803
804    /**
805     * Return a {@code SortedMap} of the {@code ServiceReference}s and service
806     * objects for all services collected by this {@code ServiceCollector}.
807     * The map is sorted in reverse natural order of {@code ServiceReference}.
808     * That is, the first entry is the service with the highest ranking and the
809     * lowest service id.
810     * 
811     * @return A {@code SortedMap} with the {@code ServiceReference}s and
812     *         service objects for all services collected by this
813     *         {@code ServiceCollector}. If no services have been collected,
814     *         then the returned map is empty.
815     */
816    public SortedMap<ServiceReference<S>, T> collected() {
817        synchronized (this) {
818            if (collected.isEmpty()) {
819                return new TreeMap<>(Collections.reverseOrder());
820            }
821            return Collections.unmodifiableSortedMap(new TreeMap<>(collected));
822        }
823    }

827     * 
828     * @return {@code true} if this {@code ServiceCollector} 
829     *     has not collected any services.
830     */
831    public boolean isEmpty() {
832        synchronized (this) {
833            return collected.isEmpty();
834        }

839     * 
840     * The specified service will be removed from this {@code ServiceCollector}.
841     * 
842     * @param reference The reference to the service to be removed.

844    public void remove(ServiceReference<S> reference) {
