047/**
048 * Maintains a collection of services matching some criteria.
049 * <P>
050 * An instance can be created with different criteria for matching
051 * services of a given type. While the instance is open, it 
052 * maintains a collection of all matching services provided
053 * by the framework. Changes of the collection are reported using
054 * the registered handlers.
055 * <P>
056 * Because OSGi is a threaded environment, the registered services 
057 * can vary any time. Results from queries on the collection are 
058 * therefore only reliable while synchronizing on the 
059 * {@code ServiceCollector} instance. The synchronization puts 
060 * other threads that attempt to change the collection on hold. 
061 * Note that this implies a certain risk of creating deadlocks.
062 * Also note that this does not prevent modifications by the
063 * holder of the lock.
064 * <P>
065 * Callbacks are always synchronized on this collector, else
066 * the state of the collector during the execution of the
067 * callback might no longer be the state that triggered
068 * the callback.
069 * <P>
070 * Instead of using the services from the framework directly,
071 * they may optionally be adapted to another type by setting a
072 * service getter function. The invocation of this function is 
073 * also synchronized on this collector.
074 *
075 * @param <S> the type of the service
076 * @param <T> the type of service instances returned by queries, 
077 *      usually the same type as the type of the service
078 *      (see {@link #setServiceGetter(BiFunction)})
079 */
080public class ServiceCollector<S, T> implements AutoCloseable {

168     * on the specified service reference.

171     * @param reference the reference

173    public ServiceCollector(BundleContext context,
174            ServiceReference<S> reference) {
175        this.context = context;
176        this.collectReference = reference;
177        collectClass = null;
178        listenerFilter = "(" + Constants.SERVICE_ID + "="

185             * invalid

188                "unexpected InvalidSyntaxException: " + e.getMessage());
189            iae.initCause(e);

329    private void modified() {
330        synchronized (modificationCount) {
331            modificationCount[0] = modificationCount[0] + 1;
332            cachedService = null;
333            modificationCount.notifyAll();
334        }
335    }

346        open(false);

349    /**
350     * Starts collecting services. Short for calling {@code open(false)}.
351     * 
352     * @param collectAllServices if <code>true</code>, then this 
353     *     {@code ServiceCollector} will collect all matching services 
354     *     regardless of class loader
355     *     accessibility. If <code>false</code>, then 
356     *     this {@code ServiceCollector} will only collect matching services 
357     *     which are class loader
358     *     accessible to the bundle whose <code>BundleContext</code> is 
359     *     used by this {@code ServiceCollector}.
360     * @throws IllegalStateException If the {@code BundleConetxt}
361     *     with which this {@code ServiceCollector} was created is no 
362     *     longer valid.
363     */
364    public void open(boolean collectAllServices) throws IllegalStateException {
365        synchronized (this) {
366            if (isOpen()) {
367                // Already open, don't treat as error.
368                return;
369            }
370            modificationCount[0] = 0;
371            try {
372                registerListener(collectAllServices);
373                if (collectClass != null) {
374                    initialReferences.addAll(getInitialReferences(
375                        collectAllServices, collectClass, null));
376                } else if (collectReference != null) {
377                    if (collectReference.getBundle() != null) {
378                        initialReferences.add(collectReference);
379                    }
380                } else { /* user supplied filter */
381                    initialReferences.addAll(getInitialReferences(
382                        collectAllServices, null, listenerFilter));
383                }
384                processInitial();
385            } catch (InvalidSyntaxException e) {
386                throw new RuntimeException(
387                    "unexpected InvalidSyntaxException: " + e.getMessage(),
388                    e);
389            }
390        }
391    }
392
393    private void registerListener(boolean collectAllServices)
394            throws InvalidSyntaxException {
395        if (collectAllServices) {
396            listener = new AllServiceListener() {
397                @Override
398                public void serviceChanged(ServiceEvent event) {
399                    ServiceCollector.this.serviceChanged(event);
400                }
401            };
402        } else {
403            listener = new ServiceListener() {
404
405                @Override
406                public void serviceChanged(ServiceEvent event) {
407                    ServiceCollector.this.serviceChanged(event);
408                }
409
410            };
411        }
412        context.addServiceListener(listener, listenerFilter);
413    }

415    /**
416     * Returns the list of initial {@code ServiceReference}s that will be
417     * collected by this {@code ServiceCollector}.
418     * 
419     * @param collectAllServices If {@code true}, use
420     *        {@code getAllServiceReferences}.
421     * @param className The class name with which the service was registered, or
422     *        {@code null} for all services.
423     * @param filterString The filter criteria or {@code null} for all services.
424     * @return The list of initial {@code ServiceReference}s.
425     * @throws InvalidSyntaxException If the specified filterString has an
426     *         invalid syntax.
427     */
428    private List<ServiceReference<S>> getInitialReferences(
429            boolean collectAllServices, String className, String filterString)
430            throws InvalidSyntaxException {
431        @SuppressWarnings("unchecked")
432        ServiceReference<S>[] result
433            = (ServiceReference<S>[]) ((collectAllServices)
434                ? context.getAllServiceReferences(className, filterString)
435                : context.getServiceReferences(className, filterString));
436        if (result == null) {
437            return Collections.emptyList();
438        }
439        return Arrays.asList(result);
440    }

442    private void processInitial() {
443        while (true) {
444            // Process one by one so that we do not keep holding the lock.
445            synchronized (this) {
446                if (!isOpen() || (initialReferences.size() == 0)) {
447                    return; /* we are done */
448                }
449                // Get one...
450                Iterator<ServiceReference<S>> iter
451                    = initialReferences.iterator();
452                ServiceReference<S> reference = iter.next();
453                iter.remove();
454                // Process (as if it had been registered).
455                addToCollected(reference);
456            }
457        }
458    }

460    private void serviceChanged(ServiceEvent event) {
461        /*
462         * Check if we had a delayed call (which could happen when we
463         * close).
464         */
465        if (!isOpen()) {
466            return;
467        }
468        @SuppressWarnings("unchecked")
469        final ServiceReference<S> reference
470            = (ServiceReference<S>) event.getServiceReference();
471
472        switch (event.getType()) {
473        case ServiceEvent.REGISTERED:
474            synchronized (this) {
475                initialReferences.remove(reference);
476                addToCollected(reference);
477            }
478            break;
479        case ServiceEvent.MODIFIED:
480            synchronized (this) {
481                T service = collected.get(reference);
482                if (service == null) {
483                    // Probably still in initialReferences, ignore.
484                    return;
485                }
486                modified();
487                if (onModified != null
488                    && reference.equals(collected.firstKey())) {
489                    onModified.accept(reference, service);
490                }
491            }
492            break;
493        case ServiceEvent.MODIFIED_ENDMATCH:
494        case ServiceEvent.UNREGISTERING:
495            synchronized (this) {
496                // May occur while processing the initial set of references.
497                initialReferences.remove(reference);
498                removeFromCollected(reference);
499            }
500            break;
501        }
502
503    }

505    /**
506     * Add the given reference to the collected services.
507     * Must be called from synchronized block.
508     * 
509     * @param reference reference to be collected.
510     */
511    private void addToCollected(final ServiceReference<S> reference) {
512        if (!isOpen()) {
513            // We have been closed.
514            return;
515        }
516        if (collected.get(reference) != null) {
517            /* if we are already collecting this reference */
518            return; /* skip this reference */
519        }
520
521        @SuppressWarnings("unchecked")
522        T service = (svcGetter == null)
523            ? (T) context.getService(reference)
524            : svcGetter.apply(context, reference);
525        if (service == null) {
526            // Has either vanished in the meantime (should not happen
527            // when processing a REGISTERED event, but may happen when
528            // processing a reference from initialReferences) or was
529            // filtered by the service getter.
530            return;
531        }
532        boolean wasEmpty = collected.isEmpty();
533        collected.put(reference, service);
534        modified();
535        if (wasEmpty) {
536            Optional.ofNullable(onBound)
537                .ifPresent(cb -> cb.accept(reference, service));
538        }
539        Optional.ofNullable(onAdded)
540            .ifPresent(cb -> cb.accept(reference, service));
541        // If added is first, first has changed
542        if (onModified != null && collected.size() > 1
543            && collected.firstKey().equals(reference)) {
544            onModified.accept(reference, service);
545        }
546    }
547
548    /**
549     * Removes a service reference from the collection.
550     * Must be called from synchronized block.
551     *
552     * @param reference the reference
553     */
554    private void removeFromCollected(ServiceReference<S> reference) {
555        synchronized (this) {
556            // Only proceed if collected
557            T service = collected.get(reference);
558            if (service == null) {
559                return;
560            }
561            Optional.ofNullable(onRemoving)
562                .ifPresent(cb -> cb.accept(reference, service));
563            if (collected.size() == 1) {
564                Optional.ofNullable(onUnbinding)
565                    .ifPresent(cb -> cb.accept(reference, service));
566            }
567            context.ungetService(reference);
568            boolean firstChanges = reference.equals(collected.firstKey());
569            collected.remove(reference);
570            if (isOpen()) {
571                modified();
572            }
573            // Has first changed?
574            if (onModified != null && firstChanges && !collected.isEmpty()) {
575                onModified.accept(reference, service);
576            }
577        }
578
579    }

581    /**
582     * Stops collecting services.
583     */
584    public void close() {
585        synchronized (this) {
586            context.removeServiceListener(listener);
587            synchronized (modificationCount) {
588                modificationCount[0] = -1;
589                modificationCount.notifyAll();
590            }
591        }
592        while (!collected.isEmpty()) {
593            synchronized (this) {
594                removeFromCollected(collected.lastKey());
595            }
596        }
597        cachedService = null;
598    }

600    /**
601     * Wait for at least one service to be collected by this
602     * {@code ServiceCollector}. This method will also return when this
603     * {@code ServiceCollector} is closed from another thread.
604     * <p>
605     * It is strongly recommended that {@code waitForService} is not used 
606     * during the calling of the {@code BundleActivator} methods.
607     * {@code BundleActivator} methods are expected to complete in a short
608     * period of time.
609     * 
610     * @param timeout The time interval in milliseconds to wait. If zero, the
611     *     method will wait indefinitely.
612     * @return Returns the result of {@link #service()}.
613     * @throws InterruptedException If another thread has interrupted the
614     *     current thread.
615     * @throws IllegalArgumentException If the value of timeout is negative.
616     */
617    public Optional<T> waitForService(long timeout)
618            throws InterruptedException {
619        if (timeout < 0) {
620            throw new IllegalArgumentException("timeout value is negative");
621        }
622
623        T service = service().orElse(null);
624        if (service != null) {
625            return Optional.of(service);
626        }
627
628        final long endTime
629            = (timeout == 0) ? 0 : (System.currentTimeMillis() + timeout);
630        while (true) {
631            synchronized (modificationCount) {
632                if (modificationCount() < 0) {
633                    return Optional.empty();
634                }
635                modificationCount.wait(timeout);
636            }
637            Optional<T> found = service();
638            if (found.isPresent()) {
639                return found;
640            }
641            if (endTime > 0) { // if we have a timeout
642                timeout = endTime - System.currentTimeMillis();
643                if (timeout <= 0) { // that has expired
644                    return Optional.empty();
645                }
646            }
647        }
648    }

665     * by this {@code ServiceCollector}.
666     * <p>
667     * If multiple services have been collected, the service with the highest
668     * ranking (as specified in its {@code service.ranking} property) is
669     * returned. If there is a tie in ranking, the service with the lowest
670     * service id (as specified in its {@code service.id} property); that is,
671     * the service that was registered first is returned. This is the same
672     * algorithm used by {@code BundleContext.getServiceReference}.
673     * 
674     * @return an optional {@code ServiceReference}
675     */

677        try {
678            synchronized (this) {
679                return Optional.of(collected.firstKey());
680            }
681        } catch (NoSuchElementException e) {
682            return Optional.empty();
683        }
684    }

700    /**
701     * Returns a service object for one of the services collected by this
702     * {@code ServiceCollector}. This is effectively the same as 
703     * {@code serviceReference().flatMap(ref -> service(ref))}.
704     * <P>
705     * The result value is cached, so refrain from implementing another
706     * cache in the invoking code.
707     * 
708     * @return an optional service object
709     */
710    public Optional<T> service() {
711        final T cached = cachedService;
712        if (cached != null) {
713            return Optional.of(cached);
714        }
715        synchronized (this) {
716            Iterator<T> iter = collected.values().iterator();
717            if (iter.hasNext()) {
718                cachedService = iter.next();
719                return Optional.of(cachedService);
720            }
721        }
722        return Optional.empty();
723    }

757     * modified or removed from this {@code ServiceCollector}, 
758     * the modification count is incremented.
759     * <p>
760     * The modification count can be used to determine if this
761     * {@code ServiceCollector} has added, modified or removed a service by
762     * comparing a modification count value previously collected with the 
763     * current modification count value. If the value has not changed, 
764     * then no service has been added, modified or removed from this 
765     * {@code ServiceCollector} since the previous modification count 
766     * was collected.
767     * 
768     * @return The modification count for this {@code ServiceCollector} or
769     *      -1 if this {@code ServiceCollector} is not open.
770     */
771    public int modificationCount() {
772        return modificationCount[0];
773    }
774
775    /**
776     * Checks if this {@code ServiceCollector} is open.
777     *
778     * @return true, if is open
779     */
780    public boolean isOpen() {
781        return modificationCount[0] >= 0;
782    }
783
784    /**
785     * Return a {@code SortedMap} of the {@code ServiceReference}s and service
786     * objects for all services collected by this {@code ServiceCollector}.
787     * The map is sorted in reverse natural order of {@code ServiceReference}.
788     * That is, the first entry is the service with the highest ranking and the
789     * lowest service id.
790     * 
791     * @return A {@code SortedMap} with the {@code ServiceReference}s and
792     *         service objects for all services collected by this
793     *         {@code ServiceCollector}. If no services have been collected,
794     *         then the returned map is empty.
795     */
796    public SortedMap<ServiceReference<S>, T> collected() {
797        synchronized (this) {
798            if (collected.isEmpty()) {
799                return new TreeMap<>(Collections.reverseOrder());
800            }
801            return Collections.unmodifiableSortedMap(new TreeMap<>(collected));
802        }
803    }
804
805    /**
806     * Return if this {@code ServiceCollector} is empty.
807     * 
808     * @return {@code true} if this {@code ServiceCollector} 
809     *     has not collected any services.
810     */
811    public boolean isEmpty() {
812        synchronized (this) {
813            return collected.isEmpty();
814        }
815    }
816
817    /**
818     * Remove a service from this {@code ServiceCollector}.
819     * 
820     * The specified service will be removed from this {@code ServiceCollector}.
821     * 
822     * @param reference The reference to the service to be removed.
823     */
824    public void remove(ServiceReference<S> reference) {
825        synchronized (this) {
826            removeFromCollected(reference);
827        }
828    }
