017package de.mnl.osgi.coreutils;

019import java.util.HashMap;
020import java.util.Map;
021import java.util.Optional;
022import java.util.function.Consumer;
023import org.osgi.framework.BundleActivator;
024import org.osgi.framework.BundleContext;
025import org.osgi.framework.Filter;
026import org.osgi.framework.ServiceReference;

028/**
029 * Maintains and attempts to resolve dependencies on services.
030 * <P>
031 * The class supports two usage pattern. The first is to use the
032 * {@code ServiceResolver} as base class for the bundle's activator.
033 * The derived class must override {@link #configure()} to add
034 * at least one dependency.
035 * <P>
036 * The methods {@link #onResolved()}, {@link #onDissolving()} and
037 * {@link #onRebound(String)} can be overridden as required.
038 * 
039 * <pre>
040 * public class Activator extends ServiceResolver {
041 *     &#x40;Override
042 *     protected void configure() {
043 *         addDependency(...);
044 *     }
045 *     
046 *     &#x40;Override
047 *     protected void onResolved() {
048 *         // Enable usage or register as service or start worker thread ...
049 *     }
050 *     
051 *     &#x40;Override
052 *     protected void onDissolving() {
053 *         // Disable usage or unregister or stop and join worker thread ...
054 *     }
055 *     
056 *     &#x40;Override
057 *     protected void onRebound(String dependency) {
058 *         // Only required if there is a "long term" usage of a dependency.
059 *     }
060 * }
061 * </pre>
062 * 
063 * The second usage pattern is to create a {@code ServiceResolver} with
064 * a {@link BundleContext}, add dependencies and callbacks as required
065 * and call {@link #open()}. When resolving is no longer required,
066 * {@link #close()} must be called.
067 * 
068 * <pre>
069 * // Usually executed during bundle start...
070 * ServiceResolver resolver = new ServiceResolver(context);
071 * // Add dependencies
072 * resolver.addDependency(...);
073 * resolver.setOnResolved(() -&gt; ...);
074 * resolver.setOnDissolving(() -&gt; ...);
075 * resolver.setOnRebound(name -&gt; ...);
076 * resolver.open();
077 * 
078 * // Usually executed during bundle stop...
079 * resolver.close();
080 * </pre>
081 * 
082 * In addition to the management of mandatory dependencies, the
083 * {@code ServiceResolver} can also keep track of optional
084 * service dependencies. This allows the {@code ServiceResolver}
085 * to be provided as unified service access point. Apart from
086 * a single {@link #open()} and {@link #close()} function, the
087 * {@code ServiceProvider} does not implement additional functions
088 * for optional service dependencies compared to the underlying
089 * {@link ServiceCollector}s.
090 */
091public class ServiceResolver implements AutoCloseable, BundleActivator {
092
093    /**
094     * The bundle context. Made available for derived classes.
095     */
096    protected BundleContext context;
097    private volatile boolean isOpen;
098    private Map<String, ServiceCollector<?, ?>> dependencies = new HashMap<>();
099    private Map<String, ServiceCollector<?, ?>> optDependencies
100        = new HashMap<>();
101    private int resolvedCount;
102    private Runnable onResolved;
103    private Runnable onDissolving;
104    private Consumer<String> onRebound;

106    /**
107     * Creates a new resolver that uses the given context.

110     */

112        this.context = context;

122    /**

124     * base class for a bundle activator.

233    /**
234     * Adds a mandatory dependency on the service specified by the
235     * class. The name of the class is used as name of the
236     * dependency.
237     *
238     * @param clazz the class
239     * @return the service resolver
240     */
241    public ServiceResolver addDependency(Class<?> clazz) {
242        addDependency(clazz.getName(), clazz.getName());
243        return this;
244    }
245
246    /**
247     * Adds a mandatory dependency on the service specified by the
248     * filter.
249     *
250     * @param name the name of the dependency
251     * @param filter the filter
252     * @return the service resolver
253     */
254    public ServiceResolver addDependency(String name, Filter filter) {
255        synchronized (this) {
256            if (isOpen) {
257                throw new IllegalStateException(
258                    "Cannot add dependencies to open reolver.");
259            }
260            dependencies.put(name, new ServiceCollector<>(context, filter)
261                .setOnBound(this::boundCb).setOnUnbinding(this::unbindingCb)
262                .setOnModfied((ref, svc) -> modifiedCb(name)));
263            return this;
264        }
265    }
266

271     * @param name the name of the dependency
272     * @param reference the reference
273     * @return the service resolver
274     */
275    public ServiceResolver addDependency(String name,
276            ServiceReference<?> reference) {
277        synchronized (this) {
278            if (isOpen) {

365    /**
366     * Adds an optional dependency on the service specified by the
367     * service reference.
368     *
369     * @param name the name of the dependency
370     * @param className the class name
371     * @return the service resolver
372     */
373    public ServiceResolver addOptionalDependency(String name,
374            String className) {
375        synchronized (this) {
376            if (isOpen) {
377                throw new IllegalStateException(
378                    "Cannot add dependencies to open reolver.");
379            }
380            optDependencies.put(name,
381                new ServiceCollector<>(context, className)
382                    .setOnModfied((ref, svc) -> modifiedCb(name)));
383            return this;
384        }
385    }
386
387    private void boundCb(ServiceReference<?> reference, Object service) {
388        synchronized (this) {
389            resolvedCount += 1;
390            if (resolvedCount == dependencies.size()) {
391                Optional.ofNullable(onResolved).ifPresent(cb -> cb.run());
392                onResolved();
393            }
394        }
395    }
396
397    private void unbindingCb(ServiceReference<?> reference, Object service) {
398        synchronized (this) {
399            if (resolvedCount == dependencies.size()) {
400                Optional.ofNullable(onDissolving).ifPresent(cb -> cb.run());
401                onDissolving();
402            }
403            resolvedCount -= 1;
404        }
405    }
406
407    private void modifiedCb(String dependency) {
408        synchronized (this) {
409            Optional.ofNullable(onRebound)
410                .ifPresent(cb -> cb.accept(dependency));
411            onRebound(dependency);
412        }
413    }
414
415    /**
416     * Starts the resolver. While open, the resolver attempts
417     * to obtain service implementation for all registered
418     * service dependencies.
419     */
420    public void open() {
421        synchronized (this) {
422            if (isOpen) {
423                return;
424            }
425            resolvedCount = 0;
426        }
427        for (ServiceCollector<?, ?> coll : dependencies.values()) {
428            coll.open();
429        }
430        for (ServiceCollector<?, ?> coll : optDependencies.values()) {
431            coll.open();
432        }
433        isOpen = true;
434    }
435
436    /**
437     * Stops the resolver. All maintained services are released.
438     */
439    public void close() {
440        synchronized (this) {
441            if (!isOpen) {
442                return;
443            }
444            isOpen = false;
445            for (ServiceCollector<?, ?> coll : dependencies.values()) {
446                coll.close();
447            }
448            for (ServiceCollector<?, ?> coll : optDependencies.values()) {
449                coll.close();
450            }
451            dependencies.clear();
452            optDependencies.clear();
453        }
454    }
455
456    /**
457     * Checks if the resolver is open.
458     *
459     * @return true, if open
460     */
461    public boolean isOpen() {
462        return isOpen;
463    }
464
465    /**
466     * Checks if the resolver is in the resolved state.
467     *
468     * @return the result
469     */
470    public boolean isResolved() {
471        synchronized (this) {
472            return resolvedCount == dependencies.size();
473        }
474    }

476    /**
477     * Returns the service found for the mandatory dependency with the
478     * given name. This method should only be called when the resolver
479     * is in state resolved. 
480     *
481     * @param <T> the type of the service
482     * @param name the name of the dependency
483     * @param clazz the class of the service
484     * @return the service or {@code null} if called in unresolved
485     *     state and the dependency is not resolved
486     */
487    @SuppressWarnings("unchecked")
488    public <T> T get(String name, Class<T> clazz) {
489        return Optional.ofNullable(dependencies.get(name))
490            .map(coll -> ((ServiceCollector<?, T>) coll).service().get())
491            .orElse(null);
492    }
493
494    /**
495     * Returns the service found for the mandatory dependency
496     * using the name of the class as name of the dependency. 
497     * This method should only be called when 
498     * the resolver is in state resolved. 
499     *
500     * @param <T> the type of the service
501     * @param clazz the class of the service
502     * @return the service or {@code null} if called in unresolved
503     *     state and the dependency is not resolved
504     */
505    public <T> T get(Class<T> clazz) {
506        return get(clazz.getName(), clazz);
507    }
508
509    /**
510     * Returns the (optional) service found for the optional dependency 
511     * with the given name. 
512     * 
513     * @param <T> the type of the service
514     * @param name the name of the dependency
515     * @param clazz the class of the service
516     * @return the result
517     */
518    @SuppressWarnings("unchecked")
519    public <T> Optional<T> optional(String name, Class<T> clazz) {
520        return Optional.ofNullable(dependencies.get(name))
521            .map(coll -> ((ServiceCollector<?, T>) coll).service())
522            .orElse(Optional.empty());
523    }
524
525    /**
526     * Returns the (optional) service found for the optional dependency, 
527     * using the name of the class as name of the dependency. 
528     *
529     * @param <T> the type of the service
530     * @param clazz the class of the service
531     * @return the result
532     */
533    public <T> Optional<T> optional(Class<T> clazz) {
534        return optional(clazz.getName(), clazz);
535    }
