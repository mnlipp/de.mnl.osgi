039/**
040 * The {@code ServiceCollector} class simplifies using services
041 * from the OSGi Framework's service registry beyond the simplification
042 * provided by the {@link ServiceTracker} from the 
043 * <a href="https://osgi.org/specification/osgi.core/7.0.0/util.tracker.html">
044 * OSGi Core Specification</a>.
045 * <p>
046 * The {@code ServiceCollector} calls registered callbacks when the first or any 
047 * service becomes available. The implementation ensures that
048 * at least one service can be obtained from the {@code ServiceCollector}
049 * during the execution of these callbacks. At least one service will also
050 * be available after the execution of the callbacks until the 
051 * "last unavailable" callback has been invoked.
052 * <P>
053 * Functions enabled/disabled in the respective callbacks can therefore 
054 * rely on a service being available while they are enabled. This
055 * includes, as a special case, starting and stopping threads in the
056 * respective callbacks.
057 * <p>
058 * A {@code ServiceCollector} object is constructed with search 
059 * criteria. It can then be opened to begin collecting all services in 
060 * the Framework's service registry that match the specified search 
061 * criteria. The {@code ServiceCollector} is currently
062 * implemented using a standard {@link ServiceTracker} as a delegee.
063 * <p>
064 * The {@code ServiceCollector} class is thread-safe.
065 *
066 * @param <S> the type of the service to be collected
067 * @param <W> the type of the collected items
068 *      (see {@link #setWrapper(Function)}). Usually, this is the 
069 *      same type as {@code S}, i.e. services are not wrapped before
070 *      being returned.
071 */
072public class ServiceCollector<S, W> implements AutoCloseable {
073
074    private final ServiceTracker<S, W> delegee;
075    /*
076     * The service data known from ServiceTracker#addingService
077     * but not yet known to have been added. Or the service data
078     * already removed from the ServiceTracker but provided as a fallback
079     * while calling onLastUnavailable in ServiceTracker#removedService.
080     */
081    private FallbackData fallback;
082    private BiConsumer<ServiceReference<S>, W> onFirstAvailable;
083    private BiConsumer<ServiceReference<S>, W> onAvailable;
084    private BiConsumer<ServiceReference<S>, W> onUnavailable;
085    private BiConsumer<ServiceReference<S>, W> onLastUnavailable;
086    private BiConsumer<ServiceReference<S>, W> onModified;
087    private Function<S, W> wrapper;
088    private int minModificationCount;

246    /**
247     * Sets the wrapper function. The function is invoked for
248     * every newly collected service. It allows to wrap a service
249     * obtained from the registry in some other object that is
250     * used as return value when retrieving services from this 
251     * {@code ServiceCollector}.
252     * <P>
253     * The wrapper can also serve as a filter. If the wrapper returns
254     * {@code null}, the service will not be added to the collection.
255     *
256     * @param wrapper the wrapper
257     * @return the {@code ServiceCollector}
258     * @throws IllegalStateException if the {@code ServiceCollector} is open
259     */
260    public ServiceCollector<S, W> setWrapper(
261            Function<S, W> wrapper) {
262        if (delegee.getTrackingCount() >= 0) {
263            throw new IllegalStateException(
264                "Wrapper cannot be set while service collector is open.");
265        }
266        this.wrapper = wrapper;
267        return this;
268    }

270    /**
271     * Wait for at least one service to be collected by this
272     * {@code ServiceCollector}. This method will also return when this
273     * {@code ServiceCollector} is closed from another thread.
274     * <p>
275     * It is strongly recommended that {@code waitForService} is not used 
276     * during the calling of the {@code BundleActivator} methods.
277     * {@code BundleActivator} methods are expected to complete in a short
278     * period of time.
279     * 
280     * @param timeout The time interval in milliseconds to wait. If zero, the
281     *     method will wait indefinitely.
282     * @return Returns the result of {@link #service()}.
283     * @throws InterruptedException If another thread has interrupted the
284     *     current thread.
285     * @throws IllegalArgumentException If the value of timeout is negative.
286     */
287    public W waitForService(long timeout) throws InterruptedException {
288        return delegee.waitForService(timeout);
289    }

291    /**
292     * Return an array of {@code ServiceReference}s for all services 
293     * collected by this {@code ServiceCollector}.
294     * 
295     * @return Array of {@code ServiceReference}s or an empty array if no 
296     *     services have been collected.
297     */
298    @SuppressWarnings("unchecked")
299    public ServiceReference<S>[] serviceReferences() {
300        FallbackData maybeAdd = fallback;
301        if (maybeAdd == null) {
302            return Optional.ofNullable(delegee.getServiceReferences())
303                .orElse((ServiceReference<S>[]) new ServiceReference[0]);
304        }
305        // Add fallback service reference if not already in list returned
306        // by delegee.
307        Object maybeAddId = maybeAdd.svcRef().getProperty(Constants.SERVICE_ID);
308        List<ServiceReference<S>> result = new ArrayList<>();
309        ServiceReference<S>[] knownRefs = delegee.getServiceReferences();
310        if (knownRefs != null) {
311            for (ServiceReference<S> ref : knownRefs) {
312                result.add(ref);
313                if (ref.getProperty(Constants.SERVICE_ID).equals(maybeAddId)) {
314                    // Known and therefore already added.
315                    maybeAdd = null;
316                }
317            }
318        }
319        if (maybeAdd != null) {
320            result.add(maybeAdd.svcRef());
321        }
322        return result.toArray(new ServiceReference[0]);
323    }

325    /**
326     * Returns a {@code ServiceReference} for one of the services collected
327     * by this {@code ServiceCollector}.
328     * <p>
329     * If multiple services have been collected, the service with the highest
330     * ranking (as specified in its {@code service.ranking} property) is
331     * returned. If there is a tie in ranking, the service with the lowest
332     * service id (as specified in its {@code service.id} property); that is,
333     * the service that was registered first is returned. This is the same
334     * algorithm used by {@code BundleContext.getServiceReference}.
335     * 
336     * @return an optional {@code ServiceReference}
337     */
338    public Optional<ServiceReference<S>> serviceReference() {
339        return Optional.ofNullable(delegee.getServiceReference())
340            .map(ref -> Optional.of(ref))
341            .orElse(Optional.ofNullable(fallback).map(FallbackData::svcRef));
342    }
343
344    /**
345     * Returns the service object for the specified {@code ServiceReference} 
346     * if the specified referenced service has been collected
347     * by this {@code ServiceCollector}.
348     * 
349     * @param reference the reference to the desired service.
350     * @return an optional service object
351     */
352    public Optional<W> service(ServiceReference<S> reference) {
353        W known = delegee.getService(reference);
354        if (known != null) {
355            return Optional.of(known);
356        }
357        // Maybe fallback can be used...
358        FallbackData maybe = fallback;
359        return Optional.ofNullable(maybe.svcRef())
360            .filter(ref -> ref.getProperty(Constants.SERVICE_ID)
361                .equals(reference.getProperty(Constants.SERVICE_ID)))
362            .map(found -> maybe.service());
363    }
364
365    /**
366     * Returns a service object for one of the services collected by this
367     * {@code ServiceCollector}.
368     * 
369     * @return an optional service object
370     */
371    public Optional<W> service() {
372        return Optional.ofNullable(
373            Optional.ofNullable(delegee.getService()).orElse(Optional
374                .ofNullable(fallback).map(FallbackData::service).orElse(null)));
375    }

377    /**
378     * Return an array of service objects for all services collected by this
379     * {@code ServiceCollector}.
380     * 
381     * @return an array of service objects which may be empty
382     */
383    public Object[] services() {
384        List<W> result = new ArrayList<>();
385        for (ServiceReference<S> ref : serviceReferences()) {
386            service(ref).ifPresent(s -> result.add(s));
387        }
388        return result.toArray(new Object[0]);
389    }

402    /**
403     * Returns the modification count for this {@code ServiceCollector}.
404     * 
405     * The modification count is initialized to 0 when this 
406     * {@code ServiceCollector} is opened. Every time a service is added, 
407     * modified or removed from this {@code ServiceCollector}, 
408     * the modification count is incremented.
409     * <p>
410     * The modification count can be used to determine if this
411     * {@code ServiceCollector} has added, modified or removed a service by
412     * comparing a modification count value previously collected with the 
413     * current modification count value. If the value has not changed, 
414     * then no service has been added, modified or removed from this 
415     * {@code ServiceCollector} since the previous modification count 
416     * was collected.
417     * 
418     * @return The modification count for this {@code ServiceCollector} or
419     *      -1 if this {@code ServiceCollector} is not open.
420     */
421    public int modificationCount() {
422        return Math.max(minModificationCount, delegee.getTrackingCount());
423    }

425    /**
426     * Return a {@code SortedMap} of the {@code ServiceReference}s and service
427     * objects for all services collected by this {@code ServiceCollector}.
428     * The map is sorted in reverse natural order of {@code ServiceReference}.
429     * That is, the first entry is the service with the highest ranking and the
430     * lowest service id.
431     * 
432     * @return A {@code SortedMap} with the {@code ServiceReference}s and
433     *         service objects for all services collected by this
434     *         {@code ServiceCollector}. If no services have been collected,
435     *         then the returned map is empty.
436     */
437    public SortedMap<ServiceReference<S>, W> collected() {
438        SortedMap<ServiceReference<S>, W> result = delegee.getTracked();
439        FallbackData maybe = fallback;
440        if (maybe != null) {
441            Object maybeId = maybe.svcRef().getProperty(Constants.SERVICE_ID);
442            // There's no equals/hashCode on ServiceReference, sigh.
443            boolean found = false;
444            for (ServiceReference<S> ref : result.keySet()) {
445                if (ref.getProperty(Constants.SERVICE_ID).equals(maybeId)) {
446                    found = true;
447                    break;
448                }
449            }
450            if (!found) {
451                result.put(maybe.svcRef(), maybe.service());
452            }
453        }
454        return result;
455    }

478    private class FallbackData {
479        private final ServiceReference<S> svcRef;
480        private final W service;
481
482        public FallbackData(ServiceReference<S> svcRef, W service) {
483            this.svcRef = svcRef;
484            this.service = service;
485        }
486
487        public final ServiceReference<S> svcRef() {
488            return svcRef;
489        }
490
491        public final W service() {
492            return service;
493        }
494    }

496    private class MyTracker extends ServiceTracker<S, W> {
497
498        public MyTracker(BundleContext context, Class<S> clazz,
499                ServiceTrackerCustomizer<S, W> customizer) {
500            super(context, clazz, customizer);
501        }
502
503        public MyTracker(BundleContext context, Filter filter,
504                ServiceTrackerCustomizer<S, W> customizer) {
505            super(context, filter, customizer);
506        }
507
508        public MyTracker(BundleContext context, ServiceReference<S> reference,
509                ServiceTrackerCustomizer<S, W> customizer) {
510            super(context, reference, customizer);
511        }
512
513        public MyTracker(BundleContext context, String clazz,
514                ServiceTrackerCustomizer<S, W> customizer) {
515            super(context, clazz, customizer);
516        }
517
518        @Override
519        public W addingService(ServiceReference<S> reference) {
520            synchronized (this) {
521                minModificationCount = delegee.getTrackingCount() + 1;
522                @SuppressWarnings("unchecked")
523                W newService = Optional.ofNullable(wrapper)
524                    .map(w -> w.apply(context.getService(reference)))
525                    .orElse((W) context.getService(reference));
526                fallback = new FallbackData(reference, newService);
527                if (delegee.isEmpty()) {
528                    Optional.ofNullable(onFirstAvailable)
529                        .ifPresent(h -> h.accept(reference, newService));
530                }
531                if (onAvailable != null) {
532                    Optional.ofNullable(onAvailable)
533                        .ifPresent(h -> h.accept(reference, newService));
534                }
535                return newService;
536            }
537        }
538
539        @Override
540        public void removedService(ServiceReference<S> reference, W service) {
541            synchronized (this) {
542                fallback = new FallbackData(reference, service);
543                Optional.ofNullable(onUnavailable)
544                    .ifPresent(h -> h.accept(reference, service));
545                if (delegee.isEmpty()) {
546                    Optional.ofNullable(onLastUnavailable)
547                        .ifPresent(h -> h.accept(reference, service));
548                }
549                fallback = null;
550                super.removedService(reference, service);
551            }
552        }
553
554        @Override
555        public void modifiedService(ServiceReference<S> reference, W service) {
556            Optional.ofNullable(onModified)
557                .ifPresent(h -> h.accept(reference, service));
558        }
559    }
