048/**
049 * Maintains a collection of services matching some criteria.
050 * <P>
051 * An instance can be created with different criteria for matching
052 * services of a given type. While the instance is open, it 
053 * maintains a collection of all matching services provided
054 * by the framework. Changes of the collection are reported using
055 * the registered handlers.
056 * <P>
057 * Because OSGi is a threaded environment, the registered services 
058 * can vary any time. Results from queries on the collection are 
059 * therefore only reliable while synchronizing on the 
060 * {@code ServiceCollector} instance. The synchronization puts 
061 * other threads that attempt to change the collection on hold. 
062 * Note that this implies a certain risk of creating deadlocks.
063 * Also note that this does not prevent modifications by the
064 * holder of the lock.
065 * <P>
066 * Callbacks are also synchronized on this collector, else
067 * the state of the collector during the execution of the
068 * callback might not reflect the state that is signaled by
069 * the callback.
070 * <P>
071 * Instead of using the services from the framework directly,
072 * they may optionally be adapted to another type by setting a
073 * service getter function. The invocation of this function is 
074 * also synchronized on this collector.
075 *
076 * @param <S> the type of the service
077 * @param <T> the type of service instances returned by queries, 
078 *      usually the same type as the type of the service
079 *      (see {@link #setServiceGetter(BiFunction)})
080 */
081public class ServiceCollector<S, T> implements AutoCloseable {

170     * on the specified service reference.

172     * @param context the bundle context used to interact with the framework
173     * @param reference the reference

176            ServiceReference<S> reference) {
177        this.context = context;
178        this.collectReference = reference;
179        this.collectClass = null;

190                "unexpected InvalidSyntaxException: " + e.getMessage());
191            iae.initCause(e);

322    private void modified() {
323        synchronized (modificationCount) {
324            modificationCount[0] = modificationCount[0] + 1;
325            cachedService = null;
326            modificationCount.notifyAll();
327        }
328    }

342    /**
343     * Starts collecting services. Short for calling {@code open(false)}.
344     * 
345     * @param collectAllServices if <code>true</code>, then this 
346     *     {@code ServiceCollector} will collect all matching services 
347     *     regardless of class loader
348     *     accessibility. If <code>false</code>, then 
349     *     this {@code ServiceCollector} will only collect matching services 
350     *     which are class loader
351     *     accessible to the bundle whose <code>BundleContext</code> is 
352     *     used by this {@code ServiceCollector}.
353     * @throws IllegalStateException If the {@code BundleConetxt}
354     *     with which this {@code ServiceCollector} was created is no 
355     *     longer valid.
356     */
357    public void open(boolean collectAllServices) throws IllegalStateException {
358        synchronized (this) {
359            if (isOpen()) {
360                // Already open, don't treat as error.
361                return;
362            }
363            modificationCount[0] = 0;
364            try {
365                registerListener(collectAllServices);
366                if (collectClass != null) {
367                    initialReferences.addAll(getInitialReferences(
368                        collectAllServices, collectClass, null));
369                } else if (collectReference != null) {
370                    if (collectReference.getBundle() != null) {
371                        initialReferences.add(collectReference);
372                    }
373                } else { /* user supplied filter */
374                    initialReferences.addAll(getInitialReferences(
375                        collectAllServices, null, listenerFilter));
376                }
377                processInitial();
378            } catch (InvalidSyntaxException e) {
379                throw new RuntimeException(
380                    "unexpected InvalidSyntaxException: " + e.getMessage(),
381                    e);
382            }
383        }
384    }
385
386    private void registerListener(boolean collectAllServices)
387            throws InvalidSyntaxException {
388        if (collectAllServices) {
389            listener = new AllServiceListener() {
390                @Override
391                public void serviceChanged(ServiceEvent event) {
392                    ServiceCollector.this.serviceChanged(event);
393                }
394            };
395        } else {
396            listener = new ServiceListener() {
397
398                @Override
399                public void serviceChanged(ServiceEvent event) {
400                    ServiceCollector.this.serviceChanged(event);
401                }
402
403            };
404        }
405        context.addServiceListener(listener, listenerFilter);
406    }

408    /**
409     * Returns the list of initial {@code ServiceReference}s that will be
410     * collected by this {@code ServiceCollector}.
411     * 
412     * @param collectAllServices If {@code true}, use
413     *        {@code getAllServiceReferences}.
414     * @param className The class name with which the service was registered, or
415     *        {@code null} for all services.
416     * @param filterString The filter criteria or {@code null} for all services.
417     * @return The list of initial {@code ServiceReference}s.
418     * @throws InvalidSyntaxException If the specified filterString has an
419     *         invalid syntax.
420     */
421    private List<ServiceReference<S>> getInitialReferences(
422            boolean collectAllServices, String className, String filterString)
423            throws InvalidSyntaxException {
424        @SuppressWarnings("unchecked")
425        ServiceReference<S>[] result
426            = (ServiceReference<S>[]) ((collectAllServices)
427                ? context.getAllServiceReferences(className, filterString)
428                : context.getServiceReferences(className, filterString));
429        if (result == null) {
430            return Collections.emptyList();
431        }
432        return Arrays.asList(result);
433    }

435    private void processInitial() {
436        while (true) {
437            ServiceReference<S> reference;
438            synchronized (this) {
439                if (!isOpen() || (initialReferences.size() == 0)) {
440                    return; /* we are done */
441                }
442                // Get one...
443                Iterator<ServiceReference<S>> iter
444                    = initialReferences.iterator();
445                reference = iter.next();
446                iter.remove();
447            }
448            // Process as if it was just registered.
449            addToCollected(reference);
450        }
451    }
452
453    private void serviceChanged(ServiceEvent event) {
454        /*
455         * Check if we had a delayed call (which could happen when we
456         * close).
457         */
458        if (!isOpen()) {
459            return;
460        }
461        @SuppressWarnings("unchecked")
462        final ServiceReference<S> reference
463            = (ServiceReference<S>) event.getServiceReference();
464
465        switch (event.getType()) {
466        case ServiceEvent.REGISTERED:
467            addToCollected(reference);
468            break;
469        case ServiceEvent.MODIFIED:
470            synchronized (this) {
471                T service = collected.get(reference);
472                if (service == null) {
473                    // Probably still in initialReferences, ignore.
474                    return;
475                }
476                modified();
477                Optional.ofNullable(onModified)
478                    .ifPresent(cb -> cb.accept(reference, service));
479            }
480            break;
481        case ServiceEvent.MODIFIED_ENDMATCH:
482        case ServiceEvent.UNREGISTERING:
483            removeFromCollected(reference);
484            break;
485        }
486
487    }

489    /**
490     * Add the given reference to the collected services.
491     * 
492     * @param reference reference to be collected.
493     */
494    private void addToCollected(final ServiceReference<S> reference) {
495        synchronized (this) {
496            if (!isOpen()) {
497                // We have been closed.
498                return;
499            }
500            if (collected.get(reference) != null) {
501                /* if we are already collecting this reference */
502                return; /* skip this reference */
503            }
504            if (initialReferences.contains(reference)) {
505                // Skip, will be added as initial.
506                return;
507            }
508
509            @SuppressWarnings("unchecked")
510            T service = (svcGetter == null)
511                ? (T) context.getService(reference)
512                : svcGetter.apply(context, reference);
513            if (service == null) {
514                // Has either vanished in the meantime (should not happen
515                // when processing a REGISTERED event, but may happen when
516                // processing a reference from initialReferences) or was
517                // filtered by the service getter.
518                return;
519            }
520            boolean wasEmpty = collected.isEmpty();
521            collected.put(reference, service);
522            modified();
523            if (wasEmpty) {
524                Optional.ofNullable(onBound)
525                    .ifPresent(cb -> cb.accept(reference, service));
526            }
527            Optional.ofNullable(onAdded)
528                .ifPresent(cb -> cb.accept(reference, service));
529        }
530    }
531
532    private void removeFromCollected(ServiceReference<S> reference) {
533        synchronized (this) {
534            T service = collected.get(reference);
535            if (service == null) {
536                return;
537            }
538            Optional.ofNullable(onRemoving)
539                .ifPresent(cb -> cb.accept(reference, service));
540            if (collected.size() == 1) {
541                Optional.ofNullable(onUnbinding)
542                    .ifPresent(cb -> cb.accept(reference, service));
543            }
544            context.ungetService(reference);
545            collected.remove(reference);
546            if (isOpen()) {
547                modified();
548            }
549        }
550    }

552    /**
553     * Stops collecting services.
554     */
555    public void close() {
556        synchronized (this) {
557            context.removeServiceListener(listener);
558            synchronized (modificationCount) {
559                modificationCount[0] = -1;
560                modificationCount.notifyAll();
561            }
562            while (!collected.isEmpty()) {
563                removeFromCollected(collected.lastKey());
564            }
565            cachedService = null;
566        }
567    }

569    /**
570     * Wait for at least one service to be collected by this
571     * {@code ServiceCollector}. This method will also return when this
572     * {@code ServiceCollector} is closed from another thread.
573     * <p>
574     * It is strongly recommended that {@code waitForService} is not used 
575     * during the calling of the {@code BundleActivator} methods.
576     * {@code BundleActivator} methods are expected to complete in a short
577     * period of time.
578     * 
579     * @param timeout The time interval in milliseconds to wait. If zero, the
580     *     method will wait indefinitely.
581     * @return Returns the result of {@link #service()}.
582     * @throws InterruptedException If another thread has interrupted the
583     *     current thread.
584     * @throws IllegalArgumentException If the value of timeout is negative.
585     */
586    public Optional<T> waitForService(long timeout)
587            throws InterruptedException {
588        if (timeout < 0) {
589            throw new IllegalArgumentException("timeout value is negative");
590        }
591
592        T service = service().orElse(null);
593        if (service != null) {
594            return Optional.of(service);
595        }
596
597        final long endTime
598            = (timeout == 0) ? 0 : (System.currentTimeMillis() + timeout);
599        while (true) {
600            synchronized (modificationCount) {
601                if (modificationCount() < 0) {
602                    return Optional.empty();
603                }
604                modificationCount.wait(timeout);
605            }
606            Optional<T> found = service();
607            if (found.isPresent()) {
608                return found;
609            }
610            if (endTime > 0) { // if we have a timeout
611                timeout = endTime - System.currentTimeMillis();
612                if (timeout <= 0) { // that has expired
613                    return Optional.empty();
614                }
615            }
616        }
617    }

632    /**
633     * Returns a {@code ServiceReference} for one of the services collected
634     * by this {@code ServiceCollector}.
635     * <p>
636     * If multiple services have been collected, the service with the highest
637     * ranking (as specified in its {@code service.ranking} property) is
638     * returned. If there is a tie in ranking, the service with the lowest
639     * service id (as specified in its {@code service.id} property); that is,
640     * the service that was registered first is returned. This is the same
641     * algorithm used by {@code BundleContext.getServiceReference}.
642     * 
643     * @return an optional {@code ServiceReference}
644     */
645    public Optional<ServiceReference<S>> serviceReference() {
646        try {
647            synchronized (this) {
648                return Optional.of(collected.firstKey());
649            }
650        } catch (NoSuchElementException e) {
651            return Optional.empty();
652        }
653    }

669    /**
670     * Returns a service object for one of the services collected by this
671     * {@code ServiceCollector}. This is effectively the same as 
672     * {@code serviceReference().flatMap(ref -> service(ref))}.
673     * <P>
674     * The result value is cached, so refrain from implementing another
675     * cache in the invoking code.
676     * 
677     * @return an optional service object
678     */
679    public Optional<T> service() {
680        final T cached = cachedService;
681        if (cached != null) {
682            return Optional.of(cached);
683        }
684        synchronized (this) {
685            Iterator<T> iter = collected.values().iterator();
686            if (iter.hasNext()) {
687                cachedService = iter.next();
688                return Optional.of(cachedService);
689            }
690        }
691        return Optional.empty();
692    }

756     * The map is sorted in reverse natural order of {@code ServiceReference}.
757     * That is, the first entry is the service with the highest ranking and the
758     * lowest service id.
759     * 
760     * @return A {@code SortedMap} with the {@code ServiceReference}s and
761     *         service objects for all services collected by this
762     *         {@code ServiceCollector}. If no services have been collected,
763     *         then the returned map is empty.
764     */
765    public SortedMap<ServiceReference<S>, T> collected() {
766        synchronized (this) {
767            if (collected.isEmpty()) {
768                return new TreeMap<>(Collections.reverseOrder());
769            }
770            return Collections.unmodifiableSortedMap(new TreeMap<>(collected));
771        }
772    }
773
774    /**
775     * Return if this {@code ServiceCollector} is empty.
776     * 
777     * @return {@code true} if this {@code ServiceCollector} 
778     *     has not collected any services.
779     */
780    public boolean isEmpty() {
781        synchronized (this) {
782            return collected.isEmpty();
783        }
784    }
785
786    /**
787     * Remove a service from this {@code ServiceCollector}.
788     * 
789     * The specified service will be removed from this {@code ServiceCollector}.
790     * 
791     * @param reference The reference to the service to be removed.
792     */
793    public void remove(ServiceReference<S> reference) {
794        removeFromCollected(reference);
795    }
