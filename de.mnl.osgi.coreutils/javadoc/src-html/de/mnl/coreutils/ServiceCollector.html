048/**
049 * Maintains a collection of services matching some criteria.
050 * <P>
051 * An instance can be created with different criteria for matching
052 * services of a given type. While the instance is open, it 
053 * maintains a collection of all matching services provided
054 * by the framework. Changes of the collection are reported using
055 * the registered handlers.
056 * <P>
057 * Because OSGi is a threaded environment, the registered services 
058 * can vary any time. Results from queries on the collection are 
059 * therefore only reliable while synchronizing on the 
060 * {@code ServiceCollector} instance. The synchronization puts 
061 * other threads that attempt to change the collection on hold. 
062 * Note that this implies a certain risk of creating deadlocks.
063 * Also note that this does not prevent modifications by the
064 * holder of the lock.
065 * <P>
066 * Callbacks are also synchronized on this collector, else
067 * the state of the collector during the execution of the
068 * callback might not reflect the state that is signaled by
069 * the callback.
070 * <P>
071 * Instead of using the services from the framework directly,
072 * they may optionally be adapted to another type by setting a
073 * service getter function. The invocation of this function is 
074 * also synchronized on this collector.
075 *
076 * @param <S> the type of the service
077 * @param <T> the type of service instances returned by queries, 
078 *      usually the same type as the type of the service
079 *      (see {@link #setServiceGetter(BiFunction)})
080 */
081public class ServiceCollector<S, T> implements AutoCloseable {

169     * on the specified service reference.
170     *
171     * @param context the bundle context used to interact with the framework
172     * @param reference the reference
173     */
174    public ServiceCollector(BundleContext context,
175            ServiceReference<S> reference) {
176        this.context = context;
177        this.collectReference = reference;
178        collectClass = null;
179        listenerFilter = "(" + Constants.SERVICE_ID + "="
180            + reference.getProperty(Constants.SERVICE_ID).toString() + ")";

185             * we could only get this exception if the ServiceReference was
186             * invalid

189                "unexpected InvalidSyntaxException: " + e.getMessage());
190            iae.initCause(e);

325    private void modified() {
326        synchronized (modificationCount) {
327            modificationCount[0] = modificationCount[0] + 1;
328            cachedService = null;
329            modificationCount.notifyAll();
330        }
331    }

342        open(false);

345    /**
346     * Starts collecting services. Short for calling {@code open(false)}.
347     * 
348     * @param collectAllServices if <code>true</code>, then this 
349     *     {@code ServiceCollector} will collect all matching services 
350     *     regardless of class loader
351     *     accessibility. If <code>false</code>, then 
352     *     this {@code ServiceCollector} will only collect matching services 
353     *     which are class loader
354     *     accessible to the bundle whose <code>BundleContext</code> is 
355     *     used by this {@code ServiceCollector}.
356     * @throws IllegalStateException If the {@code BundleConetxt}
357     *     with which this {@code ServiceCollector} was created is no 
358     *     longer valid.
359     */
360    public void open(boolean collectAllServices) throws IllegalStateException {
361        synchronized (this) {
362            if (isOpen()) {
363                // Already open, don't treat as error.
364                return;
365            }
366            modificationCount[0] = 0;
367            try {
368                registerListener(collectAllServices);
369                if (collectClass != null) {
370                    initialReferences.addAll(getInitialReferences(
371                        collectAllServices, collectClass, null));
372                } else if (collectReference != null) {
373                    if (collectReference.getBundle() != null) {
374                        initialReferences.add(collectReference);
375                    }
376                } else { /* user supplied filter */
377                    initialReferences.addAll(getInitialReferences(
378                        collectAllServices, null, listenerFilter));
379                }
380                processInitial();
381            } catch (InvalidSyntaxException e) {
382                throw new RuntimeException(
383                    "unexpected InvalidSyntaxException: " + e.getMessage(),
384                    e);
385            }
386        }
387    }
388
389    private void registerListener(boolean collectAllServices)
390            throws InvalidSyntaxException {
391        if (collectAllServices) {
392            listener = new AllServiceListener() {
393                @Override
394                public void serviceChanged(ServiceEvent event) {
395                    ServiceCollector.this.serviceChanged(event);
396                }
397            };
398        } else {
399            listener = new ServiceListener() {
400
401                @Override
402                public void serviceChanged(ServiceEvent event) {
403                    ServiceCollector.this.serviceChanged(event);
404                }
405
406            };
407        }
408        context.addServiceListener(listener, listenerFilter);
409    }

411    /**
412     * Returns the list of initial {@code ServiceReference}s that will be
413     * collected by this {@code ServiceCollector}.
414     * 
415     * @param collectAllServices If {@code true}, use
416     *        {@code getAllServiceReferences}.
417     * @param className The class name with which the service was registered, or
418     *        {@code null} for all services.
419     * @param filterString The filter criteria or {@code null} for all services.
420     * @return The list of initial {@code ServiceReference}s.
421     * @throws InvalidSyntaxException If the specified filterString has an
422     *         invalid syntax.
423     */
424    private List<ServiceReference<S>> getInitialReferences(
425            boolean collectAllServices, String className, String filterString)
426            throws InvalidSyntaxException {
427        @SuppressWarnings("unchecked")
428        ServiceReference<S>[] result
429            = (ServiceReference<S>[]) ((collectAllServices)
430                ? context.getAllServiceReferences(className, filterString)
431                : context.getServiceReferences(className, filterString));
432        if (result == null) {
433            return Collections.emptyList();
434        }
435        return Arrays.asList(result);
436    }

438    private void processInitial() {
439        while (true) {
440            ServiceReference<S> reference;
441            synchronized (this) {
442                if (!isOpen() || (initialReferences.size() == 0)) {
443                    return; /* we are done */
444                }
445                // Get one...
446                Iterator<ServiceReference<S>> iter
447                    = initialReferences.iterator();
448                reference = iter.next();
449                iter.remove();
450            }
451            // Process as if it was just registered.
452            addToCollected(reference);
453        }
454    }
455
456    private void serviceChanged(ServiceEvent event) {
457        /*
458         * Check if we had a delayed call (which could happen when we
459         * close).
460         */
461        if (!isOpen()) {
462            return;
463        }
464        @SuppressWarnings("unchecked")
465        final ServiceReference<S> reference
466            = (ServiceReference<S>) event.getServiceReference();
467
468        switch (event.getType()) {
469        case ServiceEvent.REGISTERED:
470            addToCollected(reference);
471            break;
472        case ServiceEvent.MODIFIED:
473            synchronized (this) {
474                T service = collected.get(reference);
475                if (service == null) {
476                    // Probably still in initialReferences, ignore.
477                    return;
478                }
479                modified();
480                Optional.ofNullable(onModified)
481                    .ifPresent(cb -> cb.accept(reference, service));
482            }
483            break;
484        case ServiceEvent.MODIFIED_ENDMATCH:
485        case ServiceEvent.UNREGISTERING:
486            removeFromCollected(reference);
487            break;
488        }
489
490    }

492    /**
493     * Add the given reference to the collected services.
494     * 
495     * @param reference reference to be collected.
496     */
497    private void addToCollected(final ServiceReference<S> reference) {
498        synchronized (this) {
499            if (!isOpen()) {
500                // We have been closed.
501                return;
502            }
503            if (collected.get(reference) != null) {
504                /* if we are already collecting this reference */
505                return; /* skip this reference */
506            }
507            if (initialReferences.contains(reference)) {
508                // Skip, will be added as initial.
509                return;
510            }
511
512            @SuppressWarnings("unchecked")
513            T service = (svcGetter == null)
514                ? (T) context.getService(reference)
515                : svcGetter.apply(context, reference);
516            if (service == null) {
517                // Has either vanished in the meantime (should not happen
518                // when processing a REGISTERED event, but may happen when
519                // processing a reference from initialReferences) or was
520                // filtered by the service getter.
521                return;
522            }
523            boolean wasEmpty = collected.isEmpty();
524            collected.put(reference, service);
525            modified();
526            if (wasEmpty) {
527                Optional.ofNullable(onBound)
528                    .ifPresent(cb -> cb.accept(reference, service));
529            }
530            Optional.ofNullable(onAdded)
531                .ifPresent(cb -> cb.accept(reference, service));
532        }
533    }
534
535    private void removeFromCollected(ServiceReference<S> reference) {
536        synchronized (this) {
537            T service = collected.get(reference);
538            if (service == null) {
539                return;
540            }
541            Optional.ofNullable(onRemoving)
542                .ifPresent(cb -> cb.accept(reference, service));
543            if (collected.size() == 1) {
544                Optional.ofNullable(onUnbinding)
545                    .ifPresent(cb -> cb.accept(reference, service));
546            }
547            context.ungetService(reference);
548            collected.remove(reference);
549            if (isOpen()) {
550                modified();
551            }
552        }
553    }

555    /**
556     * Stops collecting services.
557     */
558    public void close() {
559        synchronized (this) {
560            context.removeServiceListener(listener);
561            synchronized (modificationCount) {
562                modificationCount[0] = -1;
563                modificationCount.notifyAll();
564            }
565            while (!collected.isEmpty()) {
566                removeFromCollected(collected.lastKey());
567            }
568            cachedService = null;
569        }
570    }

572    /**
573     * Wait for at least one service to be collected by this
574     * {@code ServiceCollector}. This method will also return when this
575     * {@code ServiceCollector} is closed from another thread.
576     * <p>
577     * It is strongly recommended that {@code waitForService} is not used 
578     * during the calling of the {@code BundleActivator} methods.
579     * {@code BundleActivator} methods are expected to complete in a short
580     * period of time.
581     * 
582     * @param timeout The time interval in milliseconds to wait. If zero, the
583     *     method will wait indefinitely.
584     * @return Returns the result of {@link #service()}.
585     * @throws InterruptedException If another thread has interrupted the
586     *     current thread.
587     * @throws IllegalArgumentException If the value of timeout is negative.
588     */
589    public Optional<T> waitForService(long timeout)
590            throws InterruptedException {
591        if (timeout < 0) {
592            throw new IllegalArgumentException("timeout value is negative");
593        }
594
595        T service = service().orElse(null);
596        if (service != null) {
597            return Optional.of(service);
598        }
599
600        final long endTime
601            = (timeout == 0) ? 0 : (System.currentTimeMillis() + timeout);
602        while (true) {
603            synchronized (modificationCount) {
604                if (modificationCount() < 0) {
605                    return Optional.empty();
606                }
607                modificationCount.wait(timeout);
608            }
609            Optional<T> found = service();
610            if (found.isPresent()) {
611                return found;
612            }
613            if (endTime > 0) { // if we have a timeout
614                timeout = endTime - System.currentTimeMillis();
615                if (timeout <= 0) { // that has expired
616                    return Optional.empty();
617                }
618            }
619        }
620    }

635    /**
636     * Returns a {@code ServiceReference} for one of the services collected
637     * by this {@code ServiceCollector}.
638     * <p>
639     * If multiple services have been collected, the service with the highest
640     * ranking (as specified in its {@code service.ranking} property) is
641     * returned. If there is a tie in ranking, the service with the lowest
642     * service id (as specified in its {@code service.id} property); that is,
643     * the service that was registered first is returned. This is the same
644     * algorithm used by {@code BundleContext.getServiceReference}.
645     * 
646     * @return an optional {@code ServiceReference}
647     */
648    public Optional<ServiceReference<S>> serviceReference() {
649        try {
650            synchronized (this) {
651                return Optional.of(collected.firstKey());
652            }
653        } catch (NoSuchElementException e) {
654            return Optional.empty();
655        }
656    }

672    /**
673     * Returns a service object for one of the services collected by this
674     * {@code ServiceCollector}. This is effectively the same as 
675     * {@code serviceReference().flatMap(ref -> service(ref))}.
676     * <P>
677     * The result value is cached, so refrain from implementing another
678     * cache in the invoking code.
679     * 
680     * @return an optional service object
681     */
682    public Optional<T> service() {
683        final T cached = cachedService;
684        if (cached != null) {
685            return Optional.of(cached);
686        }
687        synchronized (this) {
688            Iterator<T> iter = collected.values().iterator();
689            if (iter.hasNext()) {
690                cachedService = iter.next();
691                return Optional.of(cachedService);
692            }
693        }
694        return Optional.empty();
695    }

758     * objects for all services collected by this {@code ServiceCollector}.
759     * The map is sorted in reverse natural order of {@code ServiceReference}.
760     * That is, the first entry is the service with the highest ranking and the
761     * lowest service id.
762     * 
763     * @return A {@code SortedMap} with the {@code ServiceReference}s and
764     *         service objects for all services collected by this
765     *         {@code ServiceCollector}. If no services have been collected,
766     *         then the returned map is empty.
767     */
768    public SortedMap<ServiceReference<S>, T> collected() {
769        synchronized (this) {
770            if (collected.isEmpty()) {
771                return new TreeMap<>(Collections.reverseOrder());
772            }
773            return Collections.unmodifiableSortedMap(new TreeMap<>(collected));
774        }
775    }
776
777    /**
778     * Return if this {@code ServiceCollector} is empty.
779     * 
780     * @return {@code true} if this {@code ServiceCollector} 
781     *     has not collected any services.
782     */
783    public boolean isEmpty() {
784        synchronized (this) {
785            return collected.isEmpty();
786        }
787    }

789    /**
790     * Remove a service from this {@code ServiceCollector}.
791     * 
792     * The specified service will be removed from this {@code ServiceCollector}.
793     * 
794     * @param reference The reference to the service to be removed.
795     */
796    public void remove(ServiceReference<S> reference) {
797        removeFromCollected(reference);
798    }
