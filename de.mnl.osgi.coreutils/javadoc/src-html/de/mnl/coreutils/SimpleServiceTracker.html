039/**
040 * The <code>SimpleServiceTracker</code> class simplifies using services
041 * from the OSGi Framework's service registry beyond the simplification
042 * provided by the {@link ServiceTracker} from the 
043 * <a href="https://osgi.org/specification/osgi.core/7.0.0/util.tracker.html">
044 * OSGi Core Specification</a>.
045 * <p>
046 * The tracker calls registered callbacks when the first or any 
047 * service becomes available. The implementation ensures that
048 * at least one service can be obtained from the tracker during the 
049 * execution of these callbacks. At least one service will also
050 * be available after the execution of the callbacks until the 
051 * "last unavailable" callback has been invoked.
052 * <P>
053 * Functions enabled/disabled in the respective callbacks can therefore 
054 * rely on a service being available while they are enabled. This
055 * includes, as a special case, starting and stopping threads in the
056 * respective callbacks.
057 * <p>
058 * A {@code SimpleServiceTracker} object is constructed with search 
059 * criteria. It can then be opened to begin tracking all services in 
060 * the Framework's service registry that match the specified search 
061 * criteria. The <code>SimpleServiceTracker</code> is currently
062 * implemented using a standard {@link ServiceTracker} as a delegee.
063 * <p>
064 * The <code>SimpleServiceTracker</code> class is thread-safe.
065 *
066 * @param <S> the type of the service to be tracked
067 * @param <W> the type of the objects returned as tracking results
068 *      (see {@link #setWrapper(Function)}); usually, this is the 
069 *      same type as {@code S}
070 */
071public class SimpleServiceTracker<S, W> implements AutoCloseable {
072
073    private final ServiceTracker<S, W> delegee;
074    /*
075     * The service data known from ServiceTracker#addingService
076     * but not yet known to have been added. Or the service data
077     * already removed from the ServiceTracker but provided as a fallback
078     * while calling onLastUnavailable in ServiceTracker#removedService.
079     */
080    private FallbackData fallback;
081    private BiConsumer<ServiceReference<S>, W> onFirstAvailable;
082    private BiConsumer<ServiceReference<S>, W> onAvailable;
083    private BiConsumer<ServiceReference<S>, W> onUnavailable;
084    private BiConsumer<ServiceReference<S>, W> onLastUnavailable;
085    private BiConsumer<ServiceReference<S>, W> onModified;
086    private Function<S, W> wrapper;
087    private int minTrackingCount;

258    /**
259     * Wait for at least one service to be tracked by this
260     * tracker. This method will also return when this
261     * tracked is closed from another thread.
262     * <p>
263     * It is strongly recommended that {@code waitForService} is not used 
264     * during the calling of the {@code BundleActivator} methods.
265     * {@code BundleActivator} methods are expected to complete in a short
266     * period of time.
267     * 
268     * @param timeout The time interval in milliseconds to wait. If zero, the
269     *     method will wait indefinitely.
270     * @return Returns the result of {@link #service()}.
271     * @throws InterruptedException If another thread has interrupted the
272     *     current thread.
273     * @throws IllegalArgumentException If the value of timeout is negative.
274     */
275    public W waitForService(long timeout) throws InterruptedException {
276        return delegee.waitForService(timeout);
277    }

279    /**
280     * Return an array of {@code ServiceReference}s for all services being
281     * tracked by this tracker.
282     * 
283     * @return Array of {@code ServiceReference}s or an empty array if no 
284     *     services are being tracked.
285     */
286    @SuppressWarnings("unchecked")
287    public ServiceReference<S>[] serviceReferences() {
288        FallbackData maybeAdd = fallback;
289        if (maybeAdd == null) {
290            return Optional.ofNullable(delegee.getServiceReferences())
291                .orElse((ServiceReference<S>[]) new ServiceReference[0]);
292        }
293        // Add fallback service reference if not already in list returned
294        // by delegee.
295        Object maybeAddId = maybeAdd.svcRef().getProperty(Constants.SERVICE_ID);
296        List<ServiceReference<S>> result = new ArrayList<>();
297        ServiceReference<S>[] knownRefs = delegee.getServiceReferences();
298        if (knownRefs != null) {
299            for (ServiceReference<S> ref : knownRefs) {
300                result.add(ref);
301                if (ref.getProperty(Constants.SERVICE_ID).equals(maybeAddId)) {
302                    // Known and therefore already added.
303                    maybeAdd = null;
304                }
305            }
306        }
307        if (maybeAdd != null) {
308            result.add(maybeAdd.svcRef());
309        }
310        return result.toArray(new ServiceReference[0]);
311    }
312
313    /**
314     * Returns a {@code ServiceReference} for one of the services being tracked
315     * by this tracker.
316     * <p>
317     * If multiple services are being tracked, the service with the highest
318     * ranking (as specified in its {@code service.ranking} property) is
319     * returned. If there is a tie in ranking, the service with the lowest
320     * service id (as specified in its {@code service.id} property); that is,
321     * the service that was registered first is returned. This is the same
322     * algorithm used by {@code BundleContext.getServiceReference}.
323     * 
324     * @return an optional {@code ServiceReference}
325     */
326    public Optional<ServiceReference<S>> serviceReference() {
327        return Optional.ofNullable(delegee.getServiceReference())
328            .map(ref -> Optional.of(ref))
329            .orElse(Optional.ofNullable(fallback).map(FallbackData::svcRef));
330    }
331
332    /**
333     * Returns the service object for the specified {@code ServiceReference} 
334     * if the specified referenced service is being tracked by this tracker.
335     * 
336     * @param reference the reference to the desired service.
337     * @return an optional service object
338     */
339    public Optional<W> service(ServiceReference<S> reference) {
340        W known = delegee.getService(reference);
341        if (known != null) {
342            return Optional.of(known);
343        }
344        // Maybe fallback can be used...
345        FallbackData maybe = fallback;
346        return Optional.ofNullable(maybe.svcRef())
347            .filter(ref -> ref.getProperty(Constants.SERVICE_ID)
348                .equals(reference.getProperty(Constants.SERVICE_ID)))
349            .map(found -> maybe.service());
350    }
351
352    /**
353     * Returns a service object for one of the services being tracked by this
354     * tracker.
355     * 
356     * @return an optional service object
357     */
358    public Optional<W> service() {
359        return Optional.ofNullable(
360            Optional.ofNullable(delegee.getService()).orElse(Optional
361                .ofNullable(fallback).map(FallbackData::service).orElse(null)));
362    }

364    /**
365     * Return an array of service objects for all services being tracked by this
366     * tracker.
367     * 
368     * @return an array of service objects which may be empty
369     */
370    public Object[] services() {
371        List<W> result = new ArrayList<>();
372        for (ServiceReference<S> ref : serviceReferences()) {
373            service(ref).ifPresent(s -> result.add(s));
374        }
375        return result.toArray(new Object[0]);
376    }

389    /**
390     * Returns the tracking count for this tracker.
391     * 
392     * The tracking count is initialized to 0 when this tracker
393     * is opened. Every time a service is added, modified or removed from this
394     * tracker, the tracking count is incremented.
395     * <p>
396     * The tracking count can be used to determine if this
397     * tracker has added, modified or removed a service by
398     * comparing a tracking count value previously collected with the current
399     * tracking count value. If the value has not changed, then no service has
400     * been added, modified or removed from this tracker since
401     * the previous tracking count was collected.
402     * 
403     * @return The tracking count for this tracker or -1 if this
404     *       tracker is not open.
405     */
406    public int trackingCount() {
407        return Math.max(minTrackingCount, delegee.getTrackingCount());
408    }

410    /**
411     * Return a {@code SortedMap} of the {@code ServiceReference}s and service
412     * objects for all services being tracked by this tracker.
413     * The map is sorted in reverse natural order of {@code ServiceReference}.
414     * That is, the first entry is the service with the highest ranking and the
415     * lowest service id.
416     * 
417     * @return A {@code SortedMap} with the {@code ServiceReference}s and
418     *         service objects for all services being tracked by this
419     *         tracker. If no services are being tracked, then
420     *         the returned map is empty.
421     */
422    public SortedMap<ServiceReference<S>, W> tracked() {
423        SortedMap<ServiceReference<S>, W> result = delegee.getTracked();
424        FallbackData maybe = fallback;
425        if (maybe != null) {
426            Object maybeId = maybe.svcRef().getProperty(Constants.SERVICE_ID);
427            // There's no equals/hashCode on ServiceReference, sigh.
428            boolean found = false;
429            for (ServiceReference<S> ref : result.keySet()) {
430                if (ref.getProperty(Constants.SERVICE_ID).equals(maybeId)) {
431                    found = true;
432                    break;
433                }
434            }
435            if (!found) {
436                result.put(maybe.svcRef(), maybe.service());
437            }
438        }
439        return result;
440    }

444     * 
445     * @return {@code true} if this tracker is not tracking any
446     *         services.

448    public boolean isEmpty() {
449        return delegee.isEmpty();

452    private class FallbackData {
453        private final ServiceReference<S> svcRef;
454        private final W service;
455
456        public FallbackData(ServiceReference<S> svcRef, W service) {
457            this.svcRef = svcRef;
458            this.service = service;
459        }
460
461        public final ServiceReference<S> svcRef() {
462            return svcRef;
463        }
464
465        public final W service() {
466            return service;
467        }
468    }

470    private class MyTracker extends ServiceTracker<S, W> {
471
472        public MyTracker(BundleContext context, Class<S> clazz,
473                ServiceTrackerCustomizer<S, W> customizer) {
474            super(context, clazz, customizer);
475        }
476
477        public MyTracker(BundleContext context, Filter filter,
478                ServiceTrackerCustomizer<S, W> customizer) {
479            super(context, filter, customizer);
480        }
481
482        public MyTracker(BundleContext context, ServiceReference<S> reference,
483                ServiceTrackerCustomizer<S, W> customizer) {
484            super(context, reference, customizer);
485        }
486
487        public MyTracker(BundleContext context, String clazz,
488                ServiceTrackerCustomizer<S, W> customizer) {
489            super(context, clazz, customizer);
490        }
491
492        @Override
493        public W addingService(ServiceReference<S> reference) {
494            synchronized (this) {
495                minTrackingCount = delegee.getTrackingCount() + 1;
496                @SuppressWarnings("unchecked")
497                W newService = Optional.ofNullable(wrapper)
498                    .map(w -> w.apply(context.getService(reference)))
499                    .orElse((W) context.getService(reference));
500                fallback = new FallbackData(reference, newService);
501                if (delegee.isEmpty()) {
502                    Optional.ofNullable(onFirstAvailable)
503                        .ifPresent(h -> h.accept(reference, newService));
504                }
505                if (onAvailable != null) {
506                    Optional.ofNullable(onAvailable)
507                        .ifPresent(h -> h.accept(reference, newService));
508                }
509                return newService;
510            }
511        }
512
513        @Override
514        public void removedService(ServiceReference<S> reference, W service) {
515            synchronized (this) {
516                fallback = new FallbackData(reference, service);
517                Optional.ofNullable(onUnavailable)
518                    .ifPresent(h -> h.accept(reference, service));
519                if (delegee.isEmpty()) {
520                    Optional.ofNullable(onLastUnavailable)
521                        .ifPresent(h -> h.accept(reference, service));
522                }
523                fallback = null;
524                super.removedService(reference, service);
525            }
526        }
527
528        @Override
529        public void modifiedService(ServiceReference<S> reference, W service) {
530            Optional.ofNullable(onModified)
531                .ifPresent(h -> h.accept(reference, service));
532        }
533    }
