058    private class TestCollector<S> extends ServiceCollector<S> {
059
060        public Map<String, Integer> callbacks = new HashMap<>();
061
062        private void incCount(String counter) {
063            callbacks.put(counter, callbacks.getOrDefault(counter, 0) + 1);
064        }
065
066        public TestCollector(BundleContext context, Class<S> clazz) {
067            super(context, clazz);
068            setOnBoundFirst(
069                (ref, svc) -> {
070                    incCount("firstBound");
071                    assertTrue(service().isPresent());
072                    assertEquals(svc, service().get());
073                    assertEquals(svc, services().get(0));
074                    assertEquals(ref, serviceReference().get());
075                    assertEquals(ref, serviceReferences().get(0));
076                    assertEquals(ref, collected().keySet().iterator().next());
077                    assertEquals(svc, collected().values().iterator().next());
078                    assertEquals(1, size());
079                });
080            setOnBound((ref, svc) -> incCount("bound"));
081            setOnUnbinding((ref, svc) -> incCount("unbinding"));
082            setOnUnbindingLast((ref, svc) -> {
083                incCount("lastUnbinding");
084                assertEquals(svc, service().get());
085            });
086        }
087    }

089    /**
090     * An available service must be found.
091     * 
092     * @throws InterruptedException
093     */
094    @Test
095    @SuppressWarnings("resource")
096    public void testUseAvailable() throws InterruptedException {
097        cleanup.add(context.registerService(SampleService.class,
098            new SampleService(), null));
099        TestCollector<SampleService> coll1bck;
100        try (TestCollector<SampleService> coll1
101            = new TestCollector<SampleService>(
102                context, SampleService.class)) {
103            coll1bck = coll1;
104            coll1.open();
105            coll1.waitForService(1000);
106            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
107            assertEquals(1, coll1.callbacks.get("bound").intValue());
108        }
109        assertFalse(coll1bck.service().isPresent());
110        assertEquals(1, coll1bck.callbacks.get("unbinding").intValue());
111        assertEquals(1, coll1bck.callbacks.get("lastUnbinding").intValue());
112    }

114    /**
115     * Two services must be found and rankings must be observed.
116     * 
117     * @throws InterruptedException
118     */
119    @SuppressWarnings("resource")
120    @Test
121    public void testRanking() throws InterruptedException {
122        // Prepare
123        Hashtable<String, Object> props = new Hashtable<>();
124        props.put(Constants.SERVICE_RANKING, 1);
125        SampleService service1 = new SampleService();
126        cleanup.add(context.registerService(SampleService.class,
127            service1, props));
128        props.put(Constants.SERVICE_RANKING, 2);
129        SampleService service2 = new SampleService();
130        cleanup.add(context.registerService(SampleService.class,
131            service2, props));
132
133        // Now run
134        TestCollector<SampleService> coll1bck;
135        try (TestCollector<SampleService> coll1
136            = new TestCollector<SampleService>(context,
137                SampleService.class)) {
138            coll1bck = coll1;
139            coll1.open();
140            coll1.waitForService(1000);
141            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
142            assertEquals(2, coll1.callbacks.get("bound").intValue());
143            // Weak test, may be lucky
144            assertEquals(service2,
145                coll1.collected().values().iterator().next());
146        }
147        assertFalse(coll1bck.service().isPresent());
148        assertEquals(2, coll1bck.callbacks.get("unbinding").intValue());
149        assertEquals(1, coll1bck.callbacks.get("lastUnbinding").intValue());
150    }

152    /**
153     * Adding and removing must have effects.
154     * 
155     * @throws InterruptedException
156     */
157    @SuppressWarnings("resource")
158    @Test
159    public void testDynamic() throws InterruptedException {
160        TestCollector<SampleService> coll1bck;
161        try (TestCollector<SampleService> coll1
162            = new TestCollector<SampleService>(context,
163                SampleService.class)) {
164            coll1bck = coll1;
165            coll1.open();
166            assertTrue(coll1.isEmpty());
167
168            // Add first service
169            Hashtable<String, Object> props = new Hashtable<>();
170            props.put(Constants.SERVICE_RANKING, 1);
171            SampleService service1 = new SampleService();
172            final ServiceRegistration<SampleService> reg1
173                = context.registerService(
174                    SampleService.class, service1, props);
175            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
176            assertEquals(1, coll1.callbacks.get("bound").intValue());
177            assertEquals(service1, coll1.service().get());
178
179            // Add second
180            props.put(Constants.SERVICE_RANKING, 2);
181            SampleService service2 = new SampleService();
182            final ServiceRegistration<SampleService> reg2
183                = context.registerService(
184                    SampleService.class, service2, props);
185            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
186            assertEquals(2, coll1.callbacks.get("bound").intValue());
187            assertEquals(service2, coll1.service().get());
188
189            // Remove second
190            reg2.unregister();
191            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
192            assertEquals(2, coll1.callbacks.get("bound").intValue());
193            assertEquals(1, coll1.callbacks.get("unbinding").intValue());
194            assertEquals(service1, coll1.service().get());
195
196            // Remove first
197            reg1.unregister();
198            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
199            assertEquals(2, coll1.callbacks.get("bound").intValue());
200            assertEquals(2, coll1.callbacks.get("unbinding").intValue());
201            assertEquals(1, coll1.callbacks.get("lastUnbinding").intValue());
202            assertFalse(coll1.service().isPresent());
203
204        }
205    }
