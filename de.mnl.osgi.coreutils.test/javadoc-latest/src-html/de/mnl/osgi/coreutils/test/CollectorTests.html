056    private class TestCollector<S> extends ServiceCollector<S, S> {
057
058        public Map<String, Integer> callbacks = new HashMap<>();
059
060        private void incCount(String counter) {
061            callbacks.put(counter, callbacks.getOrDefault(counter, 0) + 1);
062        }
063
064        public TestCollector(BundleContext context, Class<S> clazz) {
065            super(context, clazz);
066            setOnBound(
067                (ref, svc) -> {
068                    incCount("firstBound");
069                    assertTrue(service().isPresent());
070                    assertEquals(svc, service().get());
071                    assertEquals(svc, services().get(0));
072                    assertEquals(ref, serviceReference().get());
073                    assertEquals(ref, serviceReferences().get(0));
074                    assertEquals(ref, collected().keySet().iterator().next());
075                    assertEquals(svc, collected().values().iterator().next());
076                    assertEquals(1, size());
077                });
078            setOnAdded((ref, svc) -> incCount("bound"));
079            setOnRemoving((ref, svc) -> incCount("unbinding"));
080            setOnUnbinding((ref, svc) -> {
081                incCount("lastUnbinding");
082                assertEquals(svc, service().get());
083            });
084            setOnModfied((ref, svc) -> incCount("modified"));
085        }
086    }

153    /**
154     * Adding and removing must have effects.
155     *
156     * @throws InterruptedException the interrupted exception
157     */
158    @Test
159    public void testDynamic() throws InterruptedException {
160        try (TestCollector<SampleService1> coll1
161            = new TestCollector<SampleService1>(context,
162                SampleService1.class)) {
163            coll1.open();
164            assertTrue(coll1.isEmpty());
165
166            // Add first service
167            Hashtable<String, Object> props = new Hashtable<>();
168            props.put(Constants.SERVICE_RANKING, 1);
169            SampleService1 service1 = new SampleService1();
170            final ServiceRegistration<SampleService1> reg1
171                = context.registerService(
172                    SampleService1.class, service1, props);
173            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
174            assertEquals(1, coll1.callbacks.get("bound").intValue());
175            assertNull(coll1.callbacks.get("modified"));
176            assertEquals(service1, coll1.service().get());
177
178            // Add second
179            props.put(Constants.SERVICE_RANKING, 2);
180            SampleService1 service2 = new SampleService1();
181            final ServiceRegistration<SampleService1> reg2
182                = context.registerService(
183                    SampleService1.class, service2, props);
184            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
185            assertEquals(2, coll1.callbacks.get("bound").intValue());
186            assertEquals(1, coll1.callbacks.get("modified").intValue());
187            assertEquals(service2, coll1.service().get());
188
189            // Remove second
190            reg2.unregister();
191            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
192            assertEquals(2, coll1.callbacks.get("bound").intValue());
193            assertEquals(1, coll1.callbacks.get("unbinding").intValue());
194            assertEquals(2, coll1.callbacks.get("modified").intValue());
195            assertEquals(service1, coll1.service().get());
196
197            // Remove first
198            reg1.unregister();
199            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
200            assertEquals(2, coll1.callbacks.get("bound").intValue());
201            assertEquals(2, coll1.callbacks.get("unbinding").intValue());
202            assertEquals(1, coll1.callbacks.get("lastUnbinding").intValue());
203            assertEquals(2, coll1.callbacks.get("modified").intValue());
204            assertFalse(coll1.service().isPresent());
205        }
206    }

208    /**
209     * Attempt to test concurrency.
210     *
211     * @throws TimeoutException the timeout exception
212     * @throws InterruptedException the interrupted exception
213     */
214    @Test
215    public void testConcurrent() throws TimeoutException, InterruptedException {
216        Object starter = new Object();
217        Waiter waiter = new Waiter();
218        final int N = 100;
219        CountDownLatch started = new CountDownLatch(N);
220        int[] services = new int[N];
221        // Threads prepared
222        try (TestCollector<SampleService1> coll1
223            = new TestCollector<SampleService1>(context,
224                SampleService1.class)) {
225            coll1.open();
226            assertTrue(coll1.isEmpty());
227            // Prepare threads
228            for (int i = 0; i < N; i++) {
229                final int count = i;
230                new Thread(() -> {
231                    started.countDown();
232                    try {
233                        started.await();
234                    } catch (InterruptedException e1) {
235                        waiter.fail();
236                    }
237                    try {
238                        Thread.sleep(100 + (count / 4) * 8);
239                    } catch (InterruptedException e) {
240                        waiter.fail(e);
241                    }
242                    runTestInThread(waiter, coll1, count, services);
243                }).start();
244            }
245            started.await();
246            synchronized (starter) {
247                starter.notifyAll();
248            }
249            waiter.await(10000, N);
250            assertEquals(N, coll1.callbacks.get("bound").intValue());
251            assertEquals(N, coll1.callbacks.get("unbinding").intValue());
252            assertFalse(coll1.service().isPresent());
253        }
254    }
255
256    private void runTestInThread(Waiter waiter,
257            TestCollector<SampleService1> coll1, int count, int[] services) {
258        // Add service
259        Hashtable<String, Object> props = new Hashtable<>();
260        props.put(Constants.SERVICE_RANKING, 1);
261        SampleService1 service1 = new SampleService1();
262        final ServiceRegistration<SampleService1> reg1
263            = context.registerService(
264                SampleService1.class, service1, props);
265        services[count] = coll1.serviceReferences().size();
266        try {
267            Thread.sleep(count % 10);
268        } catch (InterruptedException e) {
269            waiter.fail(e);
270        }
271
272        // Remove service
273        reg1.unregister();
274        waiter.resume();
275    }
