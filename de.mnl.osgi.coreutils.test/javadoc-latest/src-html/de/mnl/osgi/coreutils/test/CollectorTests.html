057    private class TestCollector<S> extends ServiceCollector<S, S> {
058
059        public Map<String, Integer> callbacks = new HashMap<>();
060
061        private void incCount(String counter) {
062            callbacks.put(counter, callbacks.getOrDefault(counter, 0) + 1);
063        }
064
065        public TestCollector(BundleContext context, Class<S> clazz) {
066            super(context, clazz);
067            setOnBound(
068                (ref, svc) -> {
069                    incCount("firstBound");
070                    assertTrue(service().isPresent());
071                    assertEquals(svc, service().get());
072                    assertEquals(svc, services().get(0));
073                    assertEquals(ref, serviceReference().get());
074                    assertEquals(ref, serviceReferences().get(0));
075                    assertEquals(ref, collected().keySet().iterator().next());
076                    assertEquals(svc, collected().values().iterator().next());
077                    assertEquals(1, size());
078                });
079            setOnAdded((ref, svc) -> incCount("bound"));
080            setOnRemoving((ref, svc) -> incCount("unbinding"));
081            setOnUnbinding((ref, svc) -> {
082                incCount("lastUnbinding");
083                assertEquals(svc, service().get());
084            });
085            setOnModfied((ref, svc) -> incCount("modified"));
086        }
087    }

154    /**
155     * Adding and removing must have effects.
156     *
157     * @throws InterruptedException the interrupted exception
158     */
159    @Test
160    public void testDynamic() throws InterruptedException {
161        try (TestCollector<SampleService1> coll1
162            = new TestCollector<SampleService1>(context,
163                SampleService1.class)) {
164            coll1.open();
165            assertTrue(coll1.isEmpty());
166
167            // Add first service
168            Hashtable<String, Object> props = new Hashtable<>();
169            props.put(Constants.SERVICE_RANKING, 1);
170            SampleService1 service1 = new SampleService1();
171            final ServiceRegistration<SampleService1> reg1
172                = context.registerService(
173                    SampleService1.class, service1, props);
174            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
175            assertEquals(1, coll1.callbacks.get("bound").intValue());
176            assertNull(coll1.callbacks.get("modified"));
177            assertEquals(service1, coll1.service().get());
178
179            // Add second
180            props.put(Constants.SERVICE_RANKING, 2);
181            SampleService1 service2 = new SampleService1();
182            final ServiceRegistration<SampleService1> reg2
183                = context.registerService(
184                    SampleService1.class, service2, props);
185            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
186            assertEquals(2, coll1.callbacks.get("bound").intValue());
187            assertEquals(1, coll1.callbacks.get("modified").intValue());
188            assertEquals(service2, coll1.service().get());
189
190            // Remove second
191            reg2.unregister();
192            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
193            assertEquals(2, coll1.callbacks.get("bound").intValue());
194            assertEquals(1, coll1.callbacks.get("unbinding").intValue());
195            assertEquals(2, coll1.callbacks.get("modified").intValue());
196            assertEquals(service1, coll1.service().get());
197
198            // Remove first
199            reg1.unregister();
200            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
201            assertEquals(2, coll1.callbacks.get("bound").intValue());
202            assertEquals(2, coll1.callbacks.get("unbinding").intValue());
203            assertEquals(1, coll1.callbacks.get("lastUnbinding").intValue());
204            assertEquals(2, coll1.callbacks.get("modified").intValue());
205            assertFalse(coll1.service().isPresent());
206        }
207    }

209    /**
210     * Attempt to test concurrency.
211     *
212     * @throws TimeoutException the timeout exception
213     * @throws InterruptedException the interrupted exception
214     */
215    @Test
216    public void testConcurrent() throws TimeoutException, InterruptedException {
217        Object starter = new Object();
218        Waiter waiter = new Waiter();
219        final int N = 100;
220        CountDownLatch started = new CountDownLatch(N);
221        int[] services = new int[N];
222        // Threads prepared
223        try (TestCollector<SampleService1> coll1
224            = new TestCollector<SampleService1>(context,
225                SampleService1.class)) {
226            coll1.open();
227            assertTrue(coll1.isEmpty());
228            // Prepare threads
229            for (int i = 0; i < N; i++) {
230                final int count = i;
231                new Thread(() -> {
232                    started.countDown();
233                    try {
234                        started.await();
235                    } catch (InterruptedException e1) {
236                        waiter.fail();
237                    }
238                    try {
239                        Thread.sleep(100 + (count / 4) * 8);
240                    } catch (InterruptedException e) {
241                        waiter.fail(e);
242                    }
243                    runTestInThread(waiter, coll1, count, services);
244                }).start();
245            }
246            started.await();
247            synchronized (starter) {
248                starter.notifyAll();
249            }
250            waiter.await(10000, N);
251            assertEquals(N, coll1.callbacks.get("bound").intValue());
252            assertEquals(N, coll1.callbacks.get("unbinding").intValue());
253            assertFalse(coll1.service().isPresent());
254        }
255    }
256
257    private void runTestInThread(Waiter waiter,
258            TestCollector<SampleService1> coll1, int count, int[] services) {
259        // Add service
260        Hashtable<String, Object> props = new Hashtable<>();
261        props.put(Constants.SERVICE_RANKING, 1);
262        SampleService1 service1 = new SampleService1();
263        final ServiceRegistration<SampleService1> reg1
264            = context.registerService(
265                SampleService1.class, service1, props);
266        services[count] = coll1.serviceReferences().size();
267        try {
268            Thread.sleep(count % 10);
269        } catch (InterruptedException e) {
270            waiter.fail(e);
271        }
272
273        // Remove service
274        reg1.unregister();
275        waiter.resume();
276    }
