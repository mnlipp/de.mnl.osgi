057    private class TestCollector<S> extends ServiceCollector<S, S> {
058
059        public Map<String, Integer> callbacks = new HashMap<>();
060
061        private void incCount(String counter) {
062            callbacks.put(counter, callbacks.getOrDefault(counter, 0) + 1);
063        }
064
065        public TestCollector(BundleContext context, Class<S> clazz) {
066            super(context, clazz);
067            setOnBound(
068                (ref, svc) -> {
069                    incCount("firstBound");
070                    assertTrue(service().isPresent());
071                    assertEquals(svc, service().get());
072                    assertEquals(svc, services().get(0));
073                    assertEquals(ref, serviceReference().get());
074                    assertEquals(ref, serviceReferences().get(0));
075                    assertEquals(ref, collected().keySet().iterator().next());
076                    assertEquals(svc, collected().values().iterator().next());
077                    assertEquals(1, size());
078                });
079            setOnAdded((ref, svc) -> incCount("bound"));
080            setOnRemoving((ref, svc) -> incCount("unbinding"));
081            setOnUnbinding((ref, svc) -> {
082                incCount("lastUnbinding");
083                assertEquals(svc, service().get());
084            });
085            setOnModfied((ref, svc) -> incCount("modified"));
086        }
087    }

136    /**
137     * A service in use cannot be unregistered
138     *
139     * @throws InterruptedException the interrupted exception
140     */
141    @Test
142    public void testDelayedRemoval() throws InterruptedException {
143        var serviceRef = context.registerService(SampleService1.class,
144            new SampleService1(), null);
145        CountDownLatch toBeStarted = new CountDownLatch(1);
146        Instant[] calcTimes = new Instant[2];
147        new Thread() {
148            public void run() {
149                try (TestCollector<SampleService1> coll1
150                    = new TestCollector<SampleService1>(
151                        context, SampleService1.class)) {
152                    coll1.open();
153                    coll1.waitForService(1000);
154                    coll1.withService(s -> {
155                        toBeStarted.countDown();
156                        try {
157                            Thread.sleep(250);
158                        } catch (InterruptedException e) {
159                            // Empty
160                        }
161                        calcTimes[0] = Instant.now();
162                        int result = s.add(20, 22);
163                        calcTimes[1] = Instant.now();
164                        return result;
165                    });
166                } catch (InterruptedException e) {
167                    // Empty
168                }
169            }
170        }.start();
171        // Wait for execution start
172        toBeStarted.await();
173        Instant unregStart = Instant.now();
174        serviceRef.unregister();
175        Instant unregEnd = Instant.now();
176
177        // Calculation must start after unreg start (else it's not a test)
178        assertTrue(calcTimes[0].isAfter(unregStart));
179        // Removal must finish after calc end (else locking doesn't work)
180        assertTrue(unregEnd.isAfter(calcTimes[1]));
181        // Make sure
182        assertTrue(unregEnd.toEpochMilli() - unregStart.toEpochMilli() > 100);
183    }

223    /**
224     * Adding and removing must have effects.
225     *
226     * @throws InterruptedException the interrupted exception
227     */
228    @Test
229    public void testDynamic() throws InterruptedException {
230        try (TestCollector<SampleService1> coll1
231            = new TestCollector<SampleService1>(context,
232                SampleService1.class)) {
233            coll1.open();
234            assertTrue(coll1.isEmpty());
235
236            // Add first service
237            Hashtable<String, Object> props = new Hashtable<>();
238            props.put(Constants.SERVICE_RANKING, 1);
239            SampleService1 service1 = new SampleService1();
240            final ServiceRegistration<SampleService1> reg1
241                = context.registerService(
242                    SampleService1.class, service1, props);
243            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
244            assertEquals(1, coll1.callbacks.get("bound").intValue());
245            assertNull(coll1.callbacks.get("modified"));
246            assertEquals(service1, coll1.service().get());
247
248            // Add second
249            props.put(Constants.SERVICE_RANKING, 2);
250            SampleService1 service2 = new SampleService1();
251            final ServiceRegistration<SampleService1> reg2
252                = context.registerService(
253                    SampleService1.class, service2, props);
254            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
255            assertEquals(2, coll1.callbacks.get("bound").intValue());
256            assertEquals(1, coll1.callbacks.get("modified").intValue());
257            assertEquals(service2, coll1.service().get());
258
259            // Remove second
260            reg2.unregister();
261            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
262            assertEquals(2, coll1.callbacks.get("bound").intValue());
263            assertEquals(1, coll1.callbacks.get("unbinding").intValue());
264            assertEquals(2, coll1.callbacks.get("modified").intValue());
265            assertEquals(service1, coll1.service().get());
266
267            // Remove first
268            reg1.unregister();
269            assertEquals(1, coll1.callbacks.get("firstBound").intValue());
270            assertEquals(2, coll1.callbacks.get("bound").intValue());
271            assertEquals(2, coll1.callbacks.get("unbinding").intValue());
272            assertEquals(1, coll1.callbacks.get("lastUnbinding").intValue());
273            assertEquals(2, coll1.callbacks.get("modified").intValue());
274            assertFalse(coll1.service().isPresent());
275        }

278    /**
279     * Attempt to test concurrency.
280     *
281     * @throws TimeoutException the timeout exception
282     * @throws InterruptedException the interrupted exception
283     */
284    @Test
285    public void testConcurrent() throws TimeoutException, InterruptedException {
286        Object starter = new Object();
287        Waiter waiter = new Waiter();
288        final int N = 100;
289        CountDownLatch started = new CountDownLatch(N);
290        int[] services = new int[N];
291        // Threads prepared
292        try (TestCollector<SampleService1> coll1
293            = new TestCollector<SampleService1>(context,
294                SampleService1.class)) {
295            coll1.open();
296            assertTrue(coll1.isEmpty());
297            // Prepare threads
298            for (int i = 0; i < N; i++) {
299                final int count = i;
300                new Thread(() -> {
301                    started.countDown();
302                    try {
303                        started.await();
304                    } catch (InterruptedException e1) {
305                        waiter.fail();
306                    }
307                    try {
308                        Thread.sleep(100 + (count / 4) * 8);
309                    } catch (InterruptedException e) {
310                        waiter.fail(e);
311                    }
312                    runTestInThread(waiter, coll1, count, services);
313                }).start();
314            }
315            started.await();
316            synchronized (starter) {
317                starter.notifyAll();
318            }
319            waiter.await(10000, N);
320            assertEquals(N, coll1.callbacks.get("bound").intValue());
321            assertEquals(N, coll1.callbacks.get("unbinding").intValue());
322            assertFalse(coll1.service().isPresent());
323        }
324    }
325
326    private void runTestInThread(Waiter waiter,
327            TestCollector<SampleService1> coll1, int count, int[] services) {
328        // Add service
329        Hashtable<String, Object> props = new Hashtable<>();
330        props.put(Constants.SERVICE_RANKING, 1);
331        SampleService1 service1 = new SampleService1();
332        final ServiceRegistration<SampleService1> reg1
333            = context.registerService(
334                SampleService1.class, service1, props);
335        services[count] = coll1.serviceReferences().size();
336        try {
337            Thread.sleep(count % 10);
338        } catch (InterruptedException e) {
339            waiter.fail(e);
340        }
341
342        // Remove service
343        reg1.unregister();
344        waiter.resume();
345    }
