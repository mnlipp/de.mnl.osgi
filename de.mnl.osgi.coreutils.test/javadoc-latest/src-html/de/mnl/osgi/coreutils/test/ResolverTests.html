054    /**
055     * Basic test.
056     *
057     * @throws InterruptedException the interrupted exception
058     */
059    @SuppressWarnings("resource")
060    @Test
061    public void testBasic() throws InterruptedException {
062        // Now run
063        Map<String, Object> results = new HashMap<>();
064        ServiceResolver resolverBck;
065        try (ServiceResolver resolver = new ServiceResolver(context)) {
066            resolverBck = resolver;
067            resolver.setOnResolved(() -> results.put("resolved", true));
068            resolver.setOnDissolving(() -> results.put("dissolving", true));
069            resolver.setOnRebound(name -> results.put("rebound", true));
070            // Add dependencies
071            resolver.addDependency(SampleService1.class);
072            resolver.addDependency(SampleService2.class);
073            resolver.open();
074            assertTrue(resolver.isOpen());
075            // Change services
076            Hashtable<String, Object> props = new Hashtable<>();
077            props.put(Constants.SERVICE_RANKING, 1);
078            SampleService1 service1 = new SampleService1();
079
080            // Register first, no effect
081            final ServiceRegistration<SampleService1> reg1 = context
082                .registerService(SampleService1.class, service1, props);
083            assertFalse(resolver.isResolved());
084            assertTrue(results.isEmpty());
085
086            // Register second, resolved
087            props.put(Constants.SERVICE_RANKING, 1);
088            SampleService2 service2 = new SampleService2();
089            final ServiceRegistration<SampleService2> reg2 = context
090                .registerService(SampleService2.class, service2, props);
091            assertTrue(resolver.isResolved());
092            assertEquals(Boolean.TRUE, results.get("resolved"));
093            assertEquals(1, results.size());
094            results.clear();
095            resolver.get(SampleService1.class).equals(service1);
096            resolver.get(SampleService2.class).equals(service2);
097
098            // Invoke service
099            assertEquals(42,
100                resolver.with(SampleService1.class, s -> s.add(20, 22)).get()
101                    .intValue());
102
103            // Unregister one
104            reg1.unregister();
105            assertFalse(resolver.isResolved());
106            assertEquals(Boolean.TRUE, results.get("dissolving"));
107            assertEquals(1, results.size());
108            results.clear();
109
110            // Register again
111            final ServiceRegistration<SampleService1> reg1a = context
112                .registerService(SampleService1.class, service1, props);
113            assertTrue(resolver.isResolved());
114            assertEquals(Boolean.TRUE, results.get("resolved"));
115            assertEquals(1, results.size());
116            results.clear();
117
118            // Replace SampleService2 with SampleService2a
119            SampleService2 service2a = new SampleService2();
120            props.put(Constants.SERVICE_RANKING, 2);
121            final ServiceRegistration<SampleService2> reg2a = context
122                .registerService(SampleService2.class, service2a, props);
123            assertTrue(resolver.isResolved());
124            assertEquals(Boolean.TRUE, results.get("rebound"));
125            assertEquals(1, results.size());
126            results.clear();
127            resolver.get(SampleService1.class).equals(service1);
128            resolver.get(SampleService2.class).equals(service2a);
129
130            // Now unregister all
131            reg1a.unregister();
132            assertFalse(resolver.isResolved());
133            assertEquals(Boolean.TRUE, results.get("dissolving"));
134            assertEquals(1, results.size());
135            results.clear();
136            reg2.unregister();
137            assertFalse(resolver.isResolved());
138            assertEquals(0, results.size());
139            reg2a.unregister();
140            assertFalse(resolver.isResolved());
141            assertEquals(0, results.size());
142        }
143        assertTrue(!resolverBck.isOpen());
144    }

146    /**
147     * A service in use cannot be unregistered
148     *
149     * @throws InterruptedException the interrupted exception
150     */
151    @Test
152    public void testDelayedRemoval() throws InterruptedException {
153        try (ServiceResolver resolver = new ServiceResolver(context)) {
154            // Add dependencies
155            resolver.addDependency(SampleService1.class);
156            resolver.open();
157            assertTrue(resolver.isOpen());
158            // Change services
159            Hashtable<String, Object> props = new Hashtable<>();
160            props.put(Constants.SERVICE_RANKING, 1);
161            SampleService1 service1 = new SampleService1();
162
163            // Register service
164            final ServiceRegistration<SampleService1> reg1 = context
165                .registerService(SampleService1.class, service1, props);
166            assertTrue(resolver.isResolved());
167
168            CountDownLatch toBeStarted = new CountDownLatch(1);
169            Instant[] calcTimes = new Instant[2];
170            new Thread() {
171                public void run() {
172                    resolver.with(SampleService1.class, (s -> {
173                        toBeStarted.countDown();
174                        try {
175                            Thread.sleep(250);
176                        } catch (InterruptedException e) {
177                            // Empty
178                        }
179                        calcTimes[0] = Instant.now();
180                        int result = s.add(20, 22);
181                        calcTimes[1] = Instant.now();
182                        return result;
183                    }));
184                }
185            }.start();
186            // Wait for execution start
187            toBeStarted.await();
188            Instant unregStart = Instant.now();
189            reg1.unregister();
190            Instant unregEnd = Instant.now();
191
192            // Calculation must start after unreg start (else it's not a test)
193            assertTrue(calcTimes[0].isAfter(unregStart));
194            // Removal must finish after calc end (else locking doesn't work)
195            assertTrue(unregEnd.isAfter(calcTimes[1]));
196            // Make sure
197            assertTrue(
198                unregEnd.toEpochMilli() - unregStart.toEpochMilli() > 100);
199        }
200    }
