017package de.mnl.osgi.coreutils.test;
018
019import de.mnl.osgi.coreutils.ServiceResolver;
020import java.util.ArrayList;
021import java.util.HashMap;
022import java.util.Hashtable;
023import java.util.List;
024import java.util.Map;
025import org.junit.jupiter.api.AfterEach;
026import static org.junit.jupiter.api.Assertions.assertEquals;
027import static org.junit.jupiter.api.Assertions.assertFalse;
028import static org.junit.jupiter.api.Assertions.assertTrue;
029import org.junit.jupiter.api.Test;
030import org.junit.jupiter.api.extension.ExtendWith;
031import org.mockito.junit.jupiter.MockitoExtension;
032import org.osgi.framework.BundleContext;
033import org.osgi.framework.Constants;
034import org.osgi.framework.FrameworkUtil;
035import org.osgi.framework.ServiceRegistration;
036
037@ExtendWith(MockitoExtension.class)
038public class ResolverTests {
039
040    private final BundleContext context = FrameworkUtil
041        .getBundle(ResolverTests.class).getBundleContext();
042
043    private List<ServiceRegistration<?>> cleanup = new ArrayList<>();
044
045    @AfterEach
046    public void tearDown() {
047        for (ServiceRegistration<?> reg : cleanup) {
048            reg.unregister();
049        }
050    }
051
052    /**
053     * Basic test.
054     *
055     * @throws InterruptedException the interrupted exception
056     */
057    @SuppressWarnings("resource")
058    @Test
059    public void testBasic() throws InterruptedException {
060        // Now run
061        Map<String, Object> results = new HashMap<>();
062        ServiceResolver resolverBck;
063        try (ServiceResolver resolver = new ServiceResolver(context)) {
064            resolverBck = resolver;
065            resolver.setOnResolved(() -> results.put("resolved", true));
066            resolver.setOnDissolving(() -> results.put("dissolving", true));
067            resolver.setOnRebound(name -> results.put("rebound", true));
068            // Add dependencies
069            resolver.addDependency(SampleService1.class);
070            resolver.addDependency(SampleService2.class);
071            resolver.open();
072            assertTrue(resolver.isOpen());
073            // Change services
074            Hashtable<String, Object> props = new Hashtable<>();
075            props.put(Constants.SERVICE_RANKING, 1);
076            SampleService1 service1 = new SampleService1();
077
078            // Register first, no effect
079            final ServiceRegistration<SampleService1> reg1 = context
080                .registerService(SampleService1.class, service1, props);
081            assertFalse(resolver.isResolved());
082            assertTrue(results.isEmpty());
083
084            // Register second, resolved
085            props.put(Constants.SERVICE_RANKING, 1);
086            SampleService2 service2 = new SampleService2();
087            final ServiceRegistration<SampleService2> reg2 = context
088                .registerService(SampleService2.class, service2, props);
089            assertTrue(resolver.isResolved());
090            assertEquals(Boolean.TRUE, results.get("resolved"));
091            assertEquals(1, results.size());
092            results.clear();
093            resolver.get(SampleService1.class).equals(service1);
094            resolver.get(SampleService2.class).equals(service2);
095
096            // Unregister one
097            reg1.unregister();
098            assertFalse(resolver.isResolved());
099            assertEquals(Boolean.TRUE, results.get("dissolving"));
100            assertEquals(1, results.size());
101            results.clear();
102
103            // Register again
104            final ServiceRegistration<SampleService1> reg1a = context
105                .registerService(SampleService1.class, service1, props);
106            assertTrue(resolver.isResolved());
107            assertEquals(Boolean.TRUE, results.get("resolved"));
108            assertEquals(1, results.size());
109            results.clear();
110
111            // Replace SampleService2 with SampleService2a
112            SampleService2 service2a = new SampleService2();
113            props.put(Constants.SERVICE_RANKING, 2);
114            final ServiceRegistration<SampleService2> reg2a = context
115                .registerService(SampleService2.class, service2a, props);
116            assertTrue(resolver.isResolved());
117            assertEquals(Boolean.TRUE, results.get("rebound"));
118            assertEquals(1, results.size());
119            results.clear();
120            resolver.get(SampleService1.class).equals(service1);
121            resolver.get(SampleService2.class).equals(service2a);
122
123            // Now unregister all
124            reg1a.unregister();
125            assertFalse(resolver.isResolved());
126            assertEquals(Boolean.TRUE, results.get("dissolving"));
127            assertEquals(1, results.size());
128            results.clear();
129            reg2.unregister();
130            assertFalse(resolver.isResolved());
131            assertEquals(0, results.size());
132            reg2a.unregister();
133            assertFalse(resolver.isResolved());
134            assertEquals(0, results.size());
135        }
136        assertTrue(!resolverBck.isOpen());
137    }
138
139}
