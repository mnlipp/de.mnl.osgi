106    /**
107     * Instantiates a new representation of group data backed
108     * by the specified directory. 
109     *
110     * @param groupId the maven groupId indexed by this repository
111     * @param directory the directory used to persist data
112     * @param requested if it is a requested group id
113     * @param indexedRepository the indexed maven repository
114     * @param client the client used for remote access
115     * @param reporter the reporter
116     * @throws IOException Signals that an I/O exception has occurred.
117     */
118    @SuppressWarnings("PMD.ConfusingTernary")
119    public MavenGroupRepository(String groupId, Path directory,
120            boolean requested, IndexedMavenRepository indexedRepository,
121            HttpClient client, Reporter reporter) throws IOException {
122        this.groupId = groupId;
123        this.groupDir = directory;
124        this.requested = requested;
125        this.indexedRepository = indexedRepository;
126        this.client = client;
127        this.reporter = reporter;
128
129        // Prepare directory and files
130        groupPropsPath = directory.resolve("group.properties");
131        groupProps = new Properties();
132        // If directory does not exist, it's not from a request.
133        if (!directory.toFile().exists()) {
134            directory.toFile().mkdir();
135            propsChanged = true;
136        } else {
137            // Directory exists, either as newly created or as "old"
138            if (groupPropsPath.toFile().canRead()) {
139                try (InputStream input = Files.newInputStream(groupPropsPath)) {
140                    groupProps.load(input);
141                }
142            } else {
143                propsChanged = true;
144            }
145        }
146
147        // Prepare OSGi repository
148        groupIndexPath = directory.resolve("index.xml");
149        if (groupIndexPath.toFile().canRead()) {
150            try (XMLResourceParser parser
151                = new XMLResourceParser(groupIndexPath.toFile())) {
152                // addAll(parser.parse());
153            } catch (Exception e) { // NOPMD
154                reporter.warning("Cannot parse %s, ignored: %s", groupIndexPath,
155                    e.getMessage());
156            }
157        } else {
158            indexChanged = true;
159        }
160        // Cache revisions for faster checks.
161        for (Capability cap : findProvider(
162            newRequirementBuilder("bnd.info").build())) {
163            indexingState.put(
164                Revision.valueOf((String) cap.getAttributes().get("from")),
165                IndexingState.INDEXED);
166        }
167        LOG.debug("Created group repository for {}.", groupId);
168    }

179    /**
180     * Writes all changes to persistent storage and removes
181     * any backup information prepared by {@link #reset(boolean)}.
182     *
183     * @throws IOException Signals that an I/O exception has occurred.
184     */
185    public void flush() throws IOException {
186        if (propsChanged) {
187            try (OutputStream out = Files.newOutputStream(groupPropsPath)) {
188                groupProps.store(out, "Group properties");
189            }
190            propsChanged = false;
191        }
192        if (indexChanged) {
193            XMLResourceGenerator generator = new XMLResourceGenerator();
194            generator.resources(getResources());
195            generator.name(indexedRepository.mavenRepository().getName());
196            try {
197                generator.save(groupIndexPath.toFile());
198            } catch (IOException e) {
199                reporter.exception(e, "Cannot save %s.", groupIndexPath);
200            }
201            indexChanged = false;
202        }
203        backupRepo = null;
204        indexingState.clear();
205    }

207    /**
208     * Removes the group directory if empty.
209     *
210     * @return true, if removed
211     * @throws IOException Signals that an I/O exception has occurred.
212     */
213    public boolean removeIfRedundant() throws IOException {
214        if (isRequested() || !groupProps.isEmpty()
215            || !getResources().isEmpty()) {
216            return false;
217        }
218        // Nothing in this group
219        Files.walk(groupDir)
220            .sorted(Comparator.reverseOrder())
221            .map(Path::toFile)
222            .forEach(File::delete);
223        return true;
224    }

236    /**
237     * Clears this repository and updates the requested flag. 
238     * Keeps the current content as backup for reuse in a 
239     * subsequent call to {@link #reload()}.
240     *
241     * @param requested whether this is a requested group
242     */
243    public void reset(boolean requested) {
244        synchronized (this) {
245            // Update "type"
246            this.requested = requested;
247            // Save current content and clear.
248            backupRepo = new ResourcesRepository(getResources());
249            set(Collections.emptyList());
250            // Clear and reload properties
251            groupProps.clear();
252            if (groupPropsPath.toFile().canRead()) {
253                try (InputStream input = Files.newInputStream(groupPropsPath)) {
254                    propQueryCache.clear();
255                    groupProps.load(input);
256                } catch (IOException e) {
257                    reporter.warning("Problem reading %s (ignored): %s",
258                        groupPropsPath, e.getMessage());
259                }
260            }
261            // Clear remaining caches.
262            indexingState.clear();
263            propQueryCache.clear();
264        }
265    }

267    /**
268     * Reload the repository. Requested repositories retrieve
269     * the list of known artifactIds from the remote repository 
270     * and add the versions. For versions already in the repository, 
271     * the backup information is re-used.
272     *
273     * @throws IOException Signals that an I/O exception has occurred.
274     */
275    @SuppressWarnings({ "PMD.AvoidReassigningLoopVariables",
276        "PMD.AvoidCatchingGenericException", "PMD.AvoidDuplicateLiterals",
277        "PMD.AvoidInstantiatingObjectsInLoops",
278        "PMD.AvoidThrowingRawExceptionTypes", "PMD.PreserveStackTrace" })
279    public void reload() throws IOException {
280        if (!isRequested()) {
281            // Will be filled with dependencies only
282            return;
283        }
284        // Actively filled.
285        synchronized (this) {
286            if (backupRepo == null) {
287                backupRepo = new ResourcesRepository(getResources());
288            }
289            propQueryCache.clear();
290        }
291        try {
292            CompletableFuture<Void> actions
293                = CompletableFuture.allOf(findArtifactIds().stream().map(
294                    artifactId -> loadProgram(Program.valueOf(groupId,
295                        artifactId)))
296                    .toArray(CompletableFuture[]::new));
297            actions.get();
298        } catch (ExecutionException e) {
299            if (e.getCause() instanceof IOException) {
300                throw (IOException) e.getCause();
301            }
302            throw new RuntimeException(e.getCause());
303        } catch (InterruptedException e) {
304            throw new RuntimeException(e);
305        }
306        indexChanged = true;
307    }

309    @SuppressWarnings("PMD.ForLoopCanBeForeach")
310    private CompletableFuture<Void> loadProgram(Program program) {
311        // Get revisions of program and process.
312        CompletableFuture<Void> result = new CompletableFuture<>();
313        IndexedMavenRepository.programLoaders.submit(() -> {
314            try {
315                Stream<CompletableFuture<Void>> revLoaderStream
316                    = indexedRepository.mavenRepository()
317                        .boundRevisions(program).parallel()
318                        .map(revision -> loadRevision(revision));
319                try {
320                    for (CompletableFuture<Void> revLoad : (Iterable<
321                            CompletableFuture<
322                                    Void>>) revLoaderStream::iterator) {
323                        revLoad.get();
324                    }
325                } catch (InterruptedException e) {
326                    result.completeExceptionally(e);
327                    return;
328                } catch (ExecutionException e) {
329                    result.completeExceptionally(e.getCause());
330                    return;
331                }
332                result.complete(null);
333            } catch (IOException e) {
334                result.completeExceptionally(e);
335            }
336        });
337        return result;
338    }

340    @SuppressWarnings("PMD.AvoidCatchingGenericException")
341    private CompletableFuture<Void> loadRevision(BoundRevision revision) {
342        return CompletableFuture.runAsync(() -> {
343            if (indexingState.getOrDefault(revision.unbound(),
344                IndexingState.NONE) != IndexingState.NONE) {
345                // Already loading or handled (as dependency)
346                return;
347            }
348            MavenVersionRange acceptedRange = Optional.ofNullable(
349                searchProperty(revision.artifactId(), "versions"))
350                .map(MavenVersionRange::parseRange).orElse(null);
351            MavenVersionRange forcedRange = Optional.ofNullable(
352                searchProperty(revision.artifactId(), "forcedVersions"))
353                .map(MavenVersionRange::parseRange).orElse(null);
354            boolean inForced = forcedRange != null
355                && forcedRange.includes(revision.version());
356            if (acceptedRange != null || forcedRange != null) {
357                boolean inVersions = acceptedRange != null
358                    && acceptedRange.includes(revision.version());
359                if (!inVersions && !inForced) {
360                    return;
361                }
362            }
363            try {
364                Set<BoundRevision> deps = new HashSet<>();
365                if (isIndexable(revision, deps, inForced)) {
366                    addRevision(revision);
367                    for (BoundRevision rev : deps) {
368                        indexedRepository.getOrCreateGroupRepository(
369                            rev.groupId()).addRevision(rev);
370                    }
371                }
372            } catch (Exception e) {
373                // Load "in isolation", don't propagate individual failure.
374                reporter.exception(e, "Failed to load %s: %s", revision,
375                    e.getMessage());
376            }
377        }, IndexedMavenRepository.revisionLoaders);
378
379    }

381    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
382        "PMD.AvoidInstantiatingObjectsInLoops" })
383    private Collection<String> findArtifactIds() {
384        Set<String> result = new HashSet<>();
385        for (MavenBackingRepository repo : indexedRepository.mavenRepository()
386            .allRepositories()) {
387            URI groupUri = null;
388            try {
389                groupUri
390                    = repo.toURI("").resolve(groupId.replace('.', '/') + "/");
391                String page = client.build().headers("User-Agent", "Bnd")
392                    .get(String.class)
393                    .go(groupUri);
394                if (page == null) {
395                    continue;
396                }
397                Matcher matcher = hrefPattern.matcher(page);
398                while (matcher.find()) {
399                    URI programUri = groupUri.resolve(matcher.group("href"));
400                    String artifactId = programUri.getPath()
401                        .substring(groupUri.getPath().length());
402                    if (artifactId.endsWith("/")) {
403                        artifactId
404                            = artifactId.substring(0, artifactId.length() - 1);
405                    }
406                    result.add(artifactId);
407                }
408            } catch (Exception e) {
409                reporter.warning("Problem retrieving %s, skipped: %s", groupUri,
410                    e.getMessage());
411            }
412        }
413        return result;
414    }

416    /**
417     * Checks if the given revision is indexable. A revision is
418     * indexable if it is not excluded and its dependencies aren't
419     * excluded either.
420     *
421     * @param revision the revision to check
422     * @param dependencies revisions that must additionally be included
423     *    when the checked revision is added
424     * @param ignoreExcludedDependencies evaluate as indexable even
425     *    if some dependencies aren't indexable
426     * @return true, if the revision can be added to the index
427     * @throws IOException Signals that an I/O exception has occurred.
428     */
429    @SuppressWarnings({ "PMD.LongVariable", "PMD.NPathComplexity" })
430    public boolean isIndexable(BoundRevision revision,
431            Set<BoundRevision> dependencies,
432            boolean ignoreExcludedDependencies) throws IOException {
433        if (!revision.groupId().equals(groupId)) {
434            throw new IllegalArgumentException("Wrong groupId "
435                + revision.groupId() + " (must be " + groupId + ").");
436        }
437        // Check if already there (or being added)
438        if (INDEXING_OR_INDEXED.contains(indexingState
439            .getOrDefault(revision.unbound(), IndexingState.NONE))) {
440            // Obviously indexable and no additional resources needed.
441            return true;
442        }
443        // Check if excluded by rule.
444        if (Optional.ofNullable(
445            searchProperty(revision.artifactId(), "exclude"))
446            .map(MavenVersionRange::parseRange)
447            .map(range -> range.includes(revision.version()))
448            .orElse(false)) {
449            return false;
450        }
451        if (!ignoreExcludedDependencies
452            && indexingState.getOrDefault(revision.unbound(),
453                IndexingState.NONE) == IndexingState.EXCL_BY_DEP) {
454            // Has been found un-indexable before
455            return false;
456        }
457        // Check whether excluded because dependency is excluded
458        Set<Dependency> childDeps = new HashSet<>();
459        collectDependencies(revision, childDeps);
460        for (IPom.Dependency dep : childDeps) {
461            Optional<BoundRevision> boundDep;
462            try {
463                boundDep = indexedRepository.mavenRepository().toBoundRevision(
464                    dep.program, dep.version);
465            } catch (IOException e) {
466                reporter.exception(e, "Cannot get revision for %s:%s.",
467                    dep.program, dep.version);
468                continue;
469            }
470            if (!boundDep.isPresent()) {
471                continue;
472            }
473            MavenGroupRepository depRepo = indexedRepository
474                .getOrCreateGroupRepository(dep.program.group);
475            if (!depRepo.isIndexable(boundDep.get(), dependencies,
476                ignoreExcludedDependencies)) {
477                // Not that the revision to check is not indexable
478                // due to a dependency that is not indexable
479                indexingState.put(revision.unbound(),
480                    IndexingState.EXCL_BY_DEP);
481                if (!ignoreExcludedDependencies) {
482                    return false;
483                }
484            }
485            dependencies.add(boundDep.get());
486        }
487        return true;
488    }

490    /**
491     * Adds the specified revision unless it matches an exclude.
492     *
493     * @param revision the revision to add
494     */
495    private void addRevision(BoundRevision revision) {
496        synchronized (indexingState) {
497            // Check, but don't keep the lock
498            if (INDEXING_OR_INDEXED.contains(indexingState
499                .getOrDefault(revision.unbound(), IndexingState.NONE))) {
500                return;
501            }
502            indexingState.put(revision.unbound(), IndexingState.INDEXING);
503        }
504        Resource resource = null;
505        if (backupRepo != null) {
506            List<Capability> cap = backupRepo.findProvider(
507                backupRepo.newRequirementBuilder("bnd.info")
508                    .addDirective("filter", String.format("(from=%s)",
509                        revision.unbound().toString()))
510                    .build());
511            if (!cap.isEmpty()) {
512                // Reuse existing
513                resource = cap.get(0).getResource();
514            }
515        }
516        if (resource == null) {
517            // Extract information from artifact.
518            resource = indexedRepository.mavenRepository()
519                .toResource(revision, deps -> {
520                }, BinaryLocation.REMOTE).orElse(null);
521        }
522        if (resource == null) {
523            // No such resource
524            synchronized (indexingState) {
525                indexingState.remove(revision.unbound());
526            }
527            return;
528        }
529        // Check again, now with lock, and add
530        synchronized (indexingState) {
531            if (indexingState.getOrDefault(revision.unbound(),
532                IndexingState.NONE) == IndexingState.INDEXED) {
533                return;
534            }
535            add(resource);
536        }
537        indexChanged = true;
538    }

540    /**
541     * Collects the revisions that are immediate dependencies 
542     * of the revision passed as argument. This method uses
543     * any cached information first before resorting to
544     * {@link CompositeMavenRepository#toResource(BoundRevision, 
545     * CompositeMavenRepository.DependencyHandler, BinaryLocation)}.
546     *
547     * @param revision the revision to examine
548     * @param dependencies the dependencies
549     */
550    @SuppressWarnings({ "PMD.AssignmentInOperand", "PMD.ConfusingTernary" })
551    public void collectDependencies(BoundRevision revision,
552            Collection<Dependency> dependencies) {
553        List<Capability> cap = null;
554        if (backupRepo != null && !(cap = backupRepo.findProvider(
555            backupRepo.newRequirementBuilder("bnd.info")
556                .addDirective("filter", String.format("(from=%s)",
557                    revision.unbound().toString()))
558                .build())).isEmpty()) {
559            restoreDependencies(cap.get(0).getResource(), dependencies);
560        } else {
561            // Extract information from artifact.
562            indexedRepository.mavenRepository().toResource(revision,
563                deps -> dependencies.addAll(deps), BinaryLocation.REMOTE);
564        }
565    }

567    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
568    private void restoreDependencies(Resource resource,
569            Collection<Dependency> dependencies) {
570        // Actually, there should be only one such capability per resource.
571        for (Capability capability : resource
572            .getCapabilities(CompositeMavenRepository.MAVEN_DEPENDENCIES_NS)) {
573            Map<String, Object> depAttrs = capability.getAttributes();
574            depAttrs.values().stream()
575                .flatMap(val -> RepositoryUtils.itemizeList((String) val))
576                .map(rev -> {
577                    String[] parts = rev.split(":");
578                    IPom.Dependency dep = new IPom.Dependency();
579                    dep.program = Program.valueOf(parts[0], parts[1]);
580                    dep.version = parts[2];
581                    return dep;
582                }).forEach(dependencies::add);
583        }
584    }

586    @SuppressWarnings("PMD.CompareObjectsWithEquals")
587    private String searchProperty(String artifactId, String qualifier) {
588        String queryKey = artifactId + ";" + qualifier;
589        // Attempt to get from cache.
590        String prop = propQueryCache.get(queryKey);
591        // Special value, because ConcurrentHashMap cannot have null values.
592        if (prop == NOT_AVAILABLE) {
593            return null;
594        }
595        // Found in cache return.
596        if (prop != null) {
597            return prop;
598        }
599        // Try query key unmodified
600        prop = groupProps.getProperty(queryKey);
601        if (prop == null) {
602            // Try <id>.*, successively removing trailing parts.
603            String rest = artifactId;
604            while (true) {
605                prop = groupProps.getProperty(rest + ".*;" + qualifier);
606                if (prop != null) {
607                    break;
608                }
609                int lastDot = rest.lastIndexOf('.');
610                if (lastDot < 0) {
611                    break;
612                }
613                rest = rest.substring(0, lastDot);
614            }
615        }
616        if (prop == null) {
617            prop = groupProps.getProperty(qualifier);
618        }
619        propQueryCache.put(queryKey, prop == null ? NOT_AVAILABLE : prop);
620        return prop;
621    }
