064/**
065 * A repository with artifacts from a single group.
066 */
067@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
068public class MavenGroupRepository extends ResourcesRepository {
069
070    private static final Logger LOG = LoggerFactory.getLogger(
071        MavenGroupRepository.class);
072    private static final String NOT_AVAILABLE = "_NOT_AVAILABLE_";
073
074    private final String groupId;
075    private boolean requested;
076    private final CompositeMavenRepository mavenRepository;
077    private final HttpClient client;
078    private final Reporter reporter;
079    private final Path groupDir;
080    private final Path groupPropsPath;
081    private final Path groupIndexPath;
082    private final Set<Revision> mavenRevisions = new HashSet<>();
083    private final Properties groupProps;
084    private final Map<String, String> propQueryCache
085        = new ConcurrentHashMap<>();
086    private boolean propsChanged;
087    private boolean indexChanged;
088    private ResourcesRepository backupRepo;
089    private final Pattern hrefPattern = Pattern.compile(
090        "<[aA]\\s+(?:[^>]*?\\s+)?href=(?<quote>[\"'])"
091            + "(?<href>[a-zA-Z].*?)\\k<quote>");

093    /**
094     * Instantiates a new representation of group data backed
095     * by the specified directory. 
096     *
097     * @param groupId the maven groupId indexed by this repository
098     * @param directory the directory used to persist data
099     * @param mavenRepository the maven repository
100     * @param client the client used for remote access
101     * @param reporter the reporter
102     * @throws IOException Signals that an I/O exception has occurred.
103     */
104    @SuppressWarnings("PMD.ConfusingTernary")
105    public MavenGroupRepository(String groupId, Path directory,
106            boolean requested, CompositeMavenRepository mavenRepository,
107            HttpClient client, Reporter reporter) throws IOException {
108        this.groupId = groupId;
109        this.groupDir = directory;
110        this.requested = requested;
111        this.mavenRepository = mavenRepository;
112        this.client = client;
113        this.reporter = reporter;
114
115        // Prepare directory and files
116        groupPropsPath = directory.resolve("group.properties");
117        groupProps = new Properties();
118        // If directory does not exist, it's not from a request.
119        if (!directory.toFile().exists()) {
120            directory.toFile().mkdir();
121            propsChanged = true;
122        } else {
123            // Directory exists, either as newly created or as "old"
124            if (groupPropsPath.toFile().canRead()) {
125                try (InputStream input = Files.newInputStream(groupPropsPath)) {
126                    groupProps.load(input);
127                }
128            } else {
129                propsChanged = true;
130            }
131        }
132
133        // Prepare OSGi repository
134        groupIndexPath = directory.resolve("index.xml");
135        if (groupIndexPath.toFile().canRead()) {
136            try (XMLResourceParser parser
137                = new XMLResourceParser(groupIndexPath.toFile())) {
138                addAll(parser.parse());
139            } catch (Exception e) { // NOPMD
140                reporter.warning("Cannot parse %s, ignored: %s", groupIndexPath,
141                    e.getMessage());
142            }
143        } else {
144            indexChanged = true;
145        }
146        // Cache revisions for faster checks.
147        for (Capability cap : findProvider(
148            newRequirementBuilder("bnd.info").build())) {
149            mavenRevisions.add(
150                Revision.valueOf((String) cap.getAttributes().get("from")));
151        }
152        LOG.debug("Created group repository for {}.", groupId);
153    }
154
155    /**
156     * Checks if is requested.
157     *
158     * @return the requested
159     */
160    public final boolean isRequested() {
161        return requested;
162    }
163
164    /**
165     * Writes all changes to persistent storage.
166     *
167     * @throws IOException Signals that an I/O exception has occurred.
168     */
169    public void flush() throws IOException {
170        if (propsChanged) {
171            try (OutputStream out = Files.newOutputStream(groupPropsPath)) {
172                groupProps.store(out, "Group properties");
173            }
174            propsChanged = false;
175        }
176        if (indexChanged) {
177            XMLResourceGenerator generator = new XMLResourceGenerator();
178            generator.resources(getResources());
179            generator.name(mavenRepository.getName());
180            try {
181                generator.save(groupIndexPath.toFile());
182            } catch (IOException e) {
183                reporter.exception(e, "Cannot save %s.", groupIndexPath);
184            }
185            indexChanged = false;
186        }
187    }
188
189    /**
190     * Removes the group directory if empty.
191     *
192     * @return true, if removed
193     * @throws IOException Signals that an I/O exception has occurred.
194     */
195    public boolean removeIfRedundant() throws IOException {
196        if (mavenRevisions.isEmpty()) {
197            // Nothing in this group
198            Files.walk(groupDir)
199                .sorted(Comparator.reverseOrder())
200                .map(Path::toFile)
201                .forEach(File::delete);
202            return true;
203        }
204        return false;
205    }

217    /**
218     * Clears this repository and updates the requested flag. 
219     * Keeps the current content as backup for reuse in a 
220     * subsequent call to {@link #reload(Consumer)}.
221     */
222    public void reset(boolean requested) {
223        synchronized (this) {
224            this.requested = requested;
225            backupRepo = new ResourcesRepository(getResources());
226            set(Collections.emptyList());
227            mavenRevisions.clear();
228            propQueryCache.clear();
229        }
230    }
231
232    /**
233     * Reload the repository. Requested repositories retrieve
234     * the list of known artifactIds from the remote repository 
235     * and add the versions. For versions already in the repository, 
236     * the backup information is re-used.
237     *
238     * @param dependencyHandler the dependency handler
239     * @throws IOException Signals that an I/O exception has occurred.
240     */
241    @SuppressWarnings({ "PMD.AvoidReassigningLoopVariables",
242        "PMD.AvoidCatchingGenericException" })
243    public void reload(Consumer<Collection<IPom.Dependency>> dependencyHandler)
244            throws IOException {
245        if (groupPropsPath.toFile().canRead()) {
246            try (InputStream input = Files.newInputStream(groupPropsPath)) {
247                groupProps.clear();
248                groupProps.load(input);
249            }
250        }
251        if (!isRequested()) {
252            return;
253        }
254        synchronized (this) {
255            if (backupRepo == null) {
256                backupRepo = new ResourcesRepository(getResources());
257            }
258            propQueryCache.clear();
259        }
260        Collection<String> artifactIds = findArtifactIds(groupId);
261        for (String artifactId : artifactIds) {
262            Program program = Program.valueOf(groupId, artifactId);
263
264            // Get revisions of program.
265            @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
266            IOException[] exc = new IOException[1];
267            MavenVersionRange range
268                = Optional.ofNullable(searchProperty(artifactId, "versions"))
269                    .map(MavenVersionRange::parseRange).orElse(null);
270            mavenRepository.boundRevisions(program).forEach(
271                revision -> {
272                    if (range != null && !range.includes(revision.version())) {
273                        return;
274                    }
275                    try {
276                        addRevision(revision, dependencyHandler);
277                    } catch (IOException e) {
278                        exc[0] = e;
279                    }
280                });
281            if (exc[0] != null) {
282                throw exc[0];
283            }
284        }
285        indexChanged = true;
286    }
287
288    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
289        "PMD.AvoidInstantiatingObjectsInLoops" })
290    private Collection<String> findArtifactIds(String dir) {
291        Set<String> result = new HashSet<>();
292        for (MavenBackingRepository repo : mavenRepository.allRepositories()) {
293            URI groupUri = null;
294            try {
295                groupUri = repo.toURI("").resolve(dir.replace('.', '/') + "/");
296                String page = client.build().headers("User-Agent", "Bnd")
297                    .get(String.class)
298                    .go(groupUri);
299                if (page == null) {
300                    continue;
301                }
302                Matcher matcher = hrefPattern.matcher(page);
303                while (matcher.find()) {
304                    URI programUri = groupUri.resolve(matcher.group("href"));
305                    String artifactId = programUri.getPath()
306                        .substring(groupUri.getPath().length());
307                    if (artifactId.endsWith("/")) {
308                        artifactId
309                            = artifactId.substring(0, artifactId.length() - 1);
310                    }
311                    result.add(artifactId);
312                }
313            } catch (Exception e) {
314                reporter.warning("Problem retrieving %s, skipped: %s", groupUri,
315                    e.getMessage());
316            }
317        }
318        return result;
319    }

321    /**
322     * Adds the specified revision.
323     *
324     * @param revision the revision
325     * @param dependencyHandler the dependency handler
326     * @throws IOException Signals that an I/O exception has occurred.
327     */
328    public void addRevision(BoundRevision revision,
329            Consumer<Collection<IPom.Dependency>> dependencyHandler)
330            throws IOException {
331        if (!revision.groupId().equals(groupId)) {
332            throw new IllegalArgumentException("Wrong groupId "
333                + revision.groupId() + " (must be " + groupId + ").");
334        }
335        synchronized (this) {
336            if (mavenRevisions.contains(revision.unbound())) {
337                return;
338            }
339        }
340        MavenVersionRange excludeRange = Optional.ofNullable(
341            searchProperty(revision.artifactId(), "exclude"))
342            .map(MavenVersionRange::parseRange).orElse(null);
343        if (excludeRange != null && excludeRange.includes(revision.version())) {
344            return;
345        }
346        Resource resource = null;
347        if (backupRepo != null) {
348            List<Capability> cap = backupRepo.findProvider(
349                backupRepo.newRequirementBuilder("bnd.info")
350                    .addDirective("filter", String.format("(from=%s)",
351                        revision.unbound().toString()))
352                    .build());
353            if (!cap.isEmpty()) {
354                // Reuse existing
355                resource = cap.get(0).getResource();
356                replayDependencies(resource, dependencyHandler);
357            }
358        }
359        if (resource == null) {
360            // Extract information from artifact.
361            resource = mavenRepository.toResource(revision,
362                dependencyHandler, BinaryLocation.REMOTE).orElse(null);
363        }
364        if (resource != null) {
365            synchronized (this) {
366                add(resource);
367                mavenRevisions.add(revision.unbound());
368                indexChanged = true;
369            }
370        }
371    }
372
373    private void replayDependencies(Resource resource,
374            Consumer<Collection<IPom.Dependency>> dependencyHandler) {
375        for (Capability capability : resource
376            .getCapabilities(CompositeMavenRepository.MAVEN_DEPENDENCIES_NS)) {
377            Map<String, Object> depAttrs = capability.getAttributes();
378            @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
379            Collection<IPom.Dependency> dependencies
380                = depAttrs.values().stream()
381                    .flatMap(val -> RepositoryUtils.itemizeList((String) val))
382                    .map(rev -> {
383                        String[] parts = rev.split(":");
384                        IPom.Dependency dep = new IPom.Dependency();
385                        dep.program = Program.valueOf(parts[0], parts[1]);
386                        dep.version = parts[2];
387                        return dep;
388                    }).collect(Collectors.toList());
389            dependencyHandler.accept(dependencies);
390        }
391    }

393    @SuppressWarnings("PMD.CompareObjectsWithEquals")
394    private String searchProperty(String artifactId, String qualifier) {
395        String queryKey = artifactId + ";" + qualifier;
396        // Attempt to get from cache.
397        String prop = propQueryCache.get(queryKey);
398        // Special value, because ConcurrentHashMap cannot have null values.
399        if (prop == NOT_AVAILABLE) {
400            return null;
401        }
402        // Found in cache return.
403        if (prop != null) {
404            return prop;
405        }
406        // Try query key unmodified
407        prop = groupProps.getProperty(queryKey);
408        if (prop == null) {
409            // Try <id>.*, successively removing trailing parts.
410            String rest = artifactId;
411            while (true) {
412                prop = groupProps.getProperty(rest + ".*;" + qualifier);
413                if (prop != null) {
414                    break;
415                }
416                int lastDot = rest.lastIndexOf('.');
417                if (lastDot < 0) {
418                    break;
419                }
420                rest = rest.substring(0, lastDot);
421            }
422        }
423        if (prop == null) {
424            prop = groupProps.getProperty(qualifier);
425        }
426        propQueryCache.put(queryKey, prop == null ? NOT_AVAILABLE : prop);
427        return prop;
428    }
