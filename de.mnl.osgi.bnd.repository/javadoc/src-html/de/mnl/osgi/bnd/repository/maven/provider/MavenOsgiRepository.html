059/**
060 * Provide an OSGi repository (a collection of {@link Resource}s, see 
061 * {@link Repository}), filled with the results from recursively resolving
062 * an initial set of artifacts.
063 */
064public abstract class MavenOsgiRepository extends ResourcesRepository {
065
066        private final static Logger logger = LoggerFactory.getLogger(
067                        MavenOsgiRepository.class);
068        private String name;
069        private File obrIndexFile;
070        private Set<Revision> toBeProcessed = new HashSet<>();
071        private Set<Revision> processing = new HashSet<>();
072        private Set<Revision> processed = new HashSet<>();
073        
074        /**
075         * Create a new instance that uses the provided information/resources to perform
076         * its work.
077         * 
078         * @param name the name
079         * @param obrIndexFile the persistent representation of this repository's content
080         * @throws Exception if a problem occurs
081         */
082        public MavenOsgiRepository (String name, File obrIndexFile) throws Exception {
083                this.name = name;
084                this.obrIndexFile = obrIndexFile;
085                
086                if (!location().exists() || !location().isFile()) {
087                        refresh();
088                } else {
089                        try (XMLResourceParser parser = new XMLResourceParser(location())) {
090                                List<Resource> resources = parser.parse();
091                                addAll(resources);
092                        }
093                }
094        }
095
096        /**
097         * Return the name of this repository.
098         * 
099         * @return the name;
100         */
101        public String name() {
102                return name;
103        }
104        
105        /**
106         * Return the representation of this repository in the local file system.
107         * 
108         * @return the location
109         */
110        public File location() {
111                return obrIndexFile;
112        }
113
114        /**
115         * Refresh this repository's content.
116         * 
117         * @return true if refreshed, false if not refreshed possibly due to error
118         * @throws Exception if a problem occurs
119         */
120        public abstract boolean refresh() throws Exception;
121        
122        /**
123         * Refresh this repository's content.
124         *
125         * @param mavenRepository the maven repository
126         * @param startArtifacts the collection of artifacts to start with
127         * @return true if refreshed, false if not refreshed possibly due to error
128         * @throws Exception if a problem occurs
129         */
130        public boolean refresh(MavenRepository mavenRepository,
131                        Collection<? extends Revision> startArtifacts) throws Exception {
132                set(new HashSet<>()); // Clears this repository
133                toBeProcessed.addAll(startArtifacts);
134                // Repository information is obtained from both querying the repositories
135                // (provides information about existing repositories) and from executing
136                // the query (provides information about actually used repositories).
137                Set<Resource> collectedResources = Collections.newSetFromMap(new ConcurrentHashMap<>());
138                synchronized (this) {
139                        while (true) {
140                                if (toBeProcessed.isEmpty() && processing.isEmpty()) {
141                                        break;
142                                }
143                                if(!toBeProcessed.isEmpty() && processing.size() < 4) {
144                                        Revision rev = toBeProcessed.iterator().next();
145                                        toBeProcessed.remove(rev);
146                                        processing.add(rev);
147                                        Processor.getScheduledExecutor().submit(
148                                                        new RevisionProcessor(mavenRepository, collectedResources, rev));
149                                }
150                                wait();
151                        }
152                }
153                processed.clear();
154                // Set this repository's content to the results...
155                addAll(collectedResources);
156                // ... and persist the content.
157                XMLResourceGenerator generator = new XMLResourceGenerator();
158                generator.resources(getResources());
159                generator.name(name());
160                generator.save(obrIndexFile);
161                return true;
162        }
163
164        /**
165         * A callable (allows it to throw an exception) that processes a single
166         * Revision.
167         */
168        public class RevisionProcessor implements Callable<Void> {
169                private MavenRepository mavenRepository;
170                private Revision revision;
171                private Set<Resource> collectedResources;
172                
173                public RevisionProcessor(MavenRepository mavenRepository, 
174                                Set<Resource> collectedResources, Revision revision) {
175                        this.mavenRepository = mavenRepository;
176                        this.collectedResources = collectedResources;
177                        this.revision = revision;
178                }
179
180                /**
181                 * Get this revision's dependencies from the POM and enqueue them as
182                 * to be processed (unless processed already) and create an entry
183                 * for the resource in this repository.
184                 */
185                @Override
186                public Void call() throws Exception {
187                        try {
188                                // Get and add this revision's OSGi information (refreshes snapshots)
189                                Archive archive = mavenRepository.getResolvedArchive(revision, "jar", "");
190                                if (archive != null) {
191                                        if (archive.isSnapshot()) {
192                                                for (MavenBackingRepository mbr : mavenRepository.getSnapshotRepositories()) {
193                                                        if (mbr.getVersion(archive.getRevision()) != null) {
194                                                                // Found backing repository
195                                                                File metaFile = mavenRepository.toLocalFile(
196                                                                                revision.metadata(mbr.getId()));
197                                                                RevisionMetadata metaData = MetadataParser.parseRevisionMetadata(
198                                                                                metaFile);
199                                                                File archiveFile = mavenRepository.toLocalFile(archive);
200                                                                if (archiveFile.lastModified() < metaData.lastUpdated) {
201                                                                        archiveFile.delete();
202                                                                }
203                                                                File pomFile = mavenRepository.toLocalFile(archive.getPomArchive());
204                                                                if (pomFile.lastModified() < metaData.lastUpdated) {
205                                                                        pomFile.delete();
206                                                                }
207                                                                break;
208                                                        }
209                                                }
210                                        }
211                                        // Get POM for dependencies
212                                        IPom pom = mavenRepository.getPom(archive.getRevision());
213                                        if (pom != null) {
214                                                // Get pom and add all dependencies as to be processed.
215                                                addDependencies(pom);
216                                        }
217                                        Resource resource = parseResource(archive);
218                                        if (resource != null) {
219                                                collectedResources.add(resource);
220                                        }
221                                }
222                                return null;
223                        } finally {
224                                // We're done witht his revision.
225                                synchronized (MavenOsgiRepository.this) {
226                                        processing.remove(revision);
227                                        processed.add(revision);
228                                        MavenOsgiRepository.this.notifyAll();
229                                }
230                        }
231                }
232
233                private void addDependencies(IPom pom) {
234                        try {
235                                Map<Program,Dependency> deps = new LinkedHashMap<>();
236                                deps.putAll(pom.getDependencies(MavenScope.compile, false));
237                                deps.putAll(pom.getDependencies(MavenScope.runtime, false));
238                                synchronized (MavenOsgiRepository.this) {
239                                        for (Map.Entry<Program, Dependency> entry : deps.entrySet()) {
240                                                bindToVersion(entry.getValue());
241                                                try {
242                                                        Revision rev = entry.getValue().getRevision();
243                                                        if (!toBeProcessed.contains(rev) && !processing.contains(rev) 
244                                                                        && !processed.contains(rev)) {
245                                                                toBeProcessed.add(rev);
246                                                                logger.debug("Added as dependency {}", rev);
247                                                        }
248                                                        MavenOsgiRepository.this.notifyAll();
249                                                } catch (Exception e) {
250                                                        logger.warn("Unbindable dependency {}", entry.getValue().toString());
251                                                        continue;
252                                                }
253                                        }
254                                }
255                        } catch (Exception e) {
256                                logger.error("Failed to get POM of " + revision + ".", e);
257                        }
258                }
259                
260                private void bindToVersion(Dependency dependency) throws Exception {
261                        if (MavenVersionRange.isRange(dependency.version)) {
262
263                                MavenVersionRangeFixed range = new MavenVersionRangeFixed(dependency.version);
264                                List<Revision> revisions = mavenRepository.getRevisions(dependency.program);
265
266                                for (Iterator<Revision> it = revisions.iterator(); it.hasNext();) {
267                                        Revision r = it.next();
268                                        if (!range.includes(r.version))
269                                                it.remove();
270                                }
271
272                                if (!revisions.isEmpty()) {
273                                        Collections.sort(revisions, new MavenRevisionComparator());
274                                        Revision highest = revisions.get(revisions.size() - 1);
275                                        dependency.version = highest.version.toString();
276                                }
277                        }
278                }
279
280                private Resource parseResource(Archive archive) throws Exception {
281                        ResourceBuilder rb = new ResourceBuilder();
282                        try {
283                                File binary = mavenRepository.get(archive).getValue();
284                                rb.addFile(binary, binary.toURI());
285                                addInformationCapability(rb, archive.toString(), 
286                                                archive.getRevision().toString(), null);
287                        } catch (Exception e) {
288                                return null;
289                        }
290                        Resource resource = rb.build();
291                        return resource;
292                }
293
294        }
295        
296        public class MavenRevisionComparator implements Comparator<Revision> {
297                @Override
298                public int compare(Revision rev1, Revision rev2) {
299                        int n = rev1.program.compareTo(rev2.program);
300                        if (n != 0) {
301                                return n;
302                        }
303
304                        ComparableVersion rev1ver = new ComparableVersion(rev1.version.toString());
305                        ComparableVersion rev2ver = new ComparableVersion(rev2.version.toString());
306                        return rev1ver.compareTo(rev2ver);
307                }
308        }
