019package de.mnl.osgi.bnd.repository.maven.provider;
020
021import aQute.bnd.http.HttpClient;
022import aQute.bnd.osgi.Processor;
023import aQute.bnd.osgi.repository.XMLResourceParser;
024import aQute.maven.api.IMavenRepo;
025import aQute.maven.api.Revision;
026import aQute.maven.provider.MavenBackingRepository;
027import aQute.maven.provider.MavenRepository;
028import aQute.service.reporter.Reporter;
029
030import java.io.File;
031import java.net.URL;
032import java.util.ArrayList;
033import java.util.List;
034import java.util.stream.Collectors;
035
036import org.osgi.resource.Resource;
037import org.osgi.service.repository.Repository;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041/**
042 * Provide an OSGi repository (a collection of {@link Resource}s, see 
043 * {@link Repository}), filled with the results of searching specific
044 * artifacts.
045 */
046public class MavenCoordinatesOsgiRepository extends MavenOsgiRepository {
047
048        private static final Logger logger = LoggerFactory.getLogger(
049                        MavenCoordinatesOsgiRepository.class);
050        private List<URL> releaseUrls;
051        private List<URL> snapshotUrls;
052        private List<String> coordinates;
053        private File localRepo = null;
054        private Reporter reporter;
055        private HttpClient client; 
056        private MavenRepository mavenRepository;
057        
058        /**
059         * Create a new instance that uses the provided information/resources to perform
060         * its work.
061         *
062         * @param name the name
063         * @param releaseUrls the release urls
064         * @param snapshotUrls the snapshot urls
065         * @param coordinates the coordinates
066         * @param localRepo the local Maven repository (cache)
067         * @param obrIndexFile the persistent representation of this repository's content
068         * @param reporter a reporter for reporting the progress
069         * @param client an HTTP client for obtaining information from the Nexus server
070         * @throws Exception the exception
071         */
072        public MavenCoordinatesOsgiRepository (String name, List<URL> releaseUrls, List<URL> snapshotUrls,
073                        List<String> coordinates, File localRepo, File obrIndexFile, 
074                        Reporter reporter, HttpClient client)
075                        throws Exception {
076                super(name, obrIndexFile);
077                this.releaseUrls = releaseUrls;
078                this.snapshotUrls = snapshotUrls;
079                this.coordinates = coordinates;
080                this.localRepo = localRepo;
081                this.reporter = reporter;
082                this.client = client;
083                
084                // load results from previous execution.
085                mavenRepository = createMavenRepository();
086                if (!location().exists() || !location().isFile()) {
087                        refresh();
088                } else {
089                        try (XMLResourceParser parser = new XMLResourceParser(location())) {
090                                List<Resource> resources = parser.parse();
091                                addAll(resources);
092                        }
093                }
094        }
095
096        /**
097         * Refresh this repository's content.
098         * 
099         * @return true if refreshed, false if not refreshed possibly due to error
100         * @throws Exception if a problem occurs
101         */
102        public boolean refresh() throws Exception {
103                if (coordinates == null) {
104                        return false;
105                }
106                List<Revision> startRevisions = coordinates.stream()
107                                .map(an -> Revision.valueOf(an)).collect(Collectors.toList());
108                // List all revisions from configuration.
109                for (Revision revision: startRevisions) {
110                        logger.debug("Found {}", revision);                                     
111                }
112                return refresh(mavenRepository, startRevisions);
113        }
114
115        private MavenRepository createMavenRepository() throws Exception {
116                // Create repository from URLs
117                List<MavenBackingRepository> releaseBackers = new ArrayList<>();
118                for(URL url: releaseUrls) {
119                        releaseBackers.addAll(MavenBackingRepository.create(
120                                        url.toString(), reporter, localRepo, client));
121                }
122                List<MavenBackingRepository> snapshotBackers = new ArrayList<>();
123                for(URL url: snapshotUrls) {
124                        snapshotBackers.addAll(MavenBackingRepository.create(
125                                        url.toString(), reporter, localRepo, client));
126                }
127                return new MavenRepository(localRepo, name(), 
128                                releaseBackers, snapshotBackers,
129                                Processor.getExecutor(), reporter);
130        }
131        
132        /**
133         * Return the Maven repository object used to back this repository.
134         * 
135         * @return the repository object
136         * @throws Exception if a problem occurs
137         */
138        public IMavenRepo mavenRepository() throws Exception {
139                if (mavenRepository == null) {
140                        refresh();
141                }
142                return mavenRepository;
143        }
144}
