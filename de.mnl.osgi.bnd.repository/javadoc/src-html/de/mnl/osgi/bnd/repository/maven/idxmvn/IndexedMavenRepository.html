072/**
073 * Provide an OSGi repository (a collection of {@link Resource}s, see 
074 * {@link Repository}), filled with resolved artifacts from given groupIds.
075 */
076@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
077public class IndexedMavenRepository extends ResourcesRepository {
078
079    /* default */ static ExecutorService programLoaders
080        = Executors.newFixedThreadPool(4);
081    /* default */ static ExecutorService revisionLoaders
082        = Executors.newFixedThreadPool(8);

084    private static final Logger LOG = LoggerFactory.getLogger(
085        IndexedMavenRepository.class);
086    private final String name;
087    private final Path indexDbDir;
088    private final Path depsDir;
089    private final List<URL> releaseUrls;
090    private final List<URL> snapshotUrls;
091    private final File localRepo;
092    private final Reporter reporter;
093    private final HttpClient client;
094    private final ExecutorService groupLoaders
095        = Executors.newFixedThreadPool(4);
096    private final CompositeMavenRepository mavenRepository;
097    private final Map<String, MavenGroupRepository> groups
098        = new ConcurrentHashMap<>();

100    /**
101     * Create a new instance that uses the provided information/resources to perform
102     * its work.
103     *
104     * @param name the name
105     * @param releaseUrls the release urls
106     * @param snapshotUrls the snapshot urls
107     * @param localRepo the local Maven repository (cache)
108     * @param indexDbDir the persistent representation of this repository's content
109     * @param reporter a reporter for reporting the progress
110     * @param client an HTTP client for obtaining information from the Nexus server
111     * @throws Exception the exception
112     */
113    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
114        "PMD.SignatureDeclareThrowsException", "PMD.AvoidDuplicateLiterals" })
115    public IndexedMavenRepository(String name, List<URL> releaseUrls,
116            List<URL> snapshotUrls, File localRepo, File indexDbDir,
117            Reporter reporter, HttpClient client) throws Exception {
118        this.name = name;
119        this.indexDbDir = indexDbDir.toPath();
120        depsDir = this.indexDbDir.resolve("dependencies");
121        this.releaseUrls = releaseUrls;
122        this.snapshotUrls = snapshotUrls;
123        this.localRepo = localRepo;
124        this.reporter = reporter;
125        this.client = client;
126
127        // Check prerequisites
128        if (indexDbDir.exists() && !indexDbDir.isDirectory()) {
129            reporter.error("%s must be a directory.", indexDbDir);
130            throw new IOException(indexDbDir + "must be a directory.");
131        }
132        if (!indexDbDir.exists()) {
133            indexDbDir.mkdirs();
134        }
135        if (!depsDir.toFile().exists()) {
136            depsDir.toFile().mkdir();
137        }
138
139        // Our repository
140        mavenRepository = createMavenRepository();
141
142        // Restore
143        @SuppressWarnings("PMD.UseConcurrentHashMap")
144        Map<String, MavenGroupRepository> oldGroups = new HashMap<>();
145        CompletableFuture
146            .allOf(scanRequested(oldGroups), scanDependencies(oldGroups)).get();
147        for (MavenGroupRepository groupRepo : groups.values()) {
148            addAll(groupRepo.getResources());
149        }
150    }
151
152    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
153    private CompositeMavenRepository createMavenRepository() throws Exception {
154        // Create repository from URLs
155        List<MavenBackingRepository> releaseBackers = new ArrayList<>();
156        for (URL url : releaseUrls) {
157            releaseBackers.addAll(MavenBackingRepository.create(
158                url.toString(), reporter, localRepo, client));
159        }
160        List<MavenBackingRepository> snapshotBackers = new ArrayList<>();
161        for (URL url : snapshotUrls) {
162            snapshotBackers.addAll(MavenBackingRepository.create(
163                url.toString(), reporter, localRepo, client));
164        }
165        return new CompositeMavenRepository(localRepo, name(), releaseBackers,
166            snapshotBackers, Processor.getExecutor(), reporter);
167    }

196    /**
197     * Get or create the group repository for the given group id.
198     * If the repository is created, it is created as a repository
199     * for dependencies.
200     *
201     * @param groupId the group id
202     * @return the repository
203     * @throws IOException Signals that an I/O exception has occurred.
204     */
205    public MavenGroupRepository getOrCreateGroupRepository(String groupId)
206            throws IOException {
207        @SuppressWarnings("PMD.PrematureDeclaration")
208        MavenGroupRepository result = rethrow(IOException.class,
209            () -> groups.computeIfAbsent(groupId,
210                grp -> unthrow(() -> new MavenGroupRepository(grp,
211                    depsDir.resolve(grp), false, this, client, reporter))));
212        return result;
213    }
214
215    /**
216     * Refresh this repository's content.
217     * 
218     * @return true if refreshed, false if not refreshed possibly due to error
219     * @throws Exception if a problem occurs
220     */
221    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
222        "PMD.AvoidInstantiatingObjectsInLoops", "PMD.AvoidDuplicateLiterals" })
223    public boolean refresh() throws Exception {
224        mavenRepository.reset();
225        @SuppressWarnings("PMD.UseConcurrentHashMap")
226        Map<String, MavenGroupRepository> oldGroups = new HashMap<>(groups);
227
228        // Reuse and clear (or create new) group repositories for the existing
229        // directories, first for explicitly requested group ids...
230        groups.clear();
231        CompletableFuture
232            .allOf(scanRequested(oldGroups), scanDependencies(oldGroups)).get();
233        // Refresh them all.
234        CompletableFuture.allOf(new ArrayList<>(groups.values()).stream()
235            .map(repo -> CompletableFuture.runAsync(() -> {
236                try {
237                    LOG.debug("Reloading %s...", repo.id());
238                    repo.reload();
239                } catch (IOException e) {
240                    throw new CompletionException(e);
241                }
242            }, groupLoaders))
243            .toArray(CompletableFuture[]::new)).get();
244        // Remove no longer required group repositories.
245        for (Iterator<Map.Entry<String, MavenGroupRepository>> iter
246            = groups.entrySet().iterator(); iter.hasNext();) {
247            Map.Entry<String, MavenGroupRepository> entry = iter.next();
248            if (entry.getValue().removeIfRedundant()) {
249                iter.remove();
250            }
251        }
252        CompletableFuture.allOf(
253            // Persist updated data.
254            CompletableFuture.allOf(new ArrayList<>(groups.values()).stream()
255                .map(repo -> CompletableFuture.runAsync(() -> {
256                    try {
257                        repo.flush();
258                    } catch (IOException e) {
259                        throw new CompletionException(e);
260                    }
261                }, groupLoaders))
262                .toArray(CompletableFuture[]::new)),
263            // Write federated index.
264            CompletableFuture.runAsync(() -> {
265                if (groups.keySet().equals(oldGroups.keySet())) {
266                    return;
267                }
268                try (OutputStream fos
269                    = Files.newOutputStream(indexDbDir.resolve("index.xml"))) {
270                    writeFederatedIndex(fos);
271                } catch (IOException e) {
272                    throw new CompletionException(e);
273                }
274            }, groupLoaders),
275            // Add collected to root (this).
276            CompletableFuture.runAsync(() -> {
277                set(Collections.emptyList());
278                for (MavenGroupRepository groupRepo : groups.values()) {
279                    addAll(groupRepo.getResources());
280                }
281            }, groupLoaders)).get();
282        return true;
283    }
284
285    private CompletableFuture<Void>
286            scanRequested(Map<String, MavenGroupRepository> oldGroups) {
287        return CompletableFuture.allOf(
288            Arrays.stream(indexDbDir.toFile().list()).parallel()
289                .filter(dir -> dir.matches("^[A-Za-z].*")
290                    && !"index.xml".equals(dir)
291                    && !"dependencies".equals(dir))
292                .map(groupId -> CompletableFuture
293                    .runAsync(() -> restoreGroup(oldGroups, groupId, true),
294                        groupLoaders))
295                .toArray(CompletableFuture[]::new));
296    }
297
298    private CompletableFuture<Void>
299            scanDependencies(Map<String, MavenGroupRepository> oldGroups) {
300        if (!depsDir.toFile().canRead() || !depsDir.toFile().isDirectory()) {
301            return CompletableFuture.completedFuture(null);
302        }
303        return CompletableFuture.allOf(
304            Arrays.stream(depsDir.toFile().list()).parallel()
305                .filter(dir -> dir.matches("^[A-Za-z].*"))
306                .map(groupId -> CompletableFuture
307                    .runAsync(() -> restoreGroup(oldGroups, groupId, false),
308                        groupLoaders))
309                .toArray(CompletableFuture[]::new));
310    }
311
312    private void restoreGroup(Map<String, MavenGroupRepository> oldGroups,
313            String groupId, boolean requested) {
314        if (oldGroups.containsKey(groupId)) {
315            // Reuse existing.
316            MavenGroupRepository groupRepo = oldGroups.get(groupId);
317            groupRepo.reset(requested);
318            groups.put(groupId, groupRepo);
319            return;
320        }
321        try {
322            MavenGroupRepository groupRepo
323                = new MavenGroupRepository(
324                    groupId,
325                    (requested ? indexDbDir : depsDir).resolve(groupId),
326                    requested, this, client, reporter);
327            groups.put(groupId, groupRepo);
328        } catch (IOException e) {
329            reporter.exception(e,
330                "Cannot create group repository for %s: %s",
331                groupId, e.getMessage());
332        }
333    }
334
335    private void writeFederatedIndex(OutputStream out) {
336        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
337        dbf.setNamespaceAware(true);
338        Document doc;
339        try {
340            doc = dbf.newDocumentBuilder().newDocument();
341        } catch (ParserConfigurationException e) {
342            return;
343        }
344        @SuppressWarnings("PMD.AvoidFinalLocalVariable")
345        final String repoNs = "http://www.osgi.org/xmlns/repository/v1.0.0";
346        // <repository name=... increment=...>
347        Element repoNode = doc.createElementNS(repoNs, "repository");
348        repoNode.setAttribute("name", name);
349        repoNode.setAttribute("increment",
350            Long.toString(System.currentTimeMillis()));
351        doc.appendChild(repoNode);
352        for (Map.Entry<String, MavenGroupRepository> repo : groups.entrySet()) {
353            // <referral url=...>
354            Element referral = doc.createElementNS(repoNs, "referral");
355            referral.setAttribute("url",
356                (repo.getValue().isRequested() ? "" : "dependencies/")
357                    + repo.getKey() + "/index.xml");
358            repoNode.appendChild(referral);
359        }
360        // Write federated index.
361        try {
362            Transformer transformer
363                = TransformerFactory.newInstance().newTransformer();
364            transformer.setOutputProperty(OutputKeys.INDENT, "yes");
365            DOMSource source = new DOMSource(doc);
366            transformer.transform(source, new StreamResult(out));
367        } catch (TransformerException e) {
368            reporter.exception(e, "Cannot write federated index: %s",
369                e.getMessage());
370        }
371    }

373    /**
374     * Gets the resource just like 
375     * {@link CompositeMavenRepository#toResource(BoundRevision, 
376     * CompositeMavenRepository.BinaryLocation)} (called with
377     * {@link BinaryLocation#REMOTE} but uses
378     * {@link MavenGroupRepository#toResource(Revision)}
379     * to profit from the group respository's caching.
380     *
381     * @param revision the version
382     * @return the resource
383     * @throws IOException Signals that an I/O exception has occurred.
384     */
385    public MavenResource toResource(BoundRevision revision)
386            throws IOException {
387        return getOrCreateGroupRepository(revision.groupId())
388            .toResource(revision);
389    }

419            program, version);
