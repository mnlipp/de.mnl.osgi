051/**
052 * Provide an OSGi repository (a collection of {@link Resource}s, see 
053 * {@link Repository}), filled with resolved artifacts from given groupIds.
054 */
055public class IndexedMavenRepository extends ResourcesRepository {
056
057    private static final Logger LOG = LoggerFactory.getLogger(
058        IndexedMavenRepository.class);
059    private final String name;
060    private final File indexDbDir;
061    private final List<URL> releaseUrls;
062    private final List<URL> snapshotUrls;
063    private final File localRepo;
064    private final Reporter reporter;
065    private final HttpClient client;
066    private final MergingMavenRepository mavenRepository;
067    private final Set<URL> repositoryUrls = new HashSet<>();
068    private final Pattern hrefPattern = Pattern.compile(
069        "<[aA]\\s+(?:[^>]*?\\s+)?href=(?<quote>[\"'])"
070            + "(?<href>[a-zA-Z].*?)\\k<quote>");
071
072    /**
073     * Create a new instance that uses the provided information/resources to perform
074     * its work.
075     *
076     * @param name the name
077     * @param releaseUrls the release urls
078     * @param snapshotUrls the snapshot urls
079     * @param localRepo the local Maven repository (cache)
080     * @param indexDbDir the persistent representation of this repository's content
081     * @param reporter a reporter for reporting the progress
082     * @param client an HTTP client for obtaining information from the Nexus server
083     * @throws Exception the exception
084     */
085    public IndexedMavenRepository(String name, List<URL> releaseUrls,
086            List<URL> snapshotUrls, File localRepo, File indexDbDir,
087            Reporter reporter, HttpClient client) throws Exception {
088        this.name = name;
089        this.indexDbDir = indexDbDir;
090        this.releaseUrls = releaseUrls;
091        this.snapshotUrls = snapshotUrls;
092        this.localRepo = localRepo;
093        this.reporter = reporter;
094        this.client = client;
095
096        // Check prerequisites
097        if (indexDbDir.exists() && !indexDbDir.isDirectory()) {
098            LOG.error(indexDbDir + "must be a directory.");
099            throw new IOException(indexDbDir + "must be a directory.");
100        }
101        if (!indexDbDir.exists()) {
102            indexDbDir.mkdirs();
103        }
104
105        // All URLs
106        repositoryUrls.addAll(releaseUrls);
107        repositoryUrls.addAll(snapshotUrls);
108
109        // Our backend
110        mavenRepository = createMavenRepository();
111        // refresh();
112    }
113
114    private MergingMavenRepository createMavenRepository() throws Exception {
115        // Create repository from URLs
116        List<MavenBackingRepository> releaseBackers = new ArrayList<>();
117        for (URL url : releaseUrls) {
118            releaseBackers.addAll(MavenBackingRepository.create(
119                url.toString(), reporter, localRepo, client));
120        }
121        List<MavenBackingRepository> snapshotBackers = new ArrayList<>();
122        for (URL url : snapshotUrls) {
123            snapshotBackers.addAll(MavenBackingRepository.create(
124                url.toString(), reporter, localRepo, client));
125        }
126        return new MergingMavenRepository(localRepo, name(), releaseBackers,
127            snapshotBackers, Processor.getExecutor(), reporter);
128    }
129
130    /**
131     * Return the name of this repository.
132     * 
133     * @return the name;
134     */
135    public final String name() {
136        return name;
137    }
138
139    /**
140     * Return the representation of this repository in the local file system.
141     * 
142     * @return the location
143     */
144    public final File location() {
145        return indexDbDir;
146    }
147
148    /**
149     * Return the Maven repository object used to implements this repository.
150     * 
151     * @return the repository object
152     * @throws Exception if a problem occurs
153     */
154    public IMavenRepo mavenRepository() throws Exception {
155        if (mavenRepository == null) {
156            refresh();
157        }
158        return mavenRepository;
159    }
160
161    /**
162     * Refresh this repository's content.
163     * 
164     * @return true if refreshed, false if not refreshed possibly due to error
165     * @throws Exception if a problem occurs
166     */
167    public boolean refresh() throws Exception {
168        for (String groupId : indexDbDir.list()) {
169            if (!groupId.matches("^[A-Za-z].*")) {
170                continue;
171            }
172            LOG.debug("Refreshing {}", groupId);
173            ResourcesRepository osgiRepo = new ResourcesRepository();
174            RevisionIndexer indexer = new RevisionIndexer(mavenRepository,
175                osgiRepo, IndexedResource.REMOTE);
176            Collection<String> artifactIds = findArtifactIds(groupId);
177            for (String artifactId : artifactIds) {
178                if (artifactId.endsWith("/")) {
179                    artifactId
180                        = artifactId.substring(0, artifactId.length() - 1);
181                }
182                Program program = Program.valueOf(groupId, artifactId);
183                List<ExtRevision> revisions
184                    = mavenRepository.getExtRevisions(program);
185                Set<IPom> dependencies = new HashSet<>();
186                indexer.indexRevisions(revisions, dependencies);
187            }
188            XMLResourceGenerator generator = new XMLResourceGenerator();
189            generator.resources(osgiRepo.getResources());
190            generator.name(name());
191            generator
192                .save(new File(new File(indexDbDir, groupId), "index.xml"));
193        }
194        return true;
195    }
196
197    private Collection<String> findArtifactIds(String dir) throws Exception {
198        Set<String> result = new HashSet<>();
199        for (URL repoUrl : repositoryUrls) {
200            String page = client.build().headers("User-Agent", "Bnd")
201                .get(String.class).go(new URL(repoUrl, dir.replace('.', '/')));
202            Matcher matcher = hrefPattern.matcher(page);
203            while (matcher.find()) {
204                result.add(matcher.group("href"));
205            }
206        }
207        return result;
208    }
