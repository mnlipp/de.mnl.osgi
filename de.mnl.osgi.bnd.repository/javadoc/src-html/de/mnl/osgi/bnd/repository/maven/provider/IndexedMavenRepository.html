066/**
067 * Provide an OSGi repository (a collection of {@link Resource}s, see 
068 * {@link Repository}), filled with resolved artifacts from given groupIds.
069 */
070public class IndexedMavenRepository extends ResourcesRepository {
071
072    private static final Logger LOG = LoggerFactory.getLogger(
073        IndexedMavenRepository.class);
074    private final String name;
075    private final File indexDbDir;
076    private final List<URL> releaseUrls;
077    private final List<URL> snapshotUrls;
078    private final File localRepo;
079    private final Reporter reporter;
080    private final HttpClient client;
081    private final MergingMavenRepository mavenRepository;
082    private final Set<URL> repositoryUrls = new HashSet<>();
083    private final Pattern hrefPattern = Pattern.compile(
084        "<[aA]\\s+(?:[^>]*?\\s+)?href=(?<quote>[\"'])"
085            + "(?<href>[a-zA-Z].*?)\\k<quote>");
086    Map<String, ResourcesRepository> groupRepos = new HashMap<>();
087
088    /**
089     * Create a new instance that uses the provided information/resources to perform
090     * its work.
091     *
092     * @param name the name
093     * @param releaseUrls the release urls
094     * @param snapshotUrls the snapshot urls
095     * @param localRepo the local Maven repository (cache)
096     * @param indexDbDir the persistent representation of this repository's content
097     * @param reporter a reporter for reporting the progress
098     * @param client an HTTP client for obtaining information from the Nexus server
099     * @throws Exception the exception
100     */
101    public IndexedMavenRepository(String name, List<URL> releaseUrls,
102            List<URL> snapshotUrls, File localRepo, File indexDbDir,
103            Reporter reporter, HttpClient client) throws Exception {
104        this.name = name;
105        this.indexDbDir = indexDbDir;
106        this.releaseUrls = releaseUrls;
107        this.snapshotUrls = snapshotUrls;
108        this.localRepo = localRepo;
109        this.reporter = reporter;
110        this.client = client;
111
112        // Check prerequisites
113        if (indexDbDir.exists() && !indexDbDir.isDirectory()) {
114            LOG.error(indexDbDir + "must be a directory.");
115            throw new IOException(indexDbDir + "must be a directory.");
116        }
117        if (!indexDbDir.exists()) {
118            indexDbDir.mkdirs();
119        }
120
121        // All URLs
122        repositoryUrls.addAll(releaseUrls);
123        repositoryUrls.addAll(snapshotUrls);
124
125        // Our backend
126        mavenRepository = createMavenRepository();
127        // refresh();
128    }
129
130    private MergingMavenRepository createMavenRepository() throws Exception {
131        // Create repository from URLs
132        List<MavenBackingRepository> releaseBackers = new ArrayList<>();
133        for (URL url : releaseUrls) {
134            releaseBackers.addAll(MavenBackingRepository.create(
135                url.toString(), reporter, localRepo, client));
136        }
137        List<MavenBackingRepository> snapshotBackers = new ArrayList<>();
138        for (URL url : snapshotUrls) {
139            snapshotBackers.addAll(MavenBackingRepository.create(
140                url.toString(), reporter, localRepo, client));
141        }
142        return new MergingMavenRepository(localRepo, name(), releaseBackers,
143            snapshotBackers, Processor.getExecutor(), reporter);
144    }
145
146    /**
147     * Return the name of this repository.
148     * 
149     * @return the name;
150     */
151    public final String name() {
152        return name;
153    }
154
155    /**
156     * Return the representation of this repository in the local file system.
157     * 
158     * @return the location
159     */
160    public final File location() {
161        return indexDbDir;
162    }
163
164    /**
165     * Return the Maven repository object used to implements this repository.
166     * 
167     * @return the repository object
168     * @throws Exception if a problem occurs
169     */
170    public IMavenRepo mavenRepository() throws Exception {
171        if (mavenRepository == null) {
172            refresh();
173        }
174        return mavenRepository;
175    }
176
177    /**
178     * Refresh this repository's content.
179     * 
180     * @return true if refreshed, false if not refreshed possibly due to error
181     * @throws Exception if a problem occurs
182     */
183    public boolean refresh() throws Exception {
184        for (String groupId : indexDbDir.list()) {
185            if (!groupId.matches("^[A-Za-z].*")
186                || groupId.equals("index.xml")) {
187                continue;
188            }
189            LOG.debug("Refreshing {}", groupId);
190            ResourcesRepository osgiRepo = new ResourcesRepository();
191            groupRepos.put(groupId, osgiRepo);
192            RevisionIndexer indexer = new RevisionIndexer(mavenRepository,
193                osgiRepo, IndexedResource.REMOTE);
194            Collection<String> artifactIds = findArtifactIds(groupId);
195            for (String artifactId : artifactIds) {
196                if (artifactId.endsWith("/")) {
197                    artifactId
198                        = artifactId.substring(0, artifactId.length() - 1);
199                }
200                Program program = Program.valueOf(groupId, artifactId);
201                List<ExtRevision> revisions
202                    = mavenRepository.getExtRevisions(program);
203                Set<IPom> dependencies = new HashSet<>();
204                indexer.indexRevisions(revisions, dependencies);
205            }
206            XMLResourceGenerator generator = new XMLResourceGenerator();
207            generator.resources(osgiRepo.getResources());
208            generator.name(name());
209            generator
210                .save(new File(new File(indexDbDir, groupId), "index.xml"));
211        }
212        try (OutputStream fos
213            = new FileOutputStream(new File(indexDbDir, "index.xml"))) {
214            writeFederatedIndex(fos);
215        }
216        return true;
217    }
218
219    private Collection<String> findArtifactIds(String dir) throws Exception {
220        Set<String> result = new HashSet<>();
221        for (URL repoUrl : repositoryUrls) {
222            String page = client.build().headers("User-Agent", "Bnd")
223                .get(String.class).go(new URL(repoUrl, dir.replace('.', '/')));
224            if (page == null) {
225                return result;
226            }
227            Matcher matcher = hrefPattern.matcher(page);
228            while (matcher.find()) {
229                result.add(matcher.group("href"));
230            }
231        }
232        return result;
233    }
234
235    private void writeFederatedIndex(OutputStream out) {
236        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
237        dbf.setNamespaceAware(true);
238        Document doc;
239        final String repoNs = "http://www.osgi.org/xmlns/repository/v1.0.0";
240        try {
241            doc = dbf.newDocumentBuilder().newDocument();
242        } catch (ParserConfigurationException e) {
243            return;
244        }
245        // <repository name=... increment=...>
246        Element repoNode = doc.createElementNS(repoNs, "repository");
247        repoNode.setAttribute("name", name);
248        repoNode.setAttribute("increment",
249            Long.toString(System.currentTimeMillis()));
250        doc.appendChild(repoNode);
251        for (String groupId : groupRepos.keySet()) {
252            // <referral url=...>
253            Element referral = doc.createElementNS(repoNs, "referral");
254            referral.setAttribute("url", groupId + "/index.xml");
255            repoNode.appendChild(referral);
256        }
257        // Write federated index.
258        try {
259            Transformer transformer
260                = TransformerFactory.newInstance().newTransformer();
261            transformer.setOutputProperty(OutputKeys.INDENT, "yes");
262            DOMSource source = new DOMSource(doc);
263            transformer.transform(source, new StreamResult(out));
264        } catch (TransformerException e) {
265            // Shouldn't happen
266        }
267    }
