112    /**
113     * Instantiates a new representation of group data backed
114     * by the specified directory. 
115     *
116     * @param groupId the maven groupId indexed by this repository
117     * @param directory the directory used to persist data
118     * @param requested if it is a requested group id
119     * @param indexedRepository the indexed maven repository
120     * @param client the client used for remote access
121     * @param reporter the reporter
122     * @throws IOException Signals that an I/O exception has occurred.
123     */
124    @SuppressWarnings({ "PMD.ConfusingTernary",
125        "PMD.AvoidCatchingGenericException", "PMD.AvoidDuplicateLiterals" })
126    public MavenGroupRepository(String groupId, Path directory,
127            boolean requested, IndexedMavenRepository indexedRepository,
128            HttpClient client, Reporter reporter) throws IOException {
129        this.groupId = groupId;
130        this.groupDir = directory;
131        this.requested = requested;
132        this.indexedRepository = indexedRepository;
133        this.client = client;
134        this.reporter = reporter;
135
136        // Prepare directory and files
137        groupPropsPath = directory.resolve("group.properties");
138        groupProps = new Properties();
139        // If directory does not exist, it's not from a request.
140        if (!directory.toFile().exists()) {
141            directory.toFile().mkdir();
142            propsChanged = true;
143        } else {
144            // Directory exists, either as newly created or as "old"
145            if (groupPropsPath.toFile().canRead()) {
146                try (InputStream input = Files.newInputStream(groupPropsPath)) {
147                    groupProps.load(input);
148                }
149            } else {
150                propsChanged = true;
151            }
152        }
153
154        // Prepare OSGi repository
155        groupIndexPath = groupDir.resolve("index.xml");
156        if (groupIndexPath.toFile().canRead()) {
157            try (XMLResourceParser parser
158                = new XMLResourceParser(groupIndexPath.toFile())) {
159                addAll(parser.parse());
160            } catch (Exception e) { // NOPMD
161                reporter.warning("Cannot parse %s, ignored: %s", groupIndexPath,
162                    e.getMessage());
163            }
164        } else {
165            indexChanged = true;
166        }
167        // Cache revisions for faster checks.
168        for (Capability cap : findProvider(
169            newRequirementBuilder("bnd.info").build())) {
170            indexingState.put(
171                Revision.valueOf((String) cap.getAttributes().get("from")),
172                IndexingState.INDEXED);
173        }
174        LOG.debug("Created group repository for {}.", groupId);
175    }

186    /**
187     * Writes all changes to persistent storage and removes
188     * any backup information prepared by {@link #reset(boolean)}.
189     *
190     * @throws IOException Signals that an I/O exception has occurred.
191     */
192    public void flush() throws IOException {
193        if (propsChanged) {
194            try (OutputStream out = Files.newOutputStream(groupPropsPath)) {
195                groupProps.store(out, "Group properties");
196            }
197            propsChanged = false;
198        }
199        if (indexChanged && backupRepo != null) {
200            // Only a rough guess, verify to avoid unnecessary new versions
201            // of the generated file.
202            Set<Resource> oldSet = new HashSet<>(backupRepo.getResources());
203            Set<Resource> newSet = new HashSet<>(getResources());
204            if (newSet.equals(oldSet)) {
205                indexChanged = false;
206            }
207        }
208        if (indexChanged) {
209            XMLResourceGenerator generator = new XMLResourceGenerator();
210            generator.resources(getResources());
211            generator.name(indexedRepository.mavenRepository().getName());
212            try {
213                generator.save(groupIndexPath.toFile());
214            } catch (IOException e) {
215                reporter.exception(e, "Cannot save %s.", groupIndexPath);
216            }
217            indexChanged = false;
218        }
219        backupRepo = null;
220        indexingState.clear();
221        resourceCache.clear();
222    }

224    /**
225     * Removes the group directory if empty.
226     *
227     * @return true, if removed
228     * @throws IOException Signals that an I/O exception has occurred.
229     */
230    public boolean removeIfRedundant() throws IOException {
231        if (isRequested() || !groupProps.isEmpty()
232            || !getResources().isEmpty()) {
233            return false;
234        }
235        // Nothing in this group
236        Files.walk(groupDir)
237            .sorted(Comparator.reverseOrder())
238            .map(Path::toFile)
239            .forEach(File::delete);
240        return true;
241    }

253    /**
254     * Clears this repository and updates the requested flag. 
255     * Keeps the current content as backup for reuse in a 
256     * subsequent call to {@link #reload()}.
257     *
258     * @param requested whether this is a requested group
259     */
260    public void reset(boolean requested) {
261        synchronized (this) {
262            // Update "type"
263            this.requested = requested;
264            // Save current content and clear.
265            backupRepo = new ResourcesRepository(getResources());
266            set(Collections.emptyList());
267            // Clear and reload properties
268            groupProps.clear();
269            if (groupPropsPath.toFile().canRead()) {
270                try (InputStream input = Files.newInputStream(groupPropsPath)) {
271                    propQueryCache.clear();
272                    groupProps.load(input);
273                } catch (IOException e) {
274                    reporter.warning("Problem reading %s (ignored): %s",
275                        groupPropsPath, e.getMessage());
276                }
277            }
278            // Clear remaining caches.
279            indexingState.clear();
280            propQueryCache.clear();
281            resourceCache.clear();
282        }
283    }

285    /**
286     * Reload the repository. Requested repositories retrieve
287     * the list of known artifactIds from the remote repository 
288     * and add the versions. For versions already in the repository, 
289     * the backup information is re-used.
290     *
291     * @throws IOException Signals that an I/O exception has occurred.
292     */
293    @SuppressWarnings({ "PMD.AvoidReassigningLoopVariables",
294        "PMD.AvoidCatchingGenericException", "PMD.AvoidDuplicateLiterals",
295        "PMD.AvoidInstantiatingObjectsInLoops",
296        "PMD.AvoidThrowingRawExceptionTypes", "PMD.PreserveStackTrace" })
297    public void reload() throws IOException {
298        if (!isRequested()) {
299            // Will be filled with dependencies only
300            return;
301        }
302        // Actively filled.
303        synchronized (this) {
304            if (backupRepo == null) {
305                backupRepo = new ResourcesRepository(getResources());
306            }
307            propQueryCache.clear();
308        }
309        try {
310            CompletableFuture<Void> actions
311                = CompletableFuture.allOf(findArtifactIds().stream().map(
312                    artifactId -> loadProgram(Program.valueOf(groupId,
313                        artifactId)))
314                    .toArray(CompletableFuture[]::new));
315            actions.get();
316        } catch (ExecutionException e) {
317            if (e.getCause() instanceof IOException) {
318                throw (IOException) e.getCause();
319            }
320            throw new RuntimeException(e.getCause());
321        } catch (InterruptedException e) {
322            throw new RuntimeException(e);
323        }
324        indexChanged = true;
325    }

327    @SuppressWarnings("PMD.ForLoopCanBeForeach")
328    private CompletableFuture<Void> loadProgram(Program program) {
329        // Get revisions of program and process.
330        CompletableFuture<Void> result = new CompletableFuture<>();
331        IndexedMavenRepository.programLoaders.submit(() -> {
332            try {
333                Stream<CompletableFuture<Void>> revLoaderStream
334                    = indexedRepository.mavenRepository()
335                        .boundRevisions(program).parallel()
336                        .map(revision -> loadRevision(revision));
337                try {
338                    for (CompletableFuture<Void> revLoad : (Iterable<
339                            CompletableFuture<
340                                    Void>>) revLoaderStream::iterator) {
341                        revLoad.get();
342                    }
343                } catch (InterruptedException e) {
344                    result.completeExceptionally(e);
345                    return;
346                } catch (ExecutionException e) {
347                    result.completeExceptionally(e.getCause());
348                    return;
349                }
350                result.complete(null);
351            } catch (IOException e) {
352                result.completeExceptionally(e);
353            }
354        });
355        return result;
356    }

358    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
359        "PMD.AvoidDuplicateLiterals",
360        "PMD.PositionLiteralsFirstInComparisons" })
361    private CompletableFuture<Void> loadRevision(BoundRevision revision) {
362        return CompletableFuture.runAsync(() -> {
363            if (indexingState.getOrDefault(revision.unbound(),
364                IndexingState.NONE) != IndexingState.NONE) {
365                // Already loading or handled (as dependency)
366                return;
367            }
368            MavenVersionRange acceptedRange = Optional.ofNullable(
369                searchProperty(revision.artifactId(), "versions"))
370                .map(MavenVersionRange::parseRange).orElse(null);
371            MavenVersionRange forcedRange = Optional.ofNullable(
372                searchProperty(revision.artifactId(), "forcedVersions"))
373                .map(MavenVersionRange::parseRange).orElse(null);
374            boolean inForced = forcedRange != null
375                && forcedRange.includes(revision.version());
376            if (acceptedRange != null || forcedRange != null) {
377                boolean inVersions = acceptedRange != null
378                    && acceptedRange.includes(revision.version());
379                if (!inVersions && !inForced) {
380                    return;
381                }
382            }
383            try {
384                MavenResource resource = toResource(revision);
385                Set<MavenResource> allDeps = new HashSet<>();
386                if (!isIndexable(resource, allDeps, inForced)) {
387                    return;
388                }
389                if (isBundle(resource)) {
390                    addResource(resource);
391                }
392                rethrow(IOException.class, () -> allDeps.stream()
393                    .filter(res -> unthrow(() -> isBundle(res)))
394                    .filter(res -> indexingState.getOrDefault(
395                        res.revision(),
396                        IndexingState.NONE) == IndexingState.NONE)
397                    .forEach(res -> unthrow(() -> indexedRepository
398                        .getOrCreateGroupRepository(res.revision().group)
399                        .addResource(res))));
400            } catch (Exception e) {
401                // Load "in isolation", don't propagate individual failure.
402                reporter.exception(e, "Failed to load %s: %s", revision,
403                    e.getMessage());
404            }
405        }, IndexedMavenRepository.revisionLoaders);
406
407    }

409    @SuppressWarnings("PMD.PositionLiteralsFirstInComparisons")
410    private boolean isBundle(MavenResource resource) throws IOException {
411        return resource.getCapabilities("osgi.identity").stream()
412            .map(cap -> cap.getAttributes().keySet().stream())
413            .flatMap(Function.identity())
414            .anyMatch(key -> key.equals("osgi.identity"));
415    }

417    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
418        "PMD.AvoidInstantiatingObjectsInLoops" })
419    private Collection<String> findArtifactIds() {
420        Set<String> result = new HashSet<>();
421        for (MavenBackingRepository repo : indexedRepository.mavenRepository()
422            .allRepositories()) {
423            URI groupUri = null;
424            try {
425                groupUri
426                    = repo.toURI("").resolve(groupId.replace('.', '/') + "/");
427                String page = client.build().headers("User-Agent", "Bnd")
428                    .get(String.class)
429                    .go(groupUri);
430                if (page == null) {
431                    continue;
432                }
433                Matcher matcher = hrefPattern.matcher(page);
434                while (matcher.find()) {
435                    URI programUri = groupUri.resolve(matcher.group("href"));
436                    String artifactId = programUri.getPath()
437                        .substring(groupUri.getPath().length());
438                    if (artifactId.endsWith("/")) {
439                        artifactId
440                            = artifactId.substring(0, artifactId.length() - 1);
441                    }
442                    result.add(artifactId);
443                }
444            } catch (Exception e) {
445                reporter.warning("Problem retrieving %s, skipped: %s", groupUri,
446                    e.getMessage());
447            }
448        }
449        return result;
450    }

452    /**
453     * Checks if the given revision is indexable. A revision is
454     * indexable if it is not excluded and its dependencies aren't
455     * excluded either.
456     *
457     * @param resource the resource to check
458     * @param dependencies revisions that must additionally be included
459     *    when the checked revision is added
460     * @param ignoreExcludedDependencies evaluate as indexable even
461     *    if some dependencies aren't indexable
462     * @return true, if the revision can be added to the index
463     */
464    @SuppressWarnings({ "PMD.LongVariable", "PMD.NPathComplexity" })
465    private boolean isIndexable(MavenResource resource,
466            Set<MavenResource> dependencies,
467            boolean ignoreExcludedDependencies) {
468        Revision revision = resource.revision();
469        if (!revision.group.equals(groupId)) {
470            throw new IllegalArgumentException("Wrong groupId "
471                + revision.group + " (must be " + groupId + ").");
472        }
473        // Check if already there (or being added)
474        if (INDEXING_OR_INDEXED.contains(indexingState
475            .getOrDefault(revision, IndexingState.NONE))) {
476            // Obviously indexable and no additional resources needed.
477            return true;
478        }
479        // Check if excluded by rule.
480        if (Optional.ofNullable(
481            searchProperty(revision.artifact, "exclude"))
482            .map(MavenVersionRange::parseRange)
483            .map(range -> range.includes(MavenVersion.from(revision.version)))
484            .orElse(false)) {
485            return false;
486        }
487        // Check if has been found un-indexable before
488        if (!ignoreExcludedDependencies
489            && indexingState.getOrDefault(revision,
490                IndexingState.NONE) == IndexingState.EXCL_BY_DEP) {
491            return false;
492        }
493        // Check whether excluded because dependency is excluded
494        Set<Dependency> deps;
495        try {
496            deps = resource.dependencies();
497        } catch (IOException e) {
498            // Failing to get the dependencies is no reason to fail.
499            return true;
500        }
501        for (Dependency dep : deps) {
502            MavenGroupRepository depRepo;
503            try {
504                depRepo = indexedRepository
505                    .getOrCreateGroupRepository(dep.program.group);
506            } catch (IOException e) {
507                continue;
508            }
509            Optional<MavenResource> depRes;
510            try {
511                depRes
512                    = indexedRepository.toResource(dep.program, dep.version);
513            } catch (IOException e) {
514                continue;
515            }
516            if (!depRes.isPresent()) {
517                continue;
518            }
519            if (!depRepo.isIndexable(depRes.get(), dependencies,
520                ignoreExcludedDependencies)) {
521                // Not that the revision to check is not indexable
522                // due to a dependency that is not indexable
523                indexingState.put(revision, IndexingState.EXCL_BY_DEP);
524                if (!ignoreExcludedDependencies) {
525                    return false;
526                }
527            }
528            dependencies.add(depRes.get());
529        }
530        return true;
531    }

533    /**
534     * Adds the specified revision.
535     *
536     * @param revision the revision to add
537     */
538    private void addResource(MavenResource resource) throws IOException {
539        synchronized (indexingState) {
540            if (indexingState.getOrDefault(resource.revision(),
541                IndexingState.NONE) == IndexingState.INDEXED) {
542                return;
543            }
544            add(resource.asResource());
545            indexingState.put(resource.revision(), IndexingState.INDEXED);
546        }
547        indexChanged = true;
548    }

550    /**
551     * Similar to 
552     * {@link CompositeMavenRepository#toResource(BoundRevision, BinaryLocation)}
553     * (with location {@link BinaryLocation#REMOTE}) but checks backup first.
554     *
555     * @param revision the revision
556     * @return the maven resource
557     * @throws IOException Signals that an I/O exception has occurred.
558     */
559    public MavenResource toResource(Revision revision) throws IOException {
560        if (!revision.group.equals(groupId)) {
561            return indexedRepository.toResource(revision);
562        }
563        return resourceCache.computeIfAbsent(revision,
564            rev -> Optional.ofNullable(backupRepo)
565                .flatMap(repo -> searchInRepository(repo, rev))
566                .map(res -> indexedRepository.mavenRepository()
567                    .toResource(rev, res))
568                .orElse(indexedRepository.mavenRepository()
569                    .toResource(rev, BinaryLocation.REMOTE)));
570    }

572    /**
573     * Similar to 
574     * {@link CompositeMavenRepository#toResource(BoundRevision, BinaryLocation)}
575     * but checks backup first.
576     *
577     * @param revision the revision
578     * @return the maven resource
579     * @throws IOException Signals that an I/O exception has occurred.
580     */
581    public MavenResource toResource(BoundRevision revision)
582            throws IOException {
583        if (!revision.groupId().equals(groupId)) {
584            indexedRepository.toResource(revision);
585        }
586        return resourceCache.computeIfAbsent(revision.unbound(),
587            rev -> Optional.ofNullable(backupRepo)
588                .flatMap(repo -> searchInRepository(repo, rev))
589                .map(res -> indexedRepository.mavenRepository()
590                    .toResource(revision, res))
591                .orElse(indexedRepository.mavenRepository()
592                    .toResource(revision, BinaryLocation.REMOTE)));
593    }

595    /**
596     * Checks indexing cache and if not found, binds to revision
597     * and calls {@link #toResource(BoundRevision)}.
598     *
599     * @param program the program
600     * @param version the version
601     * @return the maven resource
602     * @throws IOException Signals that an I/O exception has occurred.
603     */
604    public Optional<MavenResource> toResource(Program program,
605            String version) throws IOException {
606        if (indexingState.getOrDefault(cacheable(program, version),
607            IndexingState.NONE) == IndexingState.FAILED) {
608            return Optional.empty();
609        }
610        if (!program.group.equals(groupId)) {
611            indexedRepository.toResource(program, version);
612        }
613        if (!MavenVersionRange.isRange(version)) {
614            return Optional.of(toResource(program.version(version)));
615        }
616        try {
617            return rethrow(IOException.class,
618                () -> indexedRepository.mavenRepository()
619                    .toBoundRevision(program, version)
620                    .map(revision -> unthrow(() -> toResource(revision))));
621        } catch (IOException e) {
622            indexingState.put(cacheable(program, version),
623                IndexingState.FAILED);
624            throw e;
625        }
626    }

628    private Optional<Resource> searchInRepository(
629            ResourcesRepository repository, Revision revision) {
630        return repository
631            .findProvider(repository.newRequirementBuilder("bnd.info")
632                .addDirective("filter",
633                    String.format("(from=%s)", revision.toString()))
634                .build())
635            .stream().findFirst().map(Capability::getResource);
636    }

638    @SuppressWarnings("PMD.CompareObjectsWithEquals")
639    private String searchProperty(String artifactId, String qualifier) {
640        String queryKey = artifactId + ";" + qualifier;
641        // Attempt to get from cache.
642        String prop = propQueryCache.get(queryKey);
643        // Special value, because ConcurrentHashMap cannot have null values.
644        if (prop == NOT_AVAILABLE) {
645            return null;
646        }
647        // Found in cache return.
648        if (prop != null) {
649            return prop;
650        }
651        // Try query key unmodified
652        prop = groupProps.getProperty(queryKey);
653        if (prop == null) {
654            // Try <id>.*, successively removing trailing parts.
655            String rest = artifactId;
656            while (true) {
657                prop = groupProps.getProperty(rest + ".*;" + qualifier);
658                if (prop != null) {
659                    break;
660                }
661                int lastDot = rest.lastIndexOf('.');
662                if (lastDot < 0) {
663                    break;
664                }
665                rest = rest.substring(0, lastDot);
666            }
667        }
668        if (prop == null) {
669            prop = groupProps.getProperty(qualifier);
670        }
671        propQueryCache.put(queryKey, prop == null ? NOT_AVAILABLE : prop);
672        return prop;
673    }

685    private static Revision cacheable(Program program, String version) {
686        if (version == null) {
687            return program.version("[0,)");
688        }
689        return program.version(version);
690    }
