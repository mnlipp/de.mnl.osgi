053/**
054 * Provide a {@link NexusSearchOsgiRepository} as a bnd repository. The interesting
055 * work is done by the {@link NexusSearchOsgiRepository}.
056 */
057public class NexusSearchRepositoryProvider extends BaseRepository 
058        implements Repository, Plugin, RegistryPlugin, RepositoryPlugin, Refreshable {
059//      private final static Logger logger = LoggerFactory.getLogger(
060//                      NexusSearchRepositoryProvider.class);
061        private static final String MAVEN_REPO_LOCAL
062                = System.getProperty("maven.repo.local", "~/.m2/repository");
063
064        private boolean initialized = false;
065        private File localRepo;
066        private NexusSearchConfiguration configuration;
067        private String name = "OssSonatype";
068        private Registry registry = null;
069        private Reporter reporter;
070        private NexusSearchOsgiRepository osgiRepository = null;
071        private URL server = null;
072        private BridgeRepository bridge;
073        
074        @Override
075        public void setProperties(Map<String, String> properties) throws Exception {
076                configuration = Converter.cnv(NexusSearchConfiguration.class, properties);
077                name = configuration.name("OssSonatype");
078                server = new URL(configuration.server() != null 
079                                ? configuration.server() : "https://oss.sonatype.org");
080        }
081
082        @Override
083        public void setRegistry(Registry registry) {
084                this.registry = registry;
085        }
086
087        @Override
088        public void setReporter(Reporter reporter) {
089                this.reporter = reporter;
090        }
091
092        @Override
093        public String getName() {
094                return name;
095        }
096
097        @Override
098        public PutResult put(InputStream stream, PutOptions options) throws Exception {
099                throw new IllegalStateException("Read-only repository");
100        }
101
102        @Override
103        public boolean canWrite() {
104                return false;
105        }
106
107        /**
108         * Performs initialization. Initialization must be delayed because the
109         * precise sequence of injecting dependencies seems to be undefined.
110         * @throws MalformedURLException 
111         */
112        private synchronized void init() {
113                if (initialized) {
114                        return;
115                }
116                initialized = true;
117                String queryString = configuration.query();
118                Workspace workspace = registry.getPlugin(Workspace.class);
119                HttpClient client = registry.getPlugin(HttpClient.class);
120                File obrIndexFile = workspace.getFile(getLocation());
121                File mvnReposFile = workspace.getFile(
122                                "cnf/cache/nexus-search-" + name + "-repositories.xml");
123                localRepo = IO.getFile(configuration.local(MAVEN_REPO_LOCAL));
124                try {
125                        osgiRepository = new NexusSearchOsgiRepository(
126                                        name, server, localRepo, obrIndexFile, mvnReposFile, 
127                                        queryString, configuration.searchBreadth(3),
128                                        configuration.chunkSize(500), reporter, client);
129                        bridge = new BridgeRepository(osgiRepository);
130                } catch (Exception e) {
131                        throw new RuntimeException(e);
132                }
133        }
134        
135        @Override
136        public File getRoot() throws Exception {
137                return localRepo;
138        }
139
140        @Override
141        public boolean refresh() throws Exception {
142                init();
143                if (!osgiRepository.refresh()) {
144                        return false;
145                }
146                bridge = new BridgeRepository(osgiRepository);
147                for (RepositoryListenerPlugin listener : registry.getPlugins(RepositoryListenerPlugin.class)) {
148                        try {
149                                listener.repositoryRefreshed(this);
150                        } catch (Exception e) {
151                                reporter.exception(e, "Updating listener plugin %s", listener);
152                        }
153                }
154                return true;
155        }
156
157        @Override
158        public File get(String bsn, Version version, Map<String, String> properties, 
159                        DownloadListener... listeners) throws Exception {
160                init();
161                ResourceInfo resource = bridge.getInfo(bsn, version);
162                if (resource == null)
163                        return null;
164
165                String name = resource.getInfo().name();
166                Archive archive = Archive.valueOf(name);
167
168                Promise<File> p = osgiRepository.mavenRepository().get(archive);
169
170                if (listeners.length == 0)
171                        return p.getValue();
172
173                new DownloadListenerPromise(reporter, name + ": get " + bsn + ";" + version, p, listeners);
174                return osgiRepository.mavenRepository().toLocalFile(archive);
175        }
176
177        @Override
178        public List<String> list(String pattern) throws Exception {
179                init();
180                return bridge.list(pattern);
181        }
182
183        @Override
184        public SortedSet<Version> versions(String bsn) throws Exception {
185                init();
186                return bridge.versions(bsn);
187        }
188
189        @Override
190        public String getLocation() {
191                return "cnf/cache/nexus-search-" + name + ".xml";
192        }
193
194        @Override
195        public Map<Requirement, Collection<Capability>> findProviders(
196                        Collection<? extends Requirement> requirements) {
197                init();
198                return osgiRepository.findProviders(requirements);
199        }
200}
