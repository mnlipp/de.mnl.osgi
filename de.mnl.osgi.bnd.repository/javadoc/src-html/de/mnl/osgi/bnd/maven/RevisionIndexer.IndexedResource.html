019package de.mnl.osgi.bnd.maven;
020
021import static aQute.bnd.osgi.repository.BridgeRepository.addInformationCapability;
022
023import aQute.bnd.osgi.repository.ResourcesRepository;
024import aQute.bnd.osgi.resource.ResourceBuilder;
025import aQute.maven.api.Archive;
026import aQute.maven.api.IPom;
027import aQute.maven.api.MavenScope;
028import aQute.maven.provider.MavenRepository;
029import aQute.maven.provider.MetadataParser;
030import aQute.maven.provider.MetadataParser.RevisionMetadata;
031import de.mnl.osgi.bnd.repository.maven.provider.MavenGroupRepository;
032
033import java.io.File;
034import java.util.Collection;
035import java.util.List;
036import java.util.function.Consumer;
037
038import org.osgi.resource.Resource;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042/**
043 * Provides a processor for adding a maven resource to
044 * an OSGi repository.
045 */
046@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
047public class RevisionIndexer {
048
049    /**
050     * Use local or remote URL in index.
051     */
052    public enum IndexedResource {
053        LOCAL, REMOTE
054    }
055
056    private static final Logger LOG = LoggerFactory.getLogger(
057        MavenGroupRepository.class);
058
059    private final MavenRepository mavenRepository;
060    private final ResourcesRepository osgiRepository;
061    private final IndexedResource indexedResource;
062
063    /**
064     * Instantiates a new revision indexer.
065     *
066     * @param mavenRepository the maven repository
067     * @param osgiRepository the osgi repository
068     * @param indexedResource the indexed resource
069     */
070    public RevisionIndexer(MavenRepository mavenRepository,
071            ResourcesRepository osgiRepository,
072            IndexedResource indexedResource) {
073        this.mavenRepository = mavenRepository;
074        this.osgiRepository = osgiRepository;
075        this.indexedResource = indexedResource;
076    }
077
078    /**
079     * Indexes the given revisions and adds the results to
080     * the OSGi repository. The dependencies of the indexed
081     * revisions are collected in {@code dependencies}.  
082     *
083     * @param revisions the revisions
084     * @param dependencyHandler the dependency handler
085     */
086    public void indexRevisions(List<BoundRevision> revisions,
087            Consumer<Collection<IPom.Dependency>> dependencyHandler) {
088        for (BoundRevision revision : revisions) {
089            indexRevision(revision, dependencyHandler);
090        }
091    }
092
093    /**
094     * Indexes the given revision and adds the results to
095     * the OSGi repository. The dependencies of the indexed
096     * revisions are collected in {@code dependencies}.  
097     *
098     * @param revision the revision
099     * @param dependencyHandler the dependency handler
100     */
101
102    @SuppressWarnings("PMD.AvoidCatchingGenericException")
103    public void indexRevision(BoundRevision revision,
104            Consumer<Collection<IPom.Dependency>> dependencyHandler) {
105        // Get and add this maven revision's OSGi information
106        Archive archive;
107        try {
108            archive = mavenRepository.getResolvedArchive(revision.revision(),
109                "jar", "");
110            if (archive == null) {
111                return;
112            }
113        } catch (Exception e) {
114            LOG.error("Problem resolving archive for {}.", revision, e);
115            return;
116        }
117        if (archive.isSnapshot()) {
118            refreshSnapshot(archive, revision);
119        }
120        // Get POM for dependencies
121        try {
122            IPom pom = mavenRepository.getPom(archive.getRevision());
123            if (pom != null) {
124                Collection<IPom.Dependency> deps = pom
125                    .getDependencies(MavenScope.compile, false).values();
126                if (!deps.isEmpty()) {
127                    dependencyHandler.accept(deps);
128                }
129                deps = pom.getDependencies(MavenScope.runtime, false)
130                    .values();
131                if (!deps.isEmpty()) {
132                    dependencyHandler.accept(deps);
133                }
134            }
135        } catch (Exception e) {
136            LOG.error("Problem processing POM of {}.", revision, e);
137        }
138        Resource resource = parseResource(revision, archive);
139        if (resource != null) {
140            synchronized (osgiRepository) {
141                osgiRepository.add(resource);
142            }
143        }
144    }
145
146    @SuppressWarnings("PMD.AvoidCatchingGenericException")
147    private void refreshSnapshot(Archive archive, BoundRevision revision) {
148        File metaFile = mavenRepository.toLocalFile(
149            revision.metadata(revision.mavenBackingRepository().getId()));
150        RevisionMetadata metaData;
151        try {
152            metaData = MetadataParser.parseRevisionMetadata(metaFile);
153        } catch (Exception e) {
154            LOG.error("Problem accessing {}.", revision, e);
155            return;
156        }
157        File archiveFile = mavenRepository.toLocalFile(archive);
158        if (archiveFile.lastModified() < metaData.lastUpdated) {
159            archiveFile.delete();
160        }
161        File pomFile = mavenRepository.toLocalFile(archive.getPomArchive());
162        if (pomFile.lastModified() < metaData.lastUpdated) {
163            pomFile.delete();
164        }
165    }
166
167    @SuppressWarnings("PMD.AvoidCatchingGenericException")
168    private Resource parseResource(BoundRevision revision, Archive archive) {
169        ResourceBuilder builder = new ResourceBuilder();
170        try {
171            File binary = mavenRepository.get(archive).getValue();
172            if (indexedResource == IndexedResource.LOCAL) {
173                builder.addFile(binary, binary.toURI());
174            } else {
175                builder.addFile(binary, revision.mavenBackingRepository()
176                    .toURI(archive.remotePath));
177            }
178            addInformationCapability(builder, archive.toString(),
179                archive.getRevision().toString(), null);
180        } catch (Exception e) {
181            LOG.error("Problem accessing {}.", revision, e);
182            return null;
183        }
184        return builder.build();
185    }
