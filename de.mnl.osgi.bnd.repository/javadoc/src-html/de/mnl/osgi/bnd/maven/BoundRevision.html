019package de.mnl.osgi.bnd.maven;
020
021import aQute.bnd.version.MavenVersion;
022import aQute.maven.api.Archive;
023import aQute.maven.api.Revision;
024import aQute.maven.provider.MavenBackingRepository;
025
026/**
027 * A revision with a reference to the maven repository
028 * in which it was found.
029 * <P>
030 * This class resorts to a delegation pattern as second best solution,
031 * because {@link Revision} has only an invisible constructor and can 
032 * therefore not be extended. The drawback is that the information
033 * about the repository doesn't propagate, i.e. you cannot retrieve
034 * it from an {@link Archive} related to this {@link BoundRevision}.
035 * <P>
036 * @see <a href="https://github.com/bndtools/bnd/issues/3058">Related 
037 * bnd issue</a>
038 */
039public class BoundRevision {
040
041    private final MavenBackingRepository mavenBackingRepository;
042    private final Revision revision;
043
044    /**
045     * Instantiates a new bound revision.
046     *
047     * @param mavenBackingRepository the maven backing repository
048     * @param revision the revision
049     */
050    public BoundRevision(MavenBackingRepository mavenBackingRepository,
051            Revision revision) {
052        this.mavenBackingRepository = mavenBackingRepository;
053        this.revision = revision;
054    }
055
056    /**
057     * Gets the maven backing repository.
058     *
059     * @return the mavenBackingRepository
060     */
061    public final MavenBackingRepository mavenBackingRepository() {
062        return mavenBackingRepository;
063    }
064
065    /**
066     * Gets the revision.
067     *
068     * @return the revision
069     */
070    public final Revision revision() {
071        return revision;
072    }
073
074    /**
075     * Checks if is snapshot.
076     *
077     * @return true, if is snapshot
078     * @see aQute.maven.api.Revision#isSnapshot()
079     */
080    public boolean isSnapshot() {
081        return revision.isSnapshot();
082    }
083
084    /**
085     * Get an archive from this revision.
086     *
087     * @param extension the extension
088     * @param classifier the classifier
089     * @return the archive
090     * @see aQute.maven.api.Revision#archive(java.lang.String, java.lang.String)
091     */
092    public Archive archive(String extension, String classifier) {
093        return revision.archive(extension, classifier);
094    }
095
096    /**
097     * Get an archive from this revision.
098     *
099     * @param version the version
100     * @param extension the extension
101     * @param classifier the classifier
102     * @return the archive
103     */
104    public Archive archive(MavenVersion version, String extension,
105            String classifier) {
106        return revision.archive(version, extension, classifier);
107    }
108
109    /**
110     * Returns the Metadata.
111     *
112     * @return the string
113     * @see aQute.maven.api.Revision#metadata()
114     */
115    public String metadata() {
116        return revision.metadata();
117    }
118
119    /**
120     * Returns the metadata for the given id.
121     *
122     * @param id the id
123     * @return the string
124     * @see aQute.maven.api.Revision#metadata(java.lang.String)
125     */
126    @SuppressWarnings("PMD.ShortVariable")
127    public String metadata(String id) {
128        return revision.metadata(id);
129    }
130
131    /**
132     * Returns the revision with the origin repository attached in
133     * square brackets.
134     * 
135     * @return the string representation
136     * @see aQute.maven.api.Revision#toString()
137     */
138    public String toString() {
139        return revision.toString() + "[" + mavenBackingRepository.toString()
140            + "]";
141    }
142
143    /**
144     * Return the pom archive.
145     *
146     * @return the archive
147     * @see aQute.maven.api.Revision#pomArchive()
148     */
149    public Archive pomArchive() {
150        return revision.pomArchive();
151    }
152
153    /**
154     * Gets the pom archive.
155     *
156     * @return the pom archive
157     * @see aQute.maven.api.Revision#getPomArchive()
158     */
159    public Archive getPomArchive() {
160        return revision.getPomArchive();
161    }
162
163    /**
164     * Compare to other revision. Ignores repository.
165     *
166     * @param other the other
167     * @return the int
168     * @see aQute.maven.api.Revision#compareTo(aQute.maven.api.Revision)
169     */
170    public int compareTo(Revision other) {
171        return revision.compareTo(other);
172    }
173
174    /**
175     * Hash code. Ignores repository.
176     *
177     * @return the int
178     * @see aQute.maven.api.Revision#hashCode()
179     */
180    public int hashCode() {
181        return revision.hashCode();
182    }
183
184    /**
185     * Equals. Ignores respository.
186     *
187     * @param obj the obj
188     * @return true, if successful
189     * @see aQute.maven.api.Revision#equals(java.lang.Object)
190     */
191    public boolean equals(Object obj) {
192        if (obj instanceof BoundRevision) {
193            return revision.equals(((BoundRevision) obj).revision);
194        }
195        return revision.equals(obj);
196    }
197
198}
