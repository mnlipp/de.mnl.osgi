035/**
036 * Provides a model of an artifact version which can be used as a maven version.
037 * <P>
038 * The maven <a href="https://maven.apache.org/pom.html">POM reference</a> does
039 * not define a format for versions. This is presumably intentional as it allows
040 * artifacts with arbitrary versioning schemes to be referenced in a POM.
041 * <P>
042 * Maven tooling, on the other hand side, defines a rather <a href=
043 * "http://books.sonatype.com/mvnref-book/reference/pom-relationships-sect-pom-syntax.html#pom-reationships-sect-versions">restrictive
044 * version number pattern</a> for maven projects. Non-compliant version numbers
045 * are parsed as qualifier-only versions.
046 * <P>
047 * The parsing methods of this class make an attempt to interpret a version
048 * number as a
049 * &lt;major&gt;/&lt;minor&gt;/&lt;micro/incremental&gt;/&lt;qualifier&gt;
050 * pattern, even if it does not match the restrictive maven project version
051 * number pattern. The string representation of an instance of this class is
052 * always the original, unparsed (or "literal") representation because due to
053 * the permissive parsing algorithm used, the original representation cannot
054 * faithfully be reconstructed from the parsed components.
055 */
056public class MavenVersion implements ArtifactVersion {
057
058    static Pattern fuzzyVersion = Pattern
059        .compile("(\\d+)(\\.(\\d+)(\\.(\\d+))?)?([^a-zA-Z0-9](.*))?",
060            Pattern.DOTALL);
061    static Pattern fuzzyVersionRange = Pattern
062        .compile(
063            "(\\(|\\[)\\s*([-\\da-zA-Z.]+)\\s*,\\s*([-\\da-zA-Z.]+)\\s*(\\]|\\))",
064            Pattern.DOTALL);
065    static Pattern fuzzyModifier
066        = Pattern.compile("(\\d+[.-])*(.*)", Pattern.DOTALL);
067    public static final String VERSION_STRING
068        = "(\\d{1,15})(\\.(\\d{1,9})(\\.(\\d{1,9}))?)?([-\\.]?([-_\\.\\da-zA-Z]+))?";
069    static final SimpleDateFormat snapshotTimestamp
070        = new SimpleDateFormat("yyyyMMdd.HHmmss", Locale.ROOT);
071    public static final Pattern VERSIONRANGE = Pattern.compile("((\\(|\\[)"
072
073        + VERSION_STRING + "," + VERSION_STRING + "(\\]|\\)))|"
074        + VERSION_STRING);
075
076    static {
077        snapshotTimestamp.setTimeZone(TimeZone.getTimeZone("UTC"));
078    }

080    private static final Pattern VERSION = Pattern.compile(VERSION_STRING);

088    // Used as "container" for the components of the maven version number
089    private final Version version;
090    // Some maven versions are too odd to be restored after parsing, keep
091    // original.
092    private final String literal;
093    // Used for comparison, cached for efficiency.
094    private final ComparableVersion comparable;

098    /**
099     * Creates a new maven version from an osgi version. The version components
100     * are copied, the string representation is built from the components as
101     * "&lt;major&gt;.&lt;minor&gt;.&lt;micro&gt;.&lt;qualifier&gt;"
102     *
103     * @param osgiVersion the osgi version
104     */
105    public MavenVersion(Version osgiVersion) {
106        this.version = osgiVersion;
107        String qual = "";
108        if (this.version.getQualifier() != null) {
109            qual += "-" + this.version.getQualifier();
110        }
111        this.literal = osgiVersion.getWithoutQualifier().toString() + qual;
112        this.comparable = new ComparableVersion(literal);
113        this.snapshot = osgiVersion.isSnapshot();
114    }
115
116    /**
117     * Instantiates a new maven version representing the information from the
118     * argument. In addition to allowing the formats supported by
119     * {@link #parseString(String)}, this constructor supports formats such as
120     * "1.2rc1", i.e. without a separator before the qualifier.
121     *
122     * @param maven the version
123     */
124    public MavenVersion(String maven) {
125        this.version = new Version(cleanupVersion(maven));
126        this.literal = maven;
127        this.comparable = new ComparableVersion(literal);
128        this.snapshot = maven.endsWith("-" + SNAPSHOT);
129    }

131    /**
132     * Parses the string as a maven version, but allows a dot as separator
133     * before the qualifier.
134     * <P>
135     * Leading sequences of digits followed by a dot or dash are converted to
136     * the major, minor and incremental version components. A dash or a dot that
137     * is not followed by a digit or the third dot is interpreted as the start
138     * of the qualifier.
139     * <P>
140     * In particular, version numbers such as "1.2.3.4.5" are parsed as major=1,
141     * minor=2, incremental=3 and qualifier="4.5". This is closer to the
142     * (assumed) semantics of such a version number than the parsing implemented
143     * in maven tooling, which interprets the complete version as a qualifier in
144     * such cases.
145     *
146     * @param versionStr the version string
147     * @return the maven version
148     * @throws IllegalArgumentException if the version cannot be parsed
149     */
150    public static final MavenVersion parseString(String versionStr) {
151        if (versionStr == null) {
152            versionStr = "0";
153        } else {
154            versionStr = versionStr.trim();
155            if (versionStr.isEmpty()) {
156                versionStr = "0";
157            }
158        }
159        Matcher matcher = VERSION.matcher(versionStr);
160        if (!matcher.matches()) {
161            throw new IllegalArgumentException(
162                "Invalid syntax for version: " + versionStr);
163        }
164        int major = Integer.parseInt(matcher.group(1));
165        int minor
166            = (matcher.group(3) != null) ? Integer.parseInt(matcher.group(3))
167                : 0;
168        int micro
169            = (matcher.group(5) != null) ? Integer.parseInt(matcher.group(5))
170                : 0;
171        String qualifier = matcher.group(7);
172        Version version = new Version(major, minor, micro, qualifier);
173        return new MavenVersion(version);
174    }

176    /**
177     * Similar to {@link #parseString(String)}, but returns {@code null} if the
178     * version cannot be parsed.
179     * 
180     * @param versionStr the version string
181     * @return the maven version
182     */
183    public static final MavenVersion parseMavenString(String versionStr) {
184        try {
185            return new MavenVersion(versionStr);
186        } catch (Exception e) {
187            return null;
188        }
189    }

254     * Compares maven version numbers according to the rules defined in the

256     * "https://maven.apache.org/pom.html#Version_Order_Specification">POM
257     * reference</a>.

259    @Override
260    public int compareTo(ArtifactVersion other) {
261        if (other instanceof MavenVersion) {
262            return comparable.compareTo(((MavenVersion) other).comparable);
263        }
264        return comparable.compareTo(new ComparableVersion(other.toString()));
265    }

320    public static String validate(String value) {
321        if (value == null) {
322            return "Version is null";
323        }
324        if (!VERSION.matcher(value).matches()) {
325            return "Not a version";
326        }
327        return null;
328    }

330    public static String toDateStamp(long epoch) {
331        String datestamp;
332        synchronized (snapshotTimestamp) {
333            datestamp = snapshotTimestamp.format(new Date(epoch));
334        }
335        return datestamp;
336
337    }

340        String str = toDateStamp(epoch);
341        if (build != null) {

343        }

347    public static String cleanupVersion(String version) {
348        if (version == null || version.trim().isEmpty()) {
349            return "0";
350        }
351
352        Matcher matcher = VERSIONRANGE.matcher(version);
353
354        if (matcher.matches()) {
355            try {
356                // Test if version ramge can be created without problems.
357                new VersionRange(version);
358                return version;
359            } catch (Exception e) {
360                // ignore
361            }
362        }
363
364        matcher = fuzzyVersionRange.matcher(version);
365        if (matcher.matches()) {
366            String prefix = matcher.group(1);
367            String first = matcher.group(2);
368            String last = matcher.group(3);
369            String suffix = matcher.group(4);
370            return prefix + cleanupVersion(first) + "," + cleanupVersion(last)
371                + suffix;
372        }
373
374        matcher = fuzzyVersion.matcher(version);
375        if (matcher.matches()) {
376            StringBuilder result = new StringBuilder();
377            String major = removeLeadingZeroes(matcher.group(1));
378            String minor = removeLeadingZeroes(matcher.group(3));
379            String micro = removeLeadingZeroes(matcher.group(5));
380            String qualifier = matcher.group(7);
381
382            if (qualifier == null) {
383                if (!isInteger(minor)) {
384                    qualifier = minor;
385                    minor = "0";
386                } else if (!isInteger(micro)) {
387                    qualifier = micro;
388                    micro = "0";
389                }
390            }
391            if (major != null) {
392                result.append(major);
393                if (minor != null) {
394                    result.append(".");
395                    result.append(minor);
396                    if (micro != null) {
397                        result.append(".");
398                        result.append(micro);
399                        if (qualifier != null) {
400                            result.append(".");
401                            cleanupModifier(result, qualifier);
402                        }
403                    } else if (qualifier != null) {
404                        result.append(".0.");
405                        cleanupModifier(result, qualifier);
406                    }
407                } else if (qualifier != null) {
408                    result.append(".0.0.");
409                    cleanupModifier(result, qualifier);
410                }
411                return result.toString();
412            }
413        }
414        StringBuilder sb = new StringBuilder();
415        sb.append("0.0.0.");
416        cleanupModifier(sb, version);
417
418        return sb.toString();
419    }

421    /**
422     * TRhe cleanup version got confused when people used numeric dates like
423     * 201209091230120 as qualifiers. These are too large for Integers. This
424     * method checks if the all digit string fits in an integer.
425     *
426     * <pre>
427     *  maxint =
428     * 2,147,483,647 = 10 digits
429     * </pre>
430     *
431     * @param integer
432     * @return if this fits in an integer
433     */
434    private static boolean isInteger(String minor) {
435        return minor.length() < 10
436            || (minor.length() == 10 && minor.compareTo("2147483647") <= 0);
437    }
438
439    private static String removeLeadingZeroes(String group) {
440        if (group == null) {
441            return "0";
442        }
443        int count = 0;
444        while (count < group.length() - 1 && group.charAt(count) == '0') {
445            count++;
446        }
447        if (count == 0) {
448            return group;
449        }
450        return group.substring(count);
451    }
452
453    static void cleanupModifier(StringBuilder result, String modifier) {
454        int len = result.length();
455        Matcher matcher = fuzzyModifier.matcher(modifier);
456        if (matcher.matches()) {
457            modifier = matcher.group(2);
458        }
459        for (int i = 0; i < modifier.length(); i++) {
460            char chr = modifier.charAt(i);
461            if ((chr >= '0' && chr <= '9') || (chr >= 'a' && chr <= 'z')
462                || (chr >= 'A' && chr <= 'Z') || chr == '_' || chr == '-') {
463                result.append(chr);
464            }
465        }
466        if (len == result.length()) {
467            result.append("_");
468        }
469    }
