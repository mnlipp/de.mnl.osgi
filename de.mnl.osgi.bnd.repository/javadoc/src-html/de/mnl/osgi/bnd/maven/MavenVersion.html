035/**
036 * Provides a model of an artifact version which can be used as a maven version.
037 * <P>
038 * The maven <a href="https://maven.apache.org/pom.html">POM reference</a> does
039 * not define a format for versions. This is presumably intentional as it allows
040 * artifacts with arbitrary versioning schemes to be referenced in a POM.
041 * <P>
042 * Maven tooling, on the other hand side, defines a rather <a href=
043 * "http://books.sonatype.com/mvnref-book/reference/pom-relationships-sect-pom-syntax.html#pom-reationships-sect-versions">restrictive
044 * version number pattern</a> for maven projects. Non-compliant version numbers
045 * are parsed as qualifier-only versions.
046 * <P>
047 * The parsing methods of this class make an attempt to interpret a version
048 * number as a
049 * &lt;major&gt;/&lt;minor&gt;/&lt;micro/incremental&gt;/&lt;qualifier&gt;
050 * pattern, even if it does not match the restrictive maven project version
051 * number pattern. The string representation of an instance of this class is
052 * always the original, unparsed (or "literal") representation because due to
053 * the permissive parsing algorithm used, the original representation cannot
054 * faithfully be reconstructed from the parsed components.
055 */
056public class MavenVersion implements ArtifactVersion {
057
058        static Pattern                                  fuzzyVersion            = Pattern
059                        .compile("(\\d+)(\\.(\\d+)(\\.(\\d+))?)?([^a-zA-Z0-9](.*))?", Pattern.DOTALL);
060        static Pattern                                  fuzzyVersionRange       = Pattern
061                        .compile("(\\(|\\[)\\s*([-\\da-zA-Z.]+)\\s*,\\s*([-\\da-zA-Z.]+)\\s*(\\]|\\))", Pattern.DOTALL);
062        static Pattern                                  fuzzyModifier           = Pattern.compile("(\\d+[.-])*(.*)", Pattern.DOTALL);
063        public static final String              VERSION_STRING          = "(\\d{1,15})(\\.(\\d{1,9})(\\.(\\d{1,9}))?)?([-\\.]?([-_\\.\\da-zA-Z]+))?";
064        final static SimpleDateFormat   snapshotTimestamp       = new SimpleDateFormat("yyyyMMdd.HHmmss", Locale.ROOT);
065        public final static Pattern             VERSIONRANGE            = Pattern.compile("((\\(|\\[)"
066
067                        + VERSION_STRING + "," + VERSION_STRING + "(\\]|\\)))|" + VERSION_STRING);
068
069        static {
070                snapshotTimestamp.setTimeZone(TimeZone.getTimeZone("UTC"));
071        }

073        private static final Pattern            VERSION         = Pattern.compile(VERSION_STRING);
074        public static MavenVersion                      UNRESOLVED      = new MavenVersion("0-UNRESOLVED");

076        static final String                                     SNAPSHOT        = "SNAPSHOT";
077        public static final MavenVersion        HIGHEST         = new MavenVersion(Version.HIGHEST);
078        public static final MavenVersion        LOWEST          = new MavenVersion("0");

080        // Used as "container" for the components of the maven version number
081        private final Version                           version;
082        // Some maven versions are too odd to be restored after parsing, keep
083        // original.
084        private final String                            literal;
085        // Used for comparison, cached for efficiency.
086        private final ComparableVersion         comparable;
087
088        private final boolean                           snapshot;
089
090        /**
091         * Creates a new maven version from an osgi version. The version components
092         * are copied, the string representation is built from the components as
093         * "&lt;major&gt;.&lt;minor&gt;.&lt;micro&gt;.&lt;qualifier&gt;"
094         *
095         * @param osgiVersion the osgi version
096         */
097        public MavenVersion(Version osgiVersion) {
098                this.version = osgiVersion;
099                String qual = "";
100                if (this.version.getQualifier() != null)
101                        qual += "-" + this.version.getQualifier();
102
103                this.literal = osgiVersion.getWithoutQualifier().toString() + qual;
104                this.comparable = new ComparableVersion(literal);
105                this.snapshot = osgiVersion.isSnapshot();
106        }
107
108        /**
109         * Instantiates a new maven version representing the information from the
110         * argument. In addition to allowing the formats supported by
111         * {@link #parseString(String)}, this constructor supports formats such as
112         * "1.2rc1", i.e. without a separator before the qualifier.
113         *
114         * @param maven the version
115         */
116        public MavenVersion(String maven) {
117                this.version = new Version(cleanupVersion(maven));
118                this.literal = maven;
119                this.comparable = new ComparableVersion(literal);
120                this.snapshot = maven.endsWith("-" + SNAPSHOT);
121        }
122
123        /**
124         * Parses the string as a maven version, but allows a dot as separator
125         * before the qualifier.
126         * <P>
127         * Leading sequences of digits followed by a dot or dash are converted to
128         * the major, minor and incremental version components. A dash or a dot that
129         * is not followed by a digit or the third dot is interpreted as the start
130         * of the qualifier.
131         * <P>
132         * In particular, version numbers such as "1.2.3.4.5" are parsed as major=1,
133         * minor=2, incremental=3 and qualifier="4.5". This is closer to the
134         * (assumed) semantics of such a version number than the parsing implemented
135         * in maven tooling, which interprets the complete version as a qualifier in
136         * such cases.
137         *
138         * @param versionStr the version string
139         * @return the maven version
140         * @throws IllegalArgumentException if the version cannot be parsed
141         */
142        public static final MavenVersion parseString(String versionStr) {
143                if (versionStr == null) {
144                        versionStr = "0";
145                } else {
146                        versionStr = versionStr.trim();
147                        if (versionStr.isEmpty()) {
148                                versionStr = "0";
149                        }
150                }
151                Matcher m = VERSION.matcher(versionStr);
152                if (!m.matches())
153                        throw new IllegalArgumentException("Invalid syntax for version: " + versionStr);
154
155                int major = Integer.parseInt(m.group(1));
156                int minor = (m.group(3) != null) ? Integer.parseInt(m.group(3)) : 0;
157                int micro = (m.group(5) != null) ? Integer.parseInt(m.group(5)) : 0;
158                String qualifier = m.group(7);
159                Version version = new Version(major, minor, micro, qualifier);
160                return new MavenVersion(version);
161        }
162
163        /**
164         * Similar to {@link #parseString(String)}, but returns {@code null} if the
165         * version cannot be parsed.
166         * 
167         * @param versionStr the version string
168         * @return the maven version
169         */
170        public static final MavenVersion parseMavenString(String versionStr) {
171                try {
172                        return new MavenVersion(versionStr);
173                } catch (Exception e) {
174                        return null;
175                }
176        }

238         * "https://maven.apache.org/pom.html#Version_Order_Specification">POM
239         * reference</a>.

241        @Override
242        public int compareTo(ArtifactVersion other) {
243                if (other instanceof MavenVersion) {
244                        return comparable.compareTo(((MavenVersion) other).comparable);
245                }
246                return comparable.compareTo(new ComparableVersion(other.toString()));
247        }

262        @Override
263        public boolean equals(Object obj) {
264                if (this == obj)
265                        return true;
266                if (obj == null)
267                        return false;
268                if (getClass() != obj.getClass())
269                        return false;
270                MavenVersion other = (MavenVersion) obj;

279        public static String validate(String v) {
280                if (v == null)
281                        return "Version is null";
282
283                if (!VERSION.matcher(v).matches())
284                        return "Not a version";
285
286                return null;
287        }

289        public static String toDateStamp(long epoch) {
290                String datestamp;
291                synchronized (snapshotTimestamp) {
292                        datestamp = snapshotTimestamp.format(new Date(epoch));
293                }
294                return datestamp;
295
296        }

316        public MavenVersion toSnapshot(String dateStamp) {
317                // -SNAPSHOT == 9 characters
318                String clean = literal.substring(0, literal.length() - 9);
319                String result = clean + "-" + dateStamp;
320
321                return new MavenVersion(result);

324        static public String cleanupVersion(String version) {
325
326                if (version == null || version.trim().isEmpty())
327                        return "0";
328
329                Matcher m = VERSIONRANGE.matcher(version);
330
331                if (m.matches()) {
332                        try {
333                                // Test if version ramge can be created without problems.
334                                new VersionRange(version);
335                                return version;
336                        } catch (Exception e) {
337                                // ignore
338                        }
339                }
340
341                m = fuzzyVersionRange.matcher(version);
342                if (m.matches()) {
343                        String prefix = m.group(1);
344                        String first = m.group(2);
345                        String last = m.group(3);
346                        String suffix = m.group(4);
347                        return prefix + cleanupVersion(first) + "," + cleanupVersion(last) + suffix;
348                }
349
350                m = fuzzyVersion.matcher(version);
351                if (m.matches()) {
352                        StringBuilder result = new StringBuilder();
353                        String major = removeLeadingZeroes(m.group(1));
354                        String minor = removeLeadingZeroes(m.group(3));
355                        String micro = removeLeadingZeroes(m.group(5));
356                        String qualifier = m.group(7);
357
358                        if (qualifier == null) {
359                                if (!isInteger(minor)) {
360                                        qualifier = minor;
361                                        minor = "0";
362                                } else if (!isInteger(micro)) {
363                                        qualifier = micro;
364                                        micro = "0";
365                                }
366                        }
367                        if (major != null) {
368                                result.append(major);
369                                if (minor != null) {
370                                        result.append(".");
371                                        result.append(minor);
372                                        if (micro != null) {
373                                                result.append(".");
374                                                result.append(micro);
375                                                if (qualifier != null) {
376                                                        result.append(".");
377                                                        cleanupModifier(result, qualifier);
378                                                }
379                                        } else if (qualifier != null) {
380                                                result.append(".0.");
381                                                cleanupModifier(result, qualifier);
382                                        }
383                                } else if (qualifier != null) {
384                                        result.append(".0.0.");
385                                        cleanupModifier(result, qualifier);
386                                }
387                                return result.toString();
388                        }
389                }
390                StringBuilder sb = new StringBuilder();
391                sb.append("0.0.0.");
392                cleanupModifier(sb, version);
393
394                return sb.toString();
395        }

397        /**
398         * TRhe cleanup version got confused when people used numeric dates like
399         * 201209091230120 as qualifiers. These are too large for Integers. This
400         * method checks if the all digit string fits in an integer.
401         *
402         * <pre>
403         *  maxint =
404         * 2,147,483,647 = 10 digits
405         * </pre>
406         *
407         * @param integer
408         * @return if this fits in an integer
409         */
410        private static boolean isInteger(String minor) {
411                return minor.length() < 10 || (minor.length() == 10 && minor.compareTo("2147483647") <= 0);
412        }
413
414        private static String removeLeadingZeroes(String group) {
415                if (group == null)
416                        return "0";
417
418                int n = 0;
419                while (n < group.length() - 1 && group.charAt(n) == '0')
420                        n++;
421                if (n == 0)
422                        return group;
423
424                return group.substring(n);
425        }
426
427        static void cleanupModifier(StringBuilder result, String modifier) {
428                int l = result.length();
429                Matcher m = fuzzyModifier.matcher(modifier);
430                if (m.matches())
431                        modifier = m.group(2);
432
433                for (int i = 0; i < modifier.length(); i++) {
434                        char c = modifier.charAt(i);
435                        if ((c >= '0' && c <= '9') || (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == '_' || c == '-')
436                                result.append(c);
437                }
438                if (l == result.length())
439                        result.append("_");
440        }
