019package de.mnl.osgi.bnd.maven;
020
021import java.lang.reflect.UndeclaredThrowableException;
022import java.util.concurrent.Callable;
023import java.util.regex.Pattern;
024import java.util.stream.Stream;
025
026// TODO: Auto-generated Javadoc
027/**
028 * The Class Utils.
029 */
030public final class RepositoryUtils {
031
032    /** The Constant LIST_ITEM_SEPARATOR. */
033    public static final Pattern LIST_ITEM_SEPARATOR
034        = Pattern.compile("\\s*,\\s*");
035
036    private RepositoryUtils() {
037    }
038
039    /**
040     * Itemize list.
041     *
042     * @param list the list
043     * @return the stream
044     */
045    public static Stream<String> itemizeList(String list) {
046        if (list == null) {
047            return Stream.empty();
048        }
049        return LIST_ITEM_SEPARATOR.splitAsStream(list);
050    }
051
052    /**
053     * A runnable that may throw an exception.
054     */
055    @FunctionalInterface
056    public interface ThrowingRunnable {
057
058        /**
059         * The operation to run.
060         *
061         * @throws Exception the exception
062         */
063        @SuppressWarnings("PMD.SignatureDeclareThrowsException")
064        void run() throws Exception;
065    }
066
067    /**
068     * Converts any exception thrown by the supplier to an
069     * {@link UndeclaredThrowableException}.
070     *
071     * @param <T> the return value
072     * @param supplier the supplier
073     * @return the t
074     */
075    @SuppressWarnings("PMD.AvoidCatchingThrowable")
076    public static <T> T unthrow(Callable<T> supplier) {
077        try {
078            return supplier.call();
079        } catch (Throwable t) {
080            throw new UndeclaredThrowableException(t);
081        }
082    }
083
084    /**
085     * Converts any exception thrown by the runnable to an
086     * {@link UndeclaredThrowableException}.
087     *
088     * @param runnable the runnable
089     */
090    @SuppressWarnings("PMD.AvoidCatchingThrowable")
091    public static void unthrow(ThrowingRunnable runnable) {
092        try {
093            runnable.run();
094        } catch (Throwable t) {
095            throw new UndeclaredThrowableException(t);
096        }
097    }
098
099    /**
100     * Catches {@link UndeclaredThrowableException}s and unwraps
101     * any underlying exception of the given type.
102     *
103     * @param <T> the return type
104     * @param <E> the type of exception that is unwrapped
105     * @param rethrown the type of exception that is rethrown
106     * @param supplier the supplier
107     * @return the result from invoking the {@code supplier}
108     * @throws E the exception type
109     */
110    @SuppressWarnings({ "unchecked", "PMD.AvoidCatchingGenericException" })
111    public static <T, E extends Throwable> T rethrow(Class<E> rethrown,
112            Callable<T> supplier) throws E {
113        try {
114            return supplier.call();
115        } catch (UndeclaredThrowableException e) {
116            if (rethrown
117                .isAssignableFrom(e.getUndeclaredThrowable().getClass())) {
118                throw (E) e.getUndeclaredThrowable();
119            }
120            throw e;
121        } catch (Exception e) {
122            throw new UndeclaredThrowableException(e);
123        }
124    }
125
126    /**
127     * Catches {@link UndeclaredThrowableException}s and unwraps
128     * any underlying exception of the given type.
129     *
130     * @param <E> the type of exception that is unwrapped
131     * @param rethrown the type of exception that is rethrown
132     * @param runnable the runnable
133     * @throws E the exception type
134     */
135    @SuppressWarnings({ "unchecked", "PMD.AvoidCatchingGenericException" })
136    public static <E extends Throwable> void rethrow(Class<E> rethrown,
137            ThrowingRunnable runnable) throws E {
138        try {
139            runnable.run();
140        } catch (UndeclaredThrowableException e) {
141            if (rethrown
142                .isAssignableFrom(e.getUndeclaredThrowable().getClass())) {
143                throw (E) e.getUndeclaredThrowable();
144            }
145            throw e;
146        } catch (Exception e) {
147            throw new UndeclaredThrowableException(e);
148        }
149    }
150}
