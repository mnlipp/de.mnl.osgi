019package de.mnl.osgi.bnd.maven;
020
021import java.lang.reflect.UndeclaredThrowableException;
022import java.util.concurrent.Callable;
023import java.util.regex.Pattern;
024import java.util.stream.Stream;
025
026/**
027 * The Class Utils.
028 */
029public final class RepositoryUtils {
030
031    public static final Pattern LIST_ITEM_SEPARATOR
032        = Pattern.compile("\\s*,\\s*");
033
034    private RepositoryUtils() {
035    }
036
037    /**
038     * Itemize list.
039     *
040     * @param list the list
041     * @return the stream
042     */
043    public static Stream<String> itemizeList(String list) {
044        if (list == null) {
045            return Stream.empty();
046        }
047        return LIST_ITEM_SEPARATOR.splitAsStream(list);
048    }
049
050    /**
051     * Converts any exception thrown by the supplier to an
052     * {@link UndeclaredThrowableException}.
053     *
054     * @param <T> the return value
055     * @param supplier the supplier
056     * @return the t
057     */
058    @SuppressWarnings("PMD.AvoidCatchingThrowable")
059    public static <T> T unthrow(Callable<T> supplier) {
060        try {
061            return supplier.call();
062        } catch (Throwable t) {
063            throw new UndeclaredThrowableException(t);
064        }
065    }
066
067    /**
068     * Catches {@link UndeclaredThrowableException}s and unwraps
069     * any underlying exception of the given type.
070     *
071     * @param <T> the return type
072     * @param <E> the type of exception that is unwrapped
073     * @param rethrown the type of exception that is rethrown
074     * @param supplier the supplier
075     * @return the result from invoking the {@code supplier}
076     * @throws E the exception type
077     */
078    @SuppressWarnings({ "unchecked", "PMD.AvoidCatchingGenericException" })
079    public static <T, E extends Throwable> T rethrow(Class<E> rethrown,
080            Callable<T> supplier) throws E {
081        try {
082            return supplier.call();
083        } catch (UndeclaredThrowableException e) {
084            if (rethrown
085                .isAssignableFrom(e.getUndeclaredThrowable().getClass())) {
086                throw (E) e.getUndeclaredThrowable();
087            }
088            throw e;
089        } catch (Exception e) {
090            throw new UndeclaredThrowableException(e);
091        }
092    }
093}
