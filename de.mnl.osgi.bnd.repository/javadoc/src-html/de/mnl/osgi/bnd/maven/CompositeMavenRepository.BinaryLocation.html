049/**
050 * Provides a common {@link IMavenRepo} view on several 
051 * {@link MavenBackingRepository} instances.
052 * <P>
053 * The class extends {@link MavenRepository} which lacks some
054 * required functionality.
055 */
056@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
057public class CompositeMavenRepository extends MavenRepository
058        implements IMavenRepo, Closeable {
059
060    private static final Logger LOG = LoggerFactory.getLogger(
061        CompositeMavenRepository.class);
062
063    /**
064     * Instantiates a new merged maven repository.
065     *
066     * @param base the base
067     * @param repoId the repository id
068     * @param releaseRepos the backing release repositories
069     * @param snapshotRepos the backing snapshot repositories
070     * @param executor an executor
071     * @param reporter the reporter
072     * @throws Exception the exception
073     */
074    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
075        "PMD.AvoidDuplicateLiterals" })
076    public CompositeMavenRepository(File base, String repoId,
077            List<MavenBackingRepository> releaseRepos,
078            List<MavenBackingRepository> snapshotRepos, Executor executor,
079            Reporter reporter)
080            throws Exception {
081        super(base, repoId, releaseRepos, snapshotRepos, executor, reporter);
082    }
083
084    /**
085     * Returns all repositories.
086     *
087     * @return the list
088     */
089    public List<MavenBackingRepository> allRepositories() {
090        List<MavenBackingRepository> result
091            = new ArrayList<>(getReleaseRepositories());
092        result.addAll(getSnapshotRepositories());
093        return result;
094    }
095
096    /**
097     * Returns all known revisions as {@link BoundRevision}s.
098     *
099     * @param program the program
100     * @return the bound revisions
101     * @throws Exception the exception
102     */
103    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
104    public List<BoundRevision> boundRevisions(Program program)
105            throws Exception {
106        List<BoundRevision> revisions = new ArrayList<>();
107
108        for (MavenBackingRepository mbr : getReleaseRepositories()) {
109            addRevisions(mbr, program, revisions);
110        }
111        for (MavenBackingRepository mbr : getSnapshotRepositories()) {
112            if (!getReleaseRepositories().contains(mbr)) {
113                addRevisions(mbr, program, revisions);
114            }
115        }
116        return revisions;
117    }
118
119    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
120        "PMD.AvoidInstantiatingObjectsInLoops" })
121    private void addRevisions(MavenBackingRepository mbr, Program program,
122            List<BoundRevision> revisions) throws Exception {
123        List<Revision> revs = new ArrayList<>();
124        mbr.getRevisions(program, revs);
125        for (Revision rev : revs) {
126            revisions.add(new BoundRevision(mbr, rev));
127        }
128    }
129
130    /**
131     * Converts a {@link Revision} to an {@link BoundRevision}.
132     *
133     * @param revision the revision
134     * @return the bound revision
135     * @throws IOException Signals that an I/O exception has occurred.
136     */
137    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
138    public Optional<BoundRevision> toBoundRevision(Revision revision)
139            throws IOException {
140        for (MavenBackingRepository mbr : getReleaseRepositories()) {
141            Optional<BoundRevision> checkResult
142                = toBoundRevision(mbr, revision);
143            if (checkResult.isPresent()) {
144                return checkResult;
145            }
146        }
147        for (MavenBackingRepository mbr : getSnapshotRepositories()) {
148            if (!getReleaseRepositories().contains(mbr)) {
149                Optional<BoundRevision> checkResult
150                    = toBoundRevision(mbr, revision);
151                if (checkResult.isPresent()) {
152                    return checkResult;
153                }
154            }
155        }
156        return Optional.empty();
157    }
158
159    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
160        "PMD.AvoidRethrowingException", "PMD.AvoidThrowingRawExceptionTypes",
161        "PMD.AvoidCatchingGenericException", "PMD.AvoidDuplicateLiterals" })
162    private Optional<BoundRevision> toBoundRevision(MavenBackingRepository mbr,
163            Revision revision) throws IOException {
164        List<Revision> revs = new ArrayList<>();
165        try {
166            mbr.getRevisions(revision.program, revs);
167        } catch (IOException e) {
168            // Should be the only reason.
169            throw e;
170        } catch (Exception e) {
171            throw new RuntimeException(e);
172        }
173        if (revs.contains(revision)) {
174            return Optional.of(new BoundRevision(mbr, revision));
175        }
176        return Optional.empty();
177    }
178
179    @Override
180    public BoundArchive getResolvedArchive(Revision revision, String extension,
181            String classifier) throws IOException {
182        Optional<BoundRevision> bound = toBoundRevision(revision);
183        if (bound.isPresent()) {
184            return getResolvedArchive(bound.get(), extension, classifier);
185        }
186        return null;
187    }
188
189    /**
190     * Gets the resolved archive. "Resolving" an archive means finding
191     * the binaries with the specified extension and classifier belonging
192     * to the given version. While this can be done with straight forward
193     * name mapping for releases, snapshots have a timestamp that has to
194     * be looked up in the backing repository.
195     *
196     * @param revision the revision
197     * @param extension the extension
198     * @param classifier the classifier
199     * @return the resolved archive
200     * @throws IOException Signals that an I/O exception has occurred.
201     */
202    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
203        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidRethrowingException" })
204    public BoundArchive getResolvedArchive(BoundRevision revision,
205            String extension, String classifier) throws IOException {
206        if (!revision.isSnapshot()) {
207            return revision.archive(extension, classifier);
208        }
209        try {
210            MavenVersion version = revision.mavenBackingRepository()
211                .getVersion(revision.unbound());
212            if (version != null) {
213                return revision.archive(version, extension, classifier);
214            }
215        } catch (IOException e) {
216            throw e;
217        } catch (Exception e) {
218            throw new RuntimeException(e);
219        }
220        return null;
221    }
222
223    /**
224     * Converts the manifest information of a revision to a
225     * {@link Resource}. Any dependencies found in the revisions's
226     * POM are reported using the specified {@code dependencyHandler}.  
227     *
228     * @param revision the revision
229     * @param dependencyHandler the dependency handler
230     * @param location which URL to use for the binary in the {@link Resource}
231     * @return the resource
232     */
233    @SuppressWarnings("PMD.AvoidCatchingGenericException")
234    public Optional<Resource> toResource(BoundRevision revision,
235            Consumer<Collection<IPom.Dependency>> dependencyHandler,
236            BinaryLocation location) {
237        BoundArchive archive;
238        try {
239            archive = getResolvedArchive(revision, "jar", "");
240            if (archive == null) {
241                return Optional.empty();
242            }
243        } catch (Exception e) {
244            LOG.error("Problem resolving archive for {}.", revision, e);
245            return Optional.empty();
246        }
247        if (archive.isSnapshot()) {
248            refreshSnapshot(archive);
249        }
250        // Get POM for dependencies
251        try {
252            IPom pom = getPom(archive.getRevision());
253            if (pom != null) {
254                Collection<IPom.Dependency> deps = pom
255                    .getDependencies(MavenScope.compile, false).values();
256                if (!deps.isEmpty()) {
257                    dependencyHandler.accept(deps);
258                }
259                deps = pom.getDependencies(MavenScope.runtime, false).values();
260                if (!deps.isEmpty()) {
261                    dependencyHandler.accept(deps);
262                }
263            }
264        } catch (Exception e) {
265            LOG.error("Problem processing POM of {}.", revision, e);
266        }
267        return Optional.ofNullable(parseResource(archive, location));
268    }
269
270    @SuppressWarnings("PMD.AvoidCatchingGenericException")
271    private void refreshSnapshot(BoundArchive archive) {
272        File metaFile = toLocalFile(
273            archive.getRevision()
274                .metadata(archive.mavenBackingRepository().getId()));
275        RevisionMetadata metaData;
276        try {
277            metaData = MetadataParser.parseRevisionMetadata(metaFile);
278        } catch (Exception e) {
279            LOG.error("Problem accessing {}.", archive, e);
280            return;
281        }
282        File archiveFile = toLocalFile(archive);
283        if (archiveFile.lastModified() < metaData.lastUpdated) {
284            archiveFile.delete();
285        }
286        File pomFile = toLocalFile(archive.getPomArchive());
287        if (pomFile.lastModified() < metaData.lastUpdated) {
288            pomFile.delete();
289        }
290    }
291
292    @SuppressWarnings("PMD.AvoidCatchingGenericException")
293    private Resource parseResource(BoundArchive archive,
294            BinaryLocation location) {
295        ResourceBuilder builder = new ResourceBuilder();
296        try {
297            File binary = get(archive).getValue();
298            if (location == BinaryLocation.LOCAL) {
299                builder.addFile(binary, binary.toURI());
300            } else {
301                builder.addFile(binary, archive.mavenBackingRepository()
302                    .toURI(archive.remotePath));
303            }
304            addInformationCapability(builder, archive.toString(),
305                archive.getRevision().toString(), null);
306        } catch (Exception e) {
307            LOG.error("Problem accessing {}.", archive, e);
308            return null;
309        }
310        return builder.build();
311    }
312
313    /**
314     * Use local or remote URL in index.
315     */
316    public enum BinaryLocation {
317        LOCAL, REMOTE
318    }
