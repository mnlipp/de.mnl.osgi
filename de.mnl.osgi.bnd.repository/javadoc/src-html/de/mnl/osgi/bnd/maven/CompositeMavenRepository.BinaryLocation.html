056/**
057 * Provides a composite {@link IMavenRepo} view on several 
058 * {@link MavenBackingRepository} instances.
059 * <P>
060 * The class extends {@link MavenRepository} which lacks some
061 * required functionality. Besides, this class has a more
062 * appropriate name.
063 */
064@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
065public class CompositeMavenRepository extends MavenRepository
066        implements IMavenRepo, Closeable {
067
068    /** The namespace used to store the maven dependencies information. */
069    public static final String MAVEN_DEPENDENCIES_NS
070        = "maven.dependencies.info";
071    private final Reporter reporter;
072
073    /**
074     * Instantiates a new composite maven repository.
075     *
076     * @param base the base
077     * @param repoId the repository id
078     * @param releaseRepos the backing release repositories
079     * @param snapshotRepos the backing snapshot repositories
080     * @param executor an executor
081     * @param reporter the reporter
082     * @throws Exception the exception
083     */
084    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
085        "PMD.AvoidDuplicateLiterals" })
086    public CompositeMavenRepository(File base, String repoId,
087            List<MavenBackingRepository> releaseRepos,
088            List<MavenBackingRepository> snapshotRepos, Executor executor,
089            Reporter reporter)
090            throws Exception {
091        super(base, repoId, releaseRepos, snapshotRepos, executor, reporter);
092        this.reporter = reporter;
093    }

095    /**
096     * Returns all backing repositories.
097     *
098     * @return the list of all repositories
099     */
100    public List<MavenBackingRepository> allRepositories() {
101        List<MavenBackingRepository> result
102            = new ArrayList<>(getReleaseRepositories());
103        result.addAll(getSnapshotRepositories());
104        return result;
105    }

107    /**
108     * Returns all repositories as a stream.
109     *
110     * @return the repositories as stream
111     */
112    public Stream<MavenBackingRepository> repositoriesAsStream() {
113        return Stream.concat(getReleaseRepositories().stream(),
114            getSnapshotRepositories().stream()).distinct();
115    }

117    /**
118     * Wrapper for {@link MavenBackingRepository#getRevisions(Program, List)}
119     * that returns the result instead of accumulating it and maps the
120     * (too general) exception.
121     *
122     * @param mbr the backing repository
123     * @param program the program
124     * @return the revisions
125     * @throws IOException Signals that an I/O exception has occurred.
126     */
127    @SuppressWarnings({ "PMD.LinguisticNaming", "PMD.AvoidRethrowingException",
128        "PMD.AvoidCatchingGenericException",
129        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidDuplicateLiterals" })
130    private static List<Revision> getRevisions(MavenBackingRepository mbr,
131            Program program) throws IOException {
132        List<Revision> result = new ArrayList<>();
133        try {
134            mbr.getRevisions(program, result);
135        } catch (IOException e) {
136            // Should be the only reason.
137            throw e;
138        } catch (Exception e) {
139            throw new RuntimeException(e);
140        }
141        return result;
142    }
143
144    /**
145     * Wrapper for {@link MavenBackingRepository#getRevisions(Program, List)}
146     * that returns the result instead of accumulating it, maps the
147     * (too general) exception and converts the {@link Revision}s to
148     * {@link BoundRevision}s.
149     *
150     * @param mbr the mbr
151     * @param program the program
152     * @return the stream
153     * @throws IOException Signals that an I/O exception has occurred.
154     */
155    private static Stream<BoundRevision> boundRevisions(
156            MavenBackingRepository mbr, Program program) throws IOException {
157        return getRevisions(mbr, program).stream()
158            .map(revision -> new BoundRevision(mbr, revision));
159    }
160
161    /**
162     * Returns all known revisions of the specified program 
163     * as {@link BoundRevision}s.
164     *
165     * @param program the program
166     * @return the bound revisions
167     * @throws IOException Signals that an I/O exception has occurred.
168     */
169    @SuppressWarnings("PMD.ExceptionAsFlowControl")
170    public Stream<BoundRevision> boundRevisions(Program program)
171            throws IOException {
172        return rethrow(IOException.class, () -> repositoriesAsStream()
173            .flatMap(mbr -> unthrow(() -> boundRevisions(mbr, program))));
174    }
175
176    /**
177     * Converts a {@link Revision} to an {@link BoundRevision}.
178     *
179     * @param revision the revision
180     * @return the bound revision
181     * @throws IOException Signals that an I/O exception has occurred.
182     */
183    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
184    public Optional<BoundRevision> toBoundRevision(Revision revision)
185            throws IOException {
186        return boundRevisions(revision.program)
187            .filter(rev -> rev.unbound().equals(revision)).findFirst();
188    }
189
190    /**
191     * Converts a {@link Program} and a version, which
192     * may be a range, to a {@link BoundRevision}.
193     *
194     * @param program the program
195     * @param version the version
196     * @return the bound revision
197     * @throws IOException Signals that an I/O exception has occurred.
198     */
199    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
200    public Optional<BoundRevision> toBoundRevision(Program program,
201            String version) throws IOException {
202        if (!MavenVersionRange.isRange(version)) {
203            return toBoundRevision(program.version(version));
204        }
205        MavenVersionRange range = new MavenVersionRange(version);
206        return boundRevisions(program)
207            .filter(rev -> range.includes(rev.version()))
208            .max(Comparator.naturalOrder());
209    }

211    /**
212     * Converts the manifest information of a revision to a
213     * {@link Resource}. Any dependencies found in the revisions's
214     * POM are reported using the specified {@code dependencyHandler}.  
215     *
216     * @param revision the revision
217     * @param dependencyHandler the dependency handler
218     * @param location which URL to use for the binary in the {@link Resource}
219     * @return the resource
220     */
221    @SuppressWarnings("PMD.AvoidCatchingGenericException")
222    public Optional<Resource> toResource(BoundRevision revision,
223            DependencyHandler dependencyHandler, BinaryLocation location) {
224        BoundArchive archive;
225        try {
226            archive = getResolvedArchive(revision, "jar", "");
227            if (archive == null) {
228                return Optional.empty();
229            }
230        } catch (Exception e) {
231            reporter.exception(e, "Problem resolving archive for %s.",
232                revision);
233            return Optional.empty();
234        }
235        if (archive.isSnapshot()) {
236            refreshSnapshot(archive);
237        }
238        return Optional
239            .ofNullable(parseResource(archive, location, dependencyHandler));
240    }

242    /*
243     * (non-Javadoc)
244     * 
245     * @see
246     * aQute.maven.provider.MavenRepository#getResolvedArchive(aQute.maven.api.
247     * Revision, java.lang.String, java.lang.String)
248     */
249    @Override
250    public BoundArchive getResolvedArchive(Revision revision, String extension,
251            String classifier) throws IOException {
252        Optional<BoundRevision> bound = toBoundRevision(revision);
253        if (bound.isPresent()) {
254            return getResolvedArchive(bound.get(), extension, classifier);
255        }
256        return null;
257    }

259    /**
260     * Gets the resolved archive. "Resolving" an archive means finding
261     * the binaries with the specified extension and classifier belonging
262     * to the given version. While this can be done with straight forward
263     * name mapping for releases, snapshots have a timestamp that has to
264     * be looked up in the backing repository.
265     *
266     * @param revision the revision
267     * @param extension the extension
268     * @param classifier the classifier
269     * @return the resolved archive
270     * @throws IOException Signals that an I/O exception has occurred.
271     */
272    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
273        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidRethrowingException" })
274    public BoundArchive getResolvedArchive(BoundRevision revision,
275            String extension, String classifier) throws IOException {
276        if (!revision.isSnapshot()) {
277            return revision.archive(extension, classifier);
278        }
279        try {
280            MavenVersion version = MavenVersion.from(revision
281                .mavenBackingRepository().getVersion(revision.unbound()));
282            if (version != null) {
283                return revision.archive(version, extension, classifier);
284            }
285        } catch (IOException e) {
286            throw e;
287        } catch (Exception e) {
288            throw new RuntimeException(e);
289        }
290        return null;
291    }

293    @SuppressWarnings("PMD.AvoidCatchingGenericException")
294    private void refreshSnapshot(BoundArchive archive) {
295        File metaFile = toLocalFile(
296            archive.getRevision()
297                .metadata(archive.mavenBackingRepository().getId()));
298        RevisionMetadata metaData;
299        try {
300            metaData = MetadataParser.parseRevisionMetadata(metaFile);
301        } catch (Exception e) {
302            reporter.exception(e, "Problem accessing %s.", archive);
303            return;
304        }
305        File archiveFile = toLocalFile(archive);
306        if (archiveFile.lastModified() < metaData.lastUpdated) {
307            archiveFile.delete();
308        }
309        File pomFile = toLocalFile(archive.getPomArchive());
310        if (pomFile.lastModified() < metaData.lastUpdated) {
311            pomFile.delete();
312        }
313    }

315    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
316        "PMD.UselessParentheses", "PMD.AvoidInstanceofChecksInCatchClause" })
317    private Resource parseResource(BoundArchive archive,
318            BinaryLocation location, DependencyHandler dependencyHandler) {
319        ResourceBuilder builder = new ResourceBuilder();
320        try {
321            File binary = get(archive).getValue();
322            if (location == BinaryLocation.LOCAL) {
323                builder.addFile(binary, binary.toURI());
324            } else {
325                builder.addFile(binary, archive.mavenBackingRepository()
326                    .toURI(archive.remotePath));
327            }
328            addInformationCapability(builder, archive.toString(),
329                archive.getRevision().toString(), null);
330            // Add dependency infos
331            // Get POM for dependencies
332            IPom pom = getPom(archive.getRevision());
333            if (pom != null) {
334                CapabilityBuilder cap = null;
335                Collection<IPom.Dependency> rtDeps = pom.getDependencies(
336                    MavenScope.runtime, false).values();
337                Collection<IPom.Dependency> cpDeps = pom
338                    .getDependencies(MavenScope.compile, false).values();
339                if (!cpDeps.isEmpty() || !rtDeps.isEmpty()) {
340                    cap = new CapabilityBuilder(MAVEN_DEPENDENCIES_NS);
341                    if (!cpDeps.isEmpty()) {
342                        dependencyHandler.handle(cpDeps);
343                        cap.addAttribute("compile", toVersionList(cpDeps));
344                    }
345                    if (!rtDeps.isEmpty()) {
346                        dependencyHandler.handle(rtDeps);
347                        cap.addAttribute("runtime", toVersionList(rtDeps));
348                    }
349                    builder.addCapability(cap);
350                }
351            }
352        } catch (Exception e) {
353            if (e instanceof InvocationTargetException
354                && ((InvocationTargetException) e)
355                    .getTargetException() instanceof FileNotFoundException) {
356                reporter.exception(e, "POM of %s not found: %s", archive,
357                    ((InvocationTargetException) e).getTargetException()
358                        .getMessage());
359            }
360            reporter.exception(e, "Problem processing POM of %s: %s", archive,
361                e.getMessage());
362            return null;
363        }
364        return builder.build();
365    }
366
367    private String toVersionList(Collection<IPom.Dependency> deps) {
368        StringBuilder depsList = new StringBuilder("");
369        for (IPom.Dependency dep : deps) {
370            if (depsList.length() > 0) {
371                depsList.append(',');
372            }
373            depsList.append(dep.program.toString());
374            depsList.append(':');
375            depsList.append(dep.version);
376        }
377        return depsList.toString();
378    }
