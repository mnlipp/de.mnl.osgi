019package de.mnl.osgi.bnd.maven;
020
021import aQute.maven.api.IMavenRepo;
022import aQute.maven.provider.MavenBackingRepository;
023import aQute.maven.provider.MavenRepository;
024import aQute.service.reporter.Reporter;
025
026import java.io.Closeable;
027import java.io.File;
028import java.util.ArrayList;
029import java.util.Collection;
030import java.util.List;
031import java.util.concurrent.Executor;
032
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036/**
037 * Provides a common {@link IMavenRepo} view on several 
038 * {@link MavenBackingRepository} instances.
039 * <P>
040 * The class extends {@link MavenRepository} which lacks some
041 * required functionality.
042 */
043public class MergingMavenRepository extends MavenRepository
044        implements IMavenRepo, Closeable {
045    private static final Logger LOG
046        = LoggerFactory.getLogger(MergingMavenRepository.class);
047
048    /**
049     * Instantiates a new merged maven repository.
050     *
051     * @param base the base
052     * @param repoId the repository id
053     * @param releaseRepos the backing release repositories
054     * @param snapshotRepos the backing snapshot repositories
055     * @param executor an executor
056     * @throws Exception the exception
057     */
058    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
059    public MergingMavenRepository(File base, String repoId,
060            List<MavenBackingRepository> releaseRepos,
061            List<MavenBackingRepository> snapshotRepos, Executor executor,
062            Reporter reporter)
063            throws Exception {
064        super(base, repoId, releaseRepos, snapshotRepos, executor, reporter);
065    }
066
067    public List<MavenBackingRepository> allRepositories() {
068        List<MavenBackingRepository> result
069            = new ArrayList<>(getReleaseRepositories());
070        result.addAll(getSnapshotRepositories());
071        return result;
072    }
073
074}
