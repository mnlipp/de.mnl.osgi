019package de.mnl.osgi.bnd.maven;
020
021import aQute.maven.api.IMavenRepo;
022import aQute.maven.api.Program;
023import aQute.maven.api.Revision;
024import aQute.maven.provider.MavenBackingRepository;
025import aQute.maven.provider.MavenRepository;
026import aQute.service.reporter.Reporter;
027
028import java.io.Closeable;
029import java.io.File;
030import java.io.IOException;
031import java.util.ArrayList;
032import java.util.List;
033import java.util.Optional;
034import java.util.concurrent.Executor;
035
036/**
037 * Provides a common {@link IMavenRepo} view on several 
038 * {@link MavenBackingRepository} instances.
039 * <P>
040 * The class extends {@link MavenRepository} which lacks some
041 * required functionality.
042 */
043public class CompositeMavenRepository extends MavenRepository
044        implements IMavenRepo, Closeable {
045
046    /**
047     * Instantiates a new merged maven repository.
048     *
049     * @param base the base
050     * @param repoId the repository id
051     * @param releaseRepos the backing release repositories
052     * @param snapshotRepos the backing snapshot repositories
053     * @param executor an executor
054     * @param reporter the reporter
055     * @throws Exception the exception
056     */
057    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
058        "PMD.AvoidDuplicateLiterals" })
059    public CompositeMavenRepository(File base, String repoId,
060            List<MavenBackingRepository> releaseRepos,
061            List<MavenBackingRepository> snapshotRepos, Executor executor,
062            Reporter reporter)
063            throws Exception {
064        super(base, repoId, releaseRepos, snapshotRepos, executor, reporter);
065    }
066
067    /**
068     * Returns all repositories.
069     *
070     * @return the list
071     */
072    public List<MavenBackingRepository> allRepositories() {
073        List<MavenBackingRepository> result
074            = new ArrayList<>(getReleaseRepositories());
075        result.addAll(getSnapshotRepositories());
076        return result;
077    }
078
079    /**
080     * Returns all known revisions as {@link BoundRevision}s.
081     *
082     * @param program the program
083     * @return the bound revisions
084     * @throws Exception the exception
085     */
086    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
087    public List<BoundRevision> boundRevisions(Program program)
088            throws Exception {
089        List<BoundRevision> revisions = new ArrayList<>();
090
091        for (MavenBackingRepository mbr : getReleaseRepositories()) {
092            addRevisions(mbr, program, revisions);
093        }
094        for (MavenBackingRepository mbr : getSnapshotRepositories()) {
095            if (!getReleaseRepositories().contains(mbr)) {
096                addRevisions(mbr, program, revisions);
097            }
098        }
099        return revisions;
100    }
101
102    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
103        "PMD.AvoidInstantiatingObjectsInLoops" })
104    private void addRevisions(MavenBackingRepository mbr, Program program,
105            List<BoundRevision> revisions) throws Exception {
106        List<Revision> revs = new ArrayList<>();
107        mbr.getRevisions(program, revs);
108        for (Revision rev : revs) {
109            revisions.add(new BoundRevision(mbr, rev));
110        }
111    }
112
113    /**
114     * Converts a {@link Revision} to an {@link BoundRevision}.
115     *
116     * @param revision the revision
117     * @return the bound revision
118     * @throws IOException Signals that an I/O exception has occurred.
119     */
120    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
121    public Optional<BoundRevision> toBoundRevision(Revision revision)
122            throws IOException {
123        for (MavenBackingRepository mbr : getReleaseRepositories()) {
124            Optional<BoundRevision> checkResult
125                = toBoundRevision(mbr, revision);
126            if (checkResult.isPresent()) {
127                return checkResult;
128            }
129        }
130        for (MavenBackingRepository mbr : getSnapshotRepositories()) {
131            if (!getReleaseRepositories().contains(mbr)) {
132                Optional<BoundRevision> checkResult
133                    = toBoundRevision(mbr, revision);
134                if (checkResult.isPresent()) {
135                    return checkResult;
136                }
137            }
138        }
139        return Optional.empty();
140    }
141
142    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
143        "PMD.AvoidRethrowingException", "PMD.AvoidCatchingGenericException",
144        "PMD.AvoidThrowingRawExceptionTypes" })
145    private Optional<BoundRevision> toBoundRevision(MavenBackingRepository mbr,
146            Revision revision) throws IOException {
147        List<Revision> revs = new ArrayList<>();
148        try {
149            mbr.getRevisions(revision.program, revs);
150        } catch (IOException e) {
151            // Should be the only reason.
152            throw e;
153        } catch (Exception e) {
154            throw new RuntimeException(e);
155        }
156        if (revs.contains(revision)) {
157            return Optional.of(new BoundRevision(mbr, revision));
158        }
159        return Optional.empty();
160    }
161
162}
