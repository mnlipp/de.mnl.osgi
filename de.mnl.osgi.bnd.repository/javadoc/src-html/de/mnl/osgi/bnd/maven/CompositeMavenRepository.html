019package de.mnl.osgi.bnd.maven;
020
021import aQute.bnd.version.MavenVersion;
022import aQute.maven.api.IMavenRepo;
023import aQute.maven.api.Program;
024import aQute.maven.api.Revision;
025import aQute.maven.provider.MavenBackingRepository;
026import aQute.maven.provider.MavenRepository;
027import aQute.service.reporter.Reporter;
028
029import java.io.Closeable;
030import java.io.File;
031import java.io.IOException;
032import java.util.ArrayList;
033import java.util.List;
034import java.util.Optional;
035import java.util.concurrent.Executor;
036
037/**
038 * Provides a common {@link IMavenRepo} view on several 
039 * {@link MavenBackingRepository} instances.
040 * <P>
041 * The class extends {@link MavenRepository} which lacks some
042 * required functionality.
043 */
044public class CompositeMavenRepository extends MavenRepository
045        implements IMavenRepo, Closeable {
046
047    /**
048     * Instantiates a new merged maven repository.
049     *
050     * @param base the base
051     * @param repoId the repository id
052     * @param releaseRepos the backing release repositories
053     * @param snapshotRepos the backing snapshot repositories
054     * @param executor an executor
055     * @param reporter the reporter
056     * @throws Exception the exception
057     */
058    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
059        "PMD.AvoidDuplicateLiterals" })
060    public CompositeMavenRepository(File base, String repoId,
061            List<MavenBackingRepository> releaseRepos,
062            List<MavenBackingRepository> snapshotRepos, Executor executor,
063            Reporter reporter)
064            throws Exception {
065        super(base, repoId, releaseRepos, snapshotRepos, executor, reporter);
066    }
067
068    /**
069     * Returns all repositories.
070     *
071     * @return the list
072     */
073    public List<MavenBackingRepository> allRepositories() {
074        List<MavenBackingRepository> result
075            = new ArrayList<>(getReleaseRepositories());
076        result.addAll(getSnapshotRepositories());
077        return result;
078    }
079
080    /**
081     * Returns all known revisions as {@link BoundRevision}s.
082     *
083     * @param program the program
084     * @return the bound revisions
085     * @throws Exception the exception
086     */
087    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
088    public List<BoundRevision> boundRevisions(Program program)
089            throws Exception {
090        List<BoundRevision> revisions = new ArrayList<>();
091
092        for (MavenBackingRepository mbr : getReleaseRepositories()) {
093            addRevisions(mbr, program, revisions);
094        }
095        for (MavenBackingRepository mbr : getSnapshotRepositories()) {
096            if (!getReleaseRepositories().contains(mbr)) {
097                addRevisions(mbr, program, revisions);
098            }
099        }
100        return revisions;
101    }
102
103    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
104        "PMD.AvoidInstantiatingObjectsInLoops" })
105    private void addRevisions(MavenBackingRepository mbr, Program program,
106            List<BoundRevision> revisions) throws Exception {
107        List<Revision> revs = new ArrayList<>();
108        mbr.getRevisions(program, revs);
109        for (Revision rev : revs) {
110            revisions.add(new BoundRevision(mbr, rev));
111        }
112    }
113
114    /**
115     * Converts a {@link Revision} to an {@link BoundRevision}.
116     *
117     * @param revision the revision
118     * @return the bound revision
119     * @throws IOException Signals that an I/O exception has occurred.
120     */
121    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
122    public Optional<BoundRevision> toBoundRevision(Revision revision)
123            throws IOException {
124        for (MavenBackingRepository mbr : getReleaseRepositories()) {
125            Optional<BoundRevision> checkResult
126                = toBoundRevision(mbr, revision);
127            if (checkResult.isPresent()) {
128                return checkResult;
129            }
130        }
131        for (MavenBackingRepository mbr : getSnapshotRepositories()) {
132            if (!getReleaseRepositories().contains(mbr)) {
133                Optional<BoundRevision> checkResult
134                    = toBoundRevision(mbr, revision);
135                if (checkResult.isPresent()) {
136                    return checkResult;
137                }
138            }
139        }
140        return Optional.empty();
141    }
142
143    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
144        "PMD.AvoidRethrowingException", "PMD.AvoidCatchingGenericException",
145        "PMD.AvoidThrowingRawExceptionTypes" })
146    private Optional<BoundRevision> toBoundRevision(MavenBackingRepository mbr,
147            Revision revision) throws IOException {
148        List<Revision> revs = new ArrayList<>();
149        try {
150            mbr.getRevisions(revision.program, revs);
151        } catch (IOException e) {
152            // Should be the only reason.
153            throw e;
154        } catch (Exception e) {
155            throw new RuntimeException(e);
156        }
157        if (revs.contains(revision)) {
158            return Optional.of(new BoundRevision(mbr, revision));
159        }
160        return Optional.empty();
161    }
162
163    @Override
164    public BoundArchive getResolvedArchive(Revision revision, String extension,
165            String classifier) throws IOException {
166        Optional<BoundRevision> bound = toBoundRevision(revision);
167        if (bound.isPresent()) {
168            return getResolvedArchive(bound.get(), extension, classifier);
169        }
170        return null;
171    }
172
173    /**
174     * Gets the resolved archive.
175     *
176     * @param revision the revision
177     * @param extension the extension
178     * @param classifier the classifier
179     * @return the resolved archive
180     * @throws IOException Signals that an I/O exception has occurred.
181     */
182    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
183        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidRethrowingException" })
184    public BoundArchive getResolvedArchive(BoundRevision revision,
185            String extension, String classifier) throws IOException {
186        if (!revision.isSnapshot()) {
187            return revision.archive(extension, classifier);
188        }
189        try {
190            MavenVersion version = revision.mavenBackingRepository()
191                .getVersion(revision.revision());
192            if (version != null) {
193                return revision.archive(version, extension, classifier);
194            }
195        } catch (IOException e) {
196            throw e;
197        } catch (Exception e) {
198            throw new RuntimeException(e);
199        }
200        return null;
201    }
202
203}
