097    /**
098     * Create a new instance that uses the provided information/resources to perform
099     * its work.
100     *
101     * @param name the name
102     * @param releaseUrls the release urls
103     * @param snapshotUrls the snapshot urls
104     * @param localRepo the local Maven repository (cache)
105     * @param indexDbDir the persistent representation of this repository's content
106     * @param reporter a reporter for reporting the progress
107     * @param client an HTTP client for obtaining information from the Nexus server
108     * @param logIndexing the log indexing
109     * @throws Exception the exception
110     */
111    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
112        "PMD.SignatureDeclareThrowsException", "PMD.AvoidDuplicateLiterals" })
113    public IndexedMavenRepository(String name, List<URL> releaseUrls,
114            List<URL> snapshotUrls, File localRepo, File indexDbDir,
115            Reporter reporter, HttpClient client, boolean logIndexing)
116            throws Exception {
117        this.name = name;
118        this.indexDbDir = indexDbDir.toPath();
119        depsDir = this.indexDbDir.resolve("dependencies");
120        this.releaseUrls = releaseUrls;
121        this.snapshotUrls = snapshotUrls;
122        this.localRepo = localRepo;
123        this.reporter = reporter;
124        this.client = client;
125        this.logIndexing = logIndexing;
126
127        // Check prerequisites
128        if (indexDbDir.exists() && !indexDbDir.isDirectory()) {
129            reporter.error("%s must be a directory.", indexDbDir);
130            throw new IOException(indexDbDir + "must be a directory.");
131        }
132        if (!indexDbDir.exists()) {
133            indexDbDir.mkdirs();
134        }
135        if (!depsDir.toFile().exists()) {
136            depsDir.toFile().mkdir();
137        }
138
139        // Our repository
140        mavenRepository = createMavenRepository();
141
142        // Restore
143        @SuppressWarnings("PMD.UseConcurrentHashMap")
144        Map<String, MavenGroupRepository> oldGroups = new HashMap<>();
145        CompletableFuture
146            .allOf(scanRequested(oldGroups), scanDependencies(oldGroups)).get();
147        for (MavenGroupRepository groupRepo : groups.values()) {
148            addAll(groupRepo.getResources());
149        }
150        backupGroups = groups;
151    }
152
153    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException", "resource" })
154    private MavenResourceRepository createMavenRepository() throws Exception {
155        // Create repository from URLs
156        List<MavenBackingRepository> releaseBackers = new ArrayList<>();
157        for (URL url : releaseUrls) {
158            releaseBackers.addAll(MavenBackingRepository.create(
159                url.toString(), reporter, localRepo, client));
160        }
161        List<MavenBackingRepository> snapshotBackers = new ArrayList<>();
162        for (URL url : snapshotUrls) {
163            snapshotBackers.addAll(MavenBackingRepository.create(
164                url.toString(), reporter, localRepo, client));
165        }
166        return new MavenResourceRepository(
167            localRepo, name(), releaseBackers,
168            snapshotBackers, Processor.getExecutor(), reporter)
169                .setResourceSupplier(this::restoreResource);
170    }
171
172    private Optional<Resource> restoreResource(Archive archive) {
173        return Optional.ofNullable(backupGroups.get(archive.revision.group))
174            .flatMap(group -> group.searchInBackup(archive));
175    }

213    /**
214     * Get or create the group repository for the given group id.
215     * If the repository is created, it is created as a repository
216     * for dependencies.
217     *
218     * @param groupId the group id
219     * @return the repository
220     * @throws IOException Signals that an I/O exception has occurred.
221     */
222    public MavenGroupRepository getOrCreateGroupRepository(String groupId)
223            throws IOException {
224        @SuppressWarnings("PMD.PrematureDeclaration")
225        MavenGroupRepository result = rethrow(IOException.class,
226            () -> groups.computeIfAbsent(groupId,
227                grp -> unthrow(() -> new MavenGroupRepository(grp,
228                    depsDir.resolve(grp), false, this, client, reporter))));
229        return result;
230    }

232    /**
233     * Refresh this repository's content.
234     * 
235     * @return true if refreshed, false if not refreshed possibly due to error
236     * @throws Exception if a problem occurs
237     */
238    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
239    public boolean refresh() throws Exception {
240        if (!refreshing.compareAndSet(false, true)) {
241            reporter.warning("Repository is already refreshing.");
242            return false;
243        }
244        String threadName = Thread.currentThread().getName();
245        try {
246            Thread.currentThread().setName("IndexedMaven Refresher");
247            return doRefresh();
248        } finally {
249            Thread.currentThread().setName(threadName);
250            refreshing.set(false);
251        }
252    }
253
254    @SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops",
255        "PMD.AvoidDuplicateLiterals", "PMD.SignatureDeclareThrowsException" })
256    private boolean doRefresh() throws Exception {
257        mavenRepository.reset();
258
259        // Reuse and clear (or create new) group repositories for the existing
260        // directories, first for explicitly requested group ids...
261        backupGroups = new HashMap<>(groups);
262        groups.clear();
263        for (var group : backupGroups.values()) {
264            group.reset();
265        }
266
267        // Scan requested first to avoid problems if a group is moved from
268        // dependency to requested.
269        scanRequested(backupGroups).get();
270        scanDependencies(backupGroups).get();
271
272        // Refresh them all.
273        CompletableFuture<?>[] repoLoaders
274            = new ArrayList<>(groups.values()).stream()
275                .map(repo -> CompletableFuture.runAsync(() -> {
276                    String threadName = Thread.currentThread().getName();
277                    try {
278                        Thread.currentThread()
279                            .setName("RepoLoader " + repo.id());
280                        LOG.debug("Reloading %s...", repo.id());
281                        repo.reload();
282                    } catch (IOException e) {
283                        throw new CompletionException(e);
284                    } finally {
285                        Thread.currentThread().setName(threadName);
286                    }
287                }, groupLoaders))
288                .toArray(CompletableFuture[]::new);
289        CompletableFuture.allOf(repoLoaders).get();
290        // Remove no longer required group repositories.
291        for (Iterator<Map.Entry<String, MavenGroupRepository>> iter
292            = groups.entrySet().iterator(); iter.hasNext();) {
293            Map.Entry<String, MavenGroupRepository> entry = iter.next();
294            if (entry.getValue().removeIfRedundant()) {
295                iter.remove();
296            }
297        }
298        CompletableFuture.allOf(
299            // Persist updated data.
300            CompletableFuture.allOf(new ArrayList<>(groups.values()).stream()
301                .map(repo -> CompletableFuture.runAsync(() -> {
302                    try {
303                        repo.flush();
304                    } catch (IOException e) {
305                        throw new CompletionException(e);
306                    }
307                }, groupLoaders))
308                .toArray(CompletableFuture[]::new)),
309            // Write federated index.
310            CompletableFuture.runAsync(() -> {
311                if (groups.keySet().equals(backupGroups.keySet())) {
312                    return;
313                }
314                try (OutputStream fos
315                    = Files.newOutputStream(indexDbDir.resolve("index.xml"))) {
316                    writeFederatedIndex(fos);
317                } catch (IOException e) {
318                    throw new CompletionException(e);
319                }
320            }, groupLoaders),
321            // Add collected to root (this).
322            CompletableFuture.runAsync(() -> {
323                set(Collections.emptyList());
324                for (MavenGroupRepository groupRepo : groups.values()) {
325                    addAll(groupRepo.getResources());
326                }
327            }, groupLoaders)).get();
328        backupGroups = groups;
329        return true;
330    }

332    private CompletableFuture<Void>
333            scanRequested(Map<String, MavenGroupRepository> oldGroups) {
334        return CompletableFuture.allOf(
335            Arrays.stream(indexDbDir.toFile().list()).parallel()
336                .filter(dir -> dir.matches("^[A-Za-z].*")
337                    && !"index.xml".equals(dir)
338                    && !"dependencies".equals(dir))
339                .map(groupId -> CompletableFuture
340                    .runAsync(() -> restoreGroup(oldGroups, groupId, true),
341                        groupLoaders))
342                .toArray(CompletableFuture[]::new));
343    }
344
345    private CompletableFuture<Void>
346            scanDependencies(Map<String, MavenGroupRepository> oldGroups) {
347        if (!depsDir.toFile().canRead() || !depsDir.toFile().isDirectory()) {
348            return CompletableFuture.completedFuture(null);
349        }
350        return CompletableFuture.allOf(
351            Arrays.stream(depsDir.toFile().list()).parallel()
352                .filter(dir -> dir.matches("^[A-Za-z].*"))
353                .filter(dir -> {
354                    if (groups.containsKey(dir)) {
355                        // Is/has become explicitly requested
356                        try {
357                            Files.walk(depsDir.resolve(dir))
358                                .sorted(Comparator.reverseOrder())
359                                .map(Path::toFile)
360                                .forEach(File::delete);
361                        } catch (IOException e) {
362                            // TODO Auto-generated catch block
363                            e.printStackTrace();
364                        }
365                        return false;
366                    }
367                    return true;
368                })
369                .map(groupId -> CompletableFuture
370                    .runAsync(() -> restoreGroup(oldGroups, groupId, false),
371                        groupLoaders))
372                .toArray(CompletableFuture[]::new));
373    }
374
375    private void restoreGroup(Map<String, MavenGroupRepository> oldGroups,
376            String groupId, boolean requested) {
377        if (oldGroups.containsKey(groupId)) {
378            // Reuse existing.
379            MavenGroupRepository groupRepo = oldGroups.get(groupId);
380            groupRepo.reset((requested ? indexDbDir : depsDir).resolve(groupId),
381                requested);
382            groups.put(groupId, groupRepo);
383            return;
384        }
385        try {
386            MavenGroupRepository groupRepo
387                = new MavenGroupRepository(groupId,
388                    (requested ? indexDbDir : depsDir).resolve(groupId),
389                    requested, this, client, reporter);
390            groups.put(groupId, groupRepo);
391        } catch (IOException e) {
392            reporter.exception(e,
393                "Cannot create group repository for %s: %s",
394                groupId, e.getMessage());
395        }
396    }

398    private void writeFederatedIndex(OutputStream out) {
399        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
400        dbf.setNamespaceAware(true);
401        Document doc;
402        try {
403            doc = dbf.newDocumentBuilder().newDocument();
404        } catch (ParserConfigurationException e) {
405            return;
406        }
407        @SuppressWarnings("PMD.AvoidFinalLocalVariable")
408        final String repoNs = "http://www.osgi.org/xmlns/repository/v1.0.0";
409        // <repository name=... increment=...>
410        Element repoNode = doc.createElementNS(repoNs, "repository");
411        repoNode.setAttribute("name", name);
412        repoNode.setAttribute("increment",
413            Long.toString(System.currentTimeMillis()));
414        doc.appendChild(repoNode);
415        for (Map.Entry<String, MavenGroupRepository> repo : groups.entrySet()) {
416            // <referral url=...>
417            Element referral = doc.createElementNS(repoNs, "referral");
418            referral.setAttribute("url",
419                (repo.getValue().isRequested() ? "" : "dependencies/")
420                    + repo.getKey() + "/index.xml");
421            repoNode.appendChild(referral);
422        }
423        // Write federated index.
424        try {
425            Transformer transformer
426                = TransformerFactory.newInstance().newTransformer();
427            transformer.setOutputProperty(OutputKeys.INDENT, "yes");
428            DOMSource source = new DOMSource(doc);
429            transformer.transform(source, new StreamResult(out));
430        } catch (TransformerException e) {
431            reporter.exception(e, "Cannot write federated index: %s",
432                e.getMessage());
433        }
434    }
