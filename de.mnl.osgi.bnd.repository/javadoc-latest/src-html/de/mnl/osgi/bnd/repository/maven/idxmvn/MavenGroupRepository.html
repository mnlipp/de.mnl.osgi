110    /**
111     * Instantiates a new representation of group data backed
112     * by the specified directory. 
113     *
114     * @param groupId the maven groupId indexed by this repository
115     * @param directory the directory used to persist data
116     * @param requested if it is a requested group id
117     * @param indexedRepository the indexed maven repository
118     * @param client the client used for remote access
119     * @param reporter the reporter
120     * @throws IOException Signals that an I/O exception has occurred.
121     */
122    @SuppressWarnings({ "PMD.ConfusingTernary",
123        "PMD.AvoidCatchingGenericException", "PMD.AvoidDuplicateLiterals" })
124    public MavenGroupRepository(String groupId, Path directory,
125            boolean requested, IndexedMavenRepository indexedRepository,
126            HttpClient client, Reporter reporter) throws IOException {
127        this.groupId = groupId;
128        this.requested = requested;
129        this.indexedRepository = indexedRepository;
130        this.client = client;
131        this.reporter = reporter;
132
133        // Prepare directory and files
134        updatePaths(directory);
135        groupProps = new Properties();
136        if (groupPropsPath.toFile().canRead()) {
137            try (InputStream input = Files.newInputStream(groupPropsPath)) {
138                groupProps.load(input);
139                versionSpecs = VersionSpecification.parse(groupProps);
140            }
141        }
142
143        // Prepare OSGi repository
144        if (groupIndexPath.toFile().canRead()) {
145            try (XMLResourceParser parser
146                = new XMLResourceParser(groupIndexPath.toFile())) {
147                addAll(parser.parse());
148            } catch (Exception e) { // NOPMD
149                reporter.warning("Cannot parse %s, ignored: %s", groupIndexPath,
150                    e.getMessage());
151            }
152        }
153
154        // Re-use exiting resources for faster checks.
155        for (Capability cap : findProvider(
156            newRequirementBuilder("bnd.info").build())) {
157            indexingState.put(
158                Archive.valueOf((String) cap.getAttributes().get("from")),
159                IndexingState.INDEXED);
160        }
161        LOG.debug("Created group repository for {}.", groupId);
162    }
163
164    private void updatePaths(Path directory) {
165        if (!directory.toFile().exists()) {
166            directory.toFile().mkdir();
167        }
168        groupDir = directory;
169        groupPropsPath = groupDir.resolve("group.properties");
170        groupIndexPath = groupDir.resolve("index.xml");
171    }
172
173    private IndexingState indexingState(Archive archive) {
174        return indexingState.getOrDefault(archive, IndexingState.NONE);
175    }

186    /**
187     * Writes all changes to persistent storage and removes
188     * any backup information prepared by {@link #reset(Path, boolean)}.
189     *
190     * @throws IOException Signals that an I/O exception has occurred.
191     */
192    public void flush() throws IOException {
193        boolean indexChanged = true;
194        if (backupRepo != null) {
195            Set<Resource> oldSet = new HashSet<>(backupRepo.getResources());
196            Set<Resource> newSet = new HashSet<>(getResources());
197            if (newSet.equals(oldSet)) {
198                indexChanged = false;
199            }
200        }
201        if (indexChanged) {
202            XMLResourceGenerator generator = new XMLResourceGenerator();
203            generator.resources(getResources());
204            generator.name(indexedRepository.mavenRepository().name());
205            try {
206                generator.save(groupIndexPath.toFile());
207            } catch (IOException e) {
208                reporter.exception(e, "Cannot save %s.", groupIndexPath);
209            }
210        }
211        backupRepo = null;
212        if (indexingLog != null) {
213            rethrow(IOException.class, () -> loggedMessages.entrySet().stream()
214                .sorted(Map.Entry.comparingByKey())
215                .flatMap(e -> e.getValue().stream())
216                .forEach(msg -> unthrow(() -> indexingLog
217                    .write(msg + System.lineSeparator()))));
218            indexingLog.close();
219            indexingLog = null;
220        }
221        loggedMessages.clear();
222        indexingState.clear();
223    }

225    /**
226     * Removes the group directory if empty.
227     *
228     * @return true, if removed
229     * @throws IOException Signals that an I/O exception has occurred.
230     */
231    public boolean removeIfRedundant() throws IOException {
232        if (isRequested() || !groupProps.isEmpty()
233            || !getResources().isEmpty()) {
234            return false;
235        }
236        // Nothing in this group
237        Files.walk(groupDir)
238            .sorted(Comparator.reverseOrder())
239            .map(Path::toFile)
240            .forEach(File::delete);
241        return true;
242    }

254    /**
255     * Clears this repository and updates the path and the requested 
256     * flag. Keeps the current content as backup for reuse in a 
257     * subsequent call to {@link #reload()}.
258     *
259     * @param directory this group's directory
260     * @param requested whether this is a requested group
261     */
262    public void reset(Path directory, boolean requested) {
263        synchronized (this) {
264            // Update basic properties
265            this.requested = requested;
266            if (!groupDir.equals(directory)) {
267                updatePaths(directory);
268                backupRepo = null;
269            } else {
270                // Save current content and clear.
271                backupRepo = new ResourcesRepository(getResources());
272            }
273            set(Collections.emptyList());
274            // Clear and reload properties
275            groupProps.clear();
276            if (groupPropsPath.toFile().canRead()) {
277                try (InputStream input = Files.newInputStream(groupPropsPath)) {
278                    groupProps.load(input);
279                    versionSpecs = VersionSpecification.parse(groupProps);
280                } catch (IOException e) {
281                    reporter.warning("Problem reading %s (ignored): %s",
282                        groupPropsPath, e.getMessage());
283                }
284            }
285            // Clear remaining caches.
286            indexingState.clear();
287        }
288    }

290    /**
291     * Reset the repository group. Must be called for all groups before
292     * reloading. 
293     *
294     * @throws IOException Signals that an I/O exception has occurred.
295     */
296    void reset() throws IOException {
297        if (indexedRepository.logIndexing()) {
298            indexingLog = Files.newBufferedWriter(
299                groupDir.resolve("indexing.log"), Charset.defaultCharset());
300        } else {
301            // Don't keep out-dated log files, it's irritating.
302            groupDir.resolve("indexing.log").toFile().delete();
303        }
304        loggedMessages.clear();
305
306        if (!isRequested()) {
307            // Will be filled with dependencies only
308            return;
309        }
310        // Actively filled.
311        synchronized (this) {
312            if (backupRepo == null) {
313                backupRepo = new ResourcesRepository(getResources());
314            }
315        }
316    }

318    /**
319     * Reload the repository. May be called concurrently for different
320     * group repositories after resetting all. Requested repositories 
321     * retrieve the list of known artifactIds from the remote repository 
322     * and add the versions. For versions already in the repository, 
323     * the backup information is re-used.
324     *
325     * @throws IOException Signals that an I/O exception has occurred.
326     */
327    @SuppressWarnings({ "PMD.AvoidReassigningLoopVariables",
328        "PMD.AvoidCatchingGenericException", "PMD.AvoidDuplicateLiterals",
329        "PMD.AvoidInstantiatingObjectsInLoops",
330        "PMD.AvoidThrowingRawExceptionTypes", "PMD.PreserveStackTrace" })
331    void reload() throws IOException {
332        if (!isRequested()) {
333            // Will be filled with dependencies only
334            return;
335        }
336        try {
337            CompletableFuture<?>[] artifactLoaders = findArtifactIds()
338                .stream().map(
339                    artifactId -> loadProgram(
340                        Program.valueOf(groupId, artifactId)))
341                .toArray(CompletableFuture[]::new);
342            CompletableFuture.allOf(artifactLoaders).get();
343        } catch (ExecutionException e) {
344            if (e.getCause() instanceof IOException) {
345                throw (IOException) e.getCause();
346            }
347            throw new RuntimeException(e.getCause());
348        } catch (InterruptedException e) {
349            throw new RuntimeException(e);
350        }
351    }
352
353    private CompletableFuture<Void> loadProgram(Program program) {
354        // Get revisions of program and process.
355        CompletableFuture<Void> result = new CompletableFuture<>();
356        IndexedMavenRepository.programLoaders.submit(() -> {
357            String threadName = Thread.currentThread().getName();
358            try {
359                Thread.currentThread().setName("RevisionQuerier " + program);
360                Stream<CompletableFuture<Void>> revLoaderStream
361                    = indexedRepository.mavenRepository().findRevisions(program)
362                        .flatMap(revision -> {
363                            var archives = VersionSpecification
364                                .toSelected(versionSpecs, revision);
365                            if (archives.isEmpty()) {
366                                logIndexing(revision.unbound(),
367                                    () -> String.format(
368                                        "%s not selected for indexing.",
369                                        revision.unbound()));
370                            }
371                            return archives.stream();
372                        }).map(archive -> loadArchive(archive));
373
374                // Complete all completables
375                revLoaderStream.forEach(
376                    revLoad -> unthrow(() -> revLoad.handle((res, thrw) -> {
377                        return null;
378                    }).get()));
379            } catch (Throwable t) {
380                reporter.exception(t,
381                    "Failed to get list of revisions of %s: %s", program,
382                    t.getMessage());
383            } finally {
384                Thread.currentThread().setName(threadName);
385                result.complete(null);
386            }
387        });
388        return result;
389    }

391    /**
392     * Load the archive asynchronously (if selected). Calling 
393     * {@link CompletableFuture#get()} never returns an exception. 
394     *
395     * @param archive the revision
396     * @return the completable future
397     */
398    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
399        "PMD.AvoidDuplicateLiterals",
400        "PMD.PositionLiteralsFirstInComparisons" })
401    private CompletableFuture<Void> loadArchive(BoundArchive archive) {
402        return CompletableFuture.runAsync(() -> {
403            if (indexingState.putIfAbsent(archive,
404                IndexingState.INDEXING) != null) {
405                // Already handled (as dependency)
406                logIndexing(archive.revision, () -> String.format(
407                    "%s in list (already handled as dependency).", archive));
408                return;
409            }
410            LOG.debug("Loading archive {}.", archive);
411            logIndexing(archive.revision,
412                () -> String.format("%s in list, indexing...", archive));
413            String threadName = Thread.currentThread().getName();
414            try {
415                Thread.currentThread().setName("ArchiveLoader " + archive);
416                MavenResource resource = indexedRepository.mavenRepository()
417                    .resource(archive, BinaryLocation.REMOTE);
418                Set<MavenResource> allDeps = new HashSet<>();
419                if (!isIndexable(resource, allDeps)) {
420                    return;
421                }
422                addResourceAndDependencies(resource, allDeps);
423            } catch (Exception e) {
424                // Load "in isolation", don't propagate individual failure.
425                reporter.exception(e, "Failed to load revision %s: %s",
426                    archive, e.getMessage());
427                logIndexing(archive.revision, () -> String.format(
428                    "%s failed to load: %s.", archive, e.getMessage()));
429            } finally {
430                Thread.currentThread().setName(threadName);
431            }
432        }, IndexedMavenRepository.revisionLoaders);
433    }

435    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
436        "PMD.AvoidInstantiatingObjectsInLoops" })
437    private Collection<String> findArtifactIds() {
438        Set<String> result = new HashSet<>();
439        for (MavenBackingRepository repo : indexedRepository.mavenRepository()
440            .backing()) {
441            URI groupUri = null;
442            try {
443                groupUri
444                    = repo.toURI("").resolve(groupId.replace('.', '/') + "/");
445                String page = client.build().headers("User-Agent", "Bnd")
446                    .get(String.class)
447                    .go(groupUri);
448                if (page == null) {
449                    continue;
450                }
451                Matcher matcher = hrefPattern.matcher(page);
452                while (matcher.find()) {
453                    URI programUri = groupUri.resolve(matcher.group("href"));
454                    String artifactId = programUri.getPath()
455                        .substring(groupUri.getPath().length());
456                    if (artifactId.endsWith("/")) {
457                        artifactId
458                            = artifactId.substring(0, artifactId.length() - 1);
459                    }
460                    result.add(artifactId);
461                }
462            } catch (Exception e) {
463                reporter.warning("Problem retrieving %s, skipped: %s", groupUri,
464                    e.getMessage());
465            }
466        }
467        return result;
468    }

470    private void addResourceAndDependencies(MavenResource resource,
471            Set<MavenResource> allDeps) {
472        addResource(resource);
473        // Add the dependencies found while checking to the index.
474        try {
475            rethrow(IOException.class, () -> allDeps.stream()
476                .forEach(res -> runIgnoring(() -> indexedRepository
477                    .getOrCreateGroupRepository(
478                        res.archive().getRevision().group)
479                    .addResource(res))));
480        } catch (Exception e) {
481            // Load "in isolation", don't propagate individual failure.
482            reporter.exception(e, "Failed to add resource %s: %s",
483                resource.archive(), e.getMessage());
484            logIndexing(resource.archive().revision, () -> String.format(
485                "%s failed to load.", resource.archive()));
486        }
487    }

489    /**
490     * Checks if the given resource is indexable, considering its 
491     * dependencies. A resource is indexable if its dependencies aren't
492     * excluded.
493     *
494     * @param resource the resource to check
495     * @param dependencies revisions that must additionally be included
496     *    when the checked revision is added
497     * @return true, if the revision can be added to the index
498     */
499    @SuppressWarnings({ "PMD.LongVariable", "PMD.NPathComplexity",
500        "PMD.AvoidCatchingGenericException" })
501    private boolean isIndexable(MavenResource resource,
502            Set<MavenResource> dependencies) {
503        Archive archive = resource.archive();
504        if (!archive.revision.group.equals(groupId)) {
505            throw new IllegalArgumentException("Wrong groupId "
506                + archive.revision.group + " (must be " + groupId + ").");
507        }
508        // Check if already decided.
509        switch (indexingState(archive)) {
510        case INDEXED:
511            return true;
512        case EXCLUDED:
513        case EXCL_BY_DEP:
514            return false;
515        default:
516            break;
517        }
518        // Check if excluded by rule.
519        if (VersionSpecification
520            .excluded(versionSpecs, archive.revision.artifact)
521            .includes(MavenVersion.from(archive.revision.version))) {
522            if (indexingState.replace(archive, IndexingState.INDEXING,
523                IndexingState.EXCLUDED)) {
524                logIndexing(archive.revision,
525                    () -> String.format("%s is excluded by rule.", archive));
526            }
527            return false;
528        }
529        boolean isForced = VersionSpecification.isForced(versionSpecs, archive);
530
531        // Check whether excluded because dependency is excluded
532        List<Dependency> deps = evaluateDependencies(resource);
533        for (Dependency dep : deps) {
534            MavenGroupRepository depRepo;
535            try {
536                depRepo = indexedRepository
537                    .getOrCreateGroupRepository(dep.getGroupId());
538            } catch (Exception e) {
539                // Failing to get a dependency is no reason to fail.
540                continue;
541            }
542            Optional<MavenResource> optRes = depRepo.dependencyToResource(dep);
543            if (!optRes.isPresent()) {
544                // Failing to get a dependency is no reason to fail.
545                continue;
546            }
547            MavenResource depResource = optRes.get();
548            // Watch out to use the proper repository in the code following!
549            depRepo.logIndexing(depResource.archive().revision,
550                () -> String.format("%s is dependency of %s, indexing...",
551                    depResource.archive(), archive));
552            depRepo.indexingState.putIfAbsent(depResource.archive(),
553                IndexingState.INDEXING);
554            Set<MavenResource> collectedDeps = new HashSet<>();
555            if (!depRepo.isIndexable(depResource, collectedDeps)) {
556                // Note that the revision which was checked is not indexable
557                // due to a dependency that is not indexable (unless forced)
558                if (!isForced) {
559                    if (indexingState.replace(archive, IndexingState.INDEXING,
560                        IndexingState.EXCL_BY_DEP)) {
561                        logIndexing(archive.revision, () -> String.format(
562                            "%s not indexable, depends on %s.",
563                            archive, depResource.archive()));
564                    }
565                    return false;
566                }
567            }
568            if (!VersionSpecification.toSelected(depRepo.versionSpecs,
569                depResource.archive().revision).isEmpty()) {
570                // Has just been indexed as dependency, but would have
571                // been indexed (as selected) anyway, so add independent
572                // of outcome of dependency evaluation.
573                depRepo.addResourceAndDependencies(depResource, collectedDeps);
574            } else {
575                dependencies.add(depResource);
576                dependencies.addAll(collectedDeps);
577            }
578        }
579        return true;
580    }

582    private List<Dependency> evaluateDependencies(MavenResource resource) {
583        List<Dependency> deps;
584        try {
585            deps = resource.dependencies();
586        } catch (Exception e) {
587            // Failing to get the dependencies is no reason to fail.
588            return Collections.emptyList();
589        }
590        if (!deps.isEmpty()) {
591            logIndexing(resource.archive().revision,
592                () -> String.format("%s has dependencies: %s",
593                    resource.archive(), deps.stream()
594                        .map(d -> d.getGroupId() + ":" + d.getArtifactId() + ":"
595                            + d.getVersion())
596                        .collect(Collectors.joining(", "))));
597        }
598        return deps;
599    }

601    private Optional<MavenResource> dependencyToResource(Dependency dep) {
602        Program depPgm = Program.valueOf(dep.getGroupId(), dep.getArtifactId());
603        try {
604            return indexedRepository.mavenRepository().resource(
605                depPgm, narrowVersion(depPgm, MavenVersionSpecification
606                    .from(dep.getVersion())),
607                dep.getType(), dep.getClassifier(),
608                BinaryLocation.REMOTE);
609        } catch (Exception e) {
610            // Failing to get a dependency is no reason to fail.
611            return Optional.empty();
612        }
613    }

615    private MavenVersionSpecification narrowVersion(
616            Program program, MavenVersionSpecification version)
617            throws IOException {
618        if (version instanceof MavenVersion) {
619            // Specific version, leave as is.
620            return version;
621        }
622        // If it's a range, restrict it to allowed
623        MavenVersionRange excluded = VersionSpecification
624            .excluded(versionSpecs, program.artifact);
625        return excluded.complement().restrict((MavenVersionRange) version);
626    }

628    /**
629     * Adds the specified revision.
630     *
631     * @param revision the revision to add
632     */
633    private void addResource(MavenResource resource) {
634        synchronized (indexingState) {
635            if (indexingState(resource.archive()) == IndexingState.INDEXED) {
636                return;
637            }
638            try {
639                add(resource.asResource());
640                logIndexing(resource.archive().revision,
641                    () -> String.format("%s indexed.", resource.archive()));
642            } catch (MavenResourceException e) {
643                reporter.exception(e, "Failed to get %s as resource.",
644                    resource.toString());
645                logIndexing(resource.archive().revision,
646                    () -> String.format("%s could not be indexed: %s.",
647                        resource.archive(), e.getMessage()));
648            }
649            indexingState.put(resource.archive(), IndexingState.INDEXED);
650        }
651    }

653    /* package */ Optional<Resource> searchInBackup(Archive archive) {
654        if (backupRepo == null) {
655            return Optional.empty();
656        }
657        return backupRepo.findProvider(
658            backupRepo.newRequirementBuilder("bnd.info")
659                .addDirective("filter",
660                    String.format("(from=%s)", archive.toString()))
661                .build())
662            .stream().findFirst().map(Capability::getResource);
663    }

665    private void logIndexing(Revision revision, Supplier<String> msgSupplier) {
666        loggedMessages
667            .computeIfAbsent(revision,
668                rev -> Collections.synchronizedList(new ArrayList<>()))
669            .add(msgSupplier.get());
670    }
