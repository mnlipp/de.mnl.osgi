019package de.mnl.osgi.bnd.maven;
020
021import aQute.maven.api.Revision;
022
023import java.util.List;
024
025import org.apache.maven.model.Dependency;
026import org.osgi.resource.Capability;
027import org.osgi.resource.Requirement;
028import org.osgi.resource.Resource;
029
030/**
031 * A resource that is backed by a maven revision.
032 */
033public interface MavenResource {
034
035    /**
036     * Returns the revision. The revision is the only mandatory
037     * information when creating a {@link MavenResource}. The other
038     * informations that can be obtained can be made available
039     * lazily, i.e. can be loaded on demand. 
040     * <P>
041     * This implies that a {@link MavenResource} can be created 
042     * that does not exist in the backing maven repositories. 
043     * If not obvious from the context, a call to {@link #boundRevision()}
044     * can be used to verify that the resource exists.
045     *
046     * @return the revision
047     */
048    Revision revision();
049
050    /**
051     * Returns the bound revision.
052     *
053     * @return the revision
054     * @throws MavenResourceException the maven resource exception
055     */
056    BoundRevision boundRevision() throws MavenResourceException;
057
058    /**
059     * Returns the mandatory maven compile and runtime dependencies. 
060     * Dependencies in the POM that use variables are filtered, because
061     * these variable cannot be resolved. 
062     *
063     * @return the dependencies
064     * @throws MavenResourceException the maven resource exception
065     */
066    List<Dependency> dependencies() throws MavenResourceException;
067
068    /**
069     * Gets the underlying "ordinary" resource.
070     *
071     * @return the resource
072     * @throws MavenResourceException the maven resource exception
073     */
074    Resource asResource() throws MavenResourceException;
075
076    /**
077     * Gets the capabilities from the given namespace.
078     *
079     * @param namespace the namespace
080     * @return the capabilities
081     * @throws MavenResourceException the maven resource exception
082     */
083    List<Capability> getCapabilities(String namespace)
084            throws MavenResourceException;
085
086    /**
087     * Gets the requirements from the given namespace.
088     *
089     * @param namespace the namespace
090     * @return the requirements
091     * @throws MavenResourceException the maven resource exception
092     */
093    List<Requirement> getRequirements(String namespace)
094            throws MavenResourceException;
095
096    /**
097     * A {@link MavenResource} can be compared with another maven resource.
098     * In this case the objects are considered equal if their revisions 
099     * are equal. A {@link MavenResource} can also be compared with 
100     * another {@link Resource}. In this case, equality will be checked 
101     * between {@link MavenResource#asResource()} and the other resource 
102     * (see {@link Resource#equals(Object)}). 
103     *
104     * @param obj the obj
105     * @return true, if successful
106     */
107    @Override
108    boolean equals(Object obj);
109
110    /**
111     * The hash code is defined by the revision.
112     *
113     * @return the int
114     */
115    @Override
116    int hashCode();
117
118    @Override
119    String toString();
120
121}
