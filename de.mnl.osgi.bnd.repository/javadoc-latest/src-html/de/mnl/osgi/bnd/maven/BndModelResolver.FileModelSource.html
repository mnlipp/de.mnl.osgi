019package de.mnl.osgi.bnd.maven;
020
021import aQute.maven.api.Archive;
022import aQute.maven.api.Program;
023import aQute.maven.api.Revision;
024import aQute.maven.provider.MavenRepository;
025import aQute.service.reporter.Reporter;
026import java.io.File;
027import org.apache.maven.building.FileSource;
028import org.apache.maven.model.Dependency;
029import org.apache.maven.model.Parent;
030import org.apache.maven.model.Repository;
031import org.apache.maven.model.building.ModelSource;
032import org.apache.maven.model.resolution.InvalidRepositoryException;
033import org.apache.maven.model.resolution.ModelResolver;
034import org.apache.maven.model.resolution.UnresolvableModelException;
035
036/**
037 * Resolves (raw) model requests using a bnd {@link MavenRepository}
038 * as backend.  
039 */
040@SuppressWarnings("deprecation")
041public class BndModelResolver implements ModelResolver, Cloneable {
042
043    private final MavenRepository bndRepository;
044    private final Reporter reporter;
045
046    /**
047     * Instantiates a new bnd model resolver.
048     *
049     * @param bndRepository the backing repository
050     * @param reporter the reporter
051     */
052    public BndModelResolver(MavenRepository bndRepository, Reporter reporter) {
053        this.bndRepository = bndRepository;
054        this.reporter = reporter;
055    }
056
057    @Override
058    @SuppressWarnings("PMD.AvoidCatchingGenericException")
059    public ModelSource resolveModel(String groupId, String artifactId,
060            String version) throws UnresolvableModelException {
061        Revision revision
062            = Program.valueOf(groupId, artifactId).version(version);
063        Archive pomArchive = revision.getPomArchive();
064        try {
065            File pomFile = bndRepository.get(pomArchive).getValue();
066            if (pomFile == null) {
067                throw new UnresolvableModelException("Not found.", groupId,
068                    artifactId, version);
069            }
070            return new FileModelSource(pomFile);
071        } catch (Exception e) {
072            throw new UnresolvableModelException(e, groupId, artifactId,
073                version);
074        }
075    }
076
077    @Override
078    @SuppressWarnings("PMD.AvoidCatchingGenericException")
079    public ModelSource resolveModel(Parent parent)
080            throws UnresolvableModelException {
081        Revision revision
082            = Program.valueOf(parent.getGroupId(), parent.getArtifactId())
083                .version(parent.getVersion());
084        try {
085            Archive pomArchive
086                = bndRepository.getResolvedArchive(revision, "pom", "");
087            parent.setVersion(pomArchive.getRevision().version.toString());
088            File pomFile = bndRepository.get(pomArchive).getValue();
089            if (pomFile == null) {
090                throw new UnresolvableModelException("Not found.",
091                    parent.getGroupId(), parent.getArtifactId(),
092                    parent.getVersion());
093            }
094            return new FileModelSource(pomFile);
095        } catch (Exception e) {
096            throw new UnresolvableModelException(e, parent.getGroupId(),
097                parent.getArtifactId(), parent.getVersion());
098        }
099    }
100
101    @Override
102    @SuppressWarnings("PMD.AvoidCatchingGenericException")
103    public ModelSource resolveModel(Dependency dependency)
104            throws UnresolvableModelException {
105        Revision revision
106            = Program.valueOf(dependency.getGroupId(),
107                dependency.getArtifactId()).version(dependency.getVersion());
108        try {
109            Archive pomArchive
110                = bndRepository.getResolvedArchive(revision, "pom", "");
111            dependency.setVersion(pomArchive.getRevision().version.toString());
112            File pomFile = bndRepository.get(pomArchive).getValue();
113            if (pomFile == null) {
114                throw new UnresolvableModelException("Not found.",
115                    dependency.getGroupId(), dependency.getArtifactId(),
116                    dependency.getVersion());
117            }
118            return new FileModelSource(pomFile);
119        } catch (Exception e) {
120            throw new UnresolvableModelException(e, dependency.getGroupId(),
121                dependency.getArtifactId(), dependency.getVersion());
122        }
123    }
124
125    @Override
126    public void addRepository(Repository repository)
127            throws InvalidRepositoryException {
128        reporter.warning("Attempt to add repository %s (not supported).",
129            repository);
130    }
131
132    @Override
133    public void addRepository(Repository repository, boolean replace)
134            throws InvalidRepositoryException {
135        reporter.warning(
136            "Attempt to add/replace repository %s (not supported).",
137            repository);
138    }
139
140    @Override
141    public ModelResolver newCopy() {
142        try {
143            return (BndModelResolver) clone();
144        } catch (CloneNotSupportedException e) {
145            // Doesn't happen
146            return null;
147        }
148    }
149
150    /**
151     * The Class FileModelSource.
152     */
153    public static class FileModelSource extends FileSource
154            implements ModelSource {
155
156        /**
157         * Instantiates a new file model source.
158         *
159         * @param file the file
160         */
161        public FileModelSource(File file) {
162            super(file);
163        }
164    }
165}
