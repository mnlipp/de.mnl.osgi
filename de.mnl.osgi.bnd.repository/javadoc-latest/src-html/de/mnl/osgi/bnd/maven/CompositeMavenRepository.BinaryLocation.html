076/**
077 * Provides a composite {@link IMavenRepo} view on several 
078 * {@link MavenBackingRepository} instances.
079 * The class replaces {@link MavenRepository} which lacks some
080 * required functionality. (Besides, this class has a more
081 * appropriate name.)
082 * <P>
083 * The information about artifacts is provided as a maven
084 * {@link Model}. It is evaluated using the maven libraries and
085 * should therefore be consistent with the model information
086 * used in other maven repository based tools.
087 */
088@SuppressWarnings({ "PMD.DataflowAnomalyAnalysis", "deprecation" })
089public class CompositeMavenRepository implements Closeable {
090
091    public static final Pattern COORDS_SPLITTER = Pattern.compile("\\s*;\\s*");
092    private final MavenRepository bndMavenRepo;
093    private final Reporter reporter;
094    private final Map<Program, List<BoundRevision>> programCache
095        = new ConcurrentHashMap<>();
096    private final Map<Revision, Model> modelCache
097        = new ConcurrentHashMap<>();
098    private final BndModelResolver modelResolver;
099    private final ModelBuilder modelBuilder;

108    /**
109     * Instantiates a new composite maven repository.
110     *
111     * @param base the base
112     * @param repoId the repository id
113     * @param releaseRepos the backing release repositories
114     * @param snapshotRepos the backing snapshot repositories
115     * @param executor an executor
116     * @param reporter the reporter
117     * @throws Exception the exception
118     */
119    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
120        "PMD.AvoidDuplicateLiterals" })
121    public CompositeMavenRepository(File base, String repoId,
122            List<MavenBackingRepository> releaseRepos,
123            List<MavenBackingRepository> snapshotRepos, Executor executor,
124            Reporter reporter)
125            throws Exception {
126        bndMavenRepo = new MavenRepository(base, repoId, releaseRepos,
127            snapshotRepos, executor, reporter);
128        this.reporter = reporter;
129        modelResolver = new BndModelResolver(bndMavenRepo, reporter);
130
131        // Create the maven model builder. This code is ridiculous,
132        // but using maven's CDI pulls in an even more ridiculous
133        // number of libraries. To get an idea why, read
134        // https://lairdnelson.wordpress.com/2017/03/06/maven-and-the-project-formerly-known-as-aether/
135        DefaultModelBuilder builder = new DefaultModelBuilder();
136        builder.setProfileSelector(new DefaultProfileSelector());
137        DefaultModelProcessor processor = new DefaultModelProcessor();
138        processor.setModelLocator(new DefaultModelLocator());
139        processor.setModelReader(new DefaultModelReader());
140        builder.setModelProcessor(processor);
141        builder.setModelValidator(new DefaultModelValidator());
142        DefaultSuperPomProvider pomProvider = new DefaultSuperPomProvider();
143        pomProvider.setModelProcessor(processor);
144        DefaultSuperPomProvider superPomProvider
145            = new DefaultSuperPomProvider();
146        superPomProvider.setModelProcessor(processor);
147        builder.setSuperPomProvider(superPomProvider);
148        builder.setModelNormalizer(new DefaultModelNormalizer());
149        builder.setInheritanceAssembler(new DefaultInheritanceAssembler());
150        StringSearchModelInterpolator modelInterpolator
151            = new StringSearchModelInterpolator();
152        DefaultPathTranslator pathTranslator = new DefaultPathTranslator();
153        modelInterpolator.setPathTranslator(pathTranslator);
154        DefaultUrlNormalizer urlNormalizer = new DefaultUrlNormalizer();
155        modelInterpolator.setUrlNormalizer(urlNormalizer);
156        builder.setModelInterpolator(modelInterpolator);
157        DefaultModelUrlNormalizer modelUrlNormalizer
158            = new DefaultModelUrlNormalizer();
159        modelUrlNormalizer.setUrlNormalizer(urlNormalizer);
160        builder.setModelUrlNormalizer(modelUrlNormalizer);
161        DefaultModelPathTranslator modelPathTranslator
162            = new DefaultModelPathTranslator();
163        modelPathTranslator.setPathTranslator(pathTranslator);
164        builder.setModelPathTranslator(modelPathTranslator);
165        builder
166            .setPluginManagementInjector(new DefaultPluginManagementInjector());
167        builder.setDependencyManagementInjector(
168            new DefaultDependencyManagementInjector());
169        builder.setDependencyManagementImporter(
170            new DefaultDependencyManagementImporter());
171        builder.setProfileInjector(new DefaultProfileInjector());
172        modelBuilder = builder;
173    }

219    /**
220     * Retrieves the file from a remote repository into the repositories 
221     * local cache directory if it doesn't exist yet.
222     *
223     * @param archive The archive to fetch
224     * @return the file or null if not found
225     * @throws IOException Signals that an I/O exception has occurred.
226     */
227    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
228        "PMD.AvoidInstanceofChecksInCatchClause",
229        "PMD.AvoidDuplicateLiterals" })
230    public Promise<File> retrieve(Archive archive) throws IOException {
231        try {
232            return bndMavenRepo.get(archive);
233        } catch (Exception e) {
234            if (e instanceof InvocationTargetException
235                && ((InvocationTargetException) e)
236                    .getTargetException() instanceof IOException) {
237                // Should be the only possible exception here
238                throw (IOException) ((InvocationTargetException) e)
239                    .getTargetException();
240            }
241            throw new UndeclaredThrowableException(e);
242        }
243    }

287    /**
288     * Wrapper for {@link MavenBackingRepository#getRevisions(Program, List)}
289     * that returns the result instead of accumulating it and maps the
290     * (too general) exception.
291     *
292     * @param mbr the backing repository
293     * @param program the program
294     * @return the revisions
295     * @throws IOException Signals that an I/O exception has occurred.
296     */
297    @SuppressWarnings({ "PMD.LinguisticNaming", "PMD.AvoidRethrowingException",
298        "PMD.AvoidCatchingGenericException",
299        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidDuplicateLiterals" })
300    private static List<Revision> revisionsFrom(MavenBackingRepository mbr,
301            Program program) throws IOException {
302        List<Revision> result = new ArrayList<>();
303        try {
304            mbr.getRevisions(program, result);
305        } catch (IOException e) {
306            // Should be the only reason.
307            throw e;
308        } catch (Exception e) {
309            throw new RuntimeException(
310                "Cannot evaluate revisions for " + program + " from " + mbr, e);
311        }
312        return result;
313    }

315    /**
316     * Get the bound revisions of the given program.
317     *
318     * @param program the program
319     * @return the list
320     * @throws IOException Signals that an I/O exception has occurred.
321     */
322    public Stream<BoundRevision> findRevisions(Program program)
323            throws IOException {
324        return rethrow(IOException.class,
325            () -> programCache.computeIfAbsent(program,
326                prg -> unthrow(() -> backingAsStream())
327                    .flatMap(mbr -> unthrow(
328                        () -> revisionsFrom(mbr, program).stream()
329                            .map(revision -> new BoundRevision(mbr, revision))))
330                    .collect(Collectors.toList()))).stream();
331    }

333    /**
334     * Converts a {@link Revision} to a {@link BoundRevision}.
335     *
336     * @param revision the revision
337     * @return the bound revision
338     * @throws IOException Signals that an I/O exception has occurred.
339     */
340    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
341    public Optional<BoundRevision> find(Revision revision)
342            throws IOException {
343        return findRevisions(revision.program)
344            .filter(rev -> rev.unbound().equals(revision)).findFirst();
345    }

347    /**
348     * Converts an {@link Archive} to a {@link BoundArchive}.
349     *
350     * @param archive the archive
351     * @return the bound archive
352     * @throws IOException Signals that an I/O exception has occurred.
353     */
354    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
355    public Optional<BoundArchive> find(Archive archive)
356            throws IOException {
357        return rethrow(IOException.class,
358            () -> find(archive.revision).flatMap(rev -> unthrow(() -> Optional
359                .ofNullable(
360                    resolve(rev, archive.extension, archive.classifier)))));
361    }

363    /**
364     * Converts a {@link Program} and a version, which
365     * may be a range, to a {@link BoundRevision}.
366     *
367     * @param program the program
368     * @param version the version
369     * @return the bound revision
370     * @throws IOException Signals that an I/O exception has occurred.
371     */
372    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
373    public Optional<BoundRevision> find(Program program,
374            MavenVersionSpecification version) throws IOException {
375        if (version instanceof MavenVersion) {
376            return find(((MavenVersion) version).of(program));
377        }
378        MavenVersionRange range = (MavenVersionRange) version;
379        return findRevisions(program)
380            .filter(rev -> range.includes(rev.version()))
381            .max(Comparator.naturalOrder());
382    }

384    /**
385     * Get a model of the specified revision. Dependency versions
386     * remain unresolved, i.e. when specified as a range, the range
387     * is preserved.
388     *
389     * @param revision the archive
390     * @return the dependencies
391     * @throws MavenResourceException the maven resource exception
392     */
393    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
394        "PMD.AvoidInstanceofChecksInCatchClause", "PMD.PreserveStackTrace",
395        "PMD.AvoidRethrowingException" })
396    public Model model(Revision revision) throws MavenResourceException {
397        return rethrow(MavenResourceException.class,
398            () -> modelCache.computeIfAbsent(
399                revision, key -> unthrow(() -> readModel(key))));
400    }
401
402    private Model readModel(Revision revision) throws MavenResourceException {
403        DefaultModelBuildingRequest request = new DefaultModelBuildingRequest();
404        try {
405            ModelSource modelSource = modelResolver.resolveModel(revision.group,
406                revision.artifact, revision.version.toString());
407            request.setModelResolver(modelResolver)
408                .setModelSource(modelSource)
409                .setValidationLevel(
410                    ModelBuildingRequest.VALIDATION_LEVEL_MINIMAL)
411                .setTwoPhaseBuilding(false);
412            ModelBuildingResult result = modelBuilder.build(request);
413            return result.getEffectiveModel();
414        } catch (UnresolvableModelException | ModelBuildingException e) {
415            throw new MavenResourceException(e);
416        }
417    }

419    /**
420     * Gets the resolved archive. "Resolving" an archive means finding
421     * the binaries with the specified extension and classifier belonging
422     * to the given version. While this can be done with straight forward
423     * name mapping for releases, snapshots have a timestamp that has to
424     * be looked up in the backing repository.
425     *
426     * @param revision the revision
427     * @param extension the extension
428     * @param classifier the classifier
429     * @return the resolved archive
430     * @throws IOException Signals that an I/O exception has occurred.
431     */
432    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
433        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidRethrowingException" })
434    public BoundArchive resolve(BoundRevision revision,
435            String extension, String classifier) throws IOException {
436        if (!revision.isSnapshot()) {
437            return revision.archive(extension, classifier);
438        }
439        try {
440            MavenVersion version = MavenVersion.from(revision
441                .mavenBackingRepository().getVersion(revision.unbound()));
442            return revision.archive(version, extension, classifier);
443        } catch (IOException e) {
444            throw e;
445        } catch (Exception e) {
446            throw new RuntimeException(e);
447        }
448    }

450    @SuppressWarnings("PMD.AvoidCatchingGenericException")
451    protected void refreshSnapshot(BoundArchive archive) {
452        File metaFile = bndMavenRepo.toLocalFile(
453            archive.getRevision()
454                .metadata(archive.mavenBackingRepository().getId()));
455        RevisionMetadata metaData;
456        try {
457            metaData = MetadataParser.parseRevisionMetadata(metaFile);
458        } catch (Exception e) {
459            reporter.exception(e, "Problem accessing %s.", archive);
460            return;
461        }
462        File archiveFile = bndMavenRepo.toLocalFile(archive);
463        if (archiveFile.lastModified() < metaData.lastUpdated) {
464            archiveFile.delete();
465        }
466        File pomFile = bndMavenRepo.toLocalFile(archive.getPomArchive());
467        if (pomFile.lastModified() < metaData.lastUpdated) {
468            pomFile.delete();
469        }
470    }
