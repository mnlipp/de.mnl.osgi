078/**
079 * Provides a composite {@link IMavenRepo} view on several 
080 * {@link MavenBackingRepository} instances.
081 * <P>
082 * The class extends {@link MavenRepository} which lacks some
083 * required functionality. Besides, this class has a more
084 * appropriate name.
085 */
086@SuppressWarnings({ "PMD.DataflowAnomalyAnalysis", "deprecation" })
087public class CompositeMavenRepository implements Closeable {
088
089    public static final Pattern COORDS_SPLITTER = Pattern.compile("\\s*;\\s*");
090    private final MavenRepository bndMavenRepo;
091    private final Reporter reporter;
092    private final Map<Program, List<BoundRevision>> programCache
093        = new ConcurrentHashMap<>();
094    private final Map<Revision, Model> modelCache
095        = new ConcurrentHashMap<>();
096    private final BndModelResolver modelResolver;
097    private final ModelBuilder modelBuilder;

106    /**
107     * Instantiates a new composite maven repository.
108     *
109     * @param base the base
110     * @param repoId the repository id
111     * @param releaseRepos the backing release repositories
112     * @param snapshotRepos the backing snapshot repositories
113     * @param executor an executor
114     * @param reporter the reporter
115     * @throws Exception the exception
116     */
117    @SuppressWarnings({ "PMD.SignatureDeclareThrowsException",
118        "PMD.AvoidDuplicateLiterals" })
119    public CompositeMavenRepository(File base, String repoId,
120            List<MavenBackingRepository> releaseRepos,
121            List<MavenBackingRepository> snapshotRepos, Executor executor,
122            Reporter reporter)
123            throws Exception {
124        bndMavenRepo = new MavenRepository(base, repoId, releaseRepos,
125            snapshotRepos, executor, reporter);
126        this.reporter = reporter;
127        modelResolver = new BndModelResolver(bndMavenRepo, reporter);
128
129        // Create the maven model builder. This code is ridiculous,
130        // but using maven's CDI pulls in an even more ridiculous
131        // number of libraries.
132        DefaultModelBuilder builder = new DefaultModelBuilder();
133        builder.setProfileSelector(new DefaultProfileSelector());
134        DefaultModelProcessor processor = new DefaultModelProcessor();
135        processor.setModelLocator(new DefaultModelLocator());
136        processor.setModelReader(new DefaultModelReader());
137        builder.setModelProcessor(processor);
138        builder.setModelValidator(new DefaultModelValidator());
139        DefaultSuperPomProvider pomProvider = new DefaultSuperPomProvider();
140        pomProvider.setModelProcessor(processor);
141        DefaultSuperPomProvider superPomProvider
142            = new DefaultSuperPomProvider();
143        superPomProvider.setModelProcessor(processor);
144        builder.setSuperPomProvider(superPomProvider);
145        builder.setModelNormalizer(new DefaultModelNormalizer());
146        builder.setInheritanceAssembler(new DefaultInheritanceAssembler());
147        StringSearchModelInterpolator modelInterpolator
148            = new StringSearchModelInterpolator();
149        DefaultPathTranslator pathTranslator = new DefaultPathTranslator();
150        modelInterpolator.setPathTranslator(pathTranslator);
151        DefaultUrlNormalizer urlNormalizer = new DefaultUrlNormalizer();
152        modelInterpolator.setUrlNormalizer(urlNormalizer);
153        builder.setModelInterpolator(modelInterpolator);
154        DefaultModelUrlNormalizer modelUrlNormalizer
155            = new DefaultModelUrlNormalizer();
156        modelUrlNormalizer.setUrlNormalizer(urlNormalizer);
157        builder.setModelUrlNormalizer(modelUrlNormalizer);
158        DefaultModelPathTranslator modelPathTranslator
159            = new DefaultModelPathTranslator();
160        modelPathTranslator.setPathTranslator(pathTranslator);
161        builder.setModelPathTranslator(modelPathTranslator);
162        builder
163            .setPluginManagementInjector(new DefaultPluginManagementInjector());
164        builder.setDependencyManagementInjector(
165            new DefaultDependencyManagementInjector());
166        builder.setDependencyManagementImporter(
167            new DefaultDependencyManagementImporter());
168        modelBuilder = builder;
169    }

205    /**
206     * Returns all repositories as a stream.
207     *
208     * @return the repositories as stream
209     */
210    public Stream<MavenBackingRepository> backingAsStream() {
211        return Stream.concat(bndMavenRepo.getReleaseRepositories().stream(),
212            bndMavenRepo.getSnapshotRepositories().stream()).distinct();
213    }

215    /**
216     * Retrieves the file from a remote repository into the repositories 
217     * local cache directory if it doesn't exist yet.
218     *
219     * @param archive The archive to fetch
220     * @return the file or null if not found
221     * @throws IOException Signals that an I/O exception has occurred.
222     */
223    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
224        "PMD.AvoidInstanceofChecksInCatchClause",
225        "PMD.AvoidDuplicateLiterals" })
226    public Promise<File> retrieve(Archive archive) throws IOException {
227        try {
228            return bndMavenRepo.get(archive);
229        } catch (Exception e) {
230            if (e instanceof InvocationTargetException
231                && ((InvocationTargetException) e)
232                    .getTargetException() instanceof IOException) {
233                // Should be the only possible exception here
234                throw (IOException) ((InvocationTargetException) e)
235                    .getTargetException();
236            }
237            throw new UndeclaredThrowableException(e);
238        }
239    }

283    /**
284     * Wrapper for {@link MavenBackingRepository#getRevisions(Program, List)}
285     * that returns the result instead of accumulating it and maps the
286     * (too general) exception.
287     *
288     * @param mbr the backing repository
289     * @param program the program
290     * @return the revisions
291     * @throws IOException Signals that an I/O exception has occurred.
292     */
293    @SuppressWarnings({ "PMD.LinguisticNaming", "PMD.AvoidRethrowingException",
294        "PMD.AvoidCatchingGenericException",
295        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidDuplicateLiterals" })
296    private static List<Revision> revisionsFrom(MavenBackingRepository mbr,
297            Program program) throws IOException {
298        List<Revision> result = new ArrayList<>();
299        try {
300            mbr.getRevisions(program, result);
301        } catch (IOException e) {
302            // Should be the only reason.
303            throw e;
304        } catch (Exception e) {
305            throw new RuntimeException(e);
306        }
307        return result;
308    }

310    /**
311     * Get the bound revisions of the given program.
312     *
313     * @param program the program
314     * @return the list
315     * @throws IOException Signals that an I/O exception has occurred.
316     */
317    public Stream<BoundRevision> findRevisions(Program program)
318            throws IOException {
319        return rethrow(IOException.class,
320            () -> programCache.computeIfAbsent(program,
321                prg -> unthrow(() -> backingAsStream())
322                    .flatMap(mbr -> unthrow(
323                        () -> revisionsFrom(mbr, program).stream()
324                            .map(revision -> new BoundRevision(mbr, revision))))
325                    .collect(Collectors.toList()))).stream();
326    }

328    /**
329     * Converts a {@link Revision} to a {@link BoundRevision}.
330     *
331     * @param revision the revision
332     * @return the bound revision
333     * @throws IOException Signals that an I/O exception has occurred.
334     */
335    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
336    public Optional<BoundRevision> find(Revision revision)
337            throws IOException {
338        return findRevisions(revision.program)
339            .filter(rev -> rev.unbound().equals(revision)).findFirst();
340    }
341
342    /**
343     * Converts a {@link Program} and a version, which
344     * may be a range, to a {@link BoundRevision}.
345     *
346     * @param program the program
347     * @param version the version
348     * @return the bound revision
349     * @throws IOException Signals that an I/O exception has occurred.
350     */
351    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
352    public Optional<BoundRevision> find(Program program,
353            MavenVersionSpecification version) throws IOException {
354        if (version instanceof MavenVersion) {
355            return find(((MavenVersion) version).of(program));
356        }
357        MavenVersionRange range = (MavenVersionRange) version;
358        return findRevisions(program)
359            .filter(rev -> range.includes(rev.version()))
360            .max(Comparator.naturalOrder());
361    }

363    /**
364     * Get a model of the specified revision. Dependency versions
365     * remain unresolved, i.e. when specified as a range, the range
366     * is preserved.
367     *
368     * @param revision the archive
369     * @return the dependencies
370     * @throws MavenResourceException the maven resource exception
371     */
372    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
373        "PMD.AvoidInstanceofChecksInCatchClause", "PMD.PreserveStackTrace",
374        "PMD.AvoidRethrowingException" })
375    public Model model(Revision revision) throws MavenResourceException {
376        return rethrow(MavenResourceException.class,
377            () -> modelCache.computeIfAbsent(
378                revision, key -> unthrow(() -> readModel(key))));
379    }
380
381    private Model readModel(Revision revision) throws MavenResourceException {
382        DefaultModelBuildingRequest request = new DefaultModelBuildingRequest();
383        try {
384            ModelSource modelSource = modelResolver.resolveModel(revision.group,
385                revision.artifact, revision.version.toString());
386            request.setModelResolver(modelResolver)
387                .setModelSource(modelSource)
388                .setValidationLevel(
389                    ModelBuildingRequest.VALIDATION_LEVEL_MINIMAL)
390                .setTwoPhaseBuilding(false);
391            ModelBuildingResult result = modelBuilder.build(request);
392            return result.getEffectiveModel();
393        } catch (UnresolvableModelException | ModelBuildingException e) {
394            throw new MavenResourceException(e);
395        }
396    }

398    /**
399     * Gets the resolved archive. "Resolving" an archive means finding
400     * the binaries with the specified extension and classifier belonging
401     * to the given version. While this can be done with straight forward
402     * name mapping for releases, snapshots have a timestamp that has to
403     * be looked up in the backing repository.
404     *
405     * @param revision the revision
406     * @param extension the extension
407     * @param classifier the classifier
408     * @return the resolved archive
409     * @throws IOException Signals that an I/O exception has occurred.
410     */
411    @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
412        "PMD.AvoidThrowingRawExceptionTypes", "PMD.AvoidRethrowingException" })
413    public BoundArchive resolve(BoundRevision revision,
414            String extension, String classifier) throws IOException {
415        if (!revision.isSnapshot()) {
416            return revision.archive(extension, classifier);
417        }
418        try {
419            MavenVersion version = MavenVersion.from(revision
420                .mavenBackingRepository().getVersion(revision.unbound()));
421            return revision.archive(version, extension, classifier);
422        } catch (IOException e) {
423            throw e;
424        } catch (Exception e) {
425            throw new RuntimeException(e);
426        }
427    }
428
429    @SuppressWarnings("PMD.AvoidCatchingGenericException")
430    protected void refreshSnapshot(BoundArchive archive) {
431        File metaFile = bndMavenRepo.toLocalFile(
432            archive.getRevision()
433                .metadata(archive.mavenBackingRepository().getId()));
434        RevisionMetadata metaData;
435        try {
436            metaData = MetadataParser.parseRevisionMetadata(metaFile);
437        } catch (Exception e) {
438            reporter.exception(e, "Problem accessing %s.", archive);
439            return;
440        }
441        File archiveFile = bndMavenRepo.toLocalFile(archive);
442        if (archiveFile.lastModified() < metaData.lastUpdated) {
443            archiveFile.delete();
444        }
445        File pomFile = bndMavenRepo.toLocalFile(archive.getPomArchive());
446        if (pomFile.lastModified() < metaData.lastUpdated) {
447            pomFile.delete();
448        }
449    }
