034/**
035 * Provides a model of an artifact version which can be used as a maven version.
036 * <P>
037 * The maven <a href="https://maven.apache.org/pom.html">POM reference</a> does
038 * not define a format for versions. This is presumably intentional as it allows
039 * artifacts with arbitrary versioning schemes to be referenced in a POM.
040 * <P>
041 * Maven tooling, on the other hand side, defines a rather <a href=
042 * "http://books.sonatype.com/mvnref-book/reference/pom-relationships-sect-pom-syntax.html#pom-reationships-sect-versions">restrictive
043 * version number pattern</a> for maven projects. Non-compliant version numbers
044 * are parsed as qualifier-only versions.
045 * <P>
046 * The parsing methods of this class make an attempt to interpret a version
047 * number as a
048 * &lt;major&gt;/&lt;minor&gt;/&lt;micro/incremental&gt;/&lt;qualifier&gt;
049 * pattern, even if it does not match the restrictive maven project version
050 * number pattern. The string representation of an instance of this class is
051 * always the original, unparsed (or "literal") representation because due to
052 * the permissive parsing algorithm used, the original representation cannot
053 * faithfully be reconstructed from the parsed components.
054 * <P>
055 * Contrary to bnd's {@link aQute.bnd.version.MavenVersion}, this 
056 * implementation inherits from {@link ArtifactVersion}, i.e. from the
057 * version as implemented by maven.
058 */
059@SuppressWarnings("PMD.GodClass")
060public class MavenVersion extends MavenVersionSpecification
061        implements ArtifactVersion {
062
063    /** The usual format of a verson string. */
064    public static final String VERSION_STRING
065        = "(\\d{1,15})(\\.(\\d{1,9})(\\.(\\d{1,9}))?)?([-\\.]?([-_\\.\\da-zA-Z]+))?";
066
067    /** The usual format for a snapshot timestamp. */
068    public static final SimpleDateFormat SNAPSHOT_TIMESTAMP
069        = new SimpleDateFormat("yyyyMMdd.HHmmss", Locale.ROOT);
070
071    static {
072        SNAPSHOT_TIMESTAMP.setTimeZone(TimeZone.getTimeZone("UTC"));
073    }
074
075    private static final Pattern VERSION = Pattern.compile(VERSION_STRING);
076
077    /** The snapshot identifier. */
078    public static final String SNAPSHOT = "SNAPSHOT";
079
080    /** The Constant HIGHEST. */
081    public static final MavenVersion HIGHEST
082        = new MavenVersion(Version.HIGHEST);
083
084    /** The Constant LOWEST. */
085    public static final MavenVersion LOWEST = new MavenVersion("0");
086
087    // Used as "container" for the components of the maven version number
088    private final Version version;
089    // Some maven versions are too odd to be restored after parsing, keep
090    // original.
091    private final String literal;
092    // Used for comparison, cached for efficiency.
093    private final ComparableVersion comparable;
094
095    private final boolean snapshot;
096
097    /**
098     * Creates a new instance. The maven version is parsed by an instance
099     * of {@link DefaultArtifactVersion}. The parsing is thus fully
100     * maven compliant.
101     *
102     * @param maven the version
103     */
104    public MavenVersion(String maven) {
105        this.literal = maven;
106        this.comparable = new ComparableVersion(literal);
107        DefaultArtifactVersion artVer = new DefaultArtifactVersion(maven);
108        this.version
109            = new Version(artVer.getMajorVersion(), artVer.getMinorVersion(),
110                artVer.getIncrementalVersion(), artVer.getQualifier());
111        this.snapshot = version.isSnapshot();
112    }
113
114    /**
115     * Creates a new maven version from an osgi version. The version components
116     * are copied, the string representation is built from the components as
117     * "&lt;major&gt;.&lt;minor&gt;.&lt;micro&gt;.&lt;qualifier&gt;"
118     *
119     * @param osgiVersion the osgi version
120     */
121    public MavenVersion(Version osgiVersion) {
122        this.version = osgiVersion;
123        StringBuilder qual = new StringBuilder("");
124        if (this.version.getQualifier() != null) {
125            qual.append('-');
126            qual.append(this.version.getQualifier());
127        }
128        this.literal = osgiVersion.getWithoutQualifier().toString() + qual;
129        this.comparable = new ComparableVersion(literal);
130        this.snapshot = osgiVersion.isSnapshot();
131    }
132
133    /**
134     * Creates a new maven version from an osgi version and an unparsed 
135     * literal. The version components are copied, the literal is used
136     * as string representation, the snapshot property is taken from
137     * the argument.
138     *
139     * @param osgiVersion the osgi version
140     * @param literal the literal
141     * @param isSnapshot whether it is a snapshot version
142     */
143    public MavenVersion(Version osgiVersion, String literal,
144            boolean isSnapshot) {
145        this.literal = literal;
146        this.comparable = new ComparableVersion(literal);
147        this.version = osgiVersion;
148        this.snapshot = isSnapshot;
149    }
150
151    /**
152     * Parses the string as a maven version, but allows a dot as separator
153     * before the qualifier.
154     * <P>
155     * Leading sequences of digits followed by a dot or dash are converted to
156     * the major, minor and incremental version components. A dash or a dot that
157     * is not followed by a digit or the third dot is interpreted as the start
158     * of the qualifier.
159     * <P>
160     * In particular, version numbers such as "1.2.3.4.5" are parsed as major=1,
161     * minor=2, incremental=3 and qualifier="4.5". This is closer to the
162     * (assumed) semantics of such a version number than the parsing implemented
163     * in maven tooling, which interprets the complete version as a qualifier in
164     * such cases.
165     *
166     * @param versionStr the version string
167     * @return the maven version
168     * @throws IllegalArgumentException if the version cannot be parsed
169     */
170    public static final MavenVersion parseString(String versionStr) {
171        if (versionStr == null) {
172            versionStr = "0";
173        } else {
174            versionStr = versionStr.trim();
175            if (versionStr.isEmpty()) {
176                versionStr = "0";
177            }
178        }
179        Matcher matcher = VERSION.matcher(versionStr);
180        if (!matcher.matches()) {
181            throw new IllegalArgumentException(
182                "Invalid syntax for version: " + versionStr);
183        }
184        int major = Integer.parseInt(matcher.group(1));
185        @SuppressWarnings("PMD.ConfusingTernary")
186        int minor = (matcher.group(3) != null)
187            ? Integer.parseInt(matcher.group(3))
188            : 0;
189        @SuppressWarnings("PMD.ConfusingTernary")
190        int micro = (matcher.group(5) != null)
191            ? Integer.parseInt(matcher.group(5))
192            : 0;
193        String qualifier = matcher.group(7);
194        Version version = new Version(major, minor, micro, qualifier);
195        return new MavenVersion(version);
196    }

198    /**
199     * Similar to {@link #parseString(String)}, but returns {@code null} if the
200     * version cannot be parsed.
201     * 
202     * @param versionStr the version string
203     * @return the maven version
204     */
205    @SuppressWarnings("PMD.AvoidCatchingGenericException")
206    public static final MavenVersion parseMavenString(String versionStr) {
207        try {
208            return new MavenVersion(versionStr);
209        } catch (Exception e) {
210            return null;
211        }
212    }

376    /**
377     * Compares maven version numbers according to the rules defined in the
378     * <a href=
379     * "https://maven.apache.org/pom.html#Version_Order_Specification">POM
380     * reference</a>.
381     *
382     * @param other the other
383     * @return the int
384     */
385    @Override
386    public int compareTo(ArtifactVersion other) {
387        if (other instanceof MavenVersion) {
388            return comparable.compareTo(((MavenVersion) other).comparable);
389        }
390        return comparable.compareTo(new ComparableVersion(other.toString()));
391    }

403    /*
404     * (non-Javadoc)
405     * 
406     * @see java.lang.Object#hashCode()
407     */
408    @Override
409    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
410    public int hashCode() {
411        @SuppressWarnings("PMD.AvoidFinalLocalVariable")
412        final int prime = 31;
413        int result = 1;
414        result = prime * result + ((literal == null) ? 0 : literal.hashCode());
415        return result;
416    }

418    /*
419     * (non-Javadoc)
420     * 
421     * @see java.lang.Object#equals(java.lang.Object)
422     */
423    @Override
424    public boolean equals(Object obj) {
425        if (this == obj) {
426            return true;
427        }
428        if (obj == null) {
429            return false;
430        }
431        if (getClass() != obj.getClass()) {
432            return false;
433        }
434        MavenVersion other = (MavenVersion) obj;
435        return literal.equals(other.literal);
436    }

474    /**
475     * To snapshot.
476     *
477     * @param dateStamp the date stamp
478     * @return the maven version
479     */
480    public MavenVersion toSnapshot(String dateStamp) {
481        // -SNAPSHOT == 9 characters
482        String clean = literal.substring(0, literal.length() - 9);
483        String result = clean + "-" + dateStamp;
484
485        return new MavenVersion(result);
486    }

488    /**
489     * Validate.
490     *
491     * @param value the value
492     * @return the string
493     */
494    public static String validate(String value) {
495        if (value == null) {
496            return "Version is null";
497        }
498        if (!VERSION.matcher(value).matches()) {
499            return "Not a version";
500        }
501        return null;
502    }

504    /**
505     * To date stamp.
506     *
507     * @param epoch the epoch
508     * @return the string
509     */
510    public static String toDateStamp(long epoch) {
511        String datestamp;
512        synchronized (SNAPSHOT_TIMESTAMP) {
513            datestamp = SNAPSHOT_TIMESTAMP.format(new Date(epoch));
514        }
515        return datestamp;
516
517    }
518
519    /**
520     * To date stamp.
521     *
522     * @param epoch the epoch
523     * @param build the build
524     * @return the string
525     */
526    public static String toDateStamp(long epoch, String build) {
527        StringBuilder str = new StringBuilder(toDateStamp(epoch));
528        if (build != null) {
529            str.append('-');
530            str.append(build);
531        }
532        return str.toString();
533    }
