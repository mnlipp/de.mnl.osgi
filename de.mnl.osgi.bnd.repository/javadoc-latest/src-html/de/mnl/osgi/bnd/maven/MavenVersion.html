034/**
035 * Provides a model of an artifact version which can be used as a maven version.
036 * <P>
037 * The maven <a href="https://maven.apache.org/pom.html">POM reference</a> does
038 * not define a format for versions. This is presumably intentional as it allows
039 * artifacts with arbitrary versioning schemes to be referenced in a POM.
040 * <P>
041 * Maven tooling, on the other hand side, defines a rather <a href=
042 * "http://books.sonatype.com/mvnref-book/reference/pom-relationships-sect-pom-syntax.html#pom-reationships-sect-versions">restrictive
043 * version number pattern</a> for maven projects. Non-compliant version numbers
044 * are parsed as qualifier-only versions.
045 * <P>
046 * The parsing methods of this class make an attempt to interpret a version
047 * number as a
048 * &lt;major&gt;/&lt;minor&gt;/&lt;micro/incremental&gt;/&lt;qualifier&gt;
049 * pattern, even if it does not match the restrictive maven project version
050 * number pattern. The string representation of an instance of this class is
051 * always the original, unparsed (or "literal") representation because due to
052 * the permissive parsing algorithm used, the original representation cannot
053 * faithfully be reconstructed from the parsed components.
054 * <P>
055 * Contrary to bnd's {@link aQute.bnd.version.MavenVersion}, this 
056 * implementation inherits from {@link ArtifactVersion}, i.e. from the
057 * version as implemented by maven.
058 */
059@SuppressWarnings("PMD.GodClass")
060public class MavenVersion implements ArtifactVersion {
061
062    /** The usual format of a verson string. */
063    public static final String VERSION_STRING
064        = "(\\d{1,15})(\\.(\\d{1,9})(\\.(\\d{1,9}))?)?([-\\.]?([-_\\.\\da-zA-Z]+))?";
065
066    /** The usual format for a snapshot timestamp. */
067    public static final SimpleDateFormat SNAPSHOT_TIMESTAMP
068        = new SimpleDateFormat("yyyyMMdd.HHmmss", Locale.ROOT);
069
070    static {
071        SNAPSHOT_TIMESTAMP.setTimeZone(TimeZone.getTimeZone("UTC"));
072    }
073
074    private static final Pattern VERSION = Pattern.compile(VERSION_STRING);
075
076    /** The snapshot identifier. */
077    public static final String SNAPSHOT = "SNAPSHOT";
078
079    /** The Constant HIGHEST. */
080    public static final MavenVersion HIGHEST
081        = new MavenVersion(Version.HIGHEST);
082
083    /** The Constant LOWEST. */
084    public static final MavenVersion LOWEST = new MavenVersion("0");
085
086    // Used as "container" for the components of the maven version number
087    private final Version version;
088    // Some maven versions are too odd to be restored after parsing, keep
089    // original.
090    private final String literal;
091    // Used for comparison, cached for efficiency.
092    private final ComparableVersion comparable;
093
094    private final boolean snapshot;
095
096    /**
097     * Creates a new instance. The maven version is parsed by an instance
098     * of {@link DefaultArtifactVersion}. The parsing is thus fully
099     * maven compliant.
100     *
101     * @param maven the version
102     */
103    public MavenVersion(String maven) {
104        this.literal = maven;
105        this.comparable = new ComparableVersion(literal);
106        DefaultArtifactVersion artVer = new DefaultArtifactVersion(maven);
107        this.version
108            = new Version(artVer.getMajorVersion(), artVer.getMinorVersion(),
109                artVer.getIncrementalVersion(), artVer.getQualifier());
110        this.snapshot = version.isSnapshot();
111    }
112
113    /**
114     * Creates a new maven version from an osgi version. The version components
115     * are copied, the string representation is built from the components as
116     * "&lt;major&gt;.&lt;minor&gt;.&lt;micro&gt;.&lt;qualifier&gt;"
117     *
118     * @param osgiVersion the osgi version
119     */
120    public MavenVersion(Version osgiVersion) {
121        this.version = osgiVersion;
122        StringBuilder qual = new StringBuilder("");
123        if (this.version.getQualifier() != null) {
124            qual.append('-');
125            qual.append(this.version.getQualifier());
126        }
127        this.literal = osgiVersion.getWithoutQualifier().toString() + qual;
128        this.comparable = new ComparableVersion(literal);
129        this.snapshot = osgiVersion.isSnapshot();
130    }
131
132    /**
133     * Creates a new maven version from an osgi version and an unparsed 
134     * literal. The version components are copied, the literal is used
135     * as string representation, the snapshot property is taken from
136     * the argument.
137     *
138     * @param osgiVersion the osgi version
139     * @param literal the literal
140     * @param isSnapshot whether it is a snapshot version
141     */
142    public MavenVersion(Version osgiVersion, String literal,
143            boolean isSnapshot) {
144        this.literal = literal;
145        this.comparable = new ComparableVersion(literal);
146        this.version = osgiVersion;
147        this.snapshot = isSnapshot;
148    }
149
150    /**
151     * Parses the string as a maven version, but allows a dot as separator
152     * before the qualifier.
153     * <P>
154     * Leading sequences of digits followed by a dot or dash are converted to
155     * the major, minor and incremental version components. A dash or a dot that
156     * is not followed by a digit or the third dot is interpreted as the start
157     * of the qualifier.
158     * <P>
159     * In particular, version numbers such as "1.2.3.4.5" are parsed as major=1,
160     * minor=2, incremental=3 and qualifier="4.5". This is closer to the
161     * (assumed) semantics of such a version number than the parsing implemented
162     * in maven tooling, which interprets the complete version as a qualifier in
163     * such cases.
164     *
165     * @param versionStr the version string
166     * @return the maven version
167     * @throws IllegalArgumentException if the version cannot be parsed
168     */
169    public static final MavenVersion parseString(String versionStr) {
170        if (versionStr == null) {
171            versionStr = "0";
172        } else {
173            versionStr = versionStr.trim();
174            if (versionStr.isEmpty()) {
175                versionStr = "0";
176            }
177        }
178        Matcher matcher = VERSION.matcher(versionStr);
179        if (!matcher.matches()) {
180            throw new IllegalArgumentException(
181                "Invalid syntax for version: " + versionStr);
182        }
183        int major = Integer.parseInt(matcher.group(1));
184        @SuppressWarnings("PMD.ConfusingTernary")
185        int minor = (matcher.group(3) != null)
186            ? Integer.parseInt(matcher.group(3))
187            : 0;
188        @SuppressWarnings("PMD.ConfusingTernary")
189        int micro = (matcher.group(5) != null)
190            ? Integer.parseInt(matcher.group(5))
191            : 0;
192        String qualifier = matcher.group(7);
193        Version version = new Version(major, minor, micro, qualifier);
194        return new MavenVersion(version);
195    }

197    /**
198     * Similar to {@link #parseString(String)}, but returns {@code null} if the
199     * version cannot be parsed.
200     * 
201     * @param versionStr the version string
202     * @return the maven version
203     */
204    @SuppressWarnings("PMD.AvoidCatchingGenericException")
205    public static final MavenVersion parseMavenString(String versionStr) {
206        try {
207            return new MavenVersion(versionStr);
208        } catch (Exception e) {
209            return null;
210        }
211    }

363    /**
364     * Compares maven version numbers according to the rules defined in the
365     * <a href=
366     * "https://maven.apache.org/pom.html#Version_Order_Specification">POM
367     * reference</a>.
368     *
369     * @param other the other
370     * @return the int
371     */
372    @Override
373    public int compareTo(ArtifactVersion other) {
374        if (other instanceof MavenVersion) {
375            return comparable.compareTo(((MavenVersion) other).comparable);
376        }
377        return comparable.compareTo(new ComparableVersion(other.toString()));
378    }

390    /*
391     * (non-Javadoc)
392     * 
393     * @see java.lang.Object#hashCode()
394     */
395    @Override
396    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
397    public int hashCode() {
398        @SuppressWarnings("PMD.AvoidFinalLocalVariable")
399        final int prime = 31;
400        int result = 1;
401        result = prime * result + ((literal == null) ? 0 : literal.hashCode());
402        return result;
403    }

405    /*
406     * (non-Javadoc)
407     * 
408     * @see java.lang.Object#equals(java.lang.Object)
409     */
410    @Override
411    public boolean equals(Object obj) {
412        if (this == obj) {
413            return true;
414        }
415        if (obj == null) {
416            return false;
417        }
418        if (getClass() != obj.getClass()) {
419            return false;
420        }
421        MavenVersion other = (MavenVersion) obj;
422        return literal.equals(other.literal);
423    }

461    /**
462     * To snapshot.
463     *
464     * @param dateStamp the date stamp
465     * @return the maven version
466     */
467    public MavenVersion toSnapshot(String dateStamp) {
468        // -SNAPSHOT == 9 characters
469        String clean = literal.substring(0, literal.length() - 9);
470        String result = clean + "-" + dateStamp;
471
472        return new MavenVersion(result);
473    }

475    /**
476     * Validate.
477     *
478     * @param value the value
479     * @return the string
480     */
481    public static String validate(String value) {
482        if (value == null) {
483            return "Version is null";
484        }
485        if (!VERSION.matcher(value).matches()) {
486            return "Not a version";
487        }
488        return null;
489    }

491    /**
492     * To date stamp.
493     *
494     * @param epoch the epoch
495     * @return the string
496     */
497    public static String toDateStamp(long epoch) {
498        String datestamp;
499        synchronized (SNAPSHOT_TIMESTAMP) {
500            datestamp = SNAPSHOT_TIMESTAMP.format(new Date(epoch));
501        }
502        return datestamp;
503
504    }
505
506    /**
507     * To date stamp.
508     *
509     * @param epoch the epoch
510     * @param build the build
511     * @return the string
512     */
513    public static String toDateStamp(long epoch, String build) {
514        StringBuilder str = new StringBuilder(toDateStamp(epoch));
515        if (build != null) {
516            str.append('-');
517            str.append(build);
518        }
519        return str.toString();
520    }
