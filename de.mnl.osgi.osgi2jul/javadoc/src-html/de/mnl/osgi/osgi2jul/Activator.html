017package de.mnl.osgi.osgi2jul;
018
019import java.util.Collections;
020import java.util.List;
021import java.util.concurrent.CountDownLatch;
022
023import org.osgi.framework.BundleActivator;
024import org.osgi.framework.BundleContext;
025import org.osgi.framework.ServiceReference;
026import org.osgi.service.log.LogEntry;
027import org.osgi.service.log.LogReaderService;
028import org.osgi.util.tracker.ServiceTracker;
029
030/**
031 * This class provides the activator for this service. It registers
032 * (respectively unregisters) the {@link LogWriter} as LogListener 
033 * for for all log reader services and forwards any already existing 
034 * log entries to it. 
035 */
036public class Activator implements BundleActivator {
037
038    /** This tracker holds all log reader services. */
039    private ServiceTracker<LogReaderService, LogReaderService> serviceTracker
040        = null;
041
042    /**
043     * Open the log service tracker. The tracker is customized to attach the 
044     * LogWriter to all registered log reader services (and detach it on 
045     * un-registration, of course). Already existing log entries 
046     * are forwarded to the LogWriter as well. No provisions have been
047     * taken to avoid the duplicate output that can occur if a message
048     * is logged between registering the LogWriter and forwarding
049     * stored log entries.
050     */
051    @Override
052    public void start(BundleContext context) throws Exception {
053        serviceTracker = new ServiceTracker<LogReaderService, LogReaderService>(
054            context, LogReaderService.class, null) {
055
056            @Override
057            public LogReaderService addingService(
058                    ServiceReference<LogReaderService> reference) {
059                LogReaderService service = super.addingService(reference);
060                CountDownLatch enabled = new CountDownLatch(1);
061                service.addLogListener(new LogWriter(enabled));
062                List<LogEntry> entries = Collections.list(service.getLog());
063                Collections.reverse(entries);
064                LogWriter historyWriter = new LogWriter(new CountDownLatch(0));
065                for (LogEntry entry : entries) {
066                    historyWriter.logged(entry);
067                }
068                enabled.countDown();
069                return service;
070            }
071
072            @Override
073            public void removedService(
074                    ServiceReference<LogReaderService> reference,
075                    LogReaderService service) {
076                super.removedService(reference, service);
077            }
078
079        };
080        serviceTracker.open();
081    }
082
083    @Override
084    public void stop(BundleContext context) throws Exception {
085        serviceTracker.close();
086        serviceTracker = null;
087    }
088
089}
