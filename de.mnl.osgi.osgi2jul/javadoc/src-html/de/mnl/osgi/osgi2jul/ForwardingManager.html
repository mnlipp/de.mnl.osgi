017package de.mnl.osgi.osgi2jul;
018
019import java.security.AccessController;
020import java.security.PrivilegedActionException;
021import java.security.PrivilegedExceptionAction;
022import java.text.MessageFormat;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.Collections;
026import java.util.List;
027import java.util.Optional;
028import java.util.concurrent.CountDownLatch;
029import java.util.logging.Handler;
030import java.util.logging.Level;
031import java.util.logging.LogRecord;
032import java.util.regex.Matcher;
033import java.util.regex.Pattern;
034
035import org.osgi.framework.Bundle;
036import org.osgi.framework.BundleActivator;
037import org.osgi.framework.BundleContext;
038import org.osgi.framework.ServiceReference;
039import org.osgi.service.log.LogEntry;
040import org.osgi.service.log.LogReaderService;
041import org.osgi.util.tracker.ServiceTracker;
042
043/**
044 * This class provides the activator for this service. It registers
045 * (respectively unregisters) the {@link LogWriter} as LogListener 
046 * for for all log reader services and forwards any already existing 
047 * log entries to it. 
048 */
049@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
050public class ForwardingManager implements BundleActivator {
051
052    private static final Pattern HANDLER_DEF = Pattern.compile(
053        "(?:(?<bundle>[^:]+):)?(?<class>[^\\[]+)(?:\\[(?<id>\\d+)\\])?");
054
055    /** This tracker holds all log reader services. */
056    private ServiceTracker<LogReaderService, LogReaderService> logReaderTracker;
057    private final List<HandlerConfig> handlers = new ArrayList<>();
058
059    /**
060     * Open the log service tracker. The tracker is customized to attach a 
061     * {@link LogWriter} to all registered log reader services (and detach 
062     * it on un-registration, of course). Already existing log entries 
063     * are forwarded to the {@link LogWriter} as well. No provisions have been
064     * taken to avoid the duplicate output that can occur if a message
065     * is logged between registering the {@link LogWriter} and forwarding
066     * stored log entries.
067     */
068    @Override
069    public void start(BundleContext context) throws Exception {
070        createHandlers(context);
071
072        logReaderTracker
073            = new ServiceTracker<LogReaderService, LogReaderService>(
074                context, LogReaderService.class, null) {
075
076                @Override
077                public LogReaderService addingService(
078                        ServiceReference<LogReaderService> reference) {
079                    LogReaderService service = super.addingService(reference);
080                    CountDownLatch enabled = new CountDownLatch(1);
081                    service.addLogListener(
082                        new LogWriter(ForwardingManager.this, enabled));
083                    List<LogEntry> entries = Collections.list(service.getLog());
084                    Collections.reverse(entries);
085                    LogWriter historyWriter = new LogWriter(
086                        ForwardingManager.this, new CountDownLatch(0));
087                    for (LogEntry entry : entries) {
088                        historyWriter.logged(entry);
089                    }
090                    enabled.countDown();
091                    return service;
092                }
093
094                @Override
095                public void removedService(
096                        ServiceReference<LogReaderService> reference,
097                        LogReaderService service) {
098                    super.removedService(reference, service);
099                }
100
101            };
102        logReaderTracker.open();
103    }
104
105    @SuppressWarnings({ "PMD.SystemPrintln", "PMD.DataflowAnomalyAnalysis" })
106    private void createHandlers(BundleContext context) {
107        String handlerClasses = Optional.ofNullable(
108            context.getProperty(
109                ForwardingManager.class.getPackage().getName() + ".handlers"))
110            .orElse("java.util.logging.ConsoleHandler");
111        Arrays.stream(handlerClasses.split(",")).map(String::trim)
112            .forEach(name -> {
113                Matcher parts = HANDLER_DEF.matcher(name);
114                if (!parts.matches()) {
115                    System.err.println("Handler definition \""
116                        + name + "\" is invalid.");
117                    return;
118                }
119                Handler handler = null;
120                try {
121                    if (parts.group("bundle") == null) {
122                        // Only class name
123                        handler = handlerFromClassName(parts.group("class"));
124                    } else {
125                        handler = handlerFromBundledClass(context,
126                            parts.group("bundle"), parts.group("class"));
127                    }
128                } catch (Exception e) { // NOPMD
129                    System.err.println("Can't load or configure log handler \""
130                        + name + "\": " + e.getMessage());
131                    return;
132                }
133                handlers.add(createHandlerConfig(context, parts, handler));
134            });
135    }
136
137    private Handler handlerFromBundledClass(BundleContext context,
138            String bundleName, String className)
139            throws InstantiationException, IllegalAccessException,
140            ClassNotFoundException {
141        for (Bundle bundle : context.getBundles()) {
142            if (bundle.getSymbolicName().equals(bundleName)) {
143                return (Handler) bundle.loadClass(className).newInstance();
144            }
145        }
146        throw new ClassNotFoundException("Class " + className + " not found "
147            + "in bundle " + bundleName);
148    }
149
150    private Handler handlerFromClassName(String name)
151            throws PrivilegedActionException {
152        Handler handler;
153        handler = AccessController
154            .doPrivileged(new PrivilegedExceptionAction<Handler>() {
155                @Override
156                public Handler run() throws Exception {
157                    Class<?> hdlrCls;
158                    hdlrCls = ClassLoader.getSystemClassLoader()
159                        .loadClass(name);
160                    return (Handler) hdlrCls.newInstance();
161                }
162            });
163        return handler;
164    }
165
166    @SuppressWarnings("PMD.SystemPrintln")
167    private HandlerConfig createHandlerConfig(BundleContext context,
168            Matcher parts, Handler handler) {
169        String levelName = null;
170        MessageFormat outputFormat = null;
171        if (parts.group("id") != null) {
172            String handlerPrefix
173                = ForwardingManager.class.getPackage().getName()
174                    + ".handler" + (parts.group("id") == null ? ""
175                        : "." + parts.group("id"));
176            levelName = context.getProperty(handlerPrefix + ".level");
177            String format = context.getProperty(handlerPrefix + ".format");
178            if (format != null) {
179                try {
180                    outputFormat = new MessageFormat(format);
181                } catch (IllegalArgumentException e) {
182                    System.err.println("Illegal format: \"" + format + "\"");
183                }
184            }
185        }
186        if (levelName != null) {
187            Level level = Level.parse(levelName);
188            handler.setLevel(level);
189        }
190        return new HandlerConfig(handler, outputFormat);
191    }
192
193    @Override
194    public void stop(BundleContext context) throws Exception {
195        logReaderTracker.close();
196        logReaderTracker = null;
197        handlers.clear();
198    }
199
200    /**
201     * Send the record to all handlers. 
202     *
203     * @param entry the original OSGi log entry
204     * @param record the prepared JUL record
205     */
206    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
207    public void publish(LogEntry entry, LogRecord record) {
208        Object[] xtraArgs = null;
209        for (HandlerConfig cfg : handlers) {
210            if (cfg.getOutputFormat() == null) {
211                cfg.getHandler().publish(record);
212                continue;
213            }
214            String oldMessage = record.getMessage();
215            try {
216                if (xtraArgs == null) {
217                    xtraArgs = new Object[] {
218                        oldMessage,
219                        entry.getBundle().getSymbolicName(),
220                        entry.getBundle().getHeaders().get("Bundle-Name"),
221                        entry.getBundle().getVersion().toString(),
222                        entry.getThreadInfo() };
223                }
224                record.setMessage(cfg.getOutputFormat().format(xtraArgs,
225                    new StringBuffer(), null).toString());
226                cfg.getHandler().publish(record);
227            } finally {
228                record.setMessage(oldMessage);
229            }
230        }
231    }
