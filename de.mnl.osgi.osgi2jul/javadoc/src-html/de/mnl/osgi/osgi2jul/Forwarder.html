017package de.mnl.osgi.osgi2jul;
018
019import java.security.AccessController;
020import java.security.PrivilegedActionException;
021import java.security.PrivilegedExceptionAction;
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.List;
025import java.util.Optional;
026import java.util.concurrent.CountDownLatch;
027import java.util.logging.Handler;
028import java.util.logging.Level;
029import java.util.logging.LogManager;
030import java.util.logging.LogRecord;
031
032import org.osgi.framework.Bundle;
033import org.osgi.framework.BundleActivator;
034import org.osgi.framework.BundleContext;
035import org.osgi.framework.ServiceReference;
036import org.osgi.service.log.LogEntry;
037import org.osgi.service.log.LogReaderService;
038import org.osgi.util.tracker.ServiceTracker;
039
040/**
041 * This class provides the activator for this service. It registers
042 * (respectively unregisters) the {@link LogWriter} as LogListener 
043 * for for all log reader services and forwards any already existing 
044 * log entries to it. 
045 */
046public class Forwarder implements BundleActivator {
047
048    /** This tracker holds all log reader services. */
049    private ServiceTracker<LogReaderService, LogReaderService> serviceTracker
050        = null;
051    private List<Handler> handlers = new ArrayList<>();
052
053    /**
054     * Open the log service tracker. The tracker is customized to attach a 
055     * {@link LogWriter} to all registered log reader services (and detach 
056     * it on un-registration, of course). Already existing log entries 
057     * are forwarded to the {@link LogWriter} as well. No provisions have been
058     * taken to avoid the duplicate output that can occur if a message
059     * is logged between registering the {@link LogWriter} and forwarding
060     * stored log entries.
061     */
062    @Override
063    public void start(BundleContext context) throws Exception {
064        createHandlers(context);
065
066        serviceTracker = new ServiceTracker<LogReaderService, LogReaderService>(
067            context, LogReaderService.class, null) {
068
069            @Override
070            public LogReaderService addingService(
071                    ServiceReference<LogReaderService> reference) {
072                LogReaderService service = super.addingService(reference);
073                CountDownLatch enabled = new CountDownLatch(1);
074                service.addLogListener(new LogWriter(Forwarder.this, enabled));
075                List<LogEntry> entries = Collections.list(service.getLog());
076                Collections.reverse(entries);
077                LogWriter historyWriter
078                    = new LogWriter(Forwarder.this, new CountDownLatch(0));
079                for (LogEntry entry : entries) {
080                    historyWriter.logged(entry);
081                }
082                enabled.countDown();
083                return service;
084            }
085
086            @Override
087            public void removedService(
088                    ServiceReference<LogReaderService> reference,
089                    LogReaderService service) {
090                super.removedService(reference, service);
091            }
092
093        };
094        serviceTracker.open();
095    }
096
097    private void createHandlers(BundleContext context) {
098        String handlerClasses = Optional.ofNullable(
099            context.getProperty(
100                Forwarder.class.getPackage().getName() + ".handlers"))
101            .orElse("java.util.logging.ConsoleHandler");
102        for (String name : handlerClasses.split(",")) {
103            name = name.trim();
104            String[] bundleAndClass = name.split(":");
105            Handler handler = null;
106            try {
107                if (bundleAndClass.length == 1) {
108                    // Only class name
109                    handler = handlerFromClassName(name);
110                } else {
111                    handler = handlerFromBundledClass(context,
112                        bundleAndClass[0], bundleAndClass[1]);
113                }
114            } catch (Exception e) {
115                System.err.println("Can't load or configure log handler \""
116                    + name + "\": " + e.getMessage());
117                e.printStackTrace();
118            }
119            String levelName
120                = LogManager.getLogManager().getProperty(name + ".level");
121            if (levelName != null) {
122                Level level = Level.parse(levelName);
123                handler.setLevel(level);
124            }
125            handlers.add(handler);
126        }
127    }
128
129    private Handler handlerFromBundledClass(BundleContext context,
130            String bundleName, String className)
131            throws InstantiationException, IllegalAccessException,
132            ClassNotFoundException {
133        for (Bundle bundle : context.getBundles()) {
134            if (bundle.getSymbolicName().equals(bundleName)) {
135                return (Handler) bundle.loadClass(className).newInstance();
136            }
137        }
138        throw new ClassNotFoundException("Class " + className + " not found "
139            + "in bundle " + bundleName);
140    }
141
142    private Handler handlerFromClassName(String name)
143            throws PrivilegedActionException {
144        Handler handler;
145        handler = AccessController
146            .doPrivileged(new PrivilegedExceptionAction<Handler>() {
147                @Override
148                public Handler run() throws Exception {
149                    Class<?> hdlrCls;
150                    hdlrCls = ClassLoader.getSystemClassLoader()
151                        .loadClass(name);
152                    return (Handler) hdlrCls.newInstance();
153                }
154            });
155        return handler;
156    }
157
158    @Override
159    public void stop(BundleContext context) throws Exception {
160        serviceTracker.close();
161        serviceTracker = null;
162        handlers.clear();
163    }
164
165    public void publish(LogRecord record) {
166        for (Handler handler : handlers) {
167            handler.publish(record);
168        }
169    }
170
171}
