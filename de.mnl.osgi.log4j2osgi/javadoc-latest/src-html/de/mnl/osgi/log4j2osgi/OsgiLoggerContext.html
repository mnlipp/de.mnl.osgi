017package de.mnl.osgi.log4j2osgi;
018
019import de.mnl.osgi.lf4osgi.core.LoggerGroup;
020import org.apache.logging.log4j.message.MessageFactory;
021import org.apache.logging.log4j.spi.ExtendedLogger;
022import org.apache.logging.log4j.spi.LoggerContext;
023import org.apache.logging.log4j.spi.LoggerRegistry;
024import org.osgi.framework.Bundle;
025
026public class OsgiLoggerContext implements LoggerGroup, LoggerContext {
027
028    private final Bundle bundle;
029    private final LoggerRegistry<OsgiLogger> loggerRegistry;
030
031    public OsgiLoggerContext(Bundle bundle) {
032        this.bundle = bundle;
033        loggerRegistry = new LoggerRegistry<>();
034    }
035
036    @Override
037    public Bundle getBundle() {
038        return bundle;
039    }
040
041    @Override
042    public OsgiLogger getLogger(final String name) {
043        if (!loggerRegistry.hasLogger(name)) {
044            loggerRegistry.putIfAbsent(name, null,
045                new OsgiLogger(this, name));
046        }
047        return loggerRegistry.getLogger(name);
048    }
049
050    @Override
051    public ExtendedLogger getLogger(final String name,
052            final MessageFactory messageFactory) {
053        if (!loggerRegistry.hasLogger(name, messageFactory)) {
054            loggerRegistry.putIfAbsent(name, messageFactory,
055                new OsgiLogger(this, name, messageFactory));
056        }
057        return loggerRegistry.getLogger(name, messageFactory);
058    }
059
060    @Override
061    public boolean hasLogger(final String name) {
062        return loggerRegistry.hasLogger(name);
063    }
064
065    @Override
066    public boolean hasLogger(final String name,
067            final MessageFactory messageFactory) {
068        return loggerRegistry.hasLogger(name, messageFactory);
069    }
070
071    @Override
072    public boolean hasLogger(final String name,
073            final Class<? extends MessageFactory> messageFactoryClass) {
074        return loggerRegistry.hasLogger(name, messageFactoryClass);
075    }
076
077    @Override
078    public Object getExternalContext() {
079        return null;
080    }
081
082}
