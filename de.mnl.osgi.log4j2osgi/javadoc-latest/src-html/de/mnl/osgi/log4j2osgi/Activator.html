001package de.mnl.osgi.log4j2osgi;
002
003import java.util.Dictionary;
004import java.util.Hashtable;
005import org.apache.logging.log4j.spi.Provider;
006import org.osgi.framework.BundleActivator;
007import org.osgi.framework.BundleContext;
008import org.osgi.framework.ServiceRegistration;
009
010public class Activator implements BundleActivator {
011
012    private ServiceRegistration<Provider> serviceRegistration;
013
014    @Override
015    public void start(BundleContext context) throws Exception {
016        Dictionary<String, Object> props = new Hashtable<>();
017        props.put("APIVersion", "2.6.0");
018        serviceRegistration = context.registerService(Provider.class,
019            new OsgiProvider(), props);
020    }
021
022    @Override
023    public void stop(BundleContext context) throws Exception {
024        serviceRegistration.unregister();
025    }
026
027}
