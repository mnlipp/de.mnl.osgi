017package de.mnl.osgi.log4j2osgi;
018
019import de.mnl.osgi.lf4osgi.core.LoggerCatalogue;
020import java.net.URI;
021import org.apache.logging.log4j.spi.LoggerContext;
022import org.apache.logging.log4j.spi.LoggerContextFactory;
023import org.osgi.framework.Bundle;
024
025public class OsgiLoggerContextFactory extends LoggerCatalogue<OsgiLoggerContext>
026        implements LoggerContextFactory {
027
028    public OsgiLoggerContextFactory() {
029        super(b -> new OsgiLoggerContext(b));
030    }
031
032    @Override
033    public OsgiLoggerContext getContext(final String fqcn,
034            final ClassLoader loader, final Object externalContext,
035            final boolean currentContext) {
036        Bundle bundle = LoggerCatalogue.findBundle(fqcn).get();
037        return getLoggerGoup(bundle);
038    }
039
040    @Override
041    public OsgiLoggerContext getContext(final String fqcn,
042            final ClassLoader loader,
043            final Object externalContext, final boolean currentContext,
044            final URI configLocation, final String name) {
045        return getContext(fqcn, loader, externalContext, currentContext);
046    }
047
048    @Override
049    public void removeContext(final LoggerContext ignored) {
050    }
051}
