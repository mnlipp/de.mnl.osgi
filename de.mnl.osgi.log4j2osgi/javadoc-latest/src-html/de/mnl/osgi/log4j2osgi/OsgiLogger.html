017package de.mnl.osgi.log4j2osgi;
018
019import de.mnl.osgi.lf4osgi.core.LoggerFacade;
020import de.mnl.osgi.lf4osgi.core.LoggerFacadeManager;
021import org.apache.logging.log4j.Level;
022import org.apache.logging.log4j.Marker;
023import org.apache.logging.log4j.message.Message;
024import org.apache.logging.log4j.message.MessageFactory;
025import org.apache.logging.log4j.spi.AbstractLogger;
026import org.osgi.service.log.LoggerFactory;
027
028public class OsgiLogger extends AbstractLogger implements LoggerFacade {
029    private static final long serialVersionUID = 1L;
030    private OsgiLoggerContext context;
031    private org.osgi.service.log.Logger delegee;
032
033    public OsgiLogger(OsgiLoggerContext context, final String name,
034            final MessageFactory messageFactory) {
035        super(name, messageFactory);
036        this.context = context;
037        LoggerFacadeManager.registerFacade(this);
038    }
039
040    public OsgiLogger(OsgiLoggerContext context, final String name) {
041        super(name);
042        this.context = context;
043        LoggerFacadeManager.registerFacade(this);
044    }
045
046    @Override
047    public void loggerFactoryUpdated(LoggerFactory factory) {
048        delegee = factory.getLogger(context.getBundle(), name,
049            org.osgi.service.log.Logger.class);
050    }
051
052    @Override
053    public Level getLevel() {
054        if (delegee.isTraceEnabled()) {
055            return Level.TRACE;
056        }
057        if (delegee.isDebugEnabled()) {
058            return Level.DEBUG;
059        }
060        if (delegee.isInfoEnabled()) {
061            return Level.INFO;
062        }
063        if (delegee.isWarnEnabled()) {
064            return Level.WARN;
065        }
066        if (delegee.isErrorEnabled()) {
067            return Level.ERROR;
068        }
069        // Option: throw new IllegalStateException("Unknown SLF4JLevel");
070        // Option: return Level.ALL;
071        return Level.OFF;
072    }
073
074    @Override
075    public boolean isEnabled(final Level level, final Marker marker,
076            final Message data, final Throwable thr) {
077        return isEnabledFor(level, marker);
078    }
079
080    @Override
081    public boolean isEnabled(final Level level, final Marker marker,
082            final CharSequence data, final Throwable thr) {
083        return isEnabledFor(level, marker);
084    }
085
086    @Override
087    public boolean isEnabled(final Level level, final Marker marker,
088            final Object data, final Throwable thr) {
089        return isEnabledFor(level, marker);
090    }
091
092    @Override
093    public boolean isEnabled(final Level level, final Marker marker,
094            final String data) {
095        return isEnabledFor(level, marker);
096    }
097
098    @Override
099    public boolean isEnabled(final Level level, final Marker marker,
100            final String data, final Object... p1) {
101        return isEnabledFor(level, marker);
102    }
103
104    @Override
105    public boolean isEnabled(final Level level, final Marker marker,
106            final String message, final Object p0) {
107        return isEnabledFor(level, marker);
108    }
109
110    @Override
111    public boolean isEnabled(final Level level, final Marker marker,
112            final String message, final Object p0,
113            final Object p1) {
114        return isEnabledFor(level, marker);
115    }
116
117    @Override
118    public boolean isEnabled(final Level level, final Marker marker,
119            final String message, final Object p0,
120            final Object p1, final Object p2) {
121        return isEnabledFor(level, marker);
122    }
123
124    @Override
125    public boolean isEnabled(final Level level, final Marker marker,
126            final String message, final Object p0,
127            final Object p1, final Object p2, final Object p3) {
128        return isEnabledFor(level, marker);
129    }
130
131    @Override
132    public boolean isEnabled(final Level level, final Marker marker,
133            final String message, final Object p0,
134            final Object p1, final Object p2, final Object p3,
135            final Object p4) {
136        return isEnabledFor(level, marker);
137    }
138
139    @Override
140    public boolean isEnabled(final Level level, final Marker marker,
141            final String message, final Object p0,
142            final Object p1, final Object p2, final Object p3,
143            final Object p4, final Object p5) {
144        return isEnabledFor(level, marker);
145    }

148    public boolean isEnabled(final Level level, final Marker marker,
149            final String message, final Object p0,
150            final Object p1, final Object p2, final Object p3,
151            final Object p4, final Object p5, final Object p6) {
152        return isEnabledFor(level, marker);

156    public boolean isEnabled(final Level level, final Marker marker,
157            final String message, final Object p0,

160            final Object p7) {

165    public boolean isEnabled(final Level level, final Marker marker,

173    @Override
174    public boolean isEnabled(final Level level, final Marker marker,
175            final String message, final Object p0,
176            final Object p1, final Object p2, final Object p3,
177            final Object p4, final Object p5, final Object p6,
178            final Object p7, final Object p8, final Object p9) {
179        return isEnabledFor(level, marker);
180    }
181
182    @Override
183    public boolean isEnabled(final Level level, final Marker marker,
184            final String data, final Throwable thr) {
185        return isEnabledFor(level, marker);
186    }
187
188    private boolean isEnabledFor(final Level level, final Marker marker) {
189        switch (level.getStandardLevel()) {
190        case DEBUG:
191            return delegee.isDebugEnabled();
192        case TRACE:
193            return delegee.isTraceEnabled();
194        case INFO:
195            return delegee.isInfoEnabled();
196        case WARN:
197            return delegee.isWarnEnabled();
198        case ERROR:
199            return delegee.isErrorEnabled();
200        default:
201            return delegee.isErrorEnabled();
202        }
203    }
204
205    @Override
206    public void logMessage(final String fqcn, final Level level,
207            final Marker marker, final Message message, final Throwable thr) {
208        switch (level.getStandardLevel()) {
209        case TRACE:
210            delegee.trace(message.getFormattedMessage(),
211                message.getParameters(), thr);
212            break;
213        case DEBUG:
214            delegee.debug(message.getFormattedMessage(),
215                message.getParameters(), thr);
216            break;
217        case INFO:
218            delegee.info(message.getFormattedMessage(),
219                message.getParameters(), thr);
220            break;
221        case WARN:
222            delegee.warn(message.getFormattedMessage(),
223                message.getParameters(), thr);
224            break;
225        case ERROR:
226            delegee.error(message.getFormattedMessage(),
227                message.getParameters(), thr);
228            break;
229        default:
230            delegee.error(message.getFormattedMessage(),
231                message.getParameters(), thr);
232            break;
233
234        }
235    }
