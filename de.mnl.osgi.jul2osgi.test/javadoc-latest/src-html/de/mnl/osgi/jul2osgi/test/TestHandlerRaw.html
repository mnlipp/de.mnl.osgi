001package de.mnl.osgi.jul2osgi.test;
002
003import java.util.ArrayList;
004import java.util.List;
005import java.util.logging.Handler;
006import java.util.logging.LogRecord;
007
008public class TestHandlerRaw extends Handler {
009
010    public static List<LogRecord> records = new ArrayList<>();
011
012    public TestHandlerRaw() {
013    }
014
015    @Override
016    public void publish(LogRecord record) {
017        LogRecord copy = new LogRecord(record.getLevel(), record.getMessage());
018        records.add(copy);
019    }
020
021    @Override
022    public void flush() {
023    }
024
025    @Override
026    public void close() throws SecurityException {
027    }
028
029}
