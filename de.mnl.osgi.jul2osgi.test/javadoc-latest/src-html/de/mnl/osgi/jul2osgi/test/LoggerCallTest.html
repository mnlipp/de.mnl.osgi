026@ExtendWith(MockitoExtension.class)
027public class LoggerCallTest {
028
029    protected static final Logger logger = Logger.getLogger("Logger Call Test");
030
031    private final BundleContext context
032        = FrameworkUtil.getBundle(getClass()).getBundleContext();
033
034    private <T> T getService(Class<T> clazz) throws InterruptedException {
035        ServiceTracker<T, T> st = new ServiceTracker<>(context, clazz, null);
036        st.open();
037        return st.waitForService(1000);
038    }
039
040    @Test
041    public void testExample() throws InterruptedException {
042        CountDownLatch recordLatch = new CountDownLatch(1);
043        LogReaderService logService = getService(LogReaderService.class);
044        String expectedMessage = "de.mnl.osgi.jul2osgi.test.LoggerCallTest"
045            + ".testExample: Calling Logger from Test.";
046        logService.addLogListener(new LogListener() {
047            @Override
048            public void logged(LogEntry entry) {
049                if (entry.getLoggerName().equals("Logger Call Test")
050                    && entry.getMessage().equals(expectedMessage)
051                    && entry.getBundle().getSymbolicName()
052                        .equals("de.mnl.osgi.jul2osgi.test")) {
053                    recordLatch.countDown();
054                }
055            }
056        });
057        logger.log(Level.INFO, "Calling Logger from {0}.", "Test");
058        assertTrue(recordLatch.await(1000, TimeUnit.MILLISECONDS));
059        String handled = null;
060        for (LogRecord record : TestHandlerRaw.records) {
061            if (record.getMessage().equals(expectedMessage)) {
062                handled = record.getMessage();
063            }
064        }
065        assertNotNull(handled);
066        handled = null;
067        for (LogRecord record : TestHandlerExt.records) {
068            if (record.getMessage().startsWith(expectedMessage
069                + " [de.mnl.osgi.jul2osgi.test|OSGi2JUL Test Bundle|")) {
070                handled = record.getMessage();
071            }
072        }
073        assertNotNull(handled);
074    }
075
076    @Test
077    public void testBuffered() throws InterruptedException, BundleException {
078        Bundle osgiLoggerFactory = context
079            .getServiceReference(org.osgi.service.log.LoggerFactory.class)
080            .getBundle();
081        assertNotNull(osgiLoggerFactory);
082        try {
083            osgiLoggerFactory.stop();
084            logger.log(Level.INFO, "Calling Logger {0}", "Info");
085            logger.log(Level.WARNING, "Calling Logger Warn");
086            logger.log(Level.SEVERE, "Calling Logger Error", new Throwable());
087            osgiLoggerFactory.start();
088            try (ServiceCollector<LogReaderService,
089                    LogReaderService> logReaderProvider
090                        = new ServiceCollector<>(
091                            context, LogReaderService.class)) {
092                logReaderProvider.open();
093                LogReaderService logReader
094                    = logReaderProvider.waitForService(1000).get();
095                CountDownLatch latch = new CountDownLatch(3);
096                for (LogEntry entry : Collections.list(logReader.getLog())) {
097                    if (entry.getMessage().indexOf("Calling Logger Info") >= 0
098                        && entry.getLogLevel() == LogLevel.INFO) {
099                        latch.countDown();
100                    }
101                    if (entry.getMessage().indexOf("Calling Logger Warn") >= 0
102                        && entry.getLogLevel() == LogLevel.WARN) {
103                        latch.countDown();
104                    }
105                    if (entry.getMessage().indexOf("Calling Logger Error") >= 0
106                        && entry.getLogLevel() == LogLevel.ERROR) {
107                        assertNotNull(entry.getException());
108                        latch.countDown();
109                    }
110                }
111                latch.await(1000, TimeUnit.MILLISECONDS);
112                assertEquals(0, latch.getCount());
113            }
114        } finally {
115            osgiLoggerFactory.start();
116        }
117    }
