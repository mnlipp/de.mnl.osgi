001package de.mnl.osgi.jul2osgi.test;
002
003import static org.junit.Assert.assertTrue;
004
005import java.util.concurrent.CountDownLatch;
006import java.util.concurrent.TimeUnit;
007import java.util.logging.Level;
008import java.util.logging.Logger;
009
010import org.junit.Test;
011import org.junit.runner.RunWith;
012import org.mockito.junit.MockitoJUnitRunner;
013import org.osgi.framework.BundleContext;
014import org.osgi.framework.FrameworkUtil;
015import org.osgi.service.log.LogEntry;
016import org.osgi.service.log.LogListener;
017import org.osgi.service.log.LogReaderService;
018import org.osgi.util.tracker.ServiceTracker;
019
020@RunWith(MockitoJUnitRunner.class)
021public class LoggerCallTest {
022
023    protected static final Logger logger 
024        = Logger.getLogger("Logger Call Test");
025    
026        private final BundleContext context = FrameworkUtil
027                        .getBundle(LoggerCallTest.class).getBundleContext();
028
029        private <T> T getService(Class<T> clazz) throws InterruptedException {
030                ServiceTracker<T,T> st = new ServiceTracker<>(context, clazz, null);
031                st.open();
032                return st.waitForService(1000);
033        }
034        
035        @Test
036        public void testExample() throws InterruptedException {
037                CountDownLatch recordLatch = new CountDownLatch(1);
038                LogReaderService logService = getService(LogReaderService.class);
039                logService.addLogListener(new LogListener() {
040                        @Override
041                        public void logged(LogEntry entry) {
042                                if (entry.getLoggerName().equals("Logger Call Test")
043                                                && entry.getMessage().equals(
044                                                                "de.mnl.osgi.jul2osgi.test.LoggerCallTest"
045                                                                + ".testExample: Calling Logger from Test.")) {
046                                        recordLatch.countDown();
047                                }
048                        }
049                });
050                logger.log(Level.INFO, "Calling Logger from {0}.", "Test");
051                assertTrue(recordLatch.await(1000, TimeUnit.MILLISECONDS));
052        }
053}
