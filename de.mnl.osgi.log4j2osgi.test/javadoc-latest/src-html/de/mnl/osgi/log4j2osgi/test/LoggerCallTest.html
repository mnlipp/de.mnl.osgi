023import org.osgi.service.log.LogReaderService;
024
025@ExtendWith(MockitoExtension.class)
026public class LoggerCallTest {
027
028    protected static final Logger logger
029        = LogManager.getLogger("Logger Call Test");
030
031    private final BundleContext context
032        = FrameworkUtil.getBundle(LoggerCallTest.class).getBundleContext();
033
034    @Test
035    public void testBasicUsage() throws InterruptedException {
036        AtomicBoolean gotIt = new AtomicBoolean();
037        try (ServiceCollector<LogReaderService,
038                LogReaderService> logReaderProvider
039                    = new ServiceCollector<>(context,
040                        LogReaderService.class)) {
041            logReaderProvider.open();
042            LogReaderService logReader
043                = logReaderProvider.waitForService(1000).get();
044            CountDownLatch latch = new CountDownLatch(1);
045            logReader.addLogListener(new LogListener() {
046                @Override
047                public void logged(LogEntry entry) {
048                    if (entry.getMessage()
049                        .startsWith("Calling Logger from Test.")) {
050                        assertEquals("de.mnl.osgi.log4j2osgi.test",
051                            entry.getBundle().getSymbolicName());
052                        gotIt.set(true);
053                        latch.countDown();
054                    }
055                }
056            });
057            logger.info("Calling Logger from {}.", "Test");
058            latch.await(1000, TimeUnit.MILLISECONDS);
059            assertTrue(gotIt.get());
060        }
061    }
062
063    @Test
064    public void testBuffered() throws InterruptedException, BundleException {
065        Bundle osgiLoggerFactory = context
066            .getServiceReference(org.osgi.service.log.LoggerFactory.class)
067            .getBundle();
068        assertNotNull(osgiLoggerFactory);
069        try {
070            osgiLoggerFactory.stop();
071            logger.info("Calling Logger {}", "Info");
072            logger.warn(() -> "Calling Logger " + "Warn");
073            logger.error("Calling Logger Error", new Throwable());
074            osgiLoggerFactory.start();
075            try (ServiceCollector<LogReaderService,
076                    LogReaderService> logReaderProvider
077                        = new ServiceCollector<>(
078                            context, LogReaderService.class)) {
079                logReaderProvider.open();
080                LogReaderService logReader
081                    = logReaderProvider.waitForService(1000).get();
082                CountDownLatch latch = new CountDownLatch(3);
083                for (LogEntry entry : Collections.list(logReader.getLog())) {
084                    if (entry.getMessage().startsWith("Calling Logger Info")
085                        && entry.getLogLevel() == LogLevel.INFO) {
086                        latch.countDown();
087                    }
088                    if (entry.getMessage().startsWith("Calling Logger Warn")
089                        && entry.getLogLevel() == LogLevel.WARN) {
090                        latch.countDown();
091                    }
092                    if (entry.getMessage().startsWith("Calling Logger Error")
093                        && entry.getLogLevel() == LogLevel.ERROR
094                        && entry.getException() != null) {
095                        latch.countDown();
096                    }
097                }
098                latch.await(1000, TimeUnit.MILLISECONDS);
099                assertEquals(0, latch.getCount());
100            }
101        } finally {
102            osgiLoggerFactory.start();
103        }
104    }
