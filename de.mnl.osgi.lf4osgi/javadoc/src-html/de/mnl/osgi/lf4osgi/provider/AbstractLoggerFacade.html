017package de.mnl.osgi.lf4osgi.provider;
018
019import de.mnl.osgi.lf4osgi.Lf4OsgiLogger;
020
021import java.lang.ref.WeakReference;
022import java.util.Map;
023import java.util.Optional;
024import java.util.concurrent.ConcurrentHashMap;
025
026import org.osgi.framework.Bundle;
027import org.osgi.framework.FrameworkUtil;
028import org.osgi.service.log.LoggerFactory;
029
030/**
031 * The base class for OSGi logger facades.
032 */
033public abstract class AbstractLoggerFacade {
034
035    private static final ContextHelper CTX_HLPR = new ContextHelper();
036    @SuppressWarnings("PMD.FieldNamingConventions")
037    private static final Map<Class<?>, WeakReference<Bundle>> bundles
038        = new ConcurrentHashMap<>();
039    private final String name;
040    private final Bundle bundle;
041
042    /**
043     * Find the class that creates the logger. This is done by searching
044     * through the current stack trace for the invocation of the
045     * getLogger method of the class that provides loggers. The
046     * next frame in the stack trace then reveals the name of the
047     * class that requests the logger.
048     *
049     * @param providingClass the providing class
050     * @return the optional
051     */
052    private static Optional<Class<?>>
053            findRequestingClass(String providingClass) {
054        StackTraceElement[] stackTrace = new Throwable().getStackTrace();
055        for (int i = 2; i < stackTrace.length; i++) {
056            StackTraceElement ste = stackTrace[i];
057            if (ste.getClassName().equals(providingClass)
058                && ste.getMethodName().equals("getLogger")) {
059                Class<?>[] classes = CTX_HLPR.getClassContext();
060                // Next in stack should caller of getLogger(), but
061                // getClassContext() has added one level, so the
062                // caller of getLogger should be at i+2. But...
063                // from the JavaDoc: "Some virtual machines may, under
064                // some circumstances, omit one or more stack frames
065                // from the stack trace." So let's make sure that we
066                // are really there.
067                while (!classes[i + 1].getName().equals(providingClass)) {
068                    i += 1;
069                }
070                return Optional.of(classes[i + 2]);
071            }
072        }
073        return Optional.empty();
074    }
075
076    private static Optional<Bundle> findBundle(Class<?> clazz) {
077        if (clazz == null) {
078            return Optional.empty();
079        }
080        Bundle bundle = Optional.ofNullable(bundles.get(clazz))
081            .map(WeakReference::get).orElse(null);
082        if (bundle != null) {
083            return Optional.of(bundle);
084        }
085        bundle = FrameworkUtil.getBundle(clazz);
086        if (bundle != null) {
087            bundles.put(clazz, new WeakReference<>(bundle));
088            return Optional.of(bundle);
089        }
090        return Optional.empty();
091    }
092
093    /**
094     * Instantiates a new logger facade. The invoking bundle is determined
095     * from the class that invoked getLogger, which is searched for
096     * in the stacktrace as caller of the getLogger method of the class
097     * that provides the loggers from the users point of view.
098     * <P>
099     * Here, the provider is always {@link Lf4OsgiLogger}. But the name
100     * is provided as parameter in order to allow other facades to this
101     * as base class.
102     *
103     * @param name the name
104     * @param providingClass the providing class
105     */
106    public AbstractLoggerFacade(String name, String providingClass) {
107        this.name = name;
108        bundle = findRequestingClass(providingClass)
109            .flatMap(AbstractLoggerFacade::findBundle)
110            .orElse(null);
111        LogFacadeManager.addLoggerFacade(this);
112    }
113
114    /**
115     * Instantiates a new logger facade for the given bundle.
116     *
117     * @param bundle the bundle
118     * @param name the name
119     */
120    public AbstractLoggerFacade(Bundle bundle, String name) {
121        this.bundle = bundle;
122        this.name = name;
123        LogFacadeManager.addLoggerFacade(this);
124    }
125
126    /**
127     * Gets the name.
128     *
129     * @return the name
130     */
131    public String getName() {
132        return name;
133    }
134
135    /**
136     * Gets the bundle.
137     *
138     * @return the bundle
139     */
140    protected Bundle getBundle() {
141        return bundle;
142    }
143
144    /**
145     * Called when the logger factory changes. Derived classes
146     * must update the logger that they had previously obtained.
147     *
148     * @param factory the factory
149     */
150    public abstract void loggerFactoryUpdated(LoggerFactory factory);
151
152    /**
153     * The Class ContextHelper.
154     */
155    private static class ContextHelper extends SecurityManager {
156        @Override
157        @SuppressWarnings("PMD.UselessOverridingMethod")
158        public Class<?>[] getClassContext() {
159            return super.getClassContext();
160        }
161    }
162}
