017package de.mnl.osgi.lf4osgi.provider;
018
019import de.mnl.osgi.coreutils.ServiceResolver;
020
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.Collections;
024import java.util.Optional;
025import java.util.WeakHashMap;
026
027import org.osgi.service.log.LogLevel;
028import org.osgi.service.log.LoggerFactory;
029
030/**
031 * Keeps the loggers in the facades up-to-date.
032 */
033public class LoggerFacadeManager extends ServiceResolver {
034
035    @SuppressWarnings("PMD.FieldNamingConventions")
036    private static final Collection<LoggerFacade> facades
037        = Collections.newSetFromMap(new WeakHashMap<>());
038    private static BufferingLoggerFactory bufferingFactory
039        = new BufferingLoggerFactory();
040    private static LoggerFactory loggerFactory = bufferingFactory;
041
042    /**
043     * Register the given facade for receiving updates when the
044     * logger factory changes. The registration is automatically
045     * invoked by the constructor. It is also provided as method
046     * to support the usage of this class as delegee.
047     *
048     * @param loggerFacade the logger facade
049     */
050    public static void registerFacade(LoggerFacade loggerFacade) {
051        synchronized (LoggerFacadeManager.class) {
052            loggerFacade.loggerFactoryUpdated(loggerFactory);
053            facades.add(loggerFacade);
054        }
055    }
056
057    private void updateLoggerFactory(LoggerFactory factory) {
058        synchronized (LoggerFacadeManager.class) {
059            loggerFactory = factory;
060            for (LoggerFacade facade : new ArrayList<>(facades)) {
061                facade.loggerFactoryUpdated(loggerFactory);
062            }
063        }
064    }
065
066    @Override
067    protected void configure() {
068        Optional.ofNullable(
069            context.getProperty(BufferingLoggerFactory.LOG_THRESHOLD_PROPERTY))
070            .map(LogLevel::valueOf)
071            .ifPresent(th -> bufferingFactory.setThreshold(th));
072        Optional.ofNullable(
073            context.getProperty(BufferingLoggerFactory.BUFFER_SIZE_PROPERTY))
074            .map(Integer::parseInt)
075            .ifPresent(sz -> bufferingFactory.setBufferSize(sz));
076        addDependency(LoggerFactory.class);
077    }
078
079    @Override
080    protected void onResolved() {
081        LoggerFactory factory = get(LoggerFactory.class);
082        // Yes, it may happen that we get a new logging event
083        // before all buffered events have been flushed. But
084        // in order to prevent this, we'd have to put all
085        // logger usages in a synchronized block.
086        updateLoggerFactory(factory);
087        bufferingFactory.flush(factory);
088    }
089
090    @Override
091    protected void onRebound(String dependency) {
092        if (LoggerFactory.class.getName().equals(dependency)) {
093            updateLoggerFactory(get(LoggerFactory.class));
094        }
095    }
096
097    @Override
098    protected void onDissolving() {
099        updateLoggerFactory(bufferingFactory);
100    }
101
102}
