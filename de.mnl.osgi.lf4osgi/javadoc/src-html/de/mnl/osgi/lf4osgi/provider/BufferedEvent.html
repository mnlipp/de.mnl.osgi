045     * @param message the message
046     * @param arguments the parameters

048    public BufferedEvent(Bundle bundle, String name, LogLevel level,
049            String message, Object... arguments) {
050        this.bundle = bundle;

103    /**
104     * Forward the log event using the provided logger factory.
105     *
106     * @param factory the factory
107     */
108    @SuppressWarnings("PMD.AvoidLiteralsInIfCondition")
109    public void forward(LoggerFactory factory) {
110        final Logger logger = factory.getLogger(bundle, name, Logger.class);
111        switch (level) {
112        case TRACE:
113            if (arguments.length == 0) {
114                logger.trace(l -> l.trace(message));
115            } else {
116                logger.trace(l -> l.trace(message, arguments));
117            }
118            break;
119        case DEBUG:
120            if (arguments.length == 0) {
121                logger.debug(l -> l.debug(message));
122            } else {
123                logger.debug(l -> l.debug(message, arguments));
124            }
125            break;
126        case INFO:
127            if (arguments.length == 0) {
128                logger.info(l -> l.info(message));
129            } else {
130                logger.info(l -> l.info(message, arguments));
131            }
132            break;
133        case WARN:
134            if (arguments.length == 0) {
135                logger.warn(l -> l.warn(message));
136            } else {
137                logger.warn(l -> l.warn(message, arguments));
138            }
139            break;
140        case ERROR:
141            if (arguments.length == 0) {
142                logger.error(l -> l.error(message));
143            } else {
144                logger.error(l -> l.error(message, arguments));
145            }
146            break;
147        case AUDIT:
148            if (arguments.length == 0) {
149                logger.audit(message);
150            } else {
151                logger.audit(message, arguments);
152            }
153            break;
154        }
155    }
