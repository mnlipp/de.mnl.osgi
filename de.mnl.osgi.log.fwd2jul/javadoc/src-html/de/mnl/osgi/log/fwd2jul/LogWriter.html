017package de.mnl.osgi.log.fwd2jul;
018
019import java.util.logging.Level;
020import java.util.logging.LogRecord;
021import java.util.logging.Logger;
022import org.osgi.framework.Bundle;
023import org.osgi.service.log.LogEntry;
024import org.osgi.service.log.LogListener;
025import org.osgi.service.log.LogService;
026
027/**
028 * This class implements a LogListener that forwards the LogEntries .
029 *
030 * @deprecated Replaced by {@code de.mnl.osgi.osgi2jul}.
031 */
032@Deprecated
033public class LogWriter implements LogListener {
034
035    /*
036     * (non-Javadoc)
037     * 
038     * @see
039     * org.osgi.service.log.LogListener#logged(org.osgi.service.log.LogEntry)
040     */
041  @Override
042    public void logged(LogEntry entry) {
043        final Logger logger = Logger.getLogger("OSGi Logger");
044        Level level = Level.OFF;
045        switch (entry.getLevel()) {
046        case LogService.LOG_DEBUG:
047            level = Level.FINE;
048            break;
049        case LogService.LOG_INFO:
050            level = Level.INFO;
051            break;
052        case LogService.LOG_WARNING:
053            level = Level.WARNING;
054            break;
055        case LogService.LOG_ERROR:
056            level = Level.SEVERE;
057            break;
058        }
059        LogRecord record = new LogRecord(level, entry.getMessage());
060        Throwable trw = entry.getException();
061        if (trw != null) {
062            record.setThrown(trw);
063        }
064        Bundle bundle = entry.getBundle();
065        if (bundle != null) {
066            record.setSourceClassName(entry.getBundle().getSymbolicName());
067        }
068        logger.log(record);
069    }
070
071}
