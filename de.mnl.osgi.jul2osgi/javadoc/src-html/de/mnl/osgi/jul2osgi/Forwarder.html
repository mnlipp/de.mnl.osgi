051/**
052 */
053@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
054public class Forwarder implements BundleActivator, LogRecordHandler {
055
056    private ServiceCollector<LogService> logSvcTracker;
057    private ServiceTracker<LoggerAdmin, LoggerAdmin> logAdmTracker;
058    private String logPattern;
059    private boolean adaptOsgiLevel = true;
060    private final Map<Class<?>, WeakReference<Bundle>> bundles
061        = new ConcurrentHashMap<>();
062    private final Set<Bundle> adaptedBundles = Collections
063        .synchronizedSet(Collections.newSetFromMap(new WeakHashMap<>()));
064
065    @Override
066    @SuppressWarnings({ "PMD.SystemPrintln", "resource" })
067    public void start(BundleContext context) throws Exception {
068        final java.util.logging.LogManager logMgr
069            = java.util.logging.LogManager.getLogManager();
070        if (!(logMgr instanceof LogManager)) {
071            System.err.println("Configuration error: "
072                + "Bundle de.mnl.osgi.jul2osgi must be used with "
073                + "LogManager from de.mnl.osgi.jul2osgi.log.");
074            return;
075        }
076        logPattern = Optional.ofNullable(
077            context.getProperty("de.mnl.osgi.jul2osgi.logPattern"))
078            .orElse("{0}");
079        String adaptProperty = context.getProperty(
080            "de.mnl.osgi.jul2osgi.adaptOSGiLevel");
081        if (adaptProperty != null) {
082            adaptOsgiLevel = Boolean.parseBoolean(adaptProperty);
083        }
084
085        // Create the admin tracker
086        logAdmTracker = new ServiceTracker<LoggerAdmin, LoggerAdmin>(
087            context, LoggerAdmin.class, null) {
088            @Override
089            public LoggerAdmin addingService(
090                    ServiceReference<LoggerAdmin> reference) {
091                LoggerAdmin adm = super.addingService(reference);
092                // Handle this bundle specially
093                if (adaptOsgiLevel) {
094                    adaptLogLevel(adm, context.getBundle());
095                }
096                return adm;
097            }
098        };
099        logAdmTracker.open();
100
101        // Create new service tracker.
102        logSvcTracker = new ServiceCollector<LogService>(
103            context, LogService.class)
104                .setOnBound((ref, svc) -> ((LogManager) logMgr)
105                    .setForwarder(this))
106                .setOnLastUnbinding((ref, svc) -> ((LogManager) logMgr)
107                    .setForwarder(this));
108        logSvcTracker.open();
109    }
110
111    private void adaptLogLevel(LoggerAdmin logAdmin, Bundle bundle) {
112        if (adaptedBundles.contains(bundle)) {
113            return;
114        }
115        LoggerContext ctx = logAdmin.getLoggerContext(
116            bundle.getSymbolicName() + "|" + bundle.getVersion());
117        @SuppressWarnings("PMD.UseConcurrentHashMap")
118        Map<String, LogLevel> logLevels = new HashMap<>();
119        logLevels.put(Logger.ROOT_LOGGER_NAME, LogLevel.TRACE);
120        ctx.setLogLevels(logLevels);
121        adaptedBundles.add(bundle);
122    }
123
124    @Override
125    public void stop(BundleContext context) throws Exception {
126        java.util.logging.LogManager logMgr
127            = java.util.logging.LogManager.getLogManager();
128        if (logMgr instanceof LogManager) {
129            ((LogManager) logMgr).setForwarder(null);
130        }
131        logSvcTracker.close();
132        logAdmTracker.close();
133    }
134
135    @Override
136    public boolean process(LogInfo logInfo) {
137        return logSvcTracker.service().map(svc -> {
138            doProcess(logInfo, svc);
139            return true;
140        }).orElse(false);
141    }
142
143    private void doProcess(LogInfo logInfo, LogService service) {
144        final LogRecord record = logInfo.getLogRecord();
145        final String loggerName = Optional.ofNullable(record.getLoggerName())
146            .orElse(Logger.ROOT_LOGGER_NAME);
147        Logger logger = findBundle(logInfo.getCallingClass())
148            .map(b -> {
149                if (adaptOsgiLevel) {
150                    Optional.ofNullable(logAdmTracker.getService())
151                        .ifPresent(adm -> adaptLogLevel(adm, b));
152                }
153                return service.getLogger(b, loggerName, Logger.class);
154            }).orElse(service.getLogger(loggerName));
155
156        int julLevel = record.getLevel().intValue();
157        if (julLevel >= Level.SEVERE.intValue()) {
158            if (logger.isErrorEnabled()) {
159                logger.error("{}", formatMessage(logPattern, record),
160                    record.getThrown());
161            }
162        } else if (julLevel >= Level.WARNING.intValue()) {
163            if (logger.isWarnEnabled()) {
164                logger.warn("{}", formatMessage(logPattern, record),
165                    record.getThrown());
166            }
167        } else if (julLevel >= Level.INFO.intValue()) {
168            if (logger.isInfoEnabled()) {
169                logger.info("{}", formatMessage(logPattern, record),
170                    record.getThrown());
171            }
172        } else if (julLevel >= Level.FINE.intValue()) {
173            if (logger.isDebugEnabled()) {
174                logger.debug("{}", formatMessage(logPattern, record),
175                    record.getThrown());
176            }
177        } else if (logger.isTraceEnabled()) {
178            logger.trace("{}", formatMessage(logPattern, record),
179                record.getThrown());
180        }
181    }
182
183    private Optional<Bundle> findBundle(Class<?> callingClass) {
184        if (callingClass == null) {
185            return Optional.empty();
186        }
187        Bundle bundle = Optional.ofNullable(bundles.get(callingClass))
188            .map(WeakReference::get).orElse(null);
189        if (bundle != null) {
190            return Optional.of(bundle);
191        }
192        bundle = FrameworkUtil.getBundle(callingClass);
193        if (bundle != null) {
194            bundles.put(callingClass, new WeakReference<>(bundle)); // NOPMD
195            return Optional.of(bundle);
196        }
197        return Optional.empty();
198    }
199
200    private String formatMessage(String format, LogRecord record) {
201        String message = record.getMessage();
202        if (record.getResourceBundle() != null) {
203            try {
204                message = record.getResourceBundle().getString(message);
205            } catch (MissingResourceException e) { // NOPMD
206                // Leave message as it is
207            }
208        }
209        Object[] parameters = record.getParameters();
210        if (parameters != null && parameters.length > 0) {
211            message = MessageFormat.format(message, record.getParameters());
212        }
213        return MessageFormat.format(format, message, record.getMillis(),
214            record.getSequenceNumber(), record.getSourceClassName(),
215            record.getSourceMethodName(), record.getThreadID());
216    }
217
218    /**
219     * Process events that are delivered .
220     *
221     * @param logInfos the log infos
222     */
223    @SuppressWarnings({ "PMD.EmptyCatchBlock", "PMD.UseVarargs" })
224    public void processBuffered(LogInfo[] logInfos) {
225        String threadName = Thread.currentThread().getName();
226        try {
227            // Set thread name to indicate invalid context
228            try {
229                AccessController.doPrivileged(new PrivilegedAction<Void>() {
230                    @Override
231                    public Void run() {
232                        Thread.currentThread().setName("(log flusher)");
233                        return null;
234                    }
235                });
236            } catch (SecurityException e) {
237                // Ignored, was just a best effort.
238            }
239            // Now process
240            for (LogInfo info : logInfos) {
241                process(info);
242            }
243        } finally {
244            try {
245                AccessController.doPrivileged(new PrivilegedAction<Void>() {
246
247                    @Override
248                    public Void run() {
249                        Thread.currentThread().setName(threadName);
250                        return null; // NOPMD
251                    }
252                });
253            } catch (SecurityException e) {
254                // Ignored. If resetting doesn't work, setting hasn't worked
255                // neither
256            }
257        }
258    }
