017package de.mnl.osgi.jul2osgi;
018
019import de.mnl.osgi.jul2osgi.lib.LogManager;
020import de.mnl.osgi.jul2osgi.lib.LogRecordHandler;
021
022import java.text.MessageFormat;
023import java.util.HashMap;
024import java.util.Map;
025import java.util.Optional;
026import java.util.logging.Level;
027import java.util.logging.LogRecord;
028
029import org.osgi.framework.BundleActivator;
030import org.osgi.framework.BundleContext;
031import org.osgi.framework.ServiceReference;
032import org.osgi.service.log.LogLevel;
033import org.osgi.service.log.LogService;
034import org.osgi.service.log.Logger;
035import org.osgi.service.log.admin.LoggerAdmin;
036import org.osgi.service.log.admin.LoggerContext;
037import org.osgi.util.tracker.ServiceTracker;
038
039/**
040 */
041public class Forwarder implements BundleActivator, LogRecordHandler {
042
043        private ServiceTracker<LogService, LogService> logSvcTracker;
044        private ServiceTracker<LoggerAdmin, LoggerAdmin> logAdmTracker;
045        private String logPattern;
046        private LogLevel contextLevel;
047
048    @Override
049        public void start(BundleContext context) throws Exception {
050        final java.util.logging.LogManager lm
051                        = java.util.logging.LogManager.getLogManager();
052                if (!(lm instanceof LogManager)) {
053                        System.err.println("Configuration error: "
054                                        + "Bundle de.mnl.osgi.jul2osgi must be used with "
055                                        + "LogManager from de.mnl.osgi.jul2osgi.log.");
056                        return;
057                }
058                logPattern = Optional.ofNullable(
059                                context.getProperty("de.mnl.osgi.jul2osgi.logPattern"))
060                                .orElse("{0}");
061                String ctxProperty = context.getProperty(
062                                "de.mnl.osgi.jul2osgi.contextLevel");
063                if (ctxProperty == null) {
064                        contextLevel = LogLevel.TRACE;
065                } else {
066                        if (!ctxProperty.equals("NONE")) {
067                                try {
068                                        contextLevel = LogLevel.valueOf(ctxProperty);
069                                } catch (IllegalArgumentException e) {
070                                    // Ignored if invalid.
071                                }
072                        }
073                }
074                
075                // Create new service tracker.
076                logSvcTracker = new ServiceTracker<LogService, LogService>(
077                                context, LogService.class, null) {
078                        @Override
079                        public LogService addingService(
080                                        ServiceReference<LogService> reference) {
081                                ((LogManager)lm).setForwarder(Forwarder.this);
082                                return super.addingService(reference);
083                        }
084
085                        @Override
086                        public void removedService(ServiceReference<LogService> reference,
087                                        LogService service) {
088                                // TODO Auto-generated method stub
089                                super.removedService(reference, service);
090                                if (getService() == null) {
091                                        ((LogManager)lm).setForwarder(null);
092                                }
093                        }
094                };
095        logSvcTracker.open();
096                // Create the admin tracker
097                logAdmTracker = new ServiceTracker<LoggerAdmin, LoggerAdmin>(
098                                context, LoggerAdmin.class, null) {
099                        @Override
100                        public LoggerAdmin addingService(
101                                        ServiceReference<LoggerAdmin> reference) {
102                                LoggerAdmin adm = super.addingService(reference);
103                                if (contextLevel != null) {
104                                        LoggerContext ctx = adm.getLoggerContext(
105                                                        Forwarder.class.getPackage().getName());
106                                        Map<String,LogLevel> logLevels = new HashMap<>();
107                                        logLevels.put(Logger.ROOT_LOGGER_NAME, contextLevel);
108                                        ctx.setLogLevels(logLevels);
109                                }
110                                return adm;
111                        }
112                };
113                logAdmTracker.open();
114        }
115
116        @Override
117        public void stop(BundleContext context) throws Exception {
118                java.util.logging.LogManager lm
119                        = java.util.logging.LogManager.getLogManager();
120                if (lm instanceof LogManager) {
121                        ((LogManager)lm).setForwarder(null);
122                }
123                logSvcTracker.close();
124                logAdmTracker.close();
125        }
126
127        @Override
128        public boolean process(LogRecord record) {
129                LogService service = logSvcTracker.getService();
130                if (service == null) {
131                        return false;
132                }
133                Logger logger = service.getLogger(
134                                Optional.ofNullable(record.getLoggerName())
135                                .orElse(Logger.ROOT_LOGGER_NAME));
136                int julLevel = record.getLevel().intValue();
137                if (julLevel >= Level.SEVERE.intValue()) {
138                        if (logger.isErrorEnabled()) {
139                                logger.error("{}", formatMessage(logPattern, record), 
140                                                record.getThrown());
141                        }
142                } else if (julLevel >= Level.WARNING.intValue()) {
143                        if (logger.isWarnEnabled()) {
144                                logger.warn("{}", formatMessage(logPattern, record), 
145                                                record.getThrown());
146                        }
147                } else if (julLevel >= Level.INFO.intValue()) {
148                        if (logger.isInfoEnabled()) {
149                                logger.info("{}", formatMessage(logPattern, record), 
150                                                record.getThrown());
151                        }
152                } else if (julLevel >= Level.FINE.intValue()) {
153                        if (logger.isDebugEnabled()) {
154                                logger.debug("{}", formatMessage(logPattern, record), 
155                                                record.getThrown());
156                        }
157                } else if (logger.isTraceEnabled()) {
158                        logger.trace("{}", formatMessage(logPattern, record), 
159                                        record.getThrown());
160                } 
161                return true;
162        }
163
164        private String formatMessage(String format, LogRecord record) {
165                String message = record.getMessage();
166                if (record.getResourceBundle() != null
167                                && record.getResourceBundle().containsKey(message)) {
168                        message = record.getResourceBundle().getString(message);
169                }
170                message = MessageFormat.format(message, record.getParameters());
171                return MessageFormat.format(format, message, record.getMillis(),
172                                        record.getSequenceNumber(), record.getSourceClassName(),
173                                        record.getSourceMethodName(), record.getThreadID());
174        }
175}
