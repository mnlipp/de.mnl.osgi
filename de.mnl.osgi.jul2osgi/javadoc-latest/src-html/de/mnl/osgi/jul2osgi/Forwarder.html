023import java.lang.ref.WeakReference;
024import java.security.AccessController;
025import java.security.PrivilegedAction;
026import java.text.MessageFormat;
027import java.util.Collections;
028import java.util.HashMap;
029import java.util.Map;
030import java.util.MissingResourceException;
031import java.util.Optional;
032import java.util.Set;
033import java.util.WeakHashMap;
034import java.util.concurrent.ConcurrentHashMap;
035import java.util.logging.Level;
036import java.util.logging.LogRecord;
037import org.osgi.framework.Bundle;
038import org.osgi.framework.BundleContext;
039import org.osgi.framework.FrameworkUtil;
040import org.osgi.service.log.LogLevel;
041import org.osgi.service.log.LogService;
042import org.osgi.service.log.Logger;
043import org.osgi.service.log.admin.LoggerAdmin;
044import org.osgi.service.log.admin.LoggerContext;

046/**
047 */
048@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
049public class Forwarder extends ServiceResolver implements LogRecordHandler {
050
051    private String logPattern;
052    private boolean adaptOsgiLevel = true;
053    private final Map<Class<?>, WeakReference<Bundle>> bundles
054        = new ConcurrentHashMap<>();
055    private final Set<Bundle> adaptedBundles = Collections
056        .synchronizedSet(Collections.newSetFromMap(new WeakHashMap<>()));
057
058    @Override
059    @SuppressWarnings({ "PMD.SystemPrintln" })
060    protected void configure() {
061        final java.util.logging.LogManager logMgr
062            = java.util.logging.LogManager.getLogManager();
063        if (!(logMgr instanceof LogManager)) {
064            System.err.println("Configuration error: "
065                + "Bundle de.mnl.osgi.jul2osgi must be used with "
066                + "LogManager from de.mnl.osgi.jul2osgi.log.");
067            return;
068        }
069        logPattern = Optional.ofNullable(
070            context.getProperty("de.mnl.osgi.jul2osgi.logPattern"))
071            .orElse("{0}");
072        String adaptProperty = context.getProperty(
073            "de.mnl.osgi.jul2osgi.adaptOSGiLevel");
074        if (adaptProperty != null) {
075            adaptOsgiLevel = Boolean.parseBoolean(adaptProperty);
076        }
077
078        addDependency(LogService.class);
079        addDependency(LoggerAdmin.class);
080        setOnResolved(() -> {
081            if (adaptOsgiLevel) {
082                // Handle this bundle specially
083                adaptLogLevel(get(LoggerAdmin.class),
084                    context.getBundle());
085            }
086            ((LogManager) logMgr).setForwarder(this);
087        });
088        setOnDissolving(() -> {
089            ((LogManager) logMgr).setForwarder(null);
090            adaptedBundles.clear();
091        });
092    }
093
094    @Override
095    public void stop(BundleContext context) throws Exception {
096        java.util.logging.LogManager logMgr
097            = java.util.logging.LogManager.getLogManager();
098        if (logMgr instanceof LogManager) {
099            ((LogManager) logMgr).setForwarder(null);
100        }
101        super.stop(context);
102    }
103
104    private void adaptLogLevel(LoggerAdmin logAdmin, Bundle bundle) {
105        if (adaptedBundles.contains(bundle)) {
106            return;
107        }
108        LoggerContext ctx = logAdmin.getLoggerContext(
109            bundle.getSymbolicName() + "|" + bundle.getVersion());
110        @SuppressWarnings("PMD.UseConcurrentHashMap")
111        Map<String, LogLevel> logLevels = new HashMap<>();
112        logLevels.put(Logger.ROOT_LOGGER_NAME, LogLevel.TRACE);
113        ctx.setLogLevels(logLevels);
114        adaptedBundles.add(bundle);
115    }
116
117    @Override
118    public boolean process(LogInfo logInfo) {
119        LogService logSvc = get(LogService.class);
120        if (logSvc == null) {
121            return false;
122        }
123        doProcess(logInfo, logSvc);
124        return true;
125    }
126
127    private void doProcess(LogInfo logInfo, LogService service) {
128        final LogRecord record = logInfo.getLogRecord();
129        final String loggerName = Optional.ofNullable(record.getLoggerName())
130            .orElse(Logger.ROOT_LOGGER_NAME);
131        Logger logger = findBundle(logInfo.getCallingClass())
132            .map(b -> {
133                if (adaptOsgiLevel) {
134                    adaptLogLevel(get(LoggerAdmin.class), b);
135                }
136                return service.getLogger(b, loggerName, Logger.class);
137            }).orElse(service.getLogger(loggerName));
138
139        int julLevel = record.getLevel().intValue();
140        if (julLevel >= Level.SEVERE.intValue()) {
141            if (logger.isErrorEnabled()) {
142                logger.error("{}", formatMessage(logPattern, record),
143                    record.getThrown());
144            }
145        } else if (julLevel >= Level.WARNING.intValue()) {
146            if (logger.isWarnEnabled()) {
147                logger.warn("{}", formatMessage(logPattern, record),
148                    record.getThrown());
149            }
150        } else if (julLevel >= Level.INFO.intValue()) {
151            if (logger.isInfoEnabled()) {
152                logger.info("{}", formatMessage(logPattern, record),
153                    record.getThrown());
154            }
155        } else if (julLevel >= Level.FINE.intValue()) {
156            if (logger.isDebugEnabled()) {
157                logger.debug("{}", formatMessage(logPattern, record),
158                    record.getThrown());
159            }
160        } else if (logger.isTraceEnabled()) {
161            logger.trace("{}", formatMessage(logPattern, record),
162                record.getThrown());
163        }
164    }
165
166    private Optional<Bundle> findBundle(Class<?> callingClass) {
167        if (callingClass == null) {
168            return Optional.empty();
169        }
170        Bundle bundle = Optional.ofNullable(bundles.get(callingClass))
171            .map(WeakReference::get).orElse(null);
172        if (bundle != null) {
173            return Optional.of(bundle);
174        }
175        bundle = FrameworkUtil.getBundle(callingClass);
176        if (bundle != null) {
177            bundles.put(callingClass, new WeakReference<>(bundle)); // NOPMD
178            return Optional.of(bundle);
179        }
180        return Optional.empty();
181    }
182
183    private String formatMessage(String format, LogRecord record) {
184        String message = record.getMessage();
185        if (record.getResourceBundle() != null) {
186            try {
187                message = record.getResourceBundle().getString(message);
188            } catch (MissingResourceException e) { // NOPMD
189                // Leave message as it is
190            }
191        }
192        Object[] parameters = record.getParameters();
193        if (parameters != null && parameters.length > 0) {
194            message = MessageFormat.format(message, record.getParameters());
195        }
196        return MessageFormat.format(format, message, record.getMillis(),
197            record.getSequenceNumber(), record.getSourceClassName(),
198            record.getSourceMethodName(), record.getThreadID());
199    }
200
201    /**
202     * Process events that are delivered .
203     *
204     * @param logInfos the log infos
205     */
206    @SuppressWarnings({ "PMD.EmptyCatchBlock", "PMD.UseVarargs",
207        "PMD.AvoidInstantiatingObjectsInLoops" })
208    public void processBuffered(LogInfo[] logInfos) {
209        String threadName = Thread.currentThread().getName();
210        for (LogInfo info : logInfos) {
211            try {
212                // Set thread name to indicate invalid context
213                try {
214                    AccessController.doPrivileged(new PrivilegedAction<Void>() {
215                        @Override
216                        public Void run() {
217                            Thread.currentThread()
218                                .setName(info.getThreadName() + " [recorded]");
219                            return null;
220                        }
221                    });
222                } catch (SecurityException e) {
223                    // Ignored, was just a best effort.
224                }
225                // Now process
226                process(info);
227            } finally {
228                try {
229                    AccessController.doPrivileged(new PrivilegedAction<Void>() {
230
231                        @Override
232                        public Void run() {
233                            Thread.currentThread().setName(threadName);
234                            return null; // NOPMD
235                        }
236                    });
237                } catch (SecurityException e) {
238                    // Ignored. If resetting doesn't work, setting hasn't worked
239                    // neither
240                }
241            }
242        }
243    }
