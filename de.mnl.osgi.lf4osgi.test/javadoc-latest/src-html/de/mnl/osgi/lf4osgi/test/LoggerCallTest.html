028import static org.junit.jupiter.api.Assertions.assertEquals;
029import static org.junit.jupiter.api.Assertions.assertNotNull;
030import static org.junit.jupiter.api.Assertions.assertTrue;

034import org.osgi.framework.Bundle;
035import org.osgi.framework.BundleContext;
036import org.osgi.framework.BundleException;
037import org.osgi.framework.FrameworkUtil;
038import org.osgi.service.log.LogEntry;
039import org.osgi.service.log.LogLevel;
040import org.osgi.service.log.LogListener;
041import org.osgi.service.log.LogReaderService;
042
043@ExtendWith(MockitoExtension.class)
044public class LoggerCallTest {
045
046    protected static final Logger logger
047        = LoggerFactory.getLogger("Logger Call Test");
048
049    private final BundleContext context = FrameworkUtil
050        .getBundle(LoggerCallTest.class).getBundleContext();
051
052    @Test
053    public void testBasicUsage() throws InterruptedException {
054        AtomicBoolean gotIt = new AtomicBoolean();
055        try (ServiceCollector<LogReaderService,
056                LogReaderService> logReaderProvider
057                    = new ServiceCollector<>(context, LogReaderService.class)) {
058            logReaderProvider.open();
059            LogReaderService logReader
060                = logReaderProvider.waitForService(1000).get();
061            CountDownLatch latch = new CountDownLatch(1);
062            logReader.addLogListener(new LogListener() {
063                @Override
064                public void logged(LogEntry entry) {
065                    if (entry.getMessage()
066                        .startsWith("Calling Logger from Test.")) {
067                        assertEquals("de.mnl.osgi.lf4osgi.test",
068                            entry.getBundle().getSymbolicName());
069                        gotIt.set(true);
070                        latch.countDown();
071                    }
072                }
073            });
074            logger.info("Calling Logger from {}.", "Test");
075            latch.await(1000, TimeUnit.MILLISECONDS);
076            assertTrue(gotIt.get());
077        }
078    }
079
080    @Test
081    public void testBuffered() throws InterruptedException, BundleException {
082        Bundle osgiLoggerFactory = context.getServiceReference(
083            org.osgi.service.log.LoggerFactory.class).getBundle();
084        assertNotNull(osgiLoggerFactory);
085        try {
086            osgiLoggerFactory.stop();
087            logger.info("Calling Logger {}", "Info");
088            logger.warn(() -> "Calling Logger " + "Warn");
089            logger.error("Calling Logger Error", new Throwable());
090            osgiLoggerFactory.start();
091            try (ServiceCollector<LogReaderService,
092                    LogReaderService> logReaderProvider
093                        = new ServiceCollector<>(context,
094                            LogReaderService.class)) {
095                logReaderProvider.open();
096                LogReaderService logReader
097                    = logReaderProvider.waitForService(1000).get();
098                CountDownLatch latch = new CountDownLatch(3);
099                for (LogEntry entry : Collections.list(logReader.getLog())) {
100                    if (entry.getMessage().startsWith("Calling Logger Info")
101                        && entry.getLogLevel() == LogLevel.INFO) {
102                        latch.countDown();
103                    }
104                    if (entry.getMessage().startsWith("Calling Logger Warn")
105                        && entry.getLogLevel() == LogLevel.WARN) {
106                        latch.countDown();
107                    }
108                    if (entry.getMessage().startsWith("Calling Logger Error")
109                        && entry.getLogLevel() == LogLevel.ERROR
110                        && entry.getException() != null) {
111                        latch.countDown();
112                    }
113                }
114                latch.await(1000, TimeUnit.MILLISECONDS);
115                assertEquals(0, latch.getCount());
116            }
117        } finally {
118            osgiLoggerFactory.start();
119        }
120    }
121

129        assertTrue(logger1 == logger2);
130    }
