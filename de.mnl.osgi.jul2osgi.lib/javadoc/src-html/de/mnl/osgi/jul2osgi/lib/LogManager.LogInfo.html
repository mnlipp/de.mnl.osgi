017package de.mnl.osgi.jul2osgi.lib;
018
019import java.util.ArrayList;
020import java.util.Deque;
021import java.util.LinkedList;
022import java.util.List;
023import java.util.logging.LogRecord;
024import java.util.logging.Logger;
025
026/**
027 * An implementation of {@link java.util.logging.LogManager}. The 
028 * system property {@code java.util.logging.manager} must be set to 
029 * {@code de.mnl.osgi.jul2osgi.LogManager} in order for this manager to be
030 * used. The library that provides this implementation must be
031 * on the system classpath (e.g. using {@code bnd}'s
032 * <a href=https://bnd.bndtools.org/instructions/runpath.html>{@code -runpath}</a>
033 * instruction. 
034 */
035public class LogManager extends java.util.logging.LogManager {
036
037    private final Deque<LogInfo> buffered = new LinkedList<>();
038    private int bufferSize = 100;
039    private LogRecordHandler forwarder;
040
041    /**
042     * Instantiates a new log manager.
043     */
044    @SuppressWarnings("PMD.EmptyCatchBlock")
045    public LogManager() {
046        try {
047            bufferSize = Integer.parseInt(System.getProperty(
048                "de.mnl.osgi.jul2osgi.bufferSize", "100"));
049        } catch (NumberFormatException e) {
050            // Left to default if invaid.
051        }
052    }
053
054    /*
055     * (non-Javadoc)
056     * 
057     * @see java.util.logging.LogManager#getLogger(java.lang.String)
058     */
059    @Override
060    @SuppressWarnings("PMD.UselessOverridingMethod")
061    public Logger getLogger(final String name) {
062        return super.getLogger(name);
063    }
064
065    /*
066     * (non-Javadoc)
067     * 
068     * @see java.util.logging.LogManager#addLogger(java.util.logging.Logger)
069     */
070    @Override
071    public boolean addLogger(Logger logger) {
072        // It's hard to believe but this effectively overwrites the
073        // Logger#getLogger factory methods.
074        if (!(logger instanceof ForwardingLogger)) {
075            logger = new ForwardingLogger(this, logger);
076        }
077        super.addLogger(logger);
078        return false;
079    }
080
081    /**
082     * Sets the forwarder.
083     *
084     * @param forwarder the new forwarder
085     */
086    public void setForwarder(LogRecordHandler forwarder) {
087        // It may be that the forwarder is used by a call to
088        // #log before all records from the buffer have been
089        // flushed, thus changing the proper sequence.
090
091        // Protect the forwarder from being changed while used in #log
092        synchronized (this) {
093            this.forwarder = forwarder;
094            if (forwarder == null) {
095                return;
096            }
097            List<LogInfo> tbf;
098            synchronized (buffered) {
099                tbf = new ArrayList<>(buffered);
100                buffered.clear();
101            }
102            forwarder.processBuffered(tbf.toArray(new LogInfo[0]));
103        }
104    }
105
106    /**
107     * Store or forward the logged information.
108     *
109     * @param logInfo the information
110     */
111    public void log(LogInfo logInfo) {
112        // Lock the forwarder. May not change while being used.
113        synchronized (this) {
114            if (forwarder != null && forwarder.process(logInfo)) {
115                return;
116            }
117        }
118        logInfo.setThreadName(Thread.currentThread().getName());
119        // We need to infer this information now, won't be possible later.
120        logInfo.getLogRecord().getSourceClassName();
121        synchronized (buffered) {
122            if (buffered.size() == bufferSize) {
123                buffered.removeFirst();
124            }
125            buffered.add(logInfo);
126        }
127    }
128
129    /**
130     * Holds the information from a logger invocation.
131     */
132    @SuppressWarnings("PMD.DataClass")
133    public static class LogInfo {
134        private final Class<?> callingClass;
135        private final LogRecord logRecord;
136        private String threadName;
137
138        /**
139         * Instantiates a new log info.
140         *
141         * @param definingClass the calling class
142         * @param logRecord the log record
143         */
144        public LogInfo(Class<?> definingClass, LogRecord logRecord) {
145            super();
146            this.callingClass = definingClass;
147            this.logRecord = logRecord;
148        }
149
150        /**
151         * Sets the thread name. Only required if the log info
152         * has to be buffered.
153         *
154         * @param threadName the new thread name
155         */
156        public void setThreadName(String threadName) {
157            this.threadName = threadName;
158        }
159
160        /**
161         * Gets the thread name.
162         *
163         * @return the threadName
164         */
165        public final String getThreadName() {
166            return threadName;
167        }
168
169        /**
170         * Gets the calling class.
171         *
172         * @return the callingClass
173         */
174        public Class<?> getCallingClass() {
175            return callingClass;
176        }
177
178        /**
179         * Gets the log record.
180         *
181         * @return the logRecord
182         */
183        public LogRecord getLogRecord() {
184            return logRecord;
185        }
186
187    }
188}
