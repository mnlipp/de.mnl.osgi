017package de.mnl.osgi.jul2osgi.lib;
018
019import java.util.ArrayList;
020import java.util.Deque;
021import java.util.LinkedList;
022import java.util.List;
023import java.util.logging.LogRecord;
024import java.util.logging.Logger;
025
026/**
027 * An implementation of {@link java.util.logging.LogManager}. The 
028 * system property {@code java.util.logging.manager} must be set to 
029 * {@code de.mnl.osgi.jul2osgi.LogManager} in order for this manager to be
030 * used. The library that provides this implementation must be
031 * on the system classpath (e.g. using {@code bnd}'s
032 * <a href=https://bnd.bndtools.org/instructions/runpath.html>{@code -runpath}</a>
033 * instruction. 
034 */
035public class LogManager extends java.util.logging.LogManager {
036
037    private Deque<LogInfo> buffered = new LinkedList<>();
038    private int bufferSize = 100;
039    private LogRecordHandler forwarder;
040
041    /**
042     * Instantiates a new log manager.
043     */
044    public LogManager() {
045        try {
046            bufferSize = Integer.parseInt(System.getProperty(
047                "de.mnl.osgi.jul2osgi.bufferSize", "100"));
048        } catch (NumberFormatException e) {
049            // Left to default if invaid.
050        }
051    }
052
053    /*
054     * (non-Javadoc)
055     * 
056     * @see java.util.logging.LogManager#getLogger(java.lang.String)
057     */
058    @Override
059    public Logger getLogger(final String name) {
060        Logger logger = super.getLogger(name);
061        return logger;
062    }
063
064    /*
065     * (non-Javadoc)
066     * 
067     * @see java.util.logging.LogManager#addLogger(java.util.logging.Logger)
068     */
069    @Override
070    public boolean addLogger(Logger logger) {
071        // It's hard to believe but this effectively overwrites the
072        // Logger#getLogger factory methods.
073        if (!(logger instanceof ForwardingLogger)) {
074            logger = new ForwardingLogger(this, logger);
075        }
076        super.addLogger(logger);
077        return false;
078    }
079
080    /**
081     * Store or forward the logged information.
082     *
083     * @param logInfo the information
084     */
085    public void log(LogInfo logInfo) {
086        LogRecordHandler fwd = forwarder;
087        if (fwd != null && fwd.process(logInfo)) {
088            return;
089        }
090        synchronized (buffered) {
091            if (buffered.size() == bufferSize) {
092                buffered.removeFirst();
093            }
094            buffered.add(logInfo);
095        }
096    }
097
098    /**
099     * Sets the forwarder.
100     *
101     * @param forwarder the new forwarder
102     */
103    public void setForwarder(LogRecordHandler forwarder) {
104        // It may be that the forwarder is used by a call to
105        // #log before all records from the buffer have been
106        // flushed, thus changing the proper sequence.
107        synchronized (this) {
108            this.forwarder = forwarder;
109            if (forwarder == null) {
110                return;
111            }
112            List<LogInfo> tbf;
113            synchronized (buffered) {
114                tbf = new ArrayList<>(buffered);
115                buffered.clear();
116            }
117            forwarder.processBuffered(tbf.toArray(new LogInfo[0]));
118        }
119    }
120
121    /**
122     * Holds the information from a logger invocation.
123     */
124    public static class LogInfo {
125        private Class<?> callingClass;
126        private LogRecord logRecord;
127
128        /**
129         * Instantiates a new log info.
130         *
131         * @param definingClass the calling class
132         * @param logRecord the log record
133         */
134        public LogInfo(Class<?> definingClass, LogRecord logRecord) {
135            super();
136            this.callingClass = definingClass;
137            this.logRecord = logRecord;
138        }
139
140        /**
141         * Gets the calling class.
142         *
143         * @return the callingClass
144         */
145        public Class<?> getCallingClass() {
146            return callingClass;
147        }
148
149        /**
150         * Gets the log record.
151         *
152         * @return the logRecord
153         */
154        public LogRecord getLogRecord() {
155            return logRecord;
156        }
157
158    }
159}
